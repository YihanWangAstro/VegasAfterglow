10template <typename Ejecta, typename Medium>
11ForwardShockEqn<Ejecta, Medium>::ForwardShockEqn(Medium const& medium, Ejecta const& ejecta, Real phi, Real theta,
12 RadParams const& rad_params, Real theta_s)
13 : medium(medium),
14 ejecta(ejecta),
15 phi(phi),
16 theta0(theta),
17 rad(rad_params),
18 dOmega0(1 - std::cos(theta)),
19 theta_s(theta_s) {
20 m_jet0 = ejecta.eps_k(phi, theta0) / ejecta.Gamma0(phi, theta0) / con::c2;
21 if constexpr (HasSigma<Ejecta>) {
22 m_jet0 /= 1 + ejecta.sigma0(phi, theta0);
23 }
24 gamma_m_coeff_ = (rad.p - 2) / (rad.p - 1) * rad.eps_e * con::mp / con::me / rad.xi_e;
25 gamma_c_coeff_ = 6 * con::pi * con::me * con::c / con::sigmaT / rad.eps_B;
26}

28template <typename Ejecta, typename Medium>
29void ForwardShockEqn<Ejecta, Medium>::operator()(State const& state, State& diff, Real t) const noexcept {
30 const Real Gamma = state.Gamma;
31 const Real Gamma2 = Gamma * Gamma;
32 const Real u2 = Gamma2 - 1;
33 const Real u = std::sqrt(u2);
34
35 diff.r = compute_dr_dt(Gamma, u);
36 diff.t_comv = Gamma + u;
37
38 if (ejecta.spreading && state.theta < 0.5 * con::pi) {
39 diff.theta = compute_dtheta_dt(theta_s, state.theta, diff.r, state.r, Gamma, u, u2);
40 } else {
41 diff.theta = 0;
42 }
43
44 if constexpr (State::mass_inject) {
45 diff.m_jet = ejecta.dm_dt(phi, theta0, t);
46 }
47
48 if constexpr (State::energy_inject) {
49 diff.eps_jet = ejecta.deps_dt(phi, theta0, t);
50 }
51
52 const Real rho = medium.rho(phi, state.theta, state.r);
53 diff.m2 = state.r * state.r * rho * diff.r;
54 const Real e_th = (Gamma - 1) * 4 * Gamma * rho * con::c2;
55 const Real eps_rad = compute_eps_rad(state.t_comv, Gamma, e_th);
56 const Real ad_idx = physics::thermo::adiabatic_idx(Gamma);
57 Real sin_theta = 0;
58 Real cos_theta = 1;
59 if (ejecta.spreading) {
60 sin_theta = std::sin(state.theta);
61 cos_theta = std::cos(state.theta);
62 }
63 diff.Gamma = compute_dGamma_dt(state, diff, ad_idx, sin_theta, cos_theta);
64 diff.U2_th = compute_dU_dt(eps_rad, state, diff, ad_idx, sin_theta, cos_theta);
65}

67template <typename Ejecta, typename Medium>
68Real ForwardShockEqn<Ejecta, Medium>::compute_dGamma_dt(State const& state, State const& diff,
69 Real ad_idx, Real sin_theta, Real cos_theta) const noexcept {
70 Real dm_dt_swept = diff.m2;
71 Real m_swept = state.m2;
72 const Real Gamma2 = state.Gamma * state.Gamma;
73 const Real Gamma_eff = (ad_idx * (Gamma2 - 1) + 1) / state.Gamma;
74 Real dGamma_eff = (ad_idx * (Gamma2 + 1) - 1) / Gamma2;
75 Real dlnVdt = 3 / state.r * diff.r; // only r term
76
77 Real m_jet = this->m_jet0;
78 Real U = state.U2_th; // Internal energy per unit solid angle
79
80 if (ejecta.spreading) {
81 const Real f_spread = (1 - cos_theta) / dOmega0;
82 dm_dt_swept = dm_dt_swept * f_spread + m_swept / dOmega0 * sin_theta * diff.theta;
83 m_swept *= f_spread;
84 dlnVdt += sin_theta / (1 - cos_theta) * diff.theta;
85 U *= f_spread;
86 }
87
88 Real a1 = -(state.Gamma - 1) * (Gamma_eff + 1) * con::c2 * dm_dt_swept;
89 const Real a2 = (ad_idx - 1) * Gamma_eff * U * dlnVdt;
90
91 if constexpr (State::energy_inject) {
92 a1 += diff.eps_jet;
93 }
94
95 if constexpr (State::mass_inject) {
96 a1 -= state.Gamma * diff.m_jet * con::c2;
97 m_jet = state.m_jet;
98 }
99
100 const Real b1 = (m_jet + m_swept) * con::c2;
101 const Real b2 = (dGamma_eff + Gamma_eff * (ad_idx - 1) / state.Gamma) * U;
102
103 return (a1 + a2) / (b1 + b2);
104}

106template <typename Ejecta, typename Medium>
107Real ForwardShockEqn<Ejecta, Medium>::compute_dU_dt(Real eps_rad, State const& state, State const& diff,
108 Real ad_idx, Real sin_theta, Real cos_theta) const noexcept {
109 Real dm_dt_swept = diff.m2;
110 const Real m_swept = state.m2;
111 Real dlnVdt = 3 / state.r * diff.r - diff.Gamma / state.Gamma;
112 if (ejecta.spreading) {
113 const Real factor = sin_theta / (1 - cos_theta) * diff.theta;
114 dm_dt_swept = dm_dt_swept + m_swept * factor;
115 dlnVdt += factor;
116 dlnVdt += factor / (ad_idx - 1);
117 }
118
119 return (1 - eps_rad) * (state.Gamma - 1) * con::c2 * dm_dt_swept - (ad_idx - 1) * dlnVdt * state.U2_th;
120}

122template <typename Ejecta, typename Medium>
123Real ForwardShockEqn<Ejecta, Medium>::compute_eps_rad(Real t_comv, Real Gamma, Real e_th) const noexcept {
124 const Real gamma_m = gamma_m_coeff_ * (Gamma - 1) + 1;
125 const Real gamma_c = std::max(gamma_c_coeff_ / (e_th * t_comv), 1.0);
126 const Real ratio = gamma_m / gamma_c;
127 if (ratio < 1 && rad.p > 2) {
128 if (ratio < 1e-2)
129 return 0;
130 return rad.eps_e * fast_pow(ratio, rad.p - 2);
131 }
132 return rad.eps_e;
133}

135template <typename Ejecta, typename Medium>
136void ForwardShockEqn<Ejecta, Medium>::set_init_state(State& state, Real t0) const noexcept {
137 Real Gamma4 = ejecta.Gamma0(phi, theta0);
138
139 const Real beta4 = physics::relativistic::gamma_to_beta(Gamma4);
140 state.r = beta4 * con::c * t0 / (1 - beta4);
141
142 state.t_comv = state.r / std::sqrt(Gamma4 * Gamma4 - 1) / con::c;
143
144 state.theta = theta0;
145
146 state.m2 = enclosed_mass_medium(medium, phi, theta0, state.r);
147
148 state.Gamma = Gamma4;
149
150 if constexpr (State::energy_inject) {
151 state.eps_jet = ejecta.eps_k(phi, theta0);
152 }
153
154 if constexpr (State::mass_inject) {
155 state.m_jet = m_jet0;
156 }
157
158 Real ad_idx = physics::thermo::adiabatic_idx(state.Gamma);
159
160 state.U2_th = enclosed_thermal_energy_medium(medium, phi, theta0, state.r, state.Gamma, ad_idx, rad.eps_e);
161}

163template <typename Eqn, typename State>
164void save_fwd_shock_state(size_t i, size_t j, size_t k, Eqn const& eqn, State const& state, Shock& shock) {
165 // Set constant parameters for the unshocked medium
166 constexpr Real gamma1 = 1; // Lorentz factor of unshocked medium (at rest)
167 constexpr Real sigma = 0; // Magnetization of unshocked medium
168 constexpr Real B_upstr = 0;
169
170 const Real comp_ratio = compute_compression(gamma1, state.Gamma, sigma);
171 const Real rho = eqn.medium.rho(eqn.phi, state.theta, state.r);
172
173 Real U_th = 0;
174 if constexpr (HasU<State>) {
175 U_th = state.U2_th;
176 } else {
177 U_th = (state.Gamma - 1) * state.m2 * con::c2;
178 }
179
180 const Real Gamma_th = compute_Gamma_therm(U_th, state.m2);
181
182 const Real B = compute_downstr_B(shock.rad.eps_B, rho, B_upstr, Gamma_th, comp_ratio);
183
184 write_shock_state(shock, i, j, k, state.t_comv, state.r, state.theta, state.Gamma, Gamma_th, B, state.m2);
185}

187template <typename FwdEqn, typename View>
188void grid_solve_fwd_shock(size_t i, size_t j, View const& t, Shock& shock, FwdEqn const& eqn, Real rtol) {
189 using namespace boost::numeric::odeint;
190
191 typename FwdEqn::State state;
192 Real t0;
193
194 Real t_dec = compute_dec_time(eqn);
195 t0 = min(t.front(), 1 * unit::sec, 0.1 * t_dec);
196 eqn.set_init_state(state, t0);
197
198 if (state.Gamma <= con::Gamma_cut) {
199 set_stopping_shock(i, j, shock, state);
200 return;
201 }
202
203 auto stepper = make_dense_output(rtol, rtol, runge_kutta_dopri5<typename FwdEqn::State>());
204 stepper.initialize(state, t0, 0.01 * t0);
205
206 for (size_t k = 0; stepper.current_time() <= t.back();) {
207 stepper.do_step(eqn);
208
209 while (k < t.size() && stepper.current_time() > t(k)) {
210 stepper.calc_state(t(k), state);
211 save_fwd_shock_state(i, j, k, eqn, state, shock);
212 ++k;
213 }
214 }
215}

217template <typename Ejecta, typename Medium>
218Shock generate_fwd_shock(Coord const& coord, Medium const& medium, Ejecta const& jet, RadParams const& rad_params,
219 Real rtol) {
220 auto [phi_size, theta_size, t_size] = coord.shape(); // Unpack coordinate dimensions
221 const size_t phi_size_needed = coord.t.shape()[0];
222 Shock shock(phi_size_needed, theta_size, t_size, rad_params);
223
224 shock.detect_symmetry(coord, jet, medium);
225
226 for (size_t i = 0; i < phi_size_needed; ++i) {
227 Real theta_s = 0;
228 if (jet.spreading) {
229 theta_s =
230 jet_spreading_edge(jet, medium, coord.phi(i), coord.theta.front(), coord.theta.back(), coord.t.front());
231 }
232
233 for (size_t j : shock.theta_reps) {
234 auto eqn = ForwardShockEqn(medium, jet, coord.phi(i), coord.theta(j), rad_params, theta_s);
235 //auto eqn = SimpleShockEqn(medium, jet, coord.phi(i), coord.theta(j), rad_params, theta_s);
236 grid_solve_fwd_shock(i, j, xt::view(coord.t, i, j, xt::all()), shock, eqn, rtol);
237 }
238
239 if (shock.symmetry >= Symmetry::phi_symmetric) {
240 shock.broadcast_groups(coord.theta);
241 return shock;
242 }
243 }
244
245 return shock;
