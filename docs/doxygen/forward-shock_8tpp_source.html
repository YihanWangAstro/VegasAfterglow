11template <typename Ejecta, typename Medium>
12ForwardShockEqn<Ejecta, Medium>::ForwardShockEqn(Medium const& medium, Ejecta const& ejecta, Real phi, Real theta,
13 RadParams const& rad_params, Real theta_s)
14 : medium(medium),
15 ejecta(ejecta),
16 phi(phi),
17 theta0(theta),
18 rad(rad_params),
19 dOmega0(1 - std::cos(theta)),
20 theta_s(theta_s),
21 m_shell(0) {
22 m_shell = ejecta.eps_k(phi, theta0) / ejecta.Gamma0(phi, theta0) / con::c2;
23 if constexpr (HasSigma<Ejecta>) {
24 m_shell /= 1 + ejecta.sigma0(phi, theta0);
25 }
26 rad_const = 16. / 3 * con::mp * con::sigmaT * con::c / (con::me * con::me) * (rad.p - 2) / (rad.p - 1) * rad.eps_e *
27 rad.eps_B / rad.xi_e;
28}

30template <typename Ejecta, typename Medium>
31void ForwardShockEqn<Ejecta, Medium>::operator()(State const& state, State& diff, Real t) const noexcept {
32 Real beta = gamma_to_beta(state.Gamma);
33
34 diff.r = compute_dr_dt(beta);
35 diff.t_comv = compute_dt_dt_comv(state.Gamma, beta);
36
37 if (ejecta.spreading && state.theta < 0.5 * con::pi) {
38 diff.theta = compute_dtheta_dt(theta_s, state.theta, diff.r, state.r, state.Gamma);
39 } else {
40 diff.theta = 0;
41 }
42
43 if constexpr (State::mass_inject) {
44 diff.m_shell = ejecta.dm_dt(phi, theta0, t);
45 }
46
47 if constexpr (State::energy_inject) {
48 diff.eps_shell = ejecta.deps_dt(phi, theta0, t);
49 }
50
51 Real rho = medium.rho(phi, state.theta, state.r);
52 Real dm_dt_swpt = state.r * state.r * rho * diff.r;
53 Real m_swept = compute_swept_mass(*this, state);
54
55 Real eps_rad = compute_radiative_efficiency(rad_const, state.t_comv, state.Gamma, rho, rad.eps_e, rad.p);
56
57 Real ad_idx = adiabatic_idx(state.Gamma);
58
59 diff.Gamma = dGamma_dt(m_swept, dm_dt_swpt, state, diff, ad_idx);
60 diff.u = dU_dt(eps_rad, m_swept, dm_dt_swpt, state, diff, ad_idx);
61}

63template <typename Ejecta, typename Medium>
64Real ForwardShockEqn<Ejecta, Medium>::dGamma_dt(Real m_swept, Real dm_dt_swept, State const& state, State const& diff,
65 Real ad_idx) const noexcept {
66 Real Gamma2 = state.Gamma * state.Gamma;
67 Real Gamma_eff = (ad_idx * (Gamma2 - 1) + 1) / state.Gamma;
68 Real dGamma_eff = (ad_idx * (Gamma2 + 1) - 1) / Gamma2;
69 Real dlnVdt = 3 / state.r * diff.r; // only r term
70
71 Real m_shell = this->m_shell;
72 Real u = state.u; // Internal energy per unit solid angle
73
74 if (ejecta.spreading) {
75 Real cos_theta = std::cos(state.theta);
76 Real sin_theta = std::sin(state.theta);
77 Real f_spread = (1 - cos_theta) / dOmega0;
78 dm_dt_swept = dm_dt_swept * f_spread + m_swept / dOmega0 * sin_theta * diff.theta;
79 m_swept *= f_spread;
80 dlnVdt += sin_theta / (1 - cos_theta) * diff.theta;
81 u *= f_spread;
82 }
83
84 Real a1 = -(state.Gamma - 1) * (Gamma_eff + 1) * con::c2 * dm_dt_swept;
85 Real a2 = (ad_idx - 1) * Gamma_eff * u * dlnVdt;
86
87 if constexpr (State::energy_inject) {
88 a1 += diff.eps_shell;
89 }
90
91 if constexpr (State::mass_inject) {
92 a1 -= state.Gamma * diff.m_shell * con::c2;
93 m_shell = state.m_shell;
94 }
95
96 Real b1 = (m_shell + m_swept) * con::c2;
97 Real b2 = (dGamma_eff + Gamma_eff * (ad_idx - 1) / state.Gamma) * u;
98
99 return (a1 + a2) / (b1 + b2);
100}

102template <typename Ejecta, typename Medium>
103Real ForwardShockEqn<Ejecta, Medium>::dU_dt(Real eps_rad, Real m_swept, Real dm_dt_swept, State const& state,
104 State const& diff, Real ad_idx) const noexcept {
105 Real dlnVdt = 3 / state.r * diff.r - diff.Gamma / state.Gamma;
106 if (ejecta.spreading) {
107 Real factor = std::sin(state.theta) / (1 - std::cos(state.theta)) * diff.theta;
108 dm_dt_swept = dm_dt_swept + m_swept * factor;
109 dlnVdt += factor;
110 dlnVdt += factor / (ad_idx - 1);
111 }
112
113 return (1 - eps_rad) * (state.Gamma - 1) * con::c2 * dm_dt_swept - (ad_idx - 1) * dlnVdt * state.u;
114}

116template <typename Ejecta, typename Medium>
117void ForwardShockEqn<Ejecta, Medium>::set_init_state(State& state, Real t0) const noexcept {
118 state.Gamma = ejecta.Gamma0(phi, theta0);
119
120 Real beta0 = gamma_to_beta(state.Gamma);
121 state.r = beta0 * con::c * t0 / (1 - beta0);
122
123 state.t_comv = state.r / std::sqrt(state.Gamma * state.Gamma - 1) / con::c;
124
125 state.theta = theta0;
126
127 if constexpr (State::energy_inject) {
128 state.eps_shell = ejecta.eps_k(phi, theta0);
129 }
130
131 if constexpr (State::mass_inject) {
132 state.m_shell = m_shell;
133 }
134
135 state.u = (state.Gamma - 1) * medium.mass(phi, theta0, state.r) * con::c2;
136}

138template <typename Eqn, typename State>
139void save_fwd_shock_state(size_t i, size_t j, size_t k, Eqn const& eqn, State const& state, Shock& shock) {
140 // Calculate number density of the ambient medium at current position
141 Real n1 = eqn.medium.rho(eqn.phi, state.theta, state.r) / con::mp;
142 Real m2 = compute_swept_mass(eqn, state);
143
144 // Set constant parameters for the unshocked medium
145 constexpr Real gamma1 = 1; // Lorentz factor of unshocked medium (at rest)
146 constexpr Real sigma1 = 0; // Magnetization of unshocked medium
147 // Update the shock state with calculated values
148 save_shock_state(shock, i, j, k, state, state.Gamma, gamma1, m2 / con::mp, n1, sigma1);
149}

151template <typename FwdEqn, typename View>
152void grid_solve_fwd_shock(size_t i, size_t j, View const& t, Shock& shock, FwdEqn const& eqn, double rtol) {
153 using namespace boost::numeric::odeint;
154
155 // Initialize state array
156 typename FwdEqn::State state;
157
158 // Get initial time and set up initial conditions
159 Real t_dec = compute_dec_time(eqn, t.back());
160 Real t0 = min(t.front(), t_dec, 1 * unit::sec);
161 eqn.set_init_state(state, t0);
162
163 // Early exit if initial Lorentz factor is below cutoff
164 if (state.Gamma <= con::Gamma_cut) {
165 set_stopping_shock(i, j, shock, state);
166 return;
167 }
168
169 // Set up ODE solver with adaptive step size control
170 auto stepper = make_dense_output(rtol, rtol, runge_kutta_dopri5<typename FwdEqn::State>());
171
172 stepper.initialize(state, t0, 0.01 * t0);
173
174 // Solve ODE and update shock state at each requested time point
175 for (size_t k = 0; stepper.current_time() <= t.back();) {
176 // Advance solution by one adaptive step
177 stepper.do_step(eqn);
178
179 // Update shock state for all time points that have been passed in this step
180 while (k < t.size() && stepper.current_time() > t(k)) {
181 stepper.calc_state(t(k), state);
182 save_fwd_shock_state(i, j, k, eqn, state, shock);
183 ++k;
184 }
185 }
186}

188template <typename Ejecta, typename Medium>
189Shock generate_fwd_shock(Coord const& coord, Medium const& medium, Ejecta const& jet, RadParams const& rad_params,
190 Real rtol) {
191 auto [phi_size, theta_size, t_size] = coord.shape(); // Unpack coordinate dimensions
192 size_t phi_size_needed = coord.t.shape()[0];
193 Shock shock(phi_size_needed, theta_size, t_size, rad_params);
194
195 for (size_t i = 0; i < phi_size_needed; ++i) {
196 Real theta_s =
197 jet_spreading_edge(jet, medium, coord.phi(i), coord.theta.front(), coord.theta.back(), coord.t.front());
198 for (size_t j = 0; j < theta_size; ++j) {
199 // auto eqn = ForwardShockEqn(medium, jet, coord.phi(i), coord.theta(j), rad_params, theta_s);
200 auto eqn = SimpleShockEqn(medium, jet, coord.phi(i), coord.theta(j), rad_params, theta_s);
201 // Solve the shock shell for this theta slice
202 grid_solve_fwd_shock(i, j, xt::view(coord.t, i, j, xt::all()), shock, eqn, rtol);
203 }
204 }
205
206 return shock;
