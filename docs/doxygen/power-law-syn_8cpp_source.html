24void PowerLawSyn::build() {
25 log2_I_nu_max = fast_log2(I_nu_max);
26
27 inv_nu_M_ = 1.0 / nu_M;
28
29 switch (regime) {
30 case 1: // a ≤ m ≤ c
31 segments_.first_segment(std::cbrt(nu_a / nu_m), nu_a, 2.0);
32 segments_.add_segment(nu_a, 1.0 / 3);
33 segments_.add_segment(nu_m, -(p - 1) / 2);
34 segments_.add_segment(nu_c, -p / 2);
35 break;
36
37 case 2: // m ≤ a ≤ c
38 segments_.first_segment(fast_pow(nu_m / nu_a, (p + 4) / 2), nu_m, 2.0);
39 segments_.add_segment(nu_m, 2.5);
40 segments_.add_segment(nu_a, -(p - 1) / 2);
41 segments_.add_segment(nu_c, -p / 2);
42 break;
43
44 case 3: // a ≤ c ≤ m
45 segments_.first_segment(std::cbrt(nu_a / nu_c), nu_a, 2.0);
46 segments_.add_segment(nu_a, 1.0 / 3);
47 segments_.add_segment(nu_c, -0.5);
48 segments_.add_segment(nu_m, -p / 2);
49 break;
50
51 case 4: // c ≤ a ≤ m
52 segments_.first_segment(std::sqrt(nu_c / nu_a), nu_a, 2.0);
53 segments_.add_segment(nu_a, -0.5);
54 segments_.add_segment(nu_m, -p / 2);
55 break;
56
57 case 5: // m ≤ c ≤ a
58 case 6: // c ≤ m ≤ a
59 segments_.first_segment(std::sqrt(nu_c / nu_a) * fast_pow(nu_m / nu_a, (p + 4) / 2), nu_m, 2.0);
60 segments_.add_segment(nu_m, 2.5);
61 segments_.add_segment(nu_a, -p / 2);
62 break;
63
64 default:
65 segments_.clear();
66 break;
67 }

70Real PowerLawSyn::compute_I_nu(Real nu) const {
71 if (nu <= nu_c) { // Below cooling frequency, simple scaling
72 return fast_exp(-nu * inv_nu_M_) * I_nu_max * compute_spectrum(nu);
73 } else {
74 return fast_exp(-nu * inv_nu_M_) * I_nu_max * compute_spectrum(nu) * inverse_compton_correction(*this, nu);
75 }
76}

78Real PowerLawSyn::compute_log2_I_nu(Real log2_nu) const {
79 constexpr Real ln2 = 1.442695040888963407359924681001892137;
80 if (log2_nu <= log2_nu_c) { // Below cooling frequency, simple scaling
81 return log2_I_nu_max + compute_log2_spectrum(log2_nu) - ln2 * fast_exp2(log2_nu) * inv_nu_M_;
82 } else {
83 const Real nu = fast_exp2(log2_nu);
84 return log2_I_nu_max + compute_log2_spectrum(log2_nu) + fast_log2(inverse_compton_correction(*this, nu)) -
85 ln2 * nu * inv_nu_M_;
86 }
