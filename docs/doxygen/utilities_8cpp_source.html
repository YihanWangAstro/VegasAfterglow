29/**
30 * <!-- ************************************************************************************** -->
31 * @brief Point-wise linear interpolation between two points.
32 * @param x0 First x-coordinate
33 * @param x1 Second x-coordinate
34 * @param y0 First y-coordinate
35 * @param y1 Second y-coordinate
36 * @param xi The x-value at which to interpolate
37 * @return The interpolated y-value
38 * <!-- ************************************************************************************** -->
39 */
40Real point_interp(Real x0, Real x1, Real y0, Real y1, Real xi) {
41 if (x0 == x1) return y0;
42 Real slope = (y1 - y0) / (x1 - x0);
43 return y0 + slope * (xi - x0);
44}

46/**
47 * <!-- ************************************************************************************** -->
48 * @brief Point-wise logarithmic interpolation between two points.
49 * @details Performs linear interpolation in log-log space.
50 * @param x0 First x-coordinate
51 * @param x1 Second x-coordinate
52 * @param y0 First y-coordinate
53 * @param y1 Second y-coordinate
54 * @param xi The x-value at which to interpolate
55 * @return The interpolated y-value
56 * <!-- ************************************************************************************** -->
57 */
58Real point_loglog_interp(Real x0, Real x1, Real y0, Real y1, Real xi) {
59 if (y0 == 0 || y1 == 0) return 0;
60 if (x0 == x1) return y0;
61 Real log_x0 = std::log(x0);
62 Real log_x1 = std::log(x1);
63 Real log_y0 = std::log(y0);
64 Real log_y1 = std::log(y1);
65 Real slope = (log_y1 - log_y0) / (log_x1 - log_x0);
66 return std::exp(log_y0 + slope * (std::log(xi) - log_x0));
67}

69/**
70 * <!-- ************************************************************************************** -->
71 * @brief Linear interpolation for arbitrary x-values.
72 * @details Finds the appropriate interval for interpolation and uses point_interp.
73 * @param xi The x-value at which to interpolate
74 * @param x Array of x-coordinates (must be monotonically increasing)
75 * @param y Array of y-coordinates
76 * @param lo_extrap Whether to extrapolate below the minimum x-value
77 * @param hi_extrap Whether to extrapolate above the maximum x-value
78 * @return The interpolated y-value
79 * <!-- ************************************************************************************** -->
80 */
81Real interp(Real xi, Array const& x, Array const& y, bool lo_extrap, bool hi_extrap) {
82 if (x.size() < 2 || y.size() < 2 || x.size() != y.size()) {
83 std::cout << "incorrect array size for interpolation!\n";
84 return 0;
85 }
86 auto x_back = x[x.size() - 1];
87 auto y_back = y[y.size() - 1];
88
89 if (xi < x[0]) {
90 return (!lo_extrap || x[0] == xi) ? y[0] : point_interp(x[0], x[1], y[0], y[1], xi);
91 } else if (xi > x_back) {
92 return (!hi_extrap || x_back == xi) ? y_back
93 : point_interp(x[x.size() - 2], x_back, y[y.size() - 2], y_back, xi);
94 } else {
95 auto it = std::lower_bound(x.begin(), x.end(), xi);
96 size_t idx = it - x.begin();
97 if (*it == xi) return y[idx]; // Exact match
98 return point_interp(x[idx - 1], x[idx], y[idx - 1], y[idx], xi);
99 }
100}
