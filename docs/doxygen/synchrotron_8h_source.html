18 * @struct SynElectrons
19 * @brief Represents synchrotron-emitting electrons in the comoving frame along with their energy distribution
20 * and properties.

23struct SynElectrons {
24 // All values in comoving frame
25 Real gamma_m{0}; ///< Minimum electron Lorentz factor
26 Real gamma_c{0}; ///< Cooling electron Lorentz factor
27 Real gamma_a{0}; ///< Self-absorption Lorentz factor
28 Real gamma_M{0}; ///< Maximum electron Lorentz factor
29 Real p{2.3}; ///< Power-law index for the electron energy distribution
30 Real N_e{0}; ///< shock electron number PER SOLID ANGLE
31 Real column_den{0}; ///< Column number density
32 Real Y_c{0}; ///< Inverse Compton Y parameter at cooling frequency
33 size_t regime{0}; ///< Regime indicator (1-6, determines spectral shape)
34 InverseComptonY Ys; ///< InverseComptonY parameters for this electron population
35
36 /**
37 * <!-- ************************************************************************************** -->
38 * @brief Calculates the comoving electron number (PER SOLID ANGLE) spectrum at a specific Lorentz factor.
39 * @details Includes corrections for inverse Compton cooling effects above the cooling Lorentz factor.
40 * @param gamma Electron Lorentz factor
41 * @return Column number density at the specified Lorentz factor
42 * <!-- ************************************************************************************** -->
43 */
44 Real compute_N_gamma(Real gamma) const;
45
46 /**
47 * <!-- ************************************************************************************** -->
48 * @brief Calculates the column number density of the electron distribution.
49 * @details Uses the electron energy spectrum to compute the column number density.
50 * @return Column number density of the electron distribution
51 * <!-- ************************************************************************************** -->
52 */
53 Real compute_column_den(Real gamma) const;
54
55 private:
56 /**
57 * <!-- ************************************************************************************** -->
58 * @brief Calculates the comoving electron energy spectrum at a given Lorentz factor.
59 * @details Different spectral forms apply based on the current regime and relative to
60 * characteristic Lorentz factors (gamma_a, gamma_c, gamma_m, gamma_M).
61 * @param gamma Electron Lorentz factor
62 * @return The normalized electron energy spectrum value
63 * <!-- ************************************************************************************** -->
64 */
65 inline Real compute_spectrum(Real gamma) const;
66};
