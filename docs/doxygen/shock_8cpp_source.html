42Real compute_downstr_4vel(Real gamma_rel, Real sigma) {
43 Real ad_idx = adiabatic_idx(gamma_rel);
44 Real gamma_m_1 = gamma_rel - 1; // (gamma_rel - 1)
45 Real ad_idx_m_2 = ad_idx - 2; // (ad_idx - 2)
46 Real ad_idx_m_1 = ad_idx - 1; // (ad_idx - 1)
47 if (sigma <= con::sigma_cut) {
48 return std::sqrt(std::fabs(gamma_m_1 * ad_idx_m_1 * ad_idx_m_1 / (-ad_idx * ad_idx_m_2 * gamma_m_1 + 2)));
49 } else {
50 Real gamma_sq = gamma_rel * gamma_rel; // gamma_rel^2
51 Real gamma_p_1 = gamma_rel + 1; // (gamma_rel + 1)
52
53 // Precompute common terms
54 Real term1 = -ad_idx * ad_idx_m_2;
55 Real term2 = gamma_sq - 1;
56
57 // Compute coefficients
58 Real A = term1 * gamma_m_1 + 2;
59 Real B = -gamma_p_1 * (-ad_idx_m_2 * (ad_idx * gamma_sq + 1) + ad_idx * ad_idx_m_1 * gamma_rel) * sigma -
60 gamma_m_1 * (term1 * (gamma_sq - 2) + 2 * gamma_rel + 3);
61 Real C = gamma_p_1 * (ad_idx * (1 - ad_idx / 4) * term2 + 1) * sigma * sigma +
62 term2 * (2 * gamma_rel + ad_idx_m_2 * (ad_idx * gamma_rel - 1)) * sigma +
63 gamma_p_1 * gamma_m_1 * gamma_m_1 * ad_idx_m_1 * ad_idx_m_1;
64 Real D = -gamma_m_1 * gamma_p_1 * gamma_p_1 * ad_idx_m_2 * ad_idx_m_2 * sigma * sigma / 4;
65
66 Real b = B / A;
67 Real c = C / A;
68 Real d = D / A;
69 Real P = c - b * b / 3;
70 Real Q = 2 * b * b * b / 27 - b * c / 3 + d;
71 Real u = std::sqrt(-P / 3);
72 Real v = std::clamp(3 * Q / (2 * P * u), -1.0, 1.0);
73 Real uds = 2 * u * std::cos((std::acos(v) - 2 * con::pi) / 3) - b / 3;
74 return std::sqrt(uds);
75 }
