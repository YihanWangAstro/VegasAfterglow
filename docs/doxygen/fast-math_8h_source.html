27/**
28 * <!-- ************************************************************************************** -->
29 * @brief Fast approximation of the base-2 logarithm
30 * @param x The input value
31 * @return log2(x)
32 * @details Uses bit manipulation + polynomial approximation.
33 * Error < 1e-10, ~3.5x faster than std::log2
34 * <!-- ************************************************************************************** -->
35 */
36inline double fast_log2(double x) {
37#ifdef EXTREME_SPEED
38 uint64_t bits = std::bit_cast<uint64_t>(x);
39
40 // Extract exponent
41 int64_t e = static_cast<int64_t>((bits >> 52) & 0x7FF) - 1023;
42
43 // Normalize mantissa to [1, 2)
44 bits = (bits & FRAC_MASK) | (1023ull << 52);
45 double m = std::bit_cast<double>(bits);
46
47 // For better polynomial convergence, center around sqrt(2)
48 // If m > sqrt(2), divide by 2 and increment exponent
49 if (m > 1.4142135623730951) {
50 m *= 0.5;
51 e++;
52 }
53
54 // Now m is in [sqrt(2)/2, sqrt(2)] â‰ˆ [0.707, 1.414]
55 // Use u = (m-1)/(m+1) for faster convergence
56 double u = (m - 1.0) / (m + 1.0);
57 double u2 = u * u;
58
59 // log2(m) = 2 * INV_LN2 * (u + u^3/3 + u^5/5 + u^7/7 + ...)
60 // Polynomial coefficients: 2/(k*ln(2)) for odd k
61 constexpr double c1 = 2.8853900817779268; // 2/ln(2)
62 constexpr double c3 = 0.9617966939259756; // 2/(3*ln(2))
63 constexpr double c5 = 0.5770780162461006; // 2/(5*ln(2))
64 constexpr double c7 = 0.4121977821679615; // 2/(7*ln(2))
65 constexpr double c9 = 0.3219280948873623; // 2/(9*ln(2))
66
67 double poly = c1 + u2 * (c3 + u2 * (c5 + u2 * (c7 + u2 * c9)));
68
69 return static_cast<double>(e) + u * poly;
70#else
71 return std::log2(x);
72#endif
73}

75/**
76 * <!-- ************************************************************************************** -->
77 * @brief Fast approximation of 2 raised to a power
78 * @param x The exponent
79 * @return 2^x
80 * @details Uses polynomial approximation with bit manipulation.
81 * Error < 2e-7, ~1.5x faster than std::exp2
82 * <!-- ************************************************************************************** -->
83 */
84inline double fast_exp2(double x) {
85#ifdef EXTREME_SPEED
86 // Handle overflow/underflow
87 if (x >= 1024.0)
88 return std::numeric_limits<double>::infinity();
89 if (x <= -1075.0)
90 return 0.0;
91
92 // Split into integer and fractional parts (round to nearest for smaller |f|)
93 double i_part = std::floor(x + 0.5);
94 double f = x - i_part; // f in [-0.5, 0.5]
95 int64_t i = static_cast<int64_t>(i_part);
96
97 // Polynomial for 2^f, f in [-0.5, 0.5]
98 // Taylor series: 2^f = sum_{k=0}^{n} (f*ln2)^k / k!
99 constexpr double c0 = 1.0;
100 constexpr double c1 = 0.6931471805599453; // ln(2)
101 constexpr double c2 = 0.2402265069591007; // ln(2)^2 / 2!
102 constexpr double c3 = 0.0555041086648216; // ln(2)^3 / 3!
103 constexpr double c4 = 0.0096181291076285; // ln(2)^4 / 4!
104 constexpr double c5 = 0.0013333558146428; // ln(2)^5 / 5!
105 constexpr double c6 = 0.0001540353039338; // ln(2)^6 / 6!
106
107 // Horner's method
108 double poly = c0 + f * (c1 + f * (c2 + f * (c3 + f * (c4 + f * (c5 + f * c6)))));
109
110 // Multiply by 2^i using bit manipulation
111 uint64_t bits = std::bit_cast<uint64_t>(poly);
112 int64_t exp = static_cast<int64_t>((bits >> 52) & 0x7FF);
113 int64_t new_exp = exp + i;
114
115 if (new_exp <= 0)
116 return 0.0;
117 if (new_exp >= 0x7FF)
118 return std::numeric_limits<double>::infinity();
119
120 bits = (bits & ~(0x7FFull << 52)) | (static_cast<uint64_t>(new_exp) << 52);
121 return std::bit_cast<double>(bits);
122#else
123 return std::exp2(x);
124#endif
125}

184/**
185 * <!-- ************************************************************************************** -->
186 * @brief Smooth broken power-law transition in log2 space.
187 * @details Returns a correction term that smoothly changes the slope by delta_beta at the break.
188 * Far below break: correction -> 0 (original slope preserved).
189 * Far above break: correction -> -delta_beta * (log2_x - log2_x_break) / s (slope shifted).
190 * @param log2_x Log2 of the evaluation point
191 * @param log2_x_break Log2 of the break point
192 * @param s_delta_beta s * (slope_below - slope_above), where s controls transition sharpness
193 * @param s Smoothing sharpness parameter (larger = sharper transition)
194 * @return Log2-space correction to add to the base power law
195 * <!-- ************************************************************************************** -->
196 */
197inline Real log2_broken_power_ratio(Real log2_x, Real log2_x_break, Real s_delta_beta, Real s) {
198 return -log2_softplus(s_delta_beta * (log2_x - log2_x_break)) / s;
199}
