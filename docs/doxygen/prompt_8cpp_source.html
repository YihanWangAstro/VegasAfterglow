19PromptPhotonsGrid gen_prompt_photons(CoastingShock const& shock, Real R0, Real nu_0, Real alpha) {
20 auto [phi_size, theta_size, t_size] = shock.shape();
21
22 PromptPhotonsGrid ph({phi_size, theta_size, t_size});
23
24 Real Gamma_c = shock.Gamma(0, 0, 0);
25 Real beta_c = gamma_to_beta(Gamma_c);
26
27 for (size_t i = 0; i < phi_size; ++i) {
28 for (size_t j = 0; j < theta_size; ++j) {
29 Real Gamma = shock.Gamma(i, j, 0);
30 Real beta = gamma_to_beta(Gamma);
31 Real R = R0 * beta / (1 - beta) * (1 - beta_c) / beta_c;
32 Real Rmin = R * 0;
33 Real Rmax = R * 1.;
34 for (size_t k = 0; k < t_size; ++k) {
35 if (shock.r(i, j, k) >= Rmin && shock.r(i, j, k) <= Rmax) {
36 ph(i, j, k).E_nu_peak = shock.epsilon(i, j, k);
37 ph(i, j, k).nu_0 = nu_0;
38 ph(i, j, k).alpha = alpha;
39 }
40 }
41 }
42 }
43 return ph;
