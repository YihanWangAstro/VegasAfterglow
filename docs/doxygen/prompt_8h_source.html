42template <typename Ejecta>
43CoastingShock gen_coasting_shock(Coord const& coord, Ejecta const& jet) {
44 auto [phi_size, theta_size, t_size] = coord.shape();
45 CoastingShock shock(1, theta_size, t_size);
46
47 for (size_t j = 0; j < theta_size; ++j) {
48 Real Gamma = jet.Gamma0(coord.phi(0), coord.theta(j));
49 Real beta = gamma_to_beta(Gamma);
50 Real epsilon = jet.eps_k(coord.phi(0), coord.theta(j));
51 for (size_t k = 0; k < t_size; ++k) {
52 shock.Gamma(0, j, k) = Gamma;
53 shock.epsilon(0, j, k) = epsilon;
54 shock.r(0, j, k) = (beta * con::c) / std::fabs(1 - beta) * coord.t(k);
55 shock.theta(0, j, k) = coord.theta(j);
56 }
57 }
58
59 return shock;
60}
