7 const Real lg2_per_decade = log2_10;
8 const Real refine_radius = refine_radius_decades * lg2_per_decade;

15 Real lg2;

27 Real w = 1;
28 if (i < break_weights.size()) {
29 const Real wi = break_weights[i];
30 if (std::isfinite(wi) && wi >= 0) {
31 w = wi;
32 }
33 }

36 }

83 } else {

88 }
89 }

101 }
102 }

107 capped.reserve(MAX_PTS);

109 capped.push_back(merged[i * (merged.size() - 1) / (MAX_PTS - 1)]);
110 }
111 merged.swap(capped);
112 }

115 for (size_t i = 0; i < merged.size(); ++i) {
116 grid(i) = std::exp2(merged[i].lg2);
117 }

121Array jump_refinement_grid(Array const& jumps, Real theta_min, Real theta_max, Real avg_spacing) {
122 std::vector<Real> points;
123 points.reserve(jumps.size() * 3);
124 const Real tight = avg_spacing / 8;
125 for (size_t idx = 0; idx < jumps.size(); ++idx) {
126 const Real jump_theta = jumps(idx);
127 if (jump_theta >= con::pi / 2 - 0.01) {
128 continue;
129 }
130 // Tight neighbors to minimize quadrature error at discontinuity
131 if (jump_theta - tight >= theta_min) {
132 points.push_back(jump_theta - tight);
133 }
134 if (jump_theta + tight <= theta_max) {
135 points.push_back(jump_theta + tight);
136 }
137 // Add the jump position itself
138 if (jump_theta >= theta_min && jump_theta <= theta_max) {
139 points.push_back(jump_theta);
140 }
141 }
142 std::ranges::sort(points);
143 points.erase(std::ranges::unique(points).begin(), points.end());
144 return xt::adapt(points, {points.size()});

148 t_cross = std::clamp(t_cross, t_start, t_end);
149 if (t_cross <= t_start || t_cross >= t_end) {

151 }

153 const Real log_total = std::log10(t_end / t_start);
154 const Real log_after = std::log10(t_end / t_cross);
155
156 // Post-crossing: same density as base (fwd-equivalent) logspace
157 size_t n_post = std::max<size_t>(static_cast<size_t>(base_t_num * log_after / log_total), 2);
158 if (n_post >= t_num) {
159 n_post = t_num / 2;
160 }
161 // Pre-crossing: gets all remaining points (base + extra)
162 size_t n_pre = t_num + 1 - n_post;
163

165 size_t idx = 0;
166
167 auto seg1 = xt::logspace(std::log10(t_start), std::log10(t_cross), n_pre);
168 for (size_t k = 0; k < n_pre; ++k) {
169 result(idx++) = seg1(k);
170 }
171
172 auto seg2 = xt::logspace(std::log10(t_cross), std::log10(t_end), n_post);
173 for (size_t k = 1; k < n_post && idx < t_num; ++k) {
