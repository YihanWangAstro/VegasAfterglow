11template <typename Eqn, typename State>
12bool is_crossing(Eqn const& eqn, State const& state, Real t) {
13 Real dmdt = 0;
14
15 if constexpr (State::mass_inject) {
16 dmdt = eqn.ejecta.dm_dt(eqn.phi, state.theta, t);
17 }
18
19 return (state.m3 < state.m4) || dmdt > 0 || t < eqn.ejecta.T0;
20}

40template <typename Ejecta, typename Medium>
41Real FRShockEqn<Ejecta, Medium>::compute_dGamma_dt(State const& state, State const& diff, Real t) const noexcept {
42 Real Gamma34 = compute_rel_Gamma(Gamma4, state.Gamma);
43 Real ad_idx2 = adiabatic_idx(state.Gamma);
44 Real ad_idx3 = adiabatic_idx(Gamma34);
45
46 Real Gamma_eff2 = compute_effective_Gamma(ad_idx2, state.Gamma);
47 Real Gamma_eff3 = compute_effective_Gamma(ad_idx3, state.Gamma);
48
49 Real dGamma_eff2_dGamma = compute_effective_Gamma_dGamma(ad_idx2, state.Gamma);
50 Real dGamma_eff3_dGamma = compute_effective_Gamma_dGamma(ad_idx3, state.Gamma);
51
52 Real deps_dt = 0;
53
54 if constexpr (State::energy_inject) {
55 deps_dt = ejecta.deps_dt(phi, state.theta, t);
56 }
57
58 Real a = (state.Gamma - 1) * con::c2 * diff.m2 + (state.Gamma - Gamma4) * con::c2 * diff.m3 +
59 Gamma_eff2 * diff.U2_th + Gamma_eff3 * diff.U3_th - deps_dt;
60 Real b = (state.m2 + state.m3) * con::c2 + dGamma_eff2_dGamma * state.U2_th + dGamma_eff3_dGamma * state.U3_th;
61
62 if (b == 0) [[unlikely]] {
63 return 0;
64 } else [[likely]] {
65 return -a / b;
66 }
67}

69template <typename Ejecta, typename Medium>
70Real FRShockEqn<Ejecta, Medium>::compute_dU2_dt(State const& state, State const& diff, Real t) const noexcept {
71 Real e_th = (state.Gamma - 1) * 4 * state.Gamma * medium.rho(phi, state.theta, state.r) * con::c2;
72 Real eps_rad = compute_radiative_efficiency(state.t_comv, state.Gamma, e_th, rad_fwd);
73
74 Real ad_idx = adiabatic_idx(state.Gamma);
75
76 Real shock_heating = compute_shock_heating_rate(state.Gamma, diff.m2);
77
78 Real adiabatic_cooling = compute_adiabatic_cooling_rate2(ad_idx, state.r, state.x4, state.U2_th, diff.r, diff.x4);
79
80 return (1 - eps_rad) * shock_heating + adiabatic_cooling;
81}

86 Real ad_idx = adiabatic_idx(Gamma34);

88
89 Real shock_heating = compute_shock_heating_rate(Gamma34, diff.m3);
90 if (state.m3 < state.m4 || diff.m4 > 0) { // reverse shock still crossing
91 Real eps_rad = 0; ////compute_radiative_efficiency(state.t_comv, state.Gamma, e_th, rad_rvs);
92 return (1 - eps_rad) * shock_heating + adiabatic_cooling;
93 } else {
94 return adiabatic_cooling;
95 }

98template <typename Ejecta, typename Medium>
99Real FRShockEqn<Ejecta, Medium>::compute_dx3_dt(State const& state, State const& diff, Real t) const noexcept {
100 if ((state.m3 < state.m4 || diff.m4 > 0) && (state.Gamma != this->Gamma4)) {
101 Real sigma = compute_shell_sigma(state);
102 Real Gamma34 = compute_rel_Gamma(this->Gamma4, state.Gamma);
103 Real beta3 = gamma_to_beta(state.Gamma);
104 Real beta4 = gamma_to_beta(this->Gamma4);
105 Real comp_ratio = compute_4vel_jump(Gamma34, sigma);
106 Real dx3dt = (beta4 - beta3) * con::c / ((1 - beta3) * (state.Gamma * comp_ratio / this->Gamma4 - 1));
107
108 return std::fabs(dx3dt * state.Gamma);
109 } else {
110 Real Gamma34 = compute_rel_Gamma(this->Gamma4, state.Gamma);
111 return compute_shell_spreading_rate(Gamma34, diff.t_comv);
112 }
113}

129template <typename Ejecta, typename Medium>
130Real FRShockEqn<Ejecta, Medium>::compute_dm3_dt(State const& state, State const& diff, Real t) const noexcept {
131 if ((state.m3 < state.m4 || diff.m4 > 0) && (state.Gamma != this->Gamma4)) {
132 Real sigma = compute_shell_sigma(state);
133 Real Gamma34 = compute_rel_Gamma(this->Gamma4, state.Gamma);
134 Real comp_ratio = compute_4vel_jump(Gamma34, sigma);
135 Real column_den3 = state.m4 * comp_ratio / state.x4;
136 Real dm3dt = column_den3 * diff.x3;
137
138 if (state.m3 >= state.m4) {
139 return std::min(dm3dt, diff.m4);
140 } else {
141 return dm3dt;
142 }
143 } else {
144 return 0.;
145 }

178template <typename Ejecta, typename Medium>
179void FRShockEqn<Ejecta, Medium>::operator()(State const& state, State& diff, Real t) {
180 Real beta3 = gamma_to_beta(state.Gamma);
181
182 diff.r = compute_dr_dt(beta3);
183 diff.t_comv = compute_dt_dt_comv(state.Gamma, beta3);
184
185 diff.m2 = compute_dm2_dt(state, diff, t);
186
187 diff.eps4 = compute_deps4_dt(state, diff, t);
188 diff.m4 = compute_dm4_dt(state, diff, t);
189
190 diff.x4 = compute_dx4_dt(state, diff, t);
191 diff.x3 = compute_dx3_dt(state, diff, t);
192
193 diff.m3 = compute_dm3_dt(state, diff, t);
194
195 diff.U2_th = compute_dU2_dt(state, diff, t);
196 diff.U3_th = compute_dU3_dt(state, diff, t);
197
198 diff.Gamma = compute_dGamma_dt(state, diff, t);
199
200 diff.theta = 0;
201}

205template <typename Ejecta, typename Medium>
206void FRShockEqn<Ejecta, Medium>::save_cross_state(State const& state) {
207 r_x = state.r;
208 u_x = std::sqrt(state.Gamma * state.Gamma - 1);
209
210 V3_comv_x = r_x * r_x * state.x3;
211
212 Real sigma4 = compute_shell_sigma(state);
213 Real comp_ratio34 = compute_compression(Gamma4, state.Gamma, sigma4);
214 Real rho4 = state.m4 / (state.r * state.r * state.x4);
215 rho3_x = rho4 * comp_ratio34;
216
217 Real B4 = compute_upstr_B(rho4, sigma4);
218 B3_ordered_x = B4 * comp_ratio34;

221inline Real calculate_init_m3(Real Gamma4, Real Gamma3, Real m2, Real sigma) {
222 Real Gamma34 = compute_rel_Gamma(Gamma4, Gamma3);
223 Real ad_idx2 = adiabatic_idx(Gamma3);
224 Real ad_idx3 = adiabatic_idx(Gamma34);
225
226 Real Gamma_eff2 = compute_effective_Gamma(ad_idx2, Gamma3);
227 Real Gamma_eff3 = compute_effective_Gamma(ad_idx3, Gamma3);
228
229 return -m2 * (Gamma3 - 1 + Gamma_eff2 * (Gamma3 - 1)) / (Gamma3 - Gamma4 + Gamma_eff3 * (Gamma34 - 1)) /
230 (1 + sigma);
231}

233template <typename Ejecta, typename Medium>
234void FRShockEqn<Ejecta, Medium>::set_init_state(State& state, Real t0) const noexcept {
235 Real beta4 = gamma_to_beta(Gamma4);
236
237 state.r = beta4 * con::c * t0 / (1 - beta4);
238 state.t_comv = state.r / std::sqrt(Gamma4 * Gamma4 - 1) / con::c;
239 state.theta = theta0;
240
241 Real dt = std::min(t0, ejecta.T0);
242 state.eps4 = deps0_dt * dt;
243 state.m4 = dm0_dt * dt;
244 state.x4 = compute_init_comv_shell_width(Gamma4, t0, ejecta.T0);
245
246 // constexpr Real Gamma34 = 1;
247 // compute_Gamma_from_relative(Gamma4, Gamma34);
248
249 state.m2 = medium.rho(phi, theta0, state.r) * state.r * state.r * state.r / 3;
250
251 Real sigma4 = compute_shell_sigma(state);
252 state.Gamma = Gamma4;
253
254 Real ad_idx = adiabatic_idx(state.Gamma);
255 state.U2_th = (state.Gamma - 1) * state.m2 * con::c2 / ad_idx;
256
257 state.m3 = 0; // calculate_init_m3(Gamma4, state.Gamma, state.m2, sigma4);
258 state.U3_th = 0; //(Gamma34 - 1) * state.m3* con::c2;
259
260 state.x3 = 0;
261}

263/**
264 * <!-- ************************************************************************************** -->
265 * @internal
266 * @brief Calculates the power-law index for post-crossing four-velocity evolution.
267 * @details The index transitions from g_low=1.5 for low relative Lorentz factors to g_high=3.5
268 * for high relative Lorentz factors (Blandford-McKee limit).
269 * @param gamma_rel Relative Lorentz factor
270 * @param k Medium power law index (default: 0)
271 * @return The power-law index for velocity evolution
272 * <!-- ************************************************************************************** -->
273 */
274inline Real get_post_cross_g(Real gamma_rel, Real k = 0) {
275 constexpr Real g_low = 1.5; // k is the medium power law index
276 constexpr Real g_high = 3.5; // Blandford-McKee limit// TODO: need to be modified for non ISM medium
277 Real p = std::sqrt(std::sqrt(std::fabs(gamma_rel - 1)));
278 return g_low + (g_high - g_low) * p / (1 + p);
279}

281template <typename Ejecta, typename Medium>
282Real FRShockEqn<Ejecta, Medium>::compute_shell_sigma(State const& state) const {
283 Real sigma = state.eps4 / (Gamma4 * state.m4 * con::c2) - 1;
284 return (sigma > con::sigma_cut) ? sigma : 0;

291/**
292 * <!-- ************************************************************************************** -->
293 * @internal
294 * @brief Calculates the comoving shell width at initial radius.
295 * @details Accounts for both pure injection phase and shell spreading phase.
296 * @param Gamma4 Lorentz factor of the unshocked ejecta
297 * @param t0 Initial time
298 * @param T Engine duration
299 * @return The comoving shell width
300 * <!-- ************************************************************************************** -->
301 */
302inline Real compute_init_comv_shell_width(Real Gamma4, Real t0, Real T) {
303 Real beta4 = gamma_to_beta(Gamma4);
304 if (t0 < T) { // pure injection
305 return Gamma4 * t0 * beta4 * con::c;
306 } else { // injection+shell spreading
307 Real cs = compute_sound_speed(Gamma4);
308 return Gamma4 * T * beta4 * con::c + cs * (t0 - T) * Gamma4;
309 }

312/**
313 * <!-- ************************************************************************************** -->
314 * @internal
315 * @brief Saves the state of reverse shocks at a grid point.
316 * @details Updates shock properties for both shocks and checks if crossing is complete.
317 * @param i Grid index for phi
318 * @param j Grid index for theta
319 * @param k Grid index for time
320 * @param eqn The reverse shock equation system
321 * @param state Current state of the system
322 * @param shock Reverse shock object to update
323 * <!-- ************************************************************************************** -->
324 */
325template <typename Eqn, typename State>
326void save_rvs_shock_state(size_t i, size_t j, int k, Eqn const& eqn, State const& state, Shock& shock) {
327 if (k <= shock.injection_idx(i, j)) {
328 Real Gamma4 = eqn.Gamma4;
329 Real sigma4 = eqn.compute_shell_sigma(state);
330
331 Real comp_ratio34 = compute_compression(Gamma4, state.Gamma, sigma4);
332 Real rho4 = state.m4 / (state.r * state.r * state.x4);
333 Real Gamma3_th = compute_Gamma_therm(state.U3_th, state.m3, true);
334
335 Real B4 = compute_upstr_B(rho4, sigma4);
336 Real B3 = compute_downstr_B(shock.rad.eps_B, rho4, B4, Gamma3_th, comp_ratio34);
337
338 write_shock_state(shock, i, j, k, state.t_comv, state.r, state.theta, state.Gamma, Gamma3_th, B3, state.m3);
339 } else {
340 Real V3_comv = state.r * state.r * state.x3;
341 Real comp_ratio = eqn.V3_comv_x / V3_comv;
342 Real Gamma3_th = compute_Gamma_therm(state.U3_th, state.m3);
343
344 Real B3 = compute_downstr_B(shock.rad.eps_B, eqn.rho3_x, eqn.B3_ordered_x, Gamma3_th, comp_ratio);
345
346 write_shock_state(shock, i, j, k, state.t_comv, state.r, state.theta, state.Gamma, Gamma3_th, B3, state.m3);
347 }

350inline void reverse_shock_early_extrap(size_t i, size_t j, Shock& shock) {
351 auto [phi_size, theta_size, t_size] = shock.shape();
352
353 size_t idx_cut = 0;
354 for (; idx_cut < t_size; ++idx_cut) {
355 if (shock.Gamma_th(i, j, idx_cut) > con::Gamma_cut) {
356 break;
357 }
358 }
359
360 Real gamma_slope = 0;
361 Real B_slope = 0;
362 Real N_p_slope = 0;
363 Real r_lg2 = fast_log2(shock.r(i, j, idx_cut));
364 Real Gamma_th_lg2 = fast_log2(shock.Gamma_th(i, j, idx_cut) - 1);
365 Real B_lg2 = fast_log2(shock.B(i, j, idx_cut));
366 Real N_p_lg2 = fast_log2(shock.N_p(i, j, idx_cut));
367
368 constexpr size_t off_set = 2;
369
370 if (idx_cut == 0 || idx_cut >= t_size - off_set || idx_cut >= shock.injection_idx(i, j)) {
371 return;
372 } else {
373 gamma_slope = (fast_log2(shock.Gamma_th(i, j, idx_cut + off_set) - 1) - Gamma_th_lg2) /
374 (fast_log2(shock.r(i, j, idx_cut + off_set)) - r_lg2);
375
376 B_slope = (fast_log2(shock.B(i, j, idx_cut + off_set)) - B_lg2) /
377 (fast_log2(shock.r(i, j, idx_cut + off_set)) - r_lg2);
378
379 N_p_slope = (fast_log2(shock.N_p(i, j, idx_cut + off_set)) - N_p_lg2) /
380 (fast_log2(shock.r(i, j, idx_cut + off_set)) - r_lg2);
381 }
382
383 for (size_t k = 0; k < idx_cut; k++) {
384 Real dr_lg2 = fast_log2(shock.r(i, j, k)) - r_lg2;
385 shock.Gamma_th(i, j, k) = 1 + fast_exp2(Gamma_th_lg2 + gamma_slope * dr_lg2);
386 shock.B(i, j, k) = fast_exp2(B_lg2 + B_slope * dr_lg2);
387 shock.N_p(i, j, k) = fast_exp2(N_p_lg2 + N_p_slope * dr_lg2);
388 }
389}
390/**
391 * <!-- ************************************************************************************** -->
392 * @internal
393 * @brief Solves the reverse/forward shock ODE at a grid point.
394 * @details Manages the evolution of both shocks before and after crossing.
395 * @param i Grid index for phi
396 * @param j Grid index for theta
397 * @param t View of time points
398 * @param shock_fwd Forward shock object
399 * @param shock_rvs Reverse shock object
400 * @param eqn Reverse shock equation system
401 * @param rtol Relative tolerance for ODE solver
402 * <!-- ************************************************************************************** -->
403 */
404template <typename Eqn, typename View>
405void grid_solve_shock_pair(size_t i, size_t j, View const& t, Shock& shock_fwd, Shock& shock_rvs, Eqn& eqn,
406 Real rtol = 1e-6) {
407 using namespace boost::numeric::odeint;
408
409 typename Eqn::State state;
410 Real t0 = 0.1 * unit::sec;
411
412 eqn.set_init_state(state, t0);
413
414 if (state.Gamma <= con::Gamma_cut) {
415 set_stopping_shock(i, j, shock_fwd, state);
416 set_stopping_shock(i, j, shock_rvs, state);
417 return;
418 }
419
420 auto stepper = make_dense_output(rtol, rtol, runge_kutta_dopri5<typename Eqn::State>());
421 // auto stepper = bulirsch_stoer_dense_out<typename Eqn::State>{rtol, rtol};
422 stepper.initialize(state, t0, 1e-9 * t0);
423
424 size_t k = 0;
425 for (; t(k) < t0; k++) {
426 eqn.set_init_state(state, t(k));
427 save_fwd_shock_state(i, j, k, eqn, state, shock_fwd);
428 save_rvs_shock_state(i, j, k, eqn, state, shock_rvs);
429 }
430
431 bool reverse_shock_crossing = true;
432 for (; stepper.current_time() <= t.back();) {
433 stepper.do_step(eqn);
434 while (k < t.size() && stepper.current_time() > t(k)) {
435 stepper.calc_state(t(k), state);
436 if (reverse_shock_crossing && !is_crossing(eqn, state, t(k))) {
437 shock_rvs.injection_idx(i, j) = k;
438 reverse_shock_crossing = false;
439 eqn.save_cross_state(state);
440 }
441 save_fwd_shock_state(i, j, k, eqn, state, shock_fwd);
442 save_rvs_shock_state(i, j, k, eqn, state, shock_rvs);
443 ++k;
444 }
445 }
446 reverse_shock_early_extrap(i, j, shock_rvs);
447}

449template <typename Ejecta, typename Medium>
450ShockPair generate_shock_pair(Coord const& coord, Medium const& medium, Ejecta const& jet, RadParams const& rad_fwd,
451 RadParams const& rad_rvs, Real rtol) {
452 auto [phi_size, theta_size, t_size] = coord.shape();
453 size_t phi_size_needed = coord.t.shape()[0];
454 Shock f_shock(phi_size_needed, theta_size, t_size, rad_fwd);
455 Shock r_shock(phi_size_needed, theta_size, t_size, rad_rvs);
456 for (size_t i = 0; i < phi_size_needed; ++i) {
457 // Real theta_s =
458 // jet_spreading_edge(jet, medium, coord.phi(i), coord.theta.front(), coord.theta.back(), coord.t.front());
459 for (size_t j = 0; j < theta_size; ++j) {
460 auto eqn_r = FRShockEqn(medium, jet, coord.phi(i), coord.theta(j), rad_fwd, rad_rvs);
461 // Solve the forward-reverse shock shell
462 grid_solve_shock_pair(i, j, xt::view(coord.t, i, j, xt::all()), f_shock, r_shock, eqn_r, rtol);
463 }
464 }
465 return std::make_pair(std::move(f_shock), std::move(r_shock));
