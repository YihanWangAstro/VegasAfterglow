85/**
86 * <!-- ************************************************************************************** -->
87 * @brief Computes the downstream four-velocity for a given relative Lorentz factor and magnetization parameter.
88 * @param gamma_rel Relative Lorentz factor between upstream and downstream regions
89 * @param sigma Magnetization parameter (ratio of magnetic to rest-mass energy density)
90 * @return The downstream four-velocity in the shock frame
91 * @details The calculation handles both magnetized (sigma > 0) and non-magnetized (sigma = 0) cases using
92 * different equations based on jump conditions across the shock front.
93 * <!-- ************************************************************************************** -->
94 */
95Real compute_downstr_4vel(Real gamma_rel, Real sigma);
96
97/**
98 * <!-- ************************************************************************************** -->
99 * @brief Computes the upstream four-velocity from downstream four-velocity and relative Lorentz factor.
100 * @param u_down Downstream four-velocity
101 * @param gamma_rel Relative Lorentz factor
102 * @return The upstream four-velocity in the shock frame
103 * <!-- ************************************************************************************** -->
104 */
105inline Real compute_upstr_4vel(Real u_down, Real gamma_rel) {
106 return std::sqrt((1 + u_down * u_down) * std::fabs(gamma_rel * gamma_rel - 1)) + u_down * gamma_rel;
107}
108
109/**
110 * <!-- ************************************************************************************** -->
111 * @brief Computes the ratio of upstream to downstream four-velocity across the shock front.
112 * @param gamma_rel Relative Lorentz factor between upstream and downstream regions
113 * @param sigma Magnetization parameter (ratio of magnetic to rest-mass energy density)
114 * @return The ratio of upstream to downstream four-velocity
115 * @details This ratio is a key parameter in determining various shock properties, such as compression ratio
116 * and jump conditions for density, pressure, and magnetic field.
117 * <!-- ************************************************************************************** -->
118 */
119inline Real compute_4vel_jump(Real gamma_rel, Real sigma_upstr) {
120 Real u_down_s_ = compute_downstr_4vel(gamma_rel, sigma_upstr);
121 Real u_up_s_ = compute_upstr_4vel(u_down_s_, gamma_rel);
122 Real ratio_u = u_up_s_ / u_down_s_;
123 if (u_down_s_ == 0.) {
124 ratio_u = 4 * gamma_rel; // (g_hat*gamma_rel+1)/(g_hat-1)
125 }
126 return ratio_u;
127}

129/**
130 * <!-- ************************************************************************************** -->
131 * @brief Computes the sound speed in the shocked medium based on the relative Lorentz factor.
132 * @param Gamma_rel Relative Lorentz factor between upstream and downstream regions
133 * @return The sound speed as a fraction of light speed
134 * <!-- ************************************************************************************** -->
135 */
136inline Real compute_sound_speed(Real Gamma_rel) {
137 Real ad_idx = adiabatic_idx(Gamma_rel);
138 return std::sqrt(std::fabs(ad_idx * (ad_idx - 1) * (Gamma_rel - 1) / (1 + (Gamma_rel - 1) * ad_idx))) * con::c;
139}

141/**
142 * <!-- ************************************************************************************** -->
143 * @brief Computes the effective Lorentz factor accounting for the adiabatic index.
144 * @param adx Adiabatic index of the medium
145 * @param Gamma Bulk Lorentz factor
146 * @return The effective Lorentz factor
147 * <!-- ************************************************************************************** -->
148 */
149inline Real compute_effective_Gamma(Real adx, Real Gamma) {
150 return (adx * Gamma * Gamma - adx + 1) / Gamma;
151}

153/**
154 * <!-- ************************************************************************************** -->
155 * @brief Computes the derivative of the effective Lorentz factor with respect to the bulk Lorentz factor.
156 * @param adx Adiabatic index of the medium
157 * @param Gamma Bulk Lorentz factor
158 * @return The derivative of the effective Lorentz factor
159 * <!-- ************************************************************************************** -->

161inline Real compute_effective_Gamma_dGamma(Real adx, Real Gamma) {
162 Real Gamma2 = Gamma * Gamma;
163 return (adx * Gamma2 + adx - 1) / Gamma2;

178/**

189/**
190 * <!-- ************************************************************************************** -->
191 * @brief Computes the time derivative of theta (dθ/dt) for jet spreading.
192 * @param theta_s Typical spreading angle of the jet
193 * @param theta Theta of the current grid point
194 * @param drdt Time derivative of radius
195 * @param r Current radius
196 * @param Gamma Current bulk Lorentz factor
197 * @return The rate of change of the half-opening angle
198 * <!-- ************************************************************************************** -->
199 */
200inline Real compute_dtheta_dt(Real theta_s, Real theta, Real drdt, Real r, Real Gamma) {
201 constexpr Real Q = 7;
202 Real u2 = Gamma * Gamma - 1;
203 Real u = std::sqrt(u2);
204 Real f = 1 / (1 + u * theta_s * Q);
205 return drdt / (2 * Gamma * r) * std::sqrt((2 * u2 + 3) / (4 * u2 + 3)) * f;
206}

213 * @return The rate of change of comoving time with respect to observer time

232/**
233 * <!-- ************************************************************************************** -->
234 * @brief Computes the relative Lorentz factor between two frames with given Lorentz factors.
235 * @param gamma1 First frame's Lorentz factor
236 * @param gamma2 Second frame's Lorentz factor
237 * @return The relative Lorentz factor between the two frames
238 * <!-- ************************************************************************************** -->
239 */
240inline Real compute_rel_Gamma(Real gamma1, Real gamma2) {
241 return gamma1 * gamma2 - std::sqrt(std::fabs((gamma1 * gamma1 - 1) * (gamma2 * gamma2 - 1)));
242}

244/**
245 * <!-- ************************************************************************************** -->
246 * @brief Computes the relative Lorentz factor between two frames with given Lorentz factors and velocities.
247 * @param gamma1 First frame's Lorentz factor
248 * @param gamma2 Second frame's Lorentz factor
249 * @param beta1 First frame's velocity as fraction of light speed
250 * @param beta2 Second frame's velocity as fraction of light speed
251 * @return The relative Lorentz factor between the two frames
252 * <!-- ************************************************************************************** -->
253 */
254inline Real compute_rel_Gamma(Real gamma1, Real gamma2, Real beta1, Real beta2) {
255 return gamma1 * gamma2 * (1 - beta1 * beta2);
256}

258/**
259 * <!-- ************************************************************************************** -->
260 * @brief Computes a Lorentz factor from a reference Lorentz factor and relative Lorentz factor.
261 * @param gamma4 Reference Lorentz factor (typically for region 4, unshocked ejecta)
262 * @param gamma_rel Relative Lorentz factor
263 * @return The derived Lorentz factor
264 * <!-- ************************************************************************************** -->
265 */
266inline Real compute_Gamma_from_relative(Real gamma4, Real gamma_rel) {
267 Real b = -2 * gamma4 * gamma_rel;
268 Real c = gamma4 * gamma4 + gamma_rel * gamma_rel - 1;
269 return (-b - std::sqrt(std::fabs(b * b - 4 * c))) / 2;
270}

284/**
285 * <!-- ************************************************************************************** -->
286 * @brief Computes the adiabatic cooling rate.
287 * @param ad_idx Adiabatic index
288 * @param r Radius
289 * @param Gamma Lorentz factor
290 * @param u Internal energy density
291 * @param drdt Rate of change of radius
292 * @param dGammadt Rate of change of Lorentz factor
293 * @return The adiabatic cooling rate
294 * <!-- ************************************************************************************** -->
295 */
296inline Real compute_adiabatic_cooling_rate(Real ad_idx, Real r, Real Gamma, Real u, Real drdt, Real dGammadt) {
297 return -(ad_idx - 1) * (3 * drdt / r - dGammadt / Gamma) * u;
298}

300inline Real compute_adiabatic_cooling_rate2(Real ad_idx, Real r, Real x, Real u, Real drdt, Real dxdt) {
301 Real dlnvdt = 2 * drdt / r;
302 if (x > 0) {
303 dlnvdt += dxdt / x;
304 }
305 return -(ad_idx - 1) * dlnvdt * u;
306}

320/**
321 * <!-- ************************************************************************************** -->
322 * @brief Computes the radiative efficiency based on the radiative constant, comoving time, Lorentz factor, and density.
323 * @param t_comv Comoving time
324 * @param Gamma_th Thermal Lorentz factor
325 * @param u internal energy density
326 * @param rad Radiation parameters
327 * @return The radiative efficiency
328 * <!-- ************************************************************************************** -->
329 */
330inline Real compute_radiative_efficiency(Real t_comv, Real Gamma_th, Real u, RadParams const& rad) { //
331 Real gamma_m = (rad.p - 2) / (rad.p - 1) * rad.eps_e * (Gamma_th - 1) * con::mp / con::me / rad.xi_e + 1;
332 Real gamma_c = (6 * con::pi * con::me * con::c / con::sigmaT) / (rad.eps_B * u * t_comv);
333 gamma_c = 0.5 * (gamma_c + std::sqrt(gamma_c * gamma_c + 4));
334
335 Real g_m_g_c = std::fabs(gamma_m / gamma_c); // gamma_m/gamma_c
336 if (g_m_g_c < 1 && rad.p > 2) { // slow cooling
337 return rad.eps_e * fast_pow(g_m_g_c, rad.p - 2);
338 } else { // fast cooling or p<=2
339 return rad.eps_e;
340 }
341}

343/**
344 * <!-- ************************************************************************************** -->
345 * @brief Computes the thermal Lorentz factor from the thermal energy and mass.
346 * @param U_th Thermal energy
347 * @param mass Mass
348 * @return The thermal Lorentz factor
349 * <!-- ************************************************************************************** -->

351inline Real compute_Gamma_therm(Real U_th, Real mass, bool limiter = false) {
352 if (mass == 0) [[unlikely]] {
353 return 1;
354 } else [[likely]] {
355 Real Gamma_th = U_th / (mass * con::c2) + 1;
356 if (limiter && Gamma_th < con::Gamma_cut) {
357 return 1;

359 return Gamma_th;

406/**
407 * <!-- ************************************************************************************** -->
408 * @brief Computes the downstream magnetic field strength.
409 * @param eps_B Magnetic energy fraction
410 * @param rho_upstr Upstream rest mass density
411 * @param B_upstr Upstream magnetic field strength
412 * @param Gamma_th Thermal Lorentz factor
413 * @param comp_ratio Compression ratio
414 * @return The downstream magnetic field strength
415 * <!-- ************************************************************************************** -->
416 */
417inline Real compute_downstr_B(Real eps_B, Real rho_upstr, Real B_upstr, Real Gamma_th, Real comp_ratio) {
418 Real rho_downstr = rho_upstr * comp_ratio;
419
420 Real e_th = (Gamma_th - 1) * rho_downstr * con::c2;
421
422 return compute_comv_weibel_B(eps_B, e_th) + B_upstr * comp_ratio;
423}
