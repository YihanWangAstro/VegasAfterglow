94template <typename Ejecta, typename Medium>
95void Shock::detect_symmetry(Coord const& coord, Ejecta const& jet, Medium const& medium) {
96 auto [phi_size_, theta_size_, t_size_] = coord.shape();
97
98 if (jet.spreading || !medium.isotropic) {
99 symmetry = Symmetry::structured;
100 theta_reps.resize(theta_size_);
101 std::iota(theta_reps.begin(), theta_reps.end(), size_t(0));
102 return;
103 }
104
105 const Real phi0 = coord.phi(0);
106 theta_reps.clear();
107 theta_reps.reserve(theta_size_);
108 theta_reps.push_back(0);
109
110 auto jet_ic_differs = [&](size_t ja, size_t jb) {
111 const Real theta_a = coord.theta(ja), theta_b = coord.theta(jb);
112 if (jet.eps_k(phi0, theta_a) != jet.eps_k(phi0, theta_b))
113 return true;
114 if (jet.Gamma0(phi0, theta_a) != jet.Gamma0(phi0, theta_b))
115 return true;
116 if constexpr (HasSigma<Ejecta>) {
117 if (jet.sigma0(phi0, theta_a) != jet.sigma0(phi0, theta_b))
118 return true;
119 }
120 if constexpr (HasDedt<Ejecta>) {
121 for (size_t k = 0; k < t_size_; ++k) {
122 const Real ta = coord.t(0, ja, k), tb = coord.t(0, jb, k);
123 if (jet.deps_dt(phi0, theta_a, ta) != jet.deps_dt(phi0, theta_b, ta))
124 return true;
125 if (jet.deps_dt(phi0, theta_a, tb) != jet.deps_dt(phi0, theta_b, tb))
126 return true;
127 }
128 }
129 if constexpr (HasDmdt<Ejecta>) {
130 for (size_t k = 0; k < t_size_; ++k) {
131 const Real ta = coord.t(0, ja, k), tb = coord.t(0, jb, k);
132 if (jet.dm_dt(phi0, theta_a, ta) != jet.dm_dt(phi0, theta_b, ta))
133 return true;
134 if (jet.dm_dt(phi0, theta_a, tb) != jet.dm_dt(phi0, theta_b, tb))
135 return true;
136 }
137 }
138 return false;
139 };
140
141 for (size_t j = 1; j < theta_size_; ++j) {
142 if (jet_ic_differs(j - 1, j)) {
143 theta_reps.push_back(j);
144 }
145 }
146
147 if (theta_reps.size() == 1)
148 symmetry = Symmetry::isotropic;
149 else if (theta_reps.size() < theta_size_)
150 symmetry = Symmetry::piecewise;
151 else
152 symmetry = Symmetry::phi_symmetric;
153}

166/**
167 * <!-- ************************************************************************************** -->
168 * @brief Computes the downstream four-velocity for a given relative Lorentz factor and magnetization parameter.
169 * @param gamma_rel Relative Lorentz factor between upstream and downstream regions
170 * @param sigma Magnetization parameter (ratio of magnetic to rest-mass energy density)
171 * @return The downstream four-velocity in the shock frame
172 * @details The calculation handles both magnetized (sigma > 0) and non-magnetized (sigma = 0) cases using
173 * different equations based on jump conditions across the shock front.
174 * <!-- ************************************************************************************** -->
175 */
176Real compute_downstr_4vel(Real gamma_rel, Real sigma);
177
178/**
179 * <!-- ************************************************************************************** -->
180 * @brief Computes the upstream four-velocity from downstream four-velocity and relative Lorentz factor.
181 * @param u_down Downstream four-velocity
182 * @param gamma_rel Relative Lorentz factor
183 * @return The upstream four-velocity in the shock frame
184 * <!-- ************************************************************************************** -->
185 */
186inline Real compute_upstr_4vel(Real u_down, Real gamma_rel) {
187 return std::sqrt((1 + u_down * u_down) * std::fabs(gamma_rel * gamma_rel - 1)) + u_down * gamma_rel;
188}
189
190/**
191 * <!-- ************************************************************************************** -->
192 * @brief Computes the ratio of upstream to downstream four-velocity across the shock front.
193 * @param gamma_rel Relative Lorentz factor between upstream and downstream regions
194 * @param sigma_upstr Magnetization parameter (ratio of magnetic to rest-mass energy density)
195 * @return The ratio of upstream to downstream four-velocity
196 * @details This ratio is a key parameter in determining various shock properties, such as compression ratio
197 * and jump conditions for density, pressure, and magnetic field.
198 * <!-- ************************************************************************************** -->
199 */
200inline Real compute_4vel_jump(Real gamma_rel, Real sigma_upstr) {
201 const Real u_down_s_ = compute_downstr_4vel(gamma_rel, sigma_upstr);
202 const Real u_up_s_ = compute_upstr_4vel(u_down_s_, gamma_rel);
203 Real ratio_u = u_up_s_ / u_down_s_;
204 if (u_down_s_ == 0.) {
205 ratio_u = 4 * gamma_rel; // (g_hat*gamma_rel+1)/(g_hat-1)
206 }
207 return ratio_u;
208}

210/**
211 * <!-- ************************************************************************************** -->
212 * @brief Computes the sound speed in the shocked medium based on the relative Lorentz factor.
213 * @param Gamma_rel Relative Lorentz factor between upstream and downstream regions
214 * @return The sound speed as a fraction of light speed
215 * <!-- ************************************************************************************** -->
216 */
217inline Real compute_sound_speed(Real Gamma_rel) {
218 const Real ad_idx = physics::thermo::adiabatic_idx(Gamma_rel);
219 return std::sqrt(std::fabs(ad_idx * (ad_idx - 1) * (Gamma_rel - 1) / (1 + (Gamma_rel - 1) * ad_idx))) * con::c;
220}

222/**
223 * <!-- ************************************************************************************** -->
224 * @brief Computes the effective Lorentz factor accounting for the adiabatic index.
225 * @param adx Adiabatic index of the medium
226 * @param Gamma Bulk Lorentz factor
227 * @return The effective Lorentz factor
228 * <!-- ************************************************************************************** -->
229 */
230inline Real compute_effective_Gamma(Real adx, Real Gamma) {
231 return (adx * Gamma * Gamma - adx + 1) / Gamma;
232}

234/**
235 * <!-- ************************************************************************************** -->
236 * @brief Computes the derivative of the effective Lorentz factor with respect to the bulk Lorentz factor.
237 * @param adx Adiabatic index of the medium
238 * @param Gamma Bulk Lorentz factor
239 * @return The derivative of the effective Lorentz factor
240 * <!-- ************************************************************************************** -->
241 */
242inline Real compute_effective_Gamma_dGamma(Real adx, Real Gamma) {
243 const Real Gamma2 = Gamma * Gamma;
244 return (adx * Gamma2 + adx - 1) / Gamma2;
245}

259/**
260 * <!-- ************************************************************************************** -->
261 * @brief Computes the time derivative of radius (dr/dt) based on the shock velocity.
262 * @param beta Shock velocity as a fraction of light speed
263 * @return The rate of change of radius with respect to observer time
264 * <!-- ************************************************************************************** -->

270/// Overload using four-velocity u and Lorentz factor Gamma directly.
271/// dr/dt = c/(1-) = uc(+u), exploiting -u=1 to avoid divisions and 1- cancellation.
272inline Real compute_dr_dt(Real Gamma, Real u) {
273 return u * (Gamma + u) * con::c;

276/**
277 * <!-- ************************************************************************************** -->
278 * @brief Computes the time derivative of theta (d/dt) for jet spreading.
279 * @param theta_s Typical spreading angle of the jet
280 * @param theta Theta of the current grid point
281 * @param drdt Time derivative of radius
282 * @param r Current radius
283 * @param Gamma Current bulk Lorentz factor
284 * @return The rate of change of the half-opening angle
285 * <!-- ************************************************************************************** -->
286 */
287inline Real compute_dtheta_dt(Real theta_s, Real theta, Real drdt, Real r, Real Gamma) {
288 constexpr Real Q = 7;
289 const Real u2 = Gamma * Gamma - 1;
290 const Real u = std::sqrt(u2);
291 const Real f = 1 / (1 + u * theta_s * Q);
292 return drdt / (2 * Gamma * r) * std::sqrt((2 * u2 + 3) / (4 * u2 + 3)) * f;
293}

319/**
320 * <!-- ************************************************************************************** -->
321 * @brief Computes the relative Lorentz factor between two frames with given Lorentz factors.
322 * @param gamma1 First frame's Lorentz factor
323 * @param gamma2 Second frame's Lorentz factor
324 * @return The relative Lorentz factor between the two frames
325 * <!-- ************************************************************************************** -->
326 */
327inline Real compute_rel_Gamma(Real gamma1, Real gamma2) {
328 return gamma1 * gamma2 - std::sqrt(std::fabs((gamma1 * gamma1 - 1) * (gamma2 * gamma2 - 1)));
329}

331/**
332 * <!-- ************************************************************************************** -->
333 * @brief Computes the relative Lorentz factor between two frames with given Lorentz factors and velocities.
334 * @param gamma1 First frame's Lorentz factor
335 * @param gamma2 Second frame's Lorentz factor
336 * @param beta1 First frame's velocity as a fraction of light speed
337 * @param beta2 Second frame's velocity as a fraction of light speed
338 * @return The relative Lorentz factor between the two frames
339 * <!-- ************************************************************************************** -->
340 */
341inline Real compute_rel_Gamma(Real gamma1, Real gamma2, Real beta1, Real beta2) {
342 return gamma1 * gamma2 * (1 - beta1 * beta2);

345/**
346 * <!-- ************************************************************************************** -->
347 * @brief Computes a Lorentz factor from a reference Lorentz factor and relative Lorentz factor.
348 * @param gamma4 Reference Lorentz factor (typically for region 4, unshocked ejecta)
349 * @param gamma_rel Relative Lorentz factor
350 * @return The derived Lorentz factor
351 * <!-- ************************************************************************************** -->
352 */
353inline Real compute_Gamma_from_relative(Real gamma4, Real gamma_rel) {
354 const Real b = -2 * gamma4 * gamma_rel;
355 const Real c = gamma4 * gamma4 + gamma_rel * gamma_rel - 1;
356 return (-b - std::sqrt(std::fabs(b * b - 4 * c))) / 2;
357}

371/**
372 * <!-- ************************************************************************************** -->
373 * @brief Computes the adiabatic cooling rate.
374 * @param ad_idx Adiabatic index
375 * @param r Radius
376 * @param Gamma Lorentz factor
377 * @param u Internal energy density
378 * @param drdt Rate of change of radius
379 * @param dGammadt Rate of change of the Lorentz factor
380 * @return The adiabatic cooling rate
381 * <!-- ************************************************************************************** -->
382 */
383inline Real compute_adiabatic_cooling_rate(Real ad_idx, Real r, Real Gamma, Real u, Real drdt, Real dGammadt) {
384 return -(ad_idx - 1) * (3 * drdt / r - dGammadt / Gamma) * u;
385}

387inline Real compute_adiabatic_cooling_rate2(Real ad_idx, Real r, Real x, Real u, Real drdt, Real dxdt) {
388 Real dlnvdt = 2 * drdt / r;
389 if (x > 0) {
390 dlnvdt += dxdt / x;
391 }
392 return -(ad_idx - 1) * dlnvdt * u;
393}

409 * @brief Computes the radiative efficiency based on the radiative constant, comoving time, Lorentz factor, and density.
410 * @param t_comv Comoving time

412 * @param u internal energy density
413 * @param rad Radiation parameters

417inline Real compute_radiative_efficiency(Real t_comv, Real Gamma_th, Real u, RadParams const& rad) { //

540/// @brief Simpson's rule integration in log-space: f(r') dr' via u = ln(r').
541/// Integrates from r * e^{-18} to r with N=32 panels.
542template <typename Func>
543Real simpson_logspace(Func const& f, Real r) {
544 constexpr size_t N = 32; // must be even
545 const Real u_max = std::log(r);
546 const Real u_min = u_max - 18; // start from r * e^{-18} r * 1.5e-8
547 const Real h = (u_max - u_min) / N;
548
549 Real sum = f(u_min) + f(u_max);
550 for (size_t i = 1; i < N; i += 2)
551 sum += 4 * f(u_min + i * h);
552 for (size_t i = 2; i < N; i += 2)
553 sum += 2 * f(u_min + i * h);
554
555 return sum * h / 3;
556}

558/// @brief Computes enclosed mass per solid angle.
559template <typename Func>
560Real enclosed_mass(Func const& rho, Real r) {
561 return simpson_logspace(
562 [&](Real u) {
563 const Real ri = std::exp(u);
564 return rho(ri) * ri * ri * ri; // (r') * r' (Jacobian from u = ln r')
565 },
566 r);
567}

569/// @brief Compute enclosed thermal energy per solid angle,
570/// accounting for adiabatic cooling and radiative losses of material swept at earlier radii.
571template <typename Func>
572Real enclosed_thermal_energy(Func const& rho, Real r, Real Gamma, Real ad_idx, Real eps_e) {
573 const Real cooling_exp = 3 * (ad_idx - 1);
574 return (1 - eps_e) * (Gamma - 1) * con::c2 *
575 simpson_logspace(
576 [&](Real u) {
577 const Real ri = std::exp(u);
578 return rho(ri) * ri * ri * ri * std::pow(ri / r, cooling_exp);
579 },
580 r);
581}

583/// @brief Finds the deceleration observer time by integrating swept mass outward until m_swept = m_jet/Gamma0.
584/// Works for any medium density profile. Returns observer time r_dec*(1-beta)/(beta*c).
585template <typename Eqn>
586Real compute_dec_time(Eqn const& eqn) {
587 const Real gamma = eqn.ejecta.Gamma0(eqn.phi, eqn.theta0);
588 const Real beta = physics::relativistic::gamma_to_beta(gamma);
589 Real m_jet = eqn.ejecta.eps_k(eqn.phi, eqn.theta0) / (gamma * con::c2);
590 if constexpr (HasSigma<decltype(eqn.ejecta)>) {
591 m_jet /= (1.0 + eqn.ejecta.sigma0(eqn.phi, eqn.theta0));
592 }
593 const Real target = m_jet / gamma;
594
595 auto rho = [&](Real r) { return eqn.medium.rho(eqn.phi, eqn.theta0, r); };
596
597 // Trapezoidal integration in log-space with early exit at deceleration radius
598 constexpr size_t N = 256;
599 const Real u_min = std::log(1e-3);
600 const Real u_max = u_min + 40 * std::log(10.0);
601 const Real du = (u_max - u_min) / N;
602
603 Real mass = 0;
604 Real r_prev = std::exp(u_min);
605 Real f_prev = rho(r_prev) * r_prev * r_prev;
606
607 for (size_t i = 1; i <= N; ++i) {
608 const Real r_i = std::exp(u_min + i * du);
609 const Real f_i = rho(r_i) * r_i * r_i;
610 const Real dr = r_i - r_prev;
611 mass += 0.5 * (f_prev + f_i) * dr;
612
613 if (mass >= target) {
614 const Real r_dec = r_prev + (target - (mass - 0.5 * (f_prev + f_i) * dr)) / f_i;
615 return r_dec * (1 - beta) / (beta * con::c);
616 }
617 f_prev = f_i;
618 r_prev = r_i;
619 }
620 return std::exp(u_max) * (1 - beta) / (beta * con::c);
621}
