97template <typename Ejecta, typename Medium>
98void Shock::detect_symmetry(Coord const& coord, Ejecta const& jet, Medium const& medium) {
99 auto [phi_size_, theta_size_, t_size_] = coord.shape();
100
101 if (jet.spreading || !medium.isotropic) {
102 symmetry = Symmetry::structured;
103 theta_reps.resize(theta_size_);
104 std::iota(theta_reps.begin(), theta_reps.end(), size_t(0));
105 return;
106 }
107
108 const Real phi0 = coord.phi(0);
109 theta_reps.clear();
110 theta_reps.reserve(theta_size_);
111 theta_reps.push_back(0);
112
113 auto jet_ic_differs = [&](size_t ja, size_t jb) {
114 const Real theta_a = coord.theta(ja), theta_b = coord.theta(jb);
115 if (jet.eps_k(phi0, theta_a) != jet.eps_k(phi0, theta_b))
116 return true;
117 if (jet.Gamma0(phi0, theta_a) != jet.Gamma0(phi0, theta_b))
118 return true;
119 if constexpr (HasSigma<Ejecta>) {
120 if (jet.sigma0(phi0, theta_a) != jet.sigma0(phi0, theta_b))
121 return true;
122 }
123 if constexpr (HasDedt<Ejecta>) {
124 for (size_t k = 0; k < t_size_; ++k) {
125 const Real ta = coord.t(0, ja, k), tb = coord.t(0, jb, k);
126 if (jet.deps_dt(phi0, theta_a, ta) != jet.deps_dt(phi0, theta_b, ta))
127 return true;
128 if (jet.deps_dt(phi0, theta_a, tb) != jet.deps_dt(phi0, theta_b, tb))
129 return true;
130 }
131 }
132 if constexpr (HasDmdt<Ejecta>) {
133 for (size_t k = 0; k < t_size_; ++k) {
134 const Real ta = coord.t(0, ja, k), tb = coord.t(0, jb, k);
135 if (jet.dm_dt(phi0, theta_a, ta) != jet.dm_dt(phi0, theta_b, ta))
136 return true;
137 if (jet.dm_dt(phi0, theta_a, tb) != jet.dm_dt(phi0, theta_b, tb))
138 return true;
139 }
140 }
141 return false;
142 };
143
144 for (size_t j = 1; j < theta_size_; ++j) {
145 if (jet_ic_differs(j - 1, j)) {
146 theta_reps.push_back(j);
147 }
148 }
149
150 if (theta_reps.size() == 1)
151 symmetry = Symmetry::isotropic;
152 else if (theta_reps.size() < theta_size_)
153 symmetry = Symmetry::piecewise;
154 else
155 symmetry = Symmetry::phi_symmetric;
156}
