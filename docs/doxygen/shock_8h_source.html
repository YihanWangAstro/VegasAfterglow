88 * <!-- ************************************************************************************** -->
89 * @brief Computes the downstream four-velocity for a given relative Lorentz factor and magnetization parameter.

99/**
100 * <!-- ************************************************************************************** -->
101 * @brief Computes the ratio of upstream to downstream four-velocity across the shock front.
102 * @param gamma_rel Relative Lorentz factor between upstream and downstream regions
103 * @param sigma Magnetization parameter (ratio of magnetic to rest-mass energy density)
104 * @return The ratio of upstream to downstream four-velocity
105 * @details This ratio is a key parameter in determining various shock properties, such as compression ratio
106 * and jump conditions for density, pressure, and magnetic field.
107 * <!-- ************************************************************************************** -->
108 */
109Real compute_4vel_jump(Real gamma_rel, Real sigma);
110
111/**
112 * <!-- ************************************************************************************** -->
113 * @brief Computes the sound speed in the shocked medium based on the relative Lorentz factor.
114 * @param Gamma_rel Relative Lorentz factor between upstream and downstream regions
115 * @return The sound speed as a fraction of light speed
116 * <!-- ************************************************************************************** -->
117 */
118inline Real compute_sound_speed(Real Gamma_rel) {
119 Real ad_idx = adiabatic_idx(Gamma_rel);
120 return std::sqrt(ad_idx * (ad_idx - 1) * (Gamma_rel - 1) / (1 + (Gamma_rel - 1) * ad_idx));
121}
122
123/**
124 * <!-- ************************************************************************************** -->
125 * @brief Computes the effective Lorentz factor accounting for the adiabatic index.
126 * @param adx Adiabatic index of the medium
127 * @param Gamma Bulk Lorentz factor
128 * @return The effective Lorentz factor
129 * <!-- ************************************************************************************** -->
130 */
131inline Real compute_effective_Gamma(Real adx, Real Gamma) { return (adx * Gamma * Gamma - adx + 1) / Gamma; }
132
133/**
134 * <!-- ************************************************************************************** -->
135 * @brief Computes the comoving magnetic field using the Weibel instability mechanism.
136 * @param eps_B Fraction of thermal energy in magnetic fields
137 * @param e_thermal Thermal energy density
138 * @return The comoving magnetic field strength
139 * <!-- ************************************************************************************** -->
140 */
141inline Real compute_comv_weibel_B(Real eps_B, Real e_thermal) { return std::sqrt(8 * con::pi * eps_B * e_thermal); }

143/**
144 * <!-- ************************************************************************************** -->
145 * @brief Computes the time derivative of radius (dr/dt) based on the shock velocity.
146 * @param beta Shock velocity as a fraction of light speed
147 * @return The rate of change of radius with respect to observer time
148 * <!-- ************************************************************************************** -->
149 */
150inline Real compute_dr_dt(Real beta) { return (beta * con::c) / (1 - beta); }

152/**
153 * <!-- ************************************************************************************** -->
154 * @brief Computes the time derivative of theta (dÎ¸/dt) for jet spreading.
155 * @param theta_s Typical spreading angle of the jet
156 * @param theta Theta of the current grid point
157 * @param drdt Time derivative of radius
158 * @param r Current radius
159 * @param Gamma Current bulk Lorentz factor
160 * @return The rate of change of the half-opening angle
161 * <!-- ************************************************************************************** -->
162 */
163inline Real compute_dtheta_dt(Real theta_s, Real theta, Real drdt, Real r, Real Gamma) {
164 constexpr Real Q = 2.82842712475;
165 Real u2 = Gamma * Gamma - 1;
166 Real ratio = u2 / (Q * Q * theta_s * theta_s);
167 Real x = theta / theta_s;
168 Real f = 1 / (1 + ratio) * x;
169 return drdt / (2 * Gamma * r) * std::sqrt((2 * u2 + 3) / (4 * u2 + 3)) * f;
170}

172/**
173 * <!-- ************************************************************************************** -->
174 * @brief Computes the time derivative of comoving time (dt_comv/dt) based on the Lorentz factor.
175 * @param Gamma Bulk Lorentz factor
176 * @param beta Bulk velocity as a fraction of light speed
177 * @return The rate of change of comoving time with respect to observer time
178 * <!-- ************************************************************************************** -->
179 */
180inline Real compute_dt_dt_comv(Real Gamma, Real beta) { return 1 / (Gamma * (1 - beta)); };

182/**
183 * <!-- ************************************************************************************** -->
184 * @brief Computes the upstream magnetic pressure based on number density and magnetization.
185 * @param n_up Upstream number density
186 * @param sigma Magnetization parameter
187 * @return The upstream magnetic pressure
188 * <!-- ************************************************************************************** -->
189 */
190inline Real compute_upstr_mag_p(Real n_up, Real sigma) { return sigma * n_up * con::mp * con::c2 / 2; }

192/**
193 * <!-- ************************************************************************************** -->
194 * @brief Computes the upstream four-velocity from downstream four-velocity and relative Lorentz factor.
195 * @param u_down Downstream four-velocity
196 * @param gamma_rel Relative Lorentz factor
197 * @return The upstream four-velocity in the shock frame
198 * <!-- ************************************************************************************** -->
199 */
200inline Real compute_upstr_4vel(Real u_down, Real gamma_rel) {
201 return std::sqrt((1 + u_down * u_down) * (gamma_rel * gamma_rel - 1)) + u_down * gamma_rel;
202}

204/**
205 * <!-- ************************************************************************************** -->
206 * @brief Computes the downstream number density from upstream density, relative Lorentz factor, and magnetization.
207 * @param n_up_str Upstream number density
208 * @param gamma_rel Relative Lorentz factor
209 * @param sigma Magnetization parameter
210 * @return The downstream number density
211 * @details Uses the shock jump conditions to determine the density of the downstream region.
212 * <!-- ************************************************************************************** -->
213 */
214inline Real compute_downstr_num_den(Real n_up_str, Real gamma_rel, Real sigma) {
215 return n_up_str * compute_4vel_jump(gamma_rel, sigma);
216}

218/**
219 * <!-- ************************************************************************************** -->
220 * @brief Computes the relative Lorentz factor between two frames with given Lorentz factors.
221 * @param gamma1 First frame's Lorentz factor
222 * @param gamma2 Second frame's Lorentz factor
223 * @return The relative Lorentz factor between the two frames
224 * <!-- ************************************************************************************** -->
225 */
226inline Real compute_rel_Gamma(Real gamma1, Real gamma2) {
227 return gamma1 * gamma2 - std::sqrt((gamma1 * gamma1 - 1) * (gamma2 * gamma2 - 1));
228}
229
230/**
231 * <!-- ************************************************************************************** -->
232 * @brief Computes the relative Lorentz factor between two frames with given Lorentz factors and velocities.
233 * @param gamma1 First frame's Lorentz factor
234 * @param gamma2 Second frame's Lorentz factor
235 * @param beta1 First frame's velocity as fraction of light speed
236 * @param beta2 Second frame's velocity as fraction of light speed
237 * @return The relative Lorentz factor between the two frames
238 * <!-- ************************************************************************************** -->
239 */
240inline Real compute_rel_Gamma(Real gamma1, Real gamma2, Real beta1, Real beta2) {
241 return gamma1 * gamma2 * (1 - beta1 * beta2);
242}

244/**
245 * <!-- ************************************************************************************** -->
246 * @brief Computes a Lorentz factor from a reference Lorentz factor and relative Lorentz factor.
247 * @param gamma4 Reference Lorentz factor (typically for region 4, unshocked ejecta)
248 * @param gamma_rel Relative Lorentz factor
249 * @return The derived Lorentz factor
250 * <!-- ************************************************************************************** -->
251 */
252inline Real compute_Gamma_from_relative(Real gamma4, Real gamma_rel) {
253 Real b = -2 * gamma4 * gamma_rel;
254 Real c = gamma4 * gamma4 + gamma_rel * gamma_rel - 1;
255 return (-b - std::sqrt(b * b - 4 * c)) / 2;
256}

258/**
259 * <!-- ************************************************************************************** -->
260 * @brief Computes the downstream thermal energy density.
261 * @param gamma_rel Relative Lorentz factor
262 * @param n_down_str Downstream number density

264 * <!-- ************************************************************************************** -->
265 */
266inline Real compute_downstr_eth(Real gamma_rel, Real n_down_str) {

284 * @brief Computes the number density in region 4 (unshocked ejecta).
285 * @param dEdOmega Energy per solid angle

369template <typename State>
370void save_shock_state(Shock& shock, size_t i, size_t j, size_t k, State const& state, Real Gamma_downstr,
371 Real Gamma_upstr, Real N_downstr, Real n_upstr, Real sigma_upstr) {
372 Real Gamma_rel = compute_rel_Gamma(Gamma_upstr, Gamma_downstr);
373 Real ratio_u = compute_4vel_jump(Gamma_rel, sigma_upstr);
374 Real pB_upstr = compute_upstr_mag_p(n_upstr, sigma_upstr);
375 Real pB_downstr = pB_upstr * ratio_u * ratio_u;
376 Real n_downstr = n_upstr * ratio_u;
377 Real e_th = compute_downstr_eth(Gamma_rel, n_downstr);
378 shock.t_comv(i, j, k) = state.t_comv;
379 shock.r(i, j, k) = state.r;
380 shock.theta(i, j, k) = state.theta;
381 shock.Gamma(i, j, k) = Gamma_downstr;
382 shock.Gamma_rel(i, j, k) = Gamma_rel;
383 shock.B(i, j, k) = compute_comv_weibel_B(shock.eps_B, e_th) + std::sqrt(pB_downstr * 8 * con::pi);
384 shock.column_num_den(i, j, k) = N_downstr / (state.r * state.r);
385}

387template <typename Eqn>
388Real compute_dec_time(Eqn const& eqn, Real t_max) {
389 Real e_k = eqn.ejecta.eps_k(eqn.phi, eqn.theta0);
390 Real gamma = eqn.ejecta.Gamma0(eqn.phi, eqn.theta0);
391 Real beta = gamma_to_beta(gamma);
392
393 Real m_shell = e_k / (gamma * con::c2);
394
395 if constexpr (HasSigma<decltype(eqn.ejecta)>) {
396 m_shell /= 1 + eqn.ejecta.sigma0(eqn.phi, eqn.theta0);
397 }
398
399 Real r_dec = beta * con::c * t_max / (1 - beta);
400 for (size_t i = 0; i < 30; i++) {
401 Real m_swept = eqn.medium.mass(eqn.phi, eqn.theta0, r_dec);
402 if (m_swept < m_shell / gamma) {
403 break;
404 }
405 r_dec /= 3;
406 }
407 Real t_dec = r_dec * (1 - beta) / (beta * con::c);
408 return t_dec;
