12/**
13 * @file simulation-defaults.h
14 * @brief Centralized simulation configuration defaults
15 *
16 * This header consolidates all tunable simulation parameters in one place.
17 * These are numerical/algorithmic parameters, NOT physical constants.
18 *
19 * Benefits:
20 * - Single source of truth for all defaults
21 * - Easy to create different presets (fast vs. accurate)
22 * - Clear documentation of what each parameter does
23 * - No magic numbers scattered across the codebase
24 */
25
26namespace defaults {
27
28 /**
29 * @brief Physics cutoff thresholds
30 *
31 * These values determine when certain physical effects are negligible
32 * and can be safely ignored for numerical efficiency.
33 */
34 namespace cutoffs {
35 /// Lorentz factor below which the flow is considered non-relativistic
36 /// Used to terminate shock evolution when Gamma approaches 1
37 inline constexpr double gamma_cut = 1.0 + 1e-6;
38
39 /// Magnetization parameter below which magnetic effects are ignored
40 /// sigma = B^2 / (4*pi*rho*c^2) - ratio of magnetic to rest-mass energy
41 inline constexpr double sigma_cut = 1e-6;
42 } // namespace cutoffs
43
44 /**
45 * @brief Grid generation parameters
46 *
47 * These control the adaptive mesh refinement and resolution
48 * for the (phi, theta, t) computational grid.
49 */
50 namespace grid {
51 /// Azimuthal (phi) angular resolution factor
52 /// Higher values = more phi grid points near the viewing direction
53 inline constexpr double phi_resolution = 0.15;
54
55 /// Polar (theta) angular resolution factor
56 /// Higher values = more theta grid points
57 inline constexpr double theta_resolution = 0.5;
58
59 /// Temporal resolution factor
60 /// Higher values = more time grid points (logarithmically spaced)
61 inline constexpr double time_resolution = 10.0;
62
63 /// Minimum number of theta grid points
64 /// Ensures adequate angular coverage even for narrow jets
65 inline constexpr std::size_t min_theta_points = 32;
66
67 /// Forward ratio for adaptive grid refinement
68 /// Controls distribution of grid points relative to jet edge
69 inline constexpr double forward_ratio = 0.4;
70
71 /// Minimum polar angle (avoids singularity at theta=0)
72 inline constexpr double theta_min = 1e-6;
73 } // namespace grid
74
75 /**
76 * @brief Numerical solver tolerances
77 *
78 * These control the accuracy of ODE integration and root-finding.
79 */
80 namespace solver {
81 /// Relative tolerance for ODE integration (Runge-Kutta)
82 inline constexpr double ode_rtol = 1e-6;
83
84 /// Absolute tolerance for ODE integration
85 inline constexpr double ode_atol = 1e-9;
86
87 /// Tolerance for binary search (jet edge finding)
88 inline constexpr double binary_search_eps = 1e-9;
89
90 /// Tolerance for checking linear/log scale arrays
91 inline constexpr double scale_check_tol = 1e-6;
92 } // namespace solver
93
94 /**
95 * @brief Observer frame parameters
96 */
97 namespace observer {
98 /// Minimum observer time to avoid numerical issues at t=0
99 /// Value in seconds (will be converted to code units)
100 inline constexpr double min_obs_time_sec = 0.1;
101 } // namespace observer
102
103 /**
104 * @brief Sampling parameters for integration
105 */
106 namespace sampling {
107 /// Number of sample points for theta integration
108 inline constexpr std::size_t theta_samples = 200;
109 } // namespace sampling
110
111} // namespace defaults
