19std::vector<size_t> MultiBandData::logscale_screen(PyArray const& data, size_t num_order) {
20 size_t total_size = data.size();
21
22 if (num_order == 0) {
23 std::vector<size_t> indices(total_size);
24 std::iota(indices.begin(), indices.end(), 0);
25 return indices;
26 }
27
28 double log_start = std::log10(static_cast<double>(data(0)));
29 double log_end = std::log10(static_cast<double>(data(total_size - 1)));
30 double log_range = log_end - log_start;
31 size_t total_points = static_cast<size_t>(std::ceil(log_range * num_order)) + 1;
32
33 if (total_points >= total_size) {
34 std::vector<size_t> indices(total_size);
35 std::iota(indices.begin(), indices.end(), 0);
36 return indices;
37 }
38
39 std::vector<size_t> indices;
40 indices.reserve(total_points);
41
42 // Always include first point
43 indices.push_back(0);
44
45 double step = log_range / (total_points - 1);
46
47 for (size_t i = 1; i < total_points - 1; ++i) {
48 double log_target = log_start + i * step;
49 double target_value = std::pow(10.0, log_target);
50
51 // Find closest index to target value
52 size_t best_idx = 1;
53 double min_diff = std::abs(static_cast<double>(data(1)) - target_value);
54
55 for (size_t j = 2; j < total_size - 1; ++j) {
56 double diff = std::abs(static_cast<double>(data(j)) - target_value);
57 if (diff < min_diff) {
58 min_diff = diff;
59 best_idx = j;
60 }
61 }
62
63 double log_diff = std::abs(std::log10(static_cast<double>(data(best_idx))) - log_target);
64 double max_log_tolerance = 0.5 / num_order; // Scale tolerance with sampling density
65
66 if (log_diff <= max_log_tolerance && std::find(indices.begin(), indices.end(), best_idx) == indices.end()) {
67 indices.push_back(best_idx);
68 }
69 }
70
71 // Always include last point
72 if (total_size > 1) {
73 indices.push_back(total_size - 1);
74 }
75
76 std::sort(indices.begin(), indices.end());
77 indices.erase(std::unique(indices.begin(), indices.end()), indices.end());
78
79 return indices;
80}

83 double chi_square = 0;
84 for (size_t i = 0; i < t.size(); ++i) {

110Ejecta MultiBandModel::select_jet(Params const& param) {
111 Real eps_iso = param.E_iso * unit::erg / (4 * con::pi);
112 Real Gamma0 = param.Gamma0;
113 Real theta_c = param.theta_c;
114 Real theta_w = param.theta_w;
115 Real eps_iso_w = param.E_iso_w * unit::erg / (4 * con::pi);
116 Real Gamma0_w = param.Gamma0_w;
117 Ejecta jet;
118 jet.T0 = param.duration * unit::sec;
119 if (config.jet == "tophat") {
120 jet.eps_k = math::tophat(theta_c, eps_iso);
121 jet.Gamma0 = math::tophat_plus_one(theta_c, Gamma0 - 1);
122 } else if (config.jet == "gaussian") {
123 jet.eps_k = math::gaussian(theta_c, eps_iso);
124 jet.Gamma0 = math::gaussian_plus_one(theta_c, Gamma0 - 1);
125 } else if (config.jet == "powerlaw") {
126 jet.eps_k = math::powerlaw(theta_c, eps_iso, param.k_e);
127 jet.Gamma0 = math::powerlaw_plus_one(theta_c, Gamma0 - 1, param.k_g);
128 } else if (config.jet == "powerlaw_wing") {
129 jet.eps_k = math::powerlaw_wing(theta_c, eps_iso_w, param.k_e);
130 jet.Gamma0 = math::powerlaw_wing_plus_one(theta_c, Gamma0_w - 1, param.k_g);
131 } else if (config.jet == "uniform") {
132 jet.eps_k = math::tophat(con::pi / 2, eps_iso);
133 jet.Gamma0 = math::tophat_plus_one(con::pi / 2, Gamma0 - 1);
134 } else if (config.jet == "two_component") {
135 jet.eps_k = math::two_component(theta_c, theta_w, eps_iso, eps_iso_w);
136 jet.Gamma0 = math::two_component_plus_one(theta_c, theta_w, Gamma0 - 1, Gamma0_w - 1);
137 } else if (config.jet == "step_powerlaw") {
138 jet.eps_k = math::step_powerlaw(theta_c, eps_iso, eps_iso_w, param.k_e);
139 jet.Gamma0 = math::step_powerlaw_plus_one(theta_c, Gamma0 - 1, Gamma0_w - 1, param.k_g);
140 } else {
141 AFTERGLOW_ENSURE(false, "Unknown jet type");
142 }
143
144 if (config.magnetar == true) {
145 jet.deps_dt =
146 math::magnetar_injection(param.t0 * unit::sec, param.q, param.L0 * unit::erg / unit::sec, theta_c);
147 }
148 return jet;
149}

180void MultiBandData::add_flux(double nu_min, double nu_max, size_t num_points, PyArray const& t, PyArray const& Fv_obs,
181 PyArray const& Fv_err, std::optional<PyArray> weights) {
182 AFTERGLOW_REQUIRE(t.size() == Fv_obs.size() && t.size() == Fv_err.size(), "light curve array inconsistent length!");
183 AFTERGLOW_REQUIRE(is_ascending(t), "Time array must be in ascending order!");
184 AFTERGLOW_REQUIRE(nu_min < nu_max, "nu_min must be less than nu_max!");
185
186 Array w = xt::ones<Real>({t.size()});
187
188 if (weights) {
189 w = *weights;
190 AFTERGLOW_REQUIRE(t.size() == w.size(), "weights array inconsistent length!");
191
192 size_t len = w.size();
193 Real weight_sum = 0;
194 for (size_t i = 0; i < len; ++i) {
195 weight_sum += w(i);
196 }
197 w /= (weight_sum / len);
198 }
199
200 Array nu = xt::logspace(std::log10(nu_min * unit::Hz), std::log10(nu_max * unit::Hz), num_points);
201
202 flux_data.emplace_back(
203 FluxData{t * unit::sec, nu, Fv_obs * unit::flux_cgs, Fv_err * unit::flux_cgs, xt::zeros<Real>({t.size()}), w});
204}

231void MultiBandData::fill_data_arrays() {
232 const size_t len = tuple_data.size();
233 std::sort(tuple_data.begin(), tuple_data.end(),
234 [](auto const& a, auto const& b) { return std::get<0>(a) < std::get<0>(b); });
235 times = Array::from_shape({len});
236 frequencies = Array::from_shape({len});
237 fluxes = Array::from_shape({len});
238 errors = Array::from_shape({len});
239 model_fluxes = Array::from_shape({len});
240 weights = Array::from_shape({len});
241
242 Real weight_sum = 0;
243 for (size_t i = 0; i < len; ++i) {
244 times(i) = std::get<0>(tuple_data[i]);
245 frequencies(i) = std::get<1>(tuple_data[i]);
246 fluxes(i) = std::get<2>(tuple_data[i]);
247 errors(i) = std::get<3>(tuple_data[i]);
248 weights(i) = std::get<4>(tuple_data[i]);
249 model_fluxes(i) = 0; // Placeholder for model fluxes
250 weight_sum += weights(i);
251 }
252 weights /= (weight_sum / len);
253
254 if (len > 0) {
255 this->t_min = times.front();
256 this->t_max = times.back();
257 }
258
259 for (auto& d : flux_data) {
260 if (d.t.front() < t_min)
261 t_min = d.t.front();
262 if (d.t.back() > t_max)
263 t_max = d.t.back();
264 }
265}

278double MultiBandModel::estimate_chi2(Params const& param) {
279 Observer obs;
280 SynPhotonGrid f_photons;
281 SynPhotonGrid r_photons;
282 ICPhotonGrid<SynElectrons, SynPhotons> f_IC_photons;
283 ICPhotonGrid<SynElectrons, SynPhotons> r_IC_photons;
284
285 generate_photons(param, obs_data.t_min, obs_data.t_max, obs, f_photons, r_photons, f_IC_photons, r_IC_photons);
286
287 obs_data.model_fluxes = obs.specific_flux_series(obs_data.times, obs_data.frequencies, f_photons);
288 for (auto& d : obs_data.flux_data) {
289 d.Fv_model = obs.flux(d.t, d.nu, f_photons);
290 }
291
292 if (r_photons.size() > 0) {
293 obs_data.model_fluxes += obs.specific_flux_series(obs_data.times, obs_data.frequencies, r_photons);
294 for (auto& d : obs_data.flux_data) {
295 d.Fv_model += obs.flux(d.t, d.nu, r_photons);
296 }
297 }
298
299 if (f_IC_photons.size() > 0) {
300 obs_data.model_fluxes += obs.specific_flux_series(obs_data.times, obs_data.frequencies, f_IC_photons);
301 for (auto& d : obs_data.flux_data) {
302 d.Fv_model += obs.flux(d.t, d.nu, f_IC_photons);
303 }
304 }
305
306 if (r_IC_photons.size() > 0) {
307 obs_data.model_fluxes += obs.specific_flux_series(obs_data.times, obs_data.frequencies, r_IC_photons);
308 for (auto& d : obs_data.flux_data) {
309 d.Fv_model += obs.flux(d.t, d.nu, r_IC_photons);
310 }
311 }
312
313 return obs_data.estimate_chi2();
314}

317 Array t_bins = t * unit::sec;

329 F_nu = obs.specific_flux(t_bins, nu_bins, f_photons);
330
331 if (r_photons.size() > 0) {
332 F_nu += obs.specific_flux(t_bins, nu_bins, r_photons);
