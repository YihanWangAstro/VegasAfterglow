20std::vector<size_t> MultiBandData::logscale_screen(PyArray const& data, size_t num_order) {
21 const size_t total_size = data.size();
22
23 // Handle edge cases: empty or single-element arrays
24 if (total_size == 0) {
25 return {};
26 }
27 if (total_size == 1) {
28 return {0};
29 }
30
31 if (num_order == 0) {
32 std::vector<size_t> indices(total_size);
33 std::iota(indices.begin(), indices.end(), 0);
34 return indices;
35 }
36
37 const double log_start = std::log10(static_cast<double>(data(0)));
38 const double log_end = std::log10(static_cast<double>(data(total_size - 1)));
39 const double log_range = log_end - log_start;
40 const size_t total_points = static_cast<size_t>(std::ceil(log_range * static_cast<double>(num_order))) + 1;
41
42 std::vector<size_t> indices;
43 indices.reserve(total_points);
44
45 // Always include the first point
46 indices.push_back(0);
47
48 // Handle case where total_points <= 1 (avoid division by zero)
49 if (total_points <= 1) {
50 if (total_size > 1) {
51 indices.push_back(total_size - 1);
52 }
53 return indices;
54 }
55
56 const double step = log_range / static_cast<double>(total_points - 1);
57
58 for (size_t i = 1; i < total_points - 1; ++i) {
59 const double log_target = log_start + static_cast<double>(i) * step;
60 const double target_value = std::pow(10.0, log_target);
61
62 size_t best_idx = 1;
63 double min_diff = std::abs(static_cast<double>(data(1)) - target_value);
64
65 for (size_t j = 2; j < total_size - 1; ++j) {
66 const double diff = std::abs(static_cast<double>(data(j)) - target_value);
67 if (diff < min_diff) {
68 min_diff = diff;
69 best_idx = j;
70 }
71 }
72
73 if (std::ranges::find(indices, best_idx) == indices.end()) {
74 indices.push_back(best_idx);
75 }
76 }
77
78 // Always include the last point
79 if (total_size > 1) {
80 indices.push_back(total_size - 1);
81 }
82
83 std::ranges::sort(indices);
84 indices.erase(std::ranges::unique(indices).begin(), indices.end());
85
86 return indices;
87}

90 double chi_square = 0;

117Ejecta MultiBandModel::select_jet(Params const& param) const {
118 const Real eps_iso = param.E_iso * unit::erg / (4 * con::pi);
119 const Real Gamma0 = param.Gamma0;
120 const Real theta_c = param.theta_c;
121 const Real theta_w = param.theta_w;
122 const Real eps_iso_w = param.E_iso_w * unit::erg / (4 * con::pi);
123 const Real Gamma0_w = param.Gamma0_w;
124 Ejecta jet;
125 jet.T0 = param.duration * unit::sec;
126 if (config.jet == "tophat") {
127 jet.eps_k = math::tophat(theta_c, eps_iso);
128 jet.Gamma0 = math::tophat_plus_one(theta_c, Gamma0 - 1);
129 } else if (config.jet == "gaussian") {
130 jet.eps_k = math::gaussian(theta_c, eps_iso);
131 jet.Gamma0 = math::gaussian_plus_one(theta_c, Gamma0 - 1);
132 } else if (config.jet == "powerlaw") {
133 jet.eps_k = math::powerlaw(theta_c, eps_iso, param.k_e);
134 jet.Gamma0 = math::powerlaw_plus_one(theta_c, Gamma0 - 1, param.k_g);
135 } else if (config.jet == "powerlaw_wing") {
136 jet.eps_k = math::powerlaw_wing(theta_c, eps_iso_w, param.k_e);
137 jet.Gamma0 = math::powerlaw_wing_plus_one(theta_c, Gamma0_w - 1, param.k_g);
138 } else if (config.jet == "uniform") {
139 jet.eps_k = math::tophat(con::pi / 2, eps_iso);
140 jet.Gamma0 = math::tophat_plus_one(con::pi / 2, Gamma0 - 1);
141 } else if (config.jet == "two_component") {
142 jet.eps_k = math::two_component(theta_c, theta_w, eps_iso, eps_iso_w);
143 jet.Gamma0 = math::two_component_plus_one(theta_c, theta_w, Gamma0 - 1, Gamma0_w - 1);
144 } else if (config.jet == "step_powerlaw") {
145 jet.eps_k = math::step_powerlaw(theta_c, eps_iso, eps_iso_w, param.k_e);
146 jet.Gamma0 = math::step_powerlaw_plus_one(theta_c, Gamma0 - 1, Gamma0_w - 1, param.k_g);
147 } else {
148 AFTERGLOW_ENSURE(false, "Unknown jet type");
149 }
150
151 if (config.magnetar == true) {
152 jet.deps_dt =
153 math::magnetar_injection(param.t0 * unit::sec, param.q, param.L0 * unit::erg / unit::sec, theta_c);
154 }
155 return jet;
156}

187void MultiBandData::add_flux(double nu_min, double nu_max, size_t num_points, PyArray const& t, PyArray const& Fv_obs,
188 PyArray const& Fv_err, const std::optional<PyArray>& weights) {
189 AFTERGLOW_REQUIRE(t.size() == Fv_obs.size() && t.size() == Fv_err.size(), "light curve array inconsistent length!");
190 AFTERGLOW_REQUIRE(is_ascending(t), "Time array must be in ascending order!");
191 AFTERGLOW_REQUIRE(nu_min < nu_max, "nu_min must be less than nu_max!");
192
193 Array w = xt::ones<Real>({t.size()});
194
195 if (weights) {
196 w = *weights;
197 AFTERGLOW_REQUIRE(t.size() == w.size(), "weights array inconsistent length!");
198
199 const size_t len = w.size();
200 Real weight_sum = 0;
201 for (size_t i = 0; i < len; ++i) {
202 weight_sum += w(i);
203 }
204 if (weight_sum > 0) {
205 w /= (weight_sum / static_cast<double>(len));
206 }
207 }
208
209 const Array nu = xt::logspace(std::log10(nu_min * unit::Hz), std::log10(nu_max * unit::Hz), num_points);
210
211 flux_data.emplace_back(
212 FluxData{t * unit::sec, nu, Fv_obs * unit::flux_cgs, Fv_err * unit::flux_cgs, xt::zeros<Real>({t.size()}), w});
213}

240void MultiBandData::fill_data_arrays() {
241 // Skip if arrays are already filled (e.g., from pickle deserialization)
242 if (times.size() > 0 || (tuple_data.empty() && !flux_data.empty())) {
243 return;
244 }
245
246 const size_t len = tuple_data.size();
247 std::ranges::sort(tuple_data, [](auto const& a, auto const& b) { return std::get<0>(a) < std::get<0>(b); });
248 times = Array::from_shape({len});
249 frequencies = Array::from_shape({len});
250 fluxes = Array::from_shape({len});
251 errors = Array::from_shape({len});
252 model_fluxes = Array::from_shape({len});
253 weights = Array::from_shape({len});
254
255 Real weight_sum = 0;
256 for (size_t i = 0; i < len; ++i) {
257 times(i) = std::get<0>(tuple_data[i]);
258 frequencies(i) = std::get<1>(tuple_data[i]);
259 fluxes(i) = std::get<2>(tuple_data[i]);
260 errors(i) = std::get<3>(tuple_data[i]);
261 weights(i) = std::get<4>(tuple_data[i]);
262 model_fluxes(i) = 0; // Placeholder for model fluxes
263 weight_sum += weights(i);
264 }
265 if (weight_sum > 0) {
266 weights /= (weight_sum / static_cast<double>(len));
267 }
268
269 if (len > 0) {
270 this->t_min = times.front();
271 this->t_max = times.back();
272 }
273
274 for (auto& d : flux_data) {
275 if (d.t.size() == 0)
276 continue;
277 if (d.t.front() < t_min)
278 t_min = d.t.front();
279 if (d.t.back() > t_max)
280 t_max = d.t.back();
281 }
282}

294double MultiBandModel::estimate_chi2(Params const& param) {
295 Observer obs;
296 SynPhotonGrid f_photons;
297 SynPhotonGrid r_photons;
298 ICPhotonGrid<SynElectrons, SynPhotons> f_IC_photons;
299 ICPhotonGrid<SynElectrons, SynPhotons> r_IC_photons;
300
301 generate_photons(param, obs_data.t_min, obs_data.t_max, obs, f_photons, r_photons, f_IC_photons, r_IC_photons);
302
303 obs_data.model_fluxes = obs.specific_flux_series(obs_data.times, obs_data.frequencies, f_photons);
304 for (auto& d : obs_data.flux_data) {
305 d.Fv_model = obs.flux(d.t, d.nu, f_photons);
306 }
307
308 if (r_photons.size() > 0) {
309 obs_data.model_fluxes += obs.specific_flux_series(obs_data.times, obs_data.frequencies, r_photons);
310 for (auto& d : obs_data.flux_data) {
311 d.Fv_model += obs.flux(d.t, d.nu, r_photons);
312 }
313 }
314
315 if (f_IC_photons.size() > 0) {
316 obs_data.model_fluxes += obs.specific_flux_series(obs_data.times, obs_data.frequencies, f_IC_photons);
317 for (auto& d : obs_data.flux_data) {
318 d.Fv_model += obs.flux(d.t, d.nu, f_IC_photons);
319 }
320 }
321
322 if (r_IC_photons.size() > 0) {
323 obs_data.model_fluxes += obs.specific_flux_series(obs_data.times, obs_data.frequencies, r_IC_photons);
324 for (auto& d : obs_data.flux_data) {
325 d.Fv_model += obs.flux(d.t, d.nu, r_IC_photons);
326 }
327 }
328
329 return obs_data.estimate_chi2();
330}

332auto MultiBandModel::flux_density_grid(Params const& param, PyArray const& t, PyArray const& nu) -> PyGrid {
333 Array t_bins = t * unit::sec;
334 Array nu_bins = nu * unit::Hz;
335 MeshGrid F_nu = MeshGrid::from_shape({nu.size(), t.size()});
336
337 Observer obs;
338 SynPhotonGrid f_photons;
339 SynPhotonGrid r_photons;
340 ICPhotonGrid<SynElectrons, SynPhotons> f_IC_photons;
341 ICPhotonGrid<SynElectrons, SynPhotons> r_IC_photons;
342
343 generate_photons(param, t_bins.front(), t_bins.back(), obs, f_photons, r_photons, f_IC_photons, r_IC_photons);
344
345 F_nu = obs.specific_flux(t_bins, nu_bins, f_photons);
346
347 if (r_photons.size() > 0) {
348 F_nu += obs.specific_flux(t_bins, nu_bins, r_photons);
349 }
350
351 if (f_IC_photons.size() > 0) {
352 F_nu += obs.specific_flux(t_bins, nu_bins, f_IC_photons);
353 }
354
355 if (r_IC_photons.size() > 0) {
356 F_nu += obs.specific_flux(t_bins, nu_bins, r_IC_photons);
357 }
358
359 // we bind this function for GIL free. As the return will create a pyobject, we need to get the GIL.
360 pybind11::gil_scoped_acquire acquire;
361 return F_nu / unit::flux_den_cgs;
