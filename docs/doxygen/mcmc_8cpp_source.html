15template <typename Array>
16void sort_synchronized(Array& a, Array& b, Array& c) {
17 std::size_t N = a.size();
18
19 std::vector<std::size_t> idx(N);
20 std::iota(idx.begin(), idx.end(), 0);
21
22 std::sort(idx.begin(), idx.end(), [&](std::size_t i, std::size_t j) { return a(i) < a(j); });
23
24 Array a_sorted = xt::empty<double>({N});
25 Array b_sorted = xt::empty<double>({N});
26 Array c_sorted = xt::empty<double>({N});
27
28 for (std::size_t i = 0; i < N; ++i) {
29 a_sorted(i) = a(idx[i]);
30 b_sorted(i) = b(idx[i]);
31 c_sorted(i) = c(idx[i]);
32 }
33
34 a = std::move(a_sorted);
35 b = std::move(b_sorted);
36 c = std::move(c_sorted);
37}

39double LightCurveData::estimate_chi2() const {
40 double chi_square = 0;
41 for (size_t i = 0; i < t.size(); ++i) {
42 if (Fv_err[i] == 0) continue;
43 double diff = Fv_obs[i] - Fv_model[i];
44 chi_square += (diff * diff) / (Fv_err[i] * Fv_err[i]);
45 }
46 return chi_square;

70void MultiBandData::add_light_curve(double nu, List const& t, List const& Fv_obs, List const& Fv_err) {
71 assert(t.size() == Fv_obs.size() && t.size() == Fv_err.size() && "light curve array inconsistent length!");
72 LightCurveData data;
73
74 data.nu = nu * unit::Hz;
75 data.t = xt::eval(xt::adapt(t));
76 data.Fv_obs = xt::eval(xt::adapt(Fv_obs));
77 data.Fv_err = xt::eval(xt::adapt(Fv_err));
78 data.Fv_model = xt::zeros_like(data.Fv_obs);
79
80 sort_synchronized(data.t, data.Fv_obs, data.Fv_err);

83 t *= unit::sec;
84 }

86 Fv_obs *= unit::flux_den_cgs;

88 for (auto& Fv_err : data.Fv_err) {

90 }

155void MultiBandModel::build_system(Params const& param, Array const& t_eval, Observer& obs, SynElectronGrid& electrons,
156 SynPhotonGrid& photons) {
157 Real E_iso = param.E_iso * unit::erg;
158 Real Gamma0 = param.Gamma0;
159 Real theta_c = param.theta_c;
160 Real theta_v = param.theta_v;
161 Real theta_w = param.theta_w;
162 Real p = param.p;
163 Real eps_e = param.eps_e;
164 Real eps_B = param.eps_B;
165 Real xi = param.xi;
166
167 Real lumi_dist = config.lumi_dist * unit::cm;
168 Real z = config.z;
169
170 // create model
171 Medium medium;
172 if (config.medium == "ism") {
173 Real n_ism = param.n_ism / unit::cm3;
174 std::tie(medium.rho, medium.mass) = evn::ISM(n_ism);
175 } else if (config.medium == "wind") {
176 std::tie(medium.rho, medium.mass) = evn::wind(param.A_star);
177 } else {
178 std::cerr << "Error: Unknown medium type" << std::endl;
179 }
180
181 Ejecta jet;
182 if (config.jet == "tophat") {
183 jet.eps_k = math::tophat(theta_c, E_iso / (4 * con::pi));
184 jet.Gamma0 = math::tophat(theta_c, Gamma0);
185 } else if (config.jet == "gaussian") {
186 jet.eps_k = math::gaussian(theta_c, E_iso / (4 * con::pi));
187 jet.Gamma0 = math::gaussian(theta_c, Gamma0);
188 } else if (config.jet == "powerlaw") {
189 jet.eps_k = math::powerlaw(theta_c, E_iso / (4 * con::pi), param.k_jet);
190 jet.Gamma0 = math::powerlaw(theta_c, Gamma0, param.k_jet);
191 } else {
192 std::cerr << "Error: Unknown jet type" << std::endl;
193 }
194
195 size_t t_num = config.t_grid;
196 size_t theta_num = config.theta_grid;
197 size_t phi_num = config.phi_grid;
198
199 auto coord = auto_grid(jet, t_eval, theta_w, theta_v, z, phi_num, theta_num, t_num);
200
201 auto shock = generate_fwd_shock(coord, medium, jet, eps_e, eps_B, config.rtol);
202
203 // obs.observe_at(t_eval, coord, shock, lumi_dist, z);
204 obs.observe(coord, shock, lumi_dist, z);
205
206 electrons = generate_syn_electrons(shock, p, xi);
207
208 photons = generate_syn_photons(shock, electrons);
209}

229std::vector<std::vector<double>> convert_to_std_vector(MeshGrid const& grid) {
230 std::vector<std::vector<double>> out;
231 out.reserve(grid.shape()[0]);
232
233 for (size_t i = 0; i < grid.shape()[0]; ++i) {
234 std::vector<double> row;
235 row.reserve(grid.shape()[1]);
236 for (size_t j = 0; j < grid.shape()[1]; ++j) {
237 row.push_back(grid(i, j) / unit::flux_den_cgs);
238 }
239 out.push_back(std::move(row));
240 }
241 return out;
242}
