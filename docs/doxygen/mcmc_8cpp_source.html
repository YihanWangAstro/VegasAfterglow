20std::vector<size_t> MultiBandData::logscale_screen(PyArray const& data, size_t num_order) {
21 const size_t total_size = data.size();
22
23 if (num_order == 0) {
24 std::vector<size_t> indices(total_size);
25 std::iota(indices.begin(), indices.end(), 0);
26 return indices;
27 }
28
29 const double log_start = std::log10(static_cast<double>(data(0)));
30 const double log_end = std::log10(static_cast<double>(data(total_size - 1)));
31 const double log_range = log_end - log_start;
32 const size_t total_points = static_cast<size_t>(std::ceil(log_range * static_cast<double>(num_order))) + 1;
33
34 std::vector<size_t> indices;
35 indices.reserve(total_points);
36
37 // Always include first point
38 indices.push_back(0);
39
40 const double step = log_range / static_cast<double>(total_points - 1);
41
42 for (size_t i = 1; i < total_points - 1; ++i) {
43 const double log_target = log_start + i * step;
44 const double target_value = std::pow(10.0, log_target);
45
46 size_t best_idx = 1;
47 double min_diff = std::abs(static_cast<double>(data(1)) - target_value);
48
49 for (size_t j = 2; j < total_size - 1; ++j) {
50 const double diff = std::abs(static_cast<double>(data(j)) - target_value);
51 if (diff < min_diff) {
52 min_diff = diff;
53 best_idx = j;
54 }
55 }
56
57 if (std::ranges::find(indices, best_idx) == indices.end()) {
58 indices.push_back(best_idx);
59 }
60 }
61
62 // Always include the last point
63 if (total_size > 1) {
64 indices.push_back(total_size - 1);
65 }
66
67 std::ranges::sort(indices);
68 indices.erase(std::ranges::unique(indices).begin(), indices.end());
69
70 return indices;
71}

74 double chi_square = 0;
75 for (size_t i = 0; i < t.size(); ++i) {
76 const double error = Fv_err(i);

101Ejecta MultiBandModel::select_jet(Params const& param) const {
102 const Real eps_iso = param.E_iso * unit::erg / (4 * con::pi);
103 const Real Gamma0 = param.Gamma0;
104 const Real theta_c = param.theta_c;
105 const Real theta_w = param.theta_w;
106 const Real eps_iso_w = param.E_iso_w * unit::erg / (4 * con::pi);
107 const Real Gamma0_w = param.Gamma0_w;
108 Ejecta jet;
109 jet.T0 = param.duration * unit::sec;
110 if (config.jet == "tophat") {
111 jet.eps_k = math::tophat(theta_c, eps_iso);
112 jet.Gamma0 = math::tophat_plus_one(theta_c, Gamma0 - 1);
113 } else if (config.jet == "gaussian") {
114 jet.eps_k = math::gaussian(theta_c, eps_iso);
115 jet.Gamma0 = math::gaussian_plus_one(theta_c, Gamma0 - 1);
116 } else if (config.jet == "powerlaw") {
117 jet.eps_k = math::powerlaw(theta_c, eps_iso, param.k_e);
118 jet.Gamma0 = math::powerlaw_plus_one(theta_c, Gamma0 - 1, param.k_g);
119 } else if (config.jet == "powerlaw_wing") {
120 jet.eps_k = math::powerlaw_wing(theta_c, eps_iso_w, param.k_e);
121 jet.Gamma0 = math::powerlaw_wing_plus_one(theta_c, Gamma0_w - 1, param.k_g);
122 } else if (config.jet == "uniform") {
123 jet.eps_k = math::tophat(con::pi / 2, eps_iso);
124 jet.Gamma0 = math::tophat_plus_one(con::pi / 2, Gamma0 - 1);
125 } else if (config.jet == "two_component") {
126 jet.eps_k = math::two_component(theta_c, theta_w, eps_iso, eps_iso_w);
127 jet.Gamma0 = math::two_component_plus_one(theta_c, theta_w, Gamma0 - 1, Gamma0_w - 1);
128 } else if (config.jet == "step_powerlaw") {
129 jet.eps_k = math::step_powerlaw(theta_c, eps_iso, eps_iso_w, param.k_e);
130 jet.Gamma0 = math::step_powerlaw_plus_one(theta_c, Gamma0 - 1, Gamma0_w - 1, param.k_g);
131 } else {
132 AFTERGLOW_ENSURE(false, "Unknown jet type");
133 }
134
135 if (config.magnetar == true) {
136 jet.deps_dt =
137 math::magnetar_injection(param.t0 * unit::sec, param.q, param.L0 * unit::erg / unit::sec, theta_c);
138 }
139 return jet;
140}

171void MultiBandData::add_flux(double nu_min, double nu_max, size_t num_points, PyArray const& t, PyArray const& Fv_obs,
172 PyArray const& Fv_err, const std::optional<PyArray>& weights) {
173 AFTERGLOW_REQUIRE(t.size() == Fv_obs.size() && t.size() == Fv_err.size(), "light curve array inconsistent length!");
174 AFTERGLOW_REQUIRE(is_ascending(t), "Time array must be in ascending order!");
175 AFTERGLOW_REQUIRE(nu_min < nu_max, "nu_min must be less than nu_max!");
176
177 Array w = xt::ones<Real>({t.size()});
178
179 if (weights) {
180 w = *weights;
181 AFTERGLOW_REQUIRE(t.size() == w.size(), "weights array inconsistent length!");
182
183 const size_t len = w.size();
184 Real weight_sum = 0;
185 for (size_t i = 0; i < len; ++i) {
186 weight_sum += w(i);
187 }
188 w /= (weight_sum / len);
189 }
190
191 const Array nu = xt::logspace(std::log10(nu_min * unit::Hz), std::log10(nu_max * unit::Hz), num_points);
192
193 flux_data.emplace_back(
194 FluxData{t * unit::sec, nu, Fv_obs * unit::flux_cgs, Fv_err * unit::flux_cgs, xt::zeros<Real>({t.size()}), w});
195}

222void MultiBandData::fill_data_arrays() {
223 const size_t len = tuple_data.size();
224 std::ranges::sort(tuple_data, [](auto const& a, auto const& b) { return std::get<0>(a) < std::get<0>(b); });
225 times = Array::from_shape({len});
226 frequencies = Array::from_shape({len});
227 fluxes = Array::from_shape({len});
228 errors = Array::from_shape({len});
229 model_fluxes = Array::from_shape({len});
230 weights = Array::from_shape({len});
231
232 Real weight_sum = 0;
233 for (size_t i = 0; i < len; ++i) {
234 times(i) = std::get<0>(tuple_data[i]);
235 frequencies(i) = std::get<1>(tuple_data[i]);
236 fluxes(i) = std::get<2>(tuple_data[i]);
237 errors(i) = std::get<3>(tuple_data[i]);
238 weights(i) = std::get<4>(tuple_data[i]);
239 model_fluxes(i) = 0; // Placeholder for model fluxes
240 weight_sum += weights(i);
241 }
242 weights /= (weight_sum / len);
243
244 if (len > 0) {
245 this->t_min = times.front();
246 this->t_max = times.back();
247 }
248
249 for (auto& d : flux_data) {
250 if (d.t.front() < t_min)
251 t_min = d.t.front();
252 if (d.t.back() > t_max)
253 t_max = d.t.back();
254 }
255}

267double MultiBandModel::estimate_chi2(Params const& param) {
268 Observer obs;
269 SynPhotonGrid f_photons;
270 SynPhotonGrid r_photons;
271 ICPhotonGrid<SynElectrons, SynPhotons> f_IC_photons;
272 ICPhotonGrid<SynElectrons, SynPhotons> r_IC_photons;
273
274 generate_photons(param, obs_data.t_min, obs_data.t_max, obs, f_photons, r_photons, f_IC_photons, r_IC_photons);
275
276 obs_data.model_fluxes = obs.specific_flux_series(obs_data.times, obs_data.frequencies, f_photons);
277 for (auto& d : obs_data.flux_data) {
278 d.Fv_model = obs.flux(d.t, d.nu, f_photons);
279 }
280
281 if (r_photons.size() > 0) {
282 obs_data.model_fluxes += obs.specific_flux_series(obs_data.times, obs_data.frequencies, r_photons);
283 for (auto& d : obs_data.flux_data) {
284 d.Fv_model += obs.flux(d.t, d.nu, r_photons);
285 }
286 }
287
288 if (f_IC_photons.size() > 0) {
289 obs_data.model_fluxes += obs.specific_flux_series(obs_data.times, obs_data.frequencies, f_IC_photons);
290 for (auto& d : obs_data.flux_data) {
291 d.Fv_model += obs.flux(d.t, d.nu, f_IC_photons);
292 }
293 }
294
295 if (r_IC_photons.size() > 0) {
296 obs_data.model_fluxes += obs.specific_flux_series(obs_data.times, obs_data.frequencies, r_IC_photons);
297 for (auto& d : obs_data.flux_data) {
298 d.Fv_model += obs.flux(d.t, d.nu, r_IC_photons);
299 }
300 }
301
302 return obs_data.estimate_chi2();

305auto MultiBandModel::flux_density_grid(Params const& param, PyArray const& t, PyArray const& nu) -> PyGrid {
306 Array t_bins = t * unit::sec;
307 Array nu_bins = nu * unit::Hz;
308 MeshGrid F_nu = MeshGrid::from_shape({nu.size(), t.size()});
309
310 Observer obs;
311 SynPhotonGrid f_photons;
312 SynPhotonGrid r_photons;
313 ICPhotonGrid<SynElectrons, SynPhotons> f_IC_photons;
314 ICPhotonGrid<SynElectrons, SynPhotons> r_IC_photons;
315
316 generate_photons(param, t_bins.front(), t_bins.back(), obs, f_photons, r_photons, f_IC_photons, r_IC_photons);
317
318 F_nu = obs.specific_flux(t_bins, nu_bins, f_photons);
319
320 if (r_photons.size() > 0) {
321 F_nu += obs.specific_flux(t_bins, nu_bins, r_photons);
322 }
323
324 if (f_IC_photons.size() > 0) {
325 F_nu += obs.specific_flux(t_bins, nu_bins, f_IC_photons);
326 }
327
328 if (r_IC_photons.size() > 0) {
329 F_nu += obs.specific_flux(t_bins, nu_bins, r_IC_photons);
330 }
331
332 // we bind this function for GIL free. As the return will create a pyobject, we need to get the GIL.
333 pybind11::gil_scoped_acquire acquire;
334 return F_nu / unit::flux_den_cgs;
