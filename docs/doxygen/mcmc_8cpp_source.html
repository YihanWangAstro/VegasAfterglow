17double MultiBandData::estimate_chi2() const {
18 double chi_square = 0;
19 for (size_t i = 0; i < times.size(); ++i) {
20 double error = errors(i);
21 if (error == 0) continue;
22 double diff = fluxes(i) - model_fluxes(i);
23 chi_square += weights(i) * (diff * diff) / (error * error);
24 }
25 return chi_square;
26}

62void MultiBandData::fill_data_arrays() {
63 const size_t len = tuple_data.size();
64 std::sort(tuple_data.begin(), tuple_data.end(),
65 [](auto const& a, auto const& b) { return std::get<0>(a) < std::get<0>(b); });
66 times = Array::from_shape({len});
67 frequencies = Array::from_shape({len});
68 fluxes = Array::from_shape({len});
69 errors = Array::from_shape({len});
70 model_fluxes = Array::from_shape({len});
71 weights = Array::from_shape({len});
72
73 Real weight_sum = 0;
74 for (size_t i = 0; i < len; ++i) {
75 times(i) = std::get<0>(tuple_data[i]);
76 frequencies(i) = std::get<1>(tuple_data[i]);
77 fluxes(i) = std::get<2>(tuple_data[i]);
78 errors(i) = std::get<3>(tuple_data[i]);
79 weights(i) = std::get<4>(tuple_data[i]);
80 model_fluxes(i) = 0; // Placeholder for model fluxes
81 weight_sum += weights(i);
82 }
83 weights /= (weight_sum / len);
84}

102auto MultiBandModel::specific_flux(Params const& param, PyArray const& t, PyArray const& nu) -> PyGrid {
103 Array t_bins = t * unit::sec;
104
105 MeshGrid F_nu = MeshGrid::from_shape({nu.size(), t.size()});
106
107 for (size_t i = 0; i < nu.size(); ++i) {
108 Array nus = Array({t.size()}, nu(i) * unit::Hz);
109 auto view = xt::view(F_nu, i, xt::all());
110 build_system(param, t_bins, nus, view);
111 }
112
113 // we bind this function for GIL free. As the return will create a pyobject, we need to get the GIL.
114 pybind11::gil_scoped_acquire acquire;
115 return F_nu / unit::flux_den_cgs;
