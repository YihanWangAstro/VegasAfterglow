18double FluxData::estimate_chi2() const {
19 double chi_square = 0;
20 for (size_t i = 0; i < t.size(); ++i) {
21 double error = Fv_err(i);
22 if (error == 0)
23 continue;
24 double diff = Fv_obs(i) - Fv_model(i);
25 chi_square += weights(i) * (diff * diff) / (error * error);
26 }
27 return chi_square;
28}

30double MultiBandData::estimate_chi2() const {
31 double chi_square = 0;
32 for (size_t i = 0; i < times.size(); ++i) {
33 double error = errors(i);
34 if (error == 0)
35 continue;
36 double diff = fluxes(i) - model_fluxes(i);
37 chi_square += weights(i) * (diff * diff) / (error * error);
38 }
39 for (auto& d : flux_data) {
40 chi_square += d.estimate_chi2();
41 }
42 return chi_square;
43}

45Ejecta MultiBandModel::select_jet(Params const& param) {
46 Real eps_iso = param.E_iso * unit::erg / (4 * con::pi);
47 Real Gamma0 = param.Gamma0;
48 Real theta_c = param.theta_c;
49 Real theta_w = param.theta_w;
50 Real eps_iso_w = param.E_iso_w * unit::erg / (4 * con::pi);
51 Real Gamma0_w = param.Gamma0_w;
52 Ejecta jet;
53 jet.T0 = param.duration * unit::sec;
54 if (config.jet == "tophat") {
55 jet.eps_k = math::tophat(theta_c, eps_iso);
56 jet.Gamma0 = math::tophat_plus_one(theta_c, Gamma0 - 1);
57 } else if (config.jet == "gaussian") {
58 jet.eps_k = math::gaussian(theta_c, eps_iso);
59 jet.Gamma0 = math::gaussian_plus_one(theta_c, Gamma0 - 1);
60 } else if (config.jet == "powerlaw") {
61 jet.eps_k = math::powerlaw(theta_c, eps_iso, param.k_e);
62 jet.Gamma0 = math::powerlaw_plus_one(theta_c, Gamma0 - 1, param.k_g);
63 } else if (config.jet == "twocomponent") {
64 jet.eps_k = math::two_component(theta_c, theta_w, eps_iso, eps_iso_w);
65 jet.Gamma0 = math::two_component_plus_one(theta_c, theta_w, Gamma0 - 1, Gamma0_w - 1);
66 } else if (config.jet == "steppowerlaw") {
67 jet.eps_k = math::step_powerlaw(theta_c, eps_iso, eps_iso_w, param.k_e);
68 jet.Gamma0 = math::step_powerlaw_plus_one(theta_c, Gamma0 - 1, Gamma0_w - 1, param.k_g);
69 } else {
70 assert(false && "Unknown jet type");
71 }
72
73 if (config.magnetar == true) {
74 jet.deps_dt =
75 math::magnetar_injection(param.t0 * unit::sec, param.q, param.L0 * unit::erg / unit::sec, theta_c);
76 }
77 return jet;
78}

109void MultiBandData::add_light_curve(double nu_min, double nu_max, size_t num_points, PyArray const& t,
110 PyArray const& Fv_obs, PyArray const& Fv_err, std::optional<PyArray> weights) {
111 assert(t.size() == Fv_obs.size() && t.size() == Fv_err.size() && "light curve array inconsistent length!");
112 assert(is_ascending(t) && "Time array must be in ascending order!");
113 assert(nu_min < nu_max && "nu_min must be less than nu_max!");
114
115 Array w = xt::ones<Real>({t.size()});
116
117 if (weights) {
118 w = *weights;
119 assert(t.size() == w.size() && "weights array inconsistent length!");
120
121 size_t len = w.size();
122 Real weight_sum = 0;
123 for (size_t i = 0; i < len; ++i) {
124 weight_sum += w(i);
125 }
126 w /= (weight_sum / len);
127 }
128
129 Array nu = xt::logspace(std::log10(nu_min * unit::Hz), std::log10(nu_max * unit::Hz), num_points);
130
131 flux_data.emplace_back(
132 FluxData{t * unit::sec, nu, Fv_obs * unit::flux_cgs, Fv_err * unit::flux_cgs, xt::zeros<Real>({t.size()}), w});
133}

152void MultiBandData::fill_data_arrays() {
153 const size_t len = tuple_data.size();
154 std::sort(tuple_data.begin(), tuple_data.end(),
155 [](auto const& a, auto const& b) { return std::get<0>(a) < std::get<0>(b); });
156 times = Array::from_shape({len});
157 frequencies = Array::from_shape({len});
158 fluxes = Array::from_shape({len});
159 errors = Array::from_shape({len});
160 model_fluxes = Array::from_shape({len});
161 weights = Array::from_shape({len});
162
163 Real weight_sum = 0;
164 for (size_t i = 0; i < len; ++i) {
165 times(i) = std::get<0>(tuple_data[i]);
166 frequencies(i) = std::get<1>(tuple_data[i]);
167 fluxes(i) = std::get<2>(tuple_data[i]);
168 errors(i) = std::get<3>(tuple_data[i]);
169 weights(i) = std::get<4>(tuple_data[i]);
170 model_fluxes(i) = 0; // Placeholder for model fluxes
171 weight_sum += weights(i);
172 }
173 weights /= (weight_sum / len);
174
175 if (len > 0) {
176 this->t_min = times.front();
177 this->t_max = times.back();
178 }
179
180 for (auto& d : flux_data) {
181 if (d.t.front() < t_min)
182 t_min = d.t.front();
183 if (d.t.back() > t_max)
184 t_max = d.t.back();
185 }
186}

198double MultiBandModel::estimate_chi2(Params const& param) {
199 Observer obs;
200 SynPhotonGrid f_photons;
201 SynPhotonGrid r_photons;
202 ICPhotonGrid<SynElectrons, SynPhotons> f_IC_photons;
203 ICPhotonGrid<SynElectrons, SynPhotons> r_IC_photons;
204
205 generate_photons(param, obs_data.t_min, obs_data.t_max, obs, f_photons, r_photons, f_IC_photons, r_IC_photons);
206
207 obs_data.model_fluxes = obs.specific_flux_series(obs_data.times, obs_data.frequencies, f_photons);
208 for (auto& d : obs_data.flux_data) {
209 d.Fv_model = obs.flux(d.t, d.nu, f_photons);
210 }
211
212 if (r_photons.size() > 0) {
213 obs_data.model_fluxes += obs.specific_flux_series(obs_data.times, obs_data.frequencies, r_photons);
214 for (auto& d : obs_data.flux_data) {
215 d.Fv_model += obs.flux(d.t, d.nu, r_photons);
216 }
217 }
218
219 if (f_IC_photons.size() > 0) {
220 obs_data.model_fluxes += obs.specific_flux_series(obs_data.times, obs_data.frequencies, f_IC_photons);
221 for (auto& d : obs_data.flux_data) {
222 d.Fv_model += obs.flux(d.t, d.nu, f_IC_photons);
223 }
224 }
225
226 if (r_IC_photons.size() > 0) {
227 obs_data.model_fluxes += obs.specific_flux_series(obs_data.times, obs_data.frequencies, r_IC_photons);
228 for (auto& d : obs_data.flux_data) {
229 d.Fv_model += obs.flux(d.t, d.nu, r_IC_photons);
230 }
231 }
232
233 return obs_data.estimate_chi2();
234}

237 Array t_bins = t * unit::sec;

239 MeshGrid F_nu = MeshGrid::from_shape({nu.size(), t.size()});

246

248
249 F_nu = obs.specific_flux(t_bins, nu_bins, f_photons);
250
251 if (r_photons.size() > 0) {
252 F_nu += obs.specific_flux(t_bins, nu_bins, r_photons);
253 }
254
255 if (f_IC_photons.size() > 0) {
256 F_nu += obs.specific_flux(t_bins, nu_bins, f_IC_photons);

258

260 F_nu += obs.specific_flux(t_bins, nu_bins, r_IC_photons);
261 }
262
263 // we bind this function for GIL free. As the return will create a pyobject, we need to get the GIL.
264 pybind11::gil_scoped_acquire acquire;
