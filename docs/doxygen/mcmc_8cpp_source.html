27// Parameter name to index mapping (computed once, used for Python interface setup)
28int param_name_to_index(const std::string& name) {
29 static const std::unordered_map<std::string, int> name_map = {
30 {"theta_v", 0}, {"n_ism", 1}, {"n0", 2}, {"A_star", 3}, {"k_m", 4}, {"E_iso", 5},
31 {"Gamma0", 6}, {"theta_c", 7}, {"k_e", 8}, {"k_g", 9}, {"duration", 10}, {"tau", 10},
32 {"E_iso_w", 11}, {"Gamma0_w", 12}, {"theta_w", 13}, {"L0", 14}, {"t0", 15}, {"q", 16},
33 {"p", 17}, {"eps_e", 18}, {"eps_B", 19}, {"xi_e", 20}, {"p_r", 21}, {"eps_e_r", 22},
34 {"eps_B_r", 23}, {"xi_e_r", 24},
35 };
36 auto it = name_map.find(name);
37 if (it == name_map.end()) {
38 return -1; // Invalid parameter name
39 }
40 return it->second;
41}

43ParamTransformer::ParamTransformer(std::vector<ParamDef> const& param_defs) {
44 for (auto const& pd : param_defs) {
45 int idx = param_name_to_index(pd.name);
46 AFTERGLOW_REQUIRE(idx >= 0, "Invalid parameter name: " + pd.name);
47
48 if (pd.scale == 2) { // Fixed parameter
49 fixed_indices_.push_back(idx);
50 // For fixed params, use initial value directly (set by user or helper function)
51 fixed_values_.push_back(pd.initial);
52 } else { // Sampled parameter
53 sampled_indices_.push_back(idx);
54 log_mask_.push_back(pd.scale == 1); // 1 = log scale
55 }
56 }
57}

108std::vector<size_t> MultiBandData::logscale_screen(PyArray const& data, size_t num_order) {
109 const size_t total_size = data.size();
110
111 // Handle edge cases: empty or single-element arrays
112 if (total_size == 0) {
113 return {};
114 }
115 if (total_size == 1) {
116 return {0};
117 }
118
119 if (num_order == 0) {
120 std::vector<size_t> indices(total_size);
121 std::iota(indices.begin(), indices.end(), 0);
122 return indices;
123 }
124
125 const double log_start = std::log10(static_cast<double>(data(0)));
126 const double log_end = std::log10(static_cast<double>(data(total_size - 1)));
127 const double log_range = log_end - log_start;
128 const size_t total_points = static_cast<size_t>(std::ceil(log_range * static_cast<double>(num_order))) + 1;
129
130 std::vector<size_t> indices;
131 indices.reserve(total_points);
132
133 // Always include the first point
134 indices.push_back(0);
135
136 // Handle case where total_points <= 1 (avoid division by zero)
137 if (total_points <= 1) {
138 if (total_size > 1) {
139 indices.push_back(total_size - 1);
140 }
141 return indices;
142 }
143
144 const double step = log_range / static_cast<double>(total_points - 1);
145
146 for (size_t i = 1; i < total_points - 1; ++i) {
147 const double log_target = log_start + static_cast<double>(i) * step;
148 const double target_value = std::pow(10.0, log_target);
149
150 size_t best_idx = 1;
151 double min_diff = std::abs(static_cast<double>(data(1)) - target_value);
152
153 for (size_t j = 2; j < total_size - 1; ++j) {
154 const double diff = std::abs(static_cast<double>(data(j)) - target_value);
155 if (diff < min_diff) {
156 min_diff = diff;
157 best_idx = j;
158 }
159 }
160
161 if (std::ranges::find(indices, best_idx) == indices.end()) {
162 indices.push_back(best_idx);
163 }
164 }
165
166 // Always include the last point
167 if (total_size > 1) {
168 indices.push_back(total_size - 1);
169 }
170
171 std::ranges::sort(indices);
172 indices.erase(std::ranges::unique(indices).begin(), indices.end());
173
174 return indices;
175}

205JetVariant MultiBandModel::select_jet(Params const& param) const {
206 const Real Gamma0 = param.Gamma0;
207 const Real theta_c = param.theta_c;
208 const Real T0 = param.duration * unit::sec;
209
210 // Common jet types without magnetar: return concrete types for optimized ODE dispatch
211 if (!config.magnetar) {
212 if (config.jet == "tophat") {
213 return TophatJet(theta_c, param.E_iso * unit::erg, Gamma0, false, T0);
214 } else if (config.jet == "gaussian") {
215 return GaussianJet(theta_c, param.E_iso * unit::erg, Gamma0, false, T0);
216 } else if (config.jet == "powerlaw") {
217 return PowerLawJet(theta_c, param.E_iso * unit::erg, Gamma0, param.k_e, param.k_g, false, T0);
218 } else if (config.jet == "uniform") {
219 return TophatJet(con::pi / 2, param.E_iso * unit::erg, Gamma0, false, T0);
220 }
221 }
222
223 // Magnetar jets, powerlaw_wing, two_component, step_powerlaw: fall back to Ejecta
224 const Real eps_iso = param.E_iso * unit::erg / (4 * con::pi);
225 const Real theta_w = param.theta_w;
226 const Real eps_iso_w = param.E_iso_w * unit::erg / (4 * con::pi);
227 const Real Gamma0_w = param.Gamma0_w;
228 Ejecta jet;
229 jet.T0 = T0;
230 if (config.jet == "tophat") {
231 jet.eps_k = math::tophat(theta_c, eps_iso);
232 jet.Gamma0 = math::tophat_plus_one(theta_c, Gamma0 - 1);
233 } else if (config.jet == "gaussian") {
234 jet.eps_k = math::gaussian(theta_c, eps_iso);
235 jet.Gamma0 = math::gaussian_plus_one(theta_c, Gamma0 - 1);
236 } else if (config.jet == "powerlaw") {
237 jet.eps_k = math::powerlaw(theta_c, eps_iso, param.k_e);
238 jet.Gamma0 = math::powerlaw_plus_one(theta_c, Gamma0 - 1, param.k_g);
239 } else if (config.jet == "powerlaw_wing") {
240 jet.eps_k = math::powerlaw_wing(theta_c, eps_iso_w, param.k_e);
241 jet.Gamma0 = math::powerlaw_wing_plus_one(theta_c, Gamma0_w - 1, param.k_g);
242 } else if (config.jet == "uniform") {
243 jet.eps_k = math::tophat(con::pi / 2, eps_iso);
244 jet.Gamma0 = math::tophat_plus_one(con::pi / 2, Gamma0 - 1);
245 } else if (config.jet == "two_component") {
246 jet.eps_k = math::two_component(theta_c, theta_w, eps_iso, eps_iso_w);
247 jet.Gamma0 = math::two_component_plus_one(theta_c, theta_w, Gamma0 - 1, Gamma0_w - 1);
248 } else if (config.jet == "step_powerlaw") {
249 jet.eps_k = math::step_powerlaw(theta_c, eps_iso, eps_iso_w, param.k_e);
250 jet.Gamma0 = math::step_powerlaw_plus_one(theta_c, Gamma0 - 1, Gamma0_w - 1, param.k_g);
251 } else {
252 AFTERGLOW_ENSURE(false, "Unknown jet type");
253 }
254
255 if (config.magnetar) {
256 jet.deps_dt =
257 math::magnetar_injection(param.t0 * unit::sec, param.q, param.L0 * unit::erg / unit::sec, theta_c);
258 }
259 return jet;
260}

296void MultiBandData::add_flux(double nu_min, double nu_max, size_t num_points, PyArray const& t, PyArray const& Fv_obs,
297 PyArray const& Fv_err, const std::optional<PyArray>& weights) {
298 AFTERGLOW_REQUIRE(t.size() == Fv_obs.size() && t.size() == Fv_err.size(), "light curve array inconsistent length!");
299 AFTERGLOW_REQUIRE(is_ascending(t), "Time array must be in ascending order!");
300 AFTERGLOW_REQUIRE(nu_min < nu_max, "nu_min must be less than nu_max!");
301
302 Array w = xt::ones<Real>({t.size()});
303
304 if (weights) {
305 w = *weights;
306 AFTERGLOW_REQUIRE(t.size() == w.size(), "weights array inconsistent length!");
307
308 const size_t len = w.size();
309 Real weight_sum = 0;
310 for (size_t i = 0; i < len; ++i) {
311 weight_sum += w(i);
312 }
313 if (weight_sum > 0) {
314 w /= (weight_sum / static_cast<double>(len));
315 }
316 }
317
318 const Array nu = xt::logspace(std::log10(nu_min * unit::Hz), std::log10(nu_max * unit::Hz), num_points);
319
320 flux_data.emplace_back(
321 FluxData{t * unit::sec, nu, Fv_obs * unit::flux_cgs, Fv_err * unit::flux_cgs, xt::zeros<Real>({t.size()}), w});
322}

349void MultiBandData::fill_data_arrays() {
350 // Skip if arrays are already filled (e.g., from pickle deserialization)
351 if (times.size() > 0 || (tuple_data.empty() && !flux_data.empty())) {
352 return;
353 }
354
355 const size_t len = tuple_data.size();
356 std::ranges::sort(tuple_data, [](auto const& a, auto const& b) { return std::get<0>(a) < std::get<0>(b); });
357 times = Array::from_shape({len});
358 frequencies = Array::from_shape({len});
359 fluxes = Array::from_shape({len});
360 errors = Array::from_shape({len});
361 model_fluxes = Array::from_shape({len});
362 weights = Array::from_shape({len});
363
364 Real weight_sum = 0;
365 for (size_t i = 0; i < len; ++i) {
366 times(i) = std::get<0>(tuple_data[i]);
367 frequencies(i) = std::get<1>(tuple_data[i]);
368 fluxes(i) = std::get<2>(tuple_data[i]);
369 errors(i) = std::get<3>(tuple_data[i]);
370 weights(i) = std::get<4>(tuple_data[i]);
371 model_fluxes(i) = 0; // Placeholder for model fluxes
372 weight_sum += weights(i);
373 }
374 if (weight_sum > 0) {
375 weights /= (weight_sum / static_cast<double>(len));
376 }
377
378 if (len > 0) {
379 this->t_min = times.front();
380 this->t_max = times.back();
381 }
382
383 for (auto& d : flux_data) {
384 if (d.t.size() == 0)
385 continue;
386 if (d.t.front() < t_min)
387 t_min = d.t.front();
388 if (d.t.back() > t_max)
389 t_max = d.t.back();
390 }
391}

408double MultiBandModel::estimate_chi2_with_workspace(Params const& param, MultiBandData const& obs_data_ref,
409 BatchWorkspace& workspace) {
410
411 Observer obs;
412 SynPhotonGrid f_photons;
413 SynPhotonGrid r_photons;
414 ICPhotonGrid<SynElectrons, SynPhotons> f_IC_photons;
415 ICPhotonGrid<SynElectrons, SynPhotons> r_IC_photons;
416
417 generate_photons(param, obs_data_ref.t_min, obs_data_ref.t_max, obs, f_photons, r_photons, f_IC_photons,
418 r_IC_photons);
419
420 // Compute model fluxes into workspace arrays (not obs_data)
421 if (obs_data_ref.times.size() > 0) {
422 workspace.model_fluxes = obs.specific_flux_series(obs_data_ref.times, obs_data_ref.frequencies, f_photons);
423 }
424 for (size_t i = 0; i < obs_data_ref.flux_data.size(); ++i) {
425 workspace.flux_data_models[i] = obs.flux(obs_data_ref.flux_data[i].t, obs_data_ref.flux_data[i].nu, f_photons);
426 }
427
428 if (r_photons.size() > 0) {
429 if (obs_data_ref.times.size() > 0) {
430 workspace.model_fluxes += obs.specific_flux_series(obs_data_ref.times, obs_data_ref.frequencies, r_photons);
431 }
432 for (size_t i = 0; i < obs_data_ref.flux_data.size(); ++i) {
433 workspace.flux_data_models[i] +=
434 obs.flux(obs_data_ref.flux_data[i].t, obs_data_ref.flux_data[i].nu, r_photons);
435 }
436 }
437
438 if (f_IC_photons.size() > 0) {
439 if (obs_data_ref.times.size() > 0) {
440 workspace.model_fluxes +=
441 obs.specific_flux_series(obs_data_ref.times, obs_data_ref.frequencies, f_IC_photons);
442 }
443 for (size_t i = 0; i < obs_data_ref.flux_data.size(); ++i) {
444 workspace.flux_data_models[i] +=
445 obs.flux(obs_data_ref.flux_data[i].t, obs_data_ref.flux_data[i].nu, f_IC_photons);
446 }
447 }
448
449 if (r_IC_photons.size() > 0) {
450 if (obs_data_ref.times.size() > 0) {
451 workspace.model_fluxes +=
452 obs.specific_flux_series(obs_data_ref.times, obs_data_ref.frequencies, r_IC_photons);
453 }
454 for (size_t i = 0; i < obs_data_ref.flux_data.size(); ++i) {
455 workspace.flux_data_models[i] +=
456 obs.flux(obs_data_ref.flux_data[i].t, obs_data_ref.flux_data[i].nu, r_IC_photons);
457 }
458 }
459
460 // Compute chi-squared using workspace arrays and const observation data
461 double chi_square = 0;
462
463 // Chi2 from tuple data (times/frequencies/fluxes arrays)
464 for (size_t i = 0; i < obs_data_ref.times.size(); ++i) {
465 const double error = obs_data_ref.errors(i);
466 const double diff = obs_data_ref.fluxes(i) - workspace.model_fluxes(i);
467 chi_square += obs_data_ref.weights(i) * (diff * diff) / (error * error);
468 }
469
470 // Chi2 from flux_data entries
471 for (size_t d = 0; d < obs_data_ref.flux_data.size(); ++d) {
472 const auto& fd = obs_data_ref.flux_data[d];
473 const auto& model = workspace.flux_data_models[d];
474 for (size_t i = 0; i < fd.t.size(); ++i) {
475 const double error = fd.Fv_err(i);
476 const double diff = fd.Fv_obs(i) - model(i);
477 chi_square += fd.weights(i) * (diff * diff) / (error * error);
478 }
479 }
480
481 return chi_square;
482}

549PyArray MultiBandModel::batch_estimate_chi2(PyGrid const& samples, ParamTransformer const& transformer) {
550 const size_t nwalkers = samples.shape(0);
551 const size_t ndim = samples.shape(1);
552
553 AFTERGLOW_REQUIRE(ndim == transformer.ndim(), "Sample dimension must match transformer.ndim()");
554
555 Array chi2_results = Array::from_shape({nwalkers});
556
557 const MultiBandData& data_ref = obs_data;
558 const ConfigParams& config_ref = config;
559
560 // Pre-compute workspace dimensions (same for all threads)
561 const size_t tuple_size = data_ref.times.size();
562 std::vector<size_t> flux_data_sizes;
563 flux_data_sizes.reserve(data_ref.flux_data.size());
564 for (const auto& fd : data_ref.flux_data) {
565 flux_data_sizes.push_back(fd.t.size());
566 }
567
568 // Unified chi2 computation - clean and simple
569 auto compute_chi2 = [&](size_t i, MultiBandModel& local_model, BatchWorkspace& workspace) {
570 Params p;
571 transformer.transform_sample(samples.data() + i * ndim, p);
572
573 double chi2;
574 try {
575 chi2 = local_model.estimate_chi2_with_workspace(p, data_ref, workspace);
576 if (chi2 != chi2) { // NaN check
577 chi2 = std::numeric_limits<double>::infinity();
578 }
579 } catch (...) {
580 chi2 = std::numeric_limits<double>::infinity();
581 }
582 return chi2;
583 };
584
585// Smart serial/parallel selection based on batch size
586#ifdef HAVE_OPENMP
587 constexpr size_t OMP_THRESHOLD = 4;
588 const bool use_parallel = nwalkers >= OMP_THRESHOLD;
589
590 if (use_parallel) {
591 #pragma omp parallel
592 {
593 BatchWorkspace workspace;
594 workspace.initialize(tuple_size, flux_data_sizes);
595 MultiBandModel local_model(config_ref);
596
597 #pragma omp for schedule(dynamic)
598 for (int i = 0; i < static_cast<int>(nwalkers); ++i) {
599 chi2_results(i) = compute_chi2(i, local_model, workspace);
600 }
601 }
602 } else {
603 BatchWorkspace workspace;
604 workspace.initialize(tuple_size, flux_data_sizes);
605 MultiBandModel local_model(config_ref);
606
607 for (size_t i = 0; i < nwalkers; ++i) {
608 chi2_results(i) = compute_chi2(i, local_model, workspace);
609 }
610 }
611#else
612 {
613 BatchWorkspace workspace;
614 workspace.initialize(tuple_size, flux_data_sizes);
615 MultiBandModel local_model(config_ref);
616
617 for (size_t i = 0; i < nwalkers; ++i) {
618 chi2_results(i) = compute_chi2(i, local_model, workspace);
619 }
620 }
621#endif
622
623 pybind11::gil_scoped_acquire acquire;
624 return chi2_results;
625}
