18void sort_synchronized(Array& a, Array& b, Array& c) {
19 std::size_t N = a.size();
20
21 std::vector<std::size_t> idx(N);
22 std::iota(idx.begin(), idx.end(), 0);
23
24 std::sort(idx.begin(), idx.end(), [&](std::size_t i, std::size_t j) { return a(i) < a(j); });
25
26 Array a_sorted = xt::empty<double>({N});
27 Array b_sorted = xt::empty<double>({N});
28 Array c_sorted = xt::empty<double>({N});
29
30 for (std::size_t i = 0; i < N; ++i) {
31 a_sorted(i) = a(idx[i]);
32 b_sorted(i) = b(idx[i]);
33 c_sorted(i) = c(idx[i]);
34 }
35
36 a = std::move(a_sorted);
37 b = std::move(b_sorted);
38 c = std::move(c_sorted);
39}

41double LightCurveData::estimate_chi2() const {
42 double chi_square = 0;
43 for (size_t i = 0; i < t.size(); ++i) {
44 if (Fv_err[i] == 0) continue;
45 double diff = Fv_obs[i] - Fv_model[i];
46 chi_square += (diff * diff) / (Fv_err[i] * Fv_err[i]);
47 }
48 return chi_square;
49}

72void MultiBandData::add_light_curve(double nu, PyArray const& t, PyArray const& Fv_obs, PyArray const& Fv_err) {
73 assert(t.size() == Fv_obs.size() && t.size() == Fv_err.size() && "light curve array inconsistent length!");
74 LightCurveData data;
75
76 data.nu = nu * unit::Hz;

78 data.Fv_obs = Fv_obs;

157void MultiBandModel::build_system(Params const& param, Array const& t_eval, Observer& obs, SynElectronGrid& electrons,
158 SynPhotonGrid& photons) {
159 Real E_iso = param.E_iso * unit::erg;
160 Real Gamma0 = param.Gamma0;
161 Real theta_c = param.theta_c;
162 Real theta_v = param.theta_v;
163 Real theta_w = param.theta_w;
164 Real p = param.p;
165 Real eps_e = param.eps_e;
166 Real eps_B = param.eps_B;
167 Real xi = param.xi;
168
169 Real lumi_dist = config.lumi_dist * unit::cm;
170 Real z = config.z;
171
172 // create model
173 Medium medium;
174 if (config.medium == "ism") {
175 Real n_ism = param.n_ism / unit::cm3;
176 std::tie(medium.rho, medium.mass) = evn::ISM(n_ism);
177 } else if (config.medium == "wind") {
178 std::tie(medium.rho, medium.mass) = evn::wind(param.A_star);
179 } else {
180 std::cerr << "Error: Unknown medium type" << std::endl;
181 }
182
183 Ejecta jet;
184 if (config.jet == "tophat") {
185 jet.eps_k = math::tophat(theta_c, E_iso / (4 * con::pi));
186 jet.Gamma0 = math::tophat(theta_c, Gamma0);
187 } else if (config.jet == "gaussian") {
188 jet.eps_k = math::gaussian(theta_c, E_iso / (4 * con::pi));
189 jet.Gamma0 = math::gaussian(theta_c, Gamma0);
190 } else if (config.jet == "powerlaw") {
191 jet.eps_k = math::powerlaw(theta_c, E_iso / (4 * con::pi), param.k_jet);
192 jet.Gamma0 = math::powerlaw(theta_c, Gamma0, param.k_jet);
193 } else {
194 std::cerr << "Error: Unknown jet type" << std::endl;
195 }
196
197 Real t_resol = config.t_resol;
198 Real theta_resol = config.theta_resol;
199 Real phi_resol = config.phi_resol;
200
201 auto coord = auto_grid(jet, t_eval, theta_w, theta_v, z, phi_resol, theta_resol, t_resol);
202
203 auto shock = generate_fwd_shock(coord, medium, jet, eps_e, eps_B, config.rtol);
204
205 // obs.observe_at(t_eval, coord, shock, lumi_dist, z);
206 obs.observe(coord, shock, lumi_dist, z);
207
208 electrons = generate_syn_electrons(shock, p, xi);
209
210 photons = generate_syn_photons(shock, electrons);
211}

213double MultiBandModel::estimate_chi2(Params const& param) {
214 Observer obs;
215 SynElectronGrid electrons;
216 SynPhotonGrid photons;
