35template <typename Ejecta>
36Array find_jet_jumps(Ejecta const& jet, Real gamma_cut, [[maybe_unused]] bool is_axisymmetric) {
37 // Detect sharp discontinuities in the Gamma0 profile via binary-search refinement.
38 constexpr size_t n_scan = 512;
39 constexpr Real eps = defaults::solver::binary_search_eps;
40 const Real theta_lo = defaults::grid::theta_min;
41 const Real theta_hi = con::pi / 2;
42 const Real dtheta = (theta_hi - theta_lo) / (n_scan - 1);
43
44 if (jet.Gamma0(0, theta_hi) >= gamma_cut) {
45 return {theta_hi};
46 }
47
48 std::vector<Real> jumps;
49 jumps.reserve(16);
50
51 Real prev_th = theta_lo;
52 Real prev_G = jet.Gamma0(0, theta_lo);
53
54 for (size_t j = 1; j < n_scan; ++j) {
55 const Real cur_th = theta_lo + dtheta * static_cast<Real>(j);
56 const Real cur_G = jet.Gamma0(0, cur_th);
57
58 if (prev_G >= gamma_cut || cur_G >= gamma_cut) {
59 const Real dG = std::abs(cur_G - prev_G);
60 const Real scale = std::max(prev_G - 1, cur_G - 1);
61
62 if (scale > 0 && dG > 0.5 * scale) {
63 Real lo = prev_th, hi = cur_th;
64 while (hi - lo > eps) {
65 const Real mid = 0.5 * (lo + hi);
66 const Real G_mid = jet.Gamma0(0, mid);
67 if (std::abs(G_mid - prev_G) < std::abs(G_mid - cur_G)) {
68 lo = mid;
69 } else {
70 hi = mid;
71 }
72 }
73 jumps.push_back(prev_G > cur_G ? lo : hi);
74 }
75 }
76 prev_th = cur_th;
77 prev_G = cur_G;
78 }
79
80 return xt::adapt(jumps, {jumps.size()});
81}

98template <typename Ejecta, typename Medium>
99Real jet_spreading_edge(Ejecta const& jet, Medium const& /*medium*/, Real phi, Real theta_min, Real theta_max,
100 Real /*t0*/) {
101 const Real step = (theta_max - theta_min) / 256;
102 Real theta_s = theta_min;
103 Real dp_min = 0;
104
105 for (Real theta = theta_min; theta <= theta_max; theta += step) {
106 const Real th_lo = std::max(theta - step, theta_min);
107 const Real th_hi = std::min(theta + step, theta_max);
108 const Real dp = (jet.Gamma0(phi, th_hi) - jet.Gamma0(phi, th_lo)) / (th_hi - th_lo);
109
110 if (dp < dp_min) {
111 dp_min = dp;
112 theta_s = theta;
113 }
114 }
115 if (dp_min == 0) {
116 theta_s = theta_max;
117 }
118
119 return theta_s;
120}

122template <typename Func>
123Array inverse_CFD_sampling(Func&& pdf, Real min, Real max, size_t num,
124 size_t sample_num = defaults::sampling::theta_samples, bool log_sample = false) {
125 using namespace boost::numeric::odeint;
126 constexpr Real rtol = defaults::solver::ode_rtol;
127 Array x_i = log_sample ? xt::eval(xt::logspace(std::log10(min), std::log10(max), sample_num))
128 : xt::eval(xt::linspace(min, max, sample_num));
129 Array CDF_i = xt::zeros<Real>({sample_num});
130
131 auto stepper = make_dense_output(rtol, rtol, runge_kutta_dopri5<Real>());
132 stepper.initialize(0, min, (max - min) / 1e3);
133
134 for (size_t k = 1, steps = 0; stepper.current_time() <= max;) {
135 stepper.do_step(pdf);
136 if (++steps > defaults::solver::max_ode_steps) {
137 break;
138 }
139 while (k < x_i.size() && stepper.current_time() > x_i(k)) {
140 stepper.calc_state(x_i(k), CDF_i(k));
141 ++k;
142 }
143 }
144
145 Array CDF_out = xt::linspace(CDF_i.front(), CDF_i.back(), num);
146 Array x_out = xt::zeros<Real>({num});
147
148 for (size_t k = 0; k < num; ++k) {
149 for (size_t j = 0; j < sample_num; ++j) {
150 if (CDF_out(k) <= CDF_i(j)) {
151 if (j == 0) {
152 x_out(k) = x_i(j);
153 } else {
154 Real slope = (x_i(j) - x_i(j - 1)) / (CDF_i(j) - CDF_i(j - 1));
155 x_out(k) = x_i(j - 1) + slope * (CDF_out(k) - CDF_i(j - 1));
156 }
157 break;
158 }
159 }
160 }
161 return x_out;
162}

164template <typename Ejecta>
165Array adaptive_theta_grid(Ejecta const& jet, Real theta_min, Real theta_max, size_t base_pts, Real theta_v,
166 Real theta_resol, ThetaGridParams const& tgp = {}) {
167 // Pre-scan: find peak structure weight, Gamma_peak, CDF estimate, and bright edge.
168 // Bright edge = outermost angle where structure > 1% of peak; limits core beam extent
169 // so structured jets (Gaussian, power-law) don't waste beam points in dim wings.
170 constexpr size_t scan_pts = 100;
171 const Real theta_extent = theta_max - theta_min;
172 Real peak_weight = 0;
173 Real Gamma_peak = 1.0;
174 Real struct_sum = 0;
175 size_t last_bright = 0;
176 for (size_t i = 0; i <= scan_pts; ++i) {
177 const Real Gamma = jet.Gamma0(0, theta_min + theta_extent * i / scan_pts);
178 const Real w = structure_weight(Gamma);
179 struct_sum += w;
180 if (w > peak_weight) {
181 peak_weight = w;
182 Gamma_peak = Gamma;
183 last_bright = i;
184 } else if (w > 0.01 * peak_weight) {
185 last_bright = i;
186 }
187 }
188 const Real floor_weight = tgp.floor_fraction * peak_weight;
189 const Real CDF_est = (struct_sum / scan_pts + floor_weight) * theta_extent;
190 const Real theta_bright = theta_min + theta_extent * last_bright / scan_pts;
191
192 const Real Gamma_v = jet.Gamma0(0, std::clamp(theta_v, theta_min, theta_max));
193 Gamma_peak = std::max(Gamma_peak, Gamma_v);
194 const Real doppler_alpha = tgp.doppler_alpha * std::sqrt(peak_weight / std::max(structure_weight(Gamma_v), 1.0));
195
196 // Beaming density: b(θ) = Γ²θ/(1+Γ²θ²), derived from |d/dθ ln(δ³)|.
197 // Core term (centered at θ=0): resolves jet core, extent limited to theta_bright.
198 // View term (centered at θ_v): resolves observer's Doppler transition, only when θ_v
199 // is outside the core beaming cone (θ_v·Γ > 3).
200 constexpr Real beam_offset = 1.0;
201
202 auto compute_beam_pts = [&](Real log_decades, Real beam_coeff) -> size_t {
203 return static_cast<size_t>(std::max(0.0, log_decades - beam_offset) * theta_resol * beam_coeff);
204 };
205
206 const Real Gamma_peak_sq = Gamma_peak * Gamma_peak;
207 const Real Gamma_v_sq = Gamma_v * Gamma_v;
208 const size_t core_beam_pts =
209 compute_beam_pts(std::log10(std::max(1.0, Gamma_peak * (theta_bright - theta_min))), tgp.core_beam_coeff);
210 const size_t view_beam_pts =
211 (theta_v * Gamma_peak > 3.0)
212 ? compute_beam_pts(std::log10(std::max(1.0, Gamma_v * std::max(theta_v - theta_min, theta_max - theta_v))),
213 tgp.view_beam_coeff)
214 : 0;
215 const size_t total_pts = base_pts + core_beam_pts + view_beam_pts;
216
217 // Calibrate beam weights so each term attracts its target fraction of points.
218 auto calibrate_beam_weight = [&](size_t n_pts, Real beam_cdf) -> Real {
219 return (n_pts > 0 && beam_cdf > 0) ? static_cast<Real>(n_pts) / base_pts * CDF_est / beam_cdf : 0.0;
220 };
221
222 const Real core_weight =
223 calibrate_beam_weight(core_beam_pts, 0.5 * std::log((1.0 + Gamma_peak_sq * theta_max * theta_max) /
224 (1.0 + Gamma_peak_sq * theta_min * theta_min)));
225 const Real theta_v_left = theta_v - theta_min;
226 const Real theta_v_right = theta_max - theta_v;
227 const Real view_weight =
228 calibrate_beam_weight(view_beam_pts, 0.5 * (std::log(1.0 + Gamma_v_sq * theta_v_left * theta_v_left) +
229 std::log(1.0 + Gamma_v_sq * theta_v_right * theta_v_right)));
230
231 auto pdf_eqn = [=, &jet](Real const& /*cdf*/, Real& pdf, Real theta) {
232 const Real Gamma = jet.Gamma0(0, theta);
233 const Real beta = std::sqrt(std::fabs(Gamma * Gamma - 1)) / Gamma;
234 const Real doppler = (1 - beta) / (1 - beta * std::cos(theta - theta_v));
235 const Real structure = structure_weight(Gamma);
236 const Real dtheta = theta - theta_v;
237 pdf = core_weight * Gamma_peak_sq * theta / (1.0 + Gamma_peak_sq * theta * theta) +
238 view_weight * Gamma_v_sq * std::fabs(dtheta) / (1.0 + Gamma_v_sq * dtheta * dtheta) +
239 (1 + doppler_alpha * doppler) * structure + floor_weight;
240 };
241
242 return inverse_CFD_sampling(pdf_eqn, theta_min, theta_max, total_pts, defaults::sampling::theta_samples,
243 /*log_sample=*/true);
244}

248template <typename Ejecta>
249Array adaptive_phi_grid(Ejecta const& jet, size_t phi_num, Real theta_v, Array const& theta_grid,
250 bool is_axisymmetric) {
251 if (theta_v == 0 && is_axisymmetric) {
252 return xt::linspace(0., 2 * con::pi, phi_num);
253 }
254
255 const Real cos_tv = std::cos(theta_v);
256 const Real sin_tv = std::sin(theta_v);
257 const size_t n_theta = theta_grid.size();
258
259 // Precompute dcos_theta bin widths from theta grid
260 std::vector<Real> dcos(n_theta);
261 for (size_t it = 0; it < n_theta; ++it) {
262 const Real left = (it == 0) ? 0.0 : 0.5 * (theta_grid(it - 1) + theta_grid(it));
263 const Real right = (it == n_theta - 1) ? theta_grid(it) : 0.5 * (theta_grid(it) + theta_grid(it + 1));
264 dcos[it] = std::fabs(std::cos(left) - std::cos(right));
265 }
266
267 // Weight function: Doppler-weighted structure summed across theta bins
268 auto phi_weight = [=, &jet](Real phi) {
269 const Real cos_phi = std::cos(phi);
270 Real w = 0;
271 for (size_t it = 0; it < n_theta; ++it) {
272 const Real theta = theta_grid(it);
273 const Real Gamma = jet.Gamma0(phi, theta);
274 const Real beta = std::sqrt(std::fabs(Gamma * Gamma - 1)) / Gamma;
275 const Real cos_alpha = std::cos(theta) * cos_tv + std::sin(theta) * sin_tv * cos_phi;
276 const Real a = (1 - beta) / (1 - beta * cos_alpha);
277 w += a * structure_weight(Gamma) * dcos[it];
278 }
279 return w;
280 };
281
282 // Pre-scan to find peak weight for floor calculation
283 constexpr size_t scan_pts = 100;
284 Real peak_weight = 0;
285 for (size_t s = 0; s <= scan_pts; ++s) {
286 const Real phi = 2 * con::pi * static_cast<Real>(s) / scan_pts;
287 peak_weight = std::max(peak_weight, phi_weight(phi));
288 }
289 const Real floor_weight = 0.05 * peak_weight;
290
291 auto eqn = [=](Real const& /*cdf*/, Real& pdf, Real phi) { pdf = phi_weight(phi) + floor_weight; };
292
293 return inverse_CFD_sampling(eqn, 0, 2 * con::pi, phi_num);
294}

296template <typename Arr>
297Array merge_grids(Arr const& arr1, Arr const& arr2) {
298 std::vector<Real> result;
299 result.reserve(arr1.size() + arr2.size());
300
301 size_t i = 0;
302 size_t j = 0;
303 auto add_unique = [&](Real val) {
304 if (result.empty() || result.back() != val) {
305 result.push_back(val);
306 }
307 };
308
309 while (i < arr1.size() && j < arr2.size()) {
310 if (arr1[i] <= arr2[j]) {
311 add_unique(arr1[i++]);
312 if (arr1[i - 1] == arr2[j]) {
313 j++;
314 }
315 } else {
316 add_unique(arr2[j++]);
317 }
318 }
319 while (i < arr1.size()) {
320 add_unique(arr1[i++]);
321 }
322 while (j < arr2.size()) {
323 add_unique(arr2[j++]);
324 }
325
326 return xt::adapt(result);
327}

505/// Scan all cells to determine global time bounds for grid construction.
506template <typename Ejecta, typename Medium>
507TimeScanResult scan_time_bounds(Coord const& coord, Ejecta const& jet, Medium const& medium, Real t_min, Real t_end,
508 Real z, bool is_rvs, size_t phi_size) {
509 const size_t theta_size = coord.theta.size();
510 const Real cos_tv = std::cos(coord.theta_view);
511 const Real sin_tv = std::sin(coord.theta_view);
512
513 Real min_raw = t_end, min_guarded = t_end, min_cut = t_end, max_ref = 0;
514 auto t_dec_cache = xt::xtensor<Real, 2>::from_shape({phi_size, theta_size});
515 auto t_start_cache = xt::xtensor<Real, 2>::from_shape({phi_size, theta_size});
516 auto early_t_cache = xt::xtensor<Real, 2>::from_shape({phi_size, theta_size});
517
518 for (size_t i = 0; i < phi_size; ++i) {
519 for (size_t j = 0; j < theta_size; ++j) {
520 const Real b = physics::relativistic::gamma_to_beta(jet.Gamma0(coord.phi(i), coord.theta(j)));
521 const Real cos_a =
522 std::cos(coord.theta(j)) * cos_tv + std::sin(coord.theta(j)) * sin_tv * std::cos(coord.phi(i));
523 const Real ts = 0.99 * t_min * (1 - b) / (1 - cos_a * b) / (1 + z);
524
525 const Real td = estimate_t_dec(jet, medium, coord.phi(i), coord.theta(j));
526 t_dec_cache(i, j) = td;
527 Real cut = std::min(0.01 * td, 1e-2 * unit::sec);
528 if (is_rvs) {
529 cut = std::min(cut, 0.01 * jet.T0);
530 max_ref = std::max(max_ref, 10.0 * std::max(td, jet.T0));
531 }
532
533 t_start_cache(i, j) = std::max(ts, cut);
534 early_t_cache(i, j) = 0.99 * std::min(ts, cut);
535
536 min_raw = std::min(min_raw, ts);
537 min_guarded = std::min(min_guarded, std::max(ts, cut));
538 min_cut = std::min(min_cut, cut);
539 }
540 }
541
542 return {min_raw,
543 min_guarded,
544 min_raw < min_cut,
545 max_ref,
546 std::move(t_dec_cache),
547 std::move(t_start_cache),
548 std::move(early_t_cache)};

551inline std::pair<size_t, size_t> compute_time_grid_size(Real t_end, Real min_t_start, Real max_t_refine, Real t_resol,
552 bool is_rvs) {
553 constexpr Real rvs_refinement_ratio = 2;
554 // t_num_base: points a plain fwd logspace would use (reference density for post-crossing)
555 const size_t t_num_base = static_cast<size_t>(std::max(std::log10(t_end / min_t_start), 1.0) * t_resol);
556 // Extra pre-crossing points so that pre/post density ratio = rvs_refinement_ratio
557 size_t t_num_rvs_extra = 0;
558 if (is_rvs && max_t_refine > min_t_start) {
559 const Real log_pre_span = std::log10(std::min(max_t_refine, t_end) / min_t_start);
560 t_num_rvs_extra = static_cast<size_t>((rvs_refinement_ratio - 1.0) * log_pre_span * t_resol);
561 }
562 return {t_num_base, t_num_base + t_num_rvs_extra};
563}

565inline Array make_time_grid(Real ts, Real td, Real t_end, Real T0, size_t t_num_tot, size_t t_num_base, bool is_rvs) {
566 if (is_rvs) {

571 return xt::logspace(std::log10(ts), std::log10(t_end), t_num_tot);

584template <typename Ejecta, typename Medium>
585void build_time_grid(Coord& coord, Ejecta const& jet, Medium const& medium, Real t_min, Real t_max, Real z,
586 Real t_resol, bool is_axisymmetric, bool is_rvs) {
587 const size_t theta_size = coord.theta.size();
588 const size_t phi_size = is_axisymmetric ? 1 : coord.phi.size();
589 const Real t_end = 1.01 * t_max / (1 + z);
590
591 auto [min_t_early, min_t_start, has_early_point, max_t_refine, t_dec, t_start, early_t] =
592 scan_time_bounds(coord, jet, medium, t_min, t_end, z, is_rvs, phi_size);
593
594 auto [t_num_base, t_num_tot] = compute_time_grid_size(t_end, min_t_start, max_t_refine, t_resol, is_rvs);
595 const size_t t_num = t_num_tot + (has_early_point ? 1 : 0);
596
597 coord.t = xt::zeros<Real>({phi_size, theta_size, t_num});
598
599 if (coord.symmetry >= Symmetry::phi_symmetric) {
600 for (size_t r = 0; r < coord.theta_reps.size(); ++r) {
601 const size_t j_rep = coord.theta_reps[r];
602 const size_t j_end = (r + 1 < coord.theta_reps.size()) ? coord.theta_reps[r + 1] : theta_size;
603 auto grid = make_time_grid(min_t_start, t_dec(0, j_rep), t_end, jet.T0, t_num_tot, t_num_base, is_rvs);
604 for (size_t i = 0; i < phi_size; ++i) {
605 for (size_t j = j_rep; j < j_end; ++j) {
606 store_time_grid(coord, i, j, min_t_early, grid, has_early_point, t_num);
607 }
608 }
609 }
610 } else {
611 for (size_t i = 0; i < phi_size; ++i) {
612 for (size_t j = 0; j < theta_size; ++j) {
613 auto grid = make_time_grid(t_start(i, j), t_dec(i, j), t_end, jet.T0, t_num_tot, t_num_base, is_rvs);
614 store_time_grid(coord, i, j, early_t(i, j), grid, has_early_point, t_num);
615 }
616 }
617 }
618}

620template <typename Ejecta, typename Medium>
621Coord auto_grid(Ejecta const& jet, Medium const& medium, Array const& t_obs, Real theta_cut, Real theta_view, Real z,
622 bool is_rvs, Real phi_resol, Real theta_resol, Real t_resol, bool is_axisymmetric) {
623 Coord coord;
624 coord.theta_view = theta_view;
625
626 size_t min_theta_num = defaults::grid::min_theta_points;
627
628 const Array jet_jumps = find_jet_jumps(jet, con::Gamma_cut, is_axisymmetric);
629 Real jet_edge = find_theta_max(jet, con::Gamma_cut);
630 for (size_t i = 0; i < jet_jumps.size(); ++i) {
631 jet_edge = std::max(jet_edge, jet_jumps(i));
632 }
633 const Real theta_min = defaults::grid::theta_min;
634 const Real theta_max = std::min(jet_edge, theta_cut);
635
636 const size_t theta_num = min_theta_num + static_cast<size_t>((theta_max - theta_min) * 180 / con::pi * theta_resol);
637
638 constexpr ThetaGridParams tgp{};
639 const Array base_theta = adaptive_theta_grid(jet, theta_min, theta_max, theta_num, theta_view, theta_resol, tgp);
640
641 const Real avg_spacing = (theta_max - theta_min) / base_theta.size();
642 const Array feature_theta = jump_refinement_grid(jet_jumps, theta_min, theta_max, avg_spacing);
643 coord.theta = merge_grids(base_theta, feature_theta);
644
645 const size_t phi_base = std::max<size_t>(static_cast<size_t>(360 * phi_resol), 1);
646 const Real doppler_sharpness = jet.Gamma0(0, theta_view) * std::sin(theta_view);
647 const Real phi_boost = std::sqrt(std::max(doppler_sharpness / (2 * con::pi), 1.0));
648 const size_t phi_num = std::clamp(static_cast<size_t>(phi_base * phi_boost), size_t(1), phi_base * 5);
649
650 if (phi_num <= 2) {
651 coord.phi = xt::linspace(0., 2 * con::pi, phi_num);
652 } else {
653 coord.phi = adaptive_phi_grid(jet, phi_num, theta_view, coord.theta, is_axisymmetric);
654 }
655 // Shift phi grid by half the first spacing so no point lands on the
656 // Doppler peak at phi=0
657 if (phi_num >= 2) {
658 const Real shift = 0.5 * (coord.phi(1) - coord.phi(0));
659 coord.phi += shift;
660 }
661
662 if (t_obs.size() == 0) {
663 assert(false && "auto_grid: t_obs is empty");
664 return coord;
665 }
666
667 const Real t_max = *std::ranges::max_element(t_obs);
668 const Real t_min = *std::ranges::min_element(t_obs);
669
670 coord.detect_symmetry(jet, medium, t_min, t_max, z, t_resol);
671 build_time_grid(coord, jet, medium, t_min, t_max, z, t_resol, is_axisymmetric, is_rvs);
672
673 return coord;
674}
