249template <typename Electrons, typename Photons>
250void ICPhoton<Electrons, Photons>::generate_grid() {
251 Real gamma_min = std::min(electrons.gamma_m, electrons.gamma_c);
252 Real gamma_max = electrons.gamma_M * 10;
253 size_t gamma_size = static_cast<size_t>(std::log10(gamma_max / gamma_min) * gamma_grid_per_order);
254
255 Real nu_min = std::min(photons.nu_a, photons.nu_m) / 10;
256 Real nu_max = photons.nu_M * 10;
257 size_t nu_size = static_cast<size_t>(std::log10(nu_max / nu_min) * nu_grid_per_order);
258
259 logspace_boundary_center(std::log2(nu_min), std::log2(nu_max), nu_size, nu0, dnu0);
260
261 logspace_boundary_center(std::log2(gamma_min), std::log2(gamma_max), gamma_size, gamma, dgamma);
262
263 // I_nu = Array({nu_size}, -1);
264 // column_den = Array({gamma_size}, -1);
265 IC_tab = MeshGrid({gamma_size, nu_size}, -1);
266 generated = true;
267
268 I_nu = Array::from_shape({nu_size});
269 column_den = Array::from_shape({gamma_size});
270
271 for (size_t i = 0; i < gamma_size; ++i) {
272 column_den(i) = electrons.compute_column_den(gamma(i));
273 }
274
275 for (size_t j = 0; j < nu_size; ++j) {
276 I_nu(j) = photons.compute_I_nu(nu0(j));
277 }
278}

280template <typename Electrons, typename Photons>
281Real ICPhoton<Electrons, Photons>::compute_I_nu(Real nu) {
282 if (generated == false) generate_grid();
283
284 Real IC_I_nu = 0;
285
286 size_t gamma_size = gamma.size();
287 size_t nu_size = nu0.size();
288
289 const auto sigma =
290 KN ? +[](Real nu_comv) { return compton_cross_section(nu_comv); } : +[](Real) { return con::sigmaT; };
291
292 for (size_t i = gamma_size; i-- > 0;) {
293 Real gamma_i = gamma(i);
294 Real upscatter = 4 * gamma_i * gamma_i * IC_x0;
295 Real Ndgamma = column_den(i) * dgamma(i);
296
297 if (nu > upscatter * nu0.back()) continue;
298
299 bool extrapolate = true;
300 for (size_t j = nu_size; j-- > 0;) {
301 Real nu0_j = nu0(j);
302
303 if (IC_tab(i, j) < 0) { // integral at (gamma(i), nu(j)) has not been evaluated
304 Real nu_comv = gamma_i * nu0_j;
305 Real inv = 1 / nu_comv;
306
307 Real grid_value = I_nu(j) * sigma(nu_comv) * inv * inv * dnu0(j);
308 IC_tab(i, j) = (j != nu_size - 1) ? (IC_tab(i, j + 1) + grid_value) : grid_value;
309 }
310
311 if (upscatter * nu0_j < nu) {
312 IC_I_nu += Ndgamma * (IC_tab(i, j + 1) + (IC_tab(i, j) - IC_tab(i, j + 1)) / (nu0(j + 1) - nu0(j)) *
313 (nu0(j + 1) - nu / upscatter));
314
315 extrapolate = false;
316 break;
317 }
318 }
319 if (extrapolate) {
320 IC_I_nu += Ndgamma *
321 (IC_tab(i, 0) + (IC_tab(i, 0) - IC_tab(i, 1)) / (nu0(1) - nu0(0)) * (nu0(0) - nu / upscatter));
322 }
323 }
324
325 return IC_I_nu * nu / 4;
326}

355template <typename Electrons>
356Real compute_Thomson_Y(Real B, Real t_com, Real eps_e, Real eps_B, Electrons const& e) {
357 Real eta_e = eta_rad(e.gamma_m, e.gamma_c, e.p);
358 Real b = eta_e * eps_e / eps_B;
359 Real Y0 = (std::sqrt(1 + 4 * b) - 1) / 2;
360 Real Y1 = 2 * Y0;
361 for (; std::fabs((Y1 - Y0) / Y0) > 1e-4;) {
362 Y1 = Y0;
363 Real gamma_c = compute_gamma_c(t_com, B, e.Ys, e.p);
364 eta_e = eta_rad(e.gamma_m, gamma_c, e.p);
365 b = eta_e * eps_e / eps_B;
366 Y0 = (std::sqrt(1 + 4 * b) - 1) / 2;
367 }
368 return Y0;
369}

371template <typename Electrons, typename Photons>
372void Thomson_cooling(ElectronGrid<Electrons>& electrons, PhotonGrid<Photons>& photons, Shock const& shock) {
373 size_t phi_size = electrons.shape()[0];
374 size_t theta_size = electrons.shape()[1];
375 size_t t_size = electrons.shape()[2];
376

378 for (size_t j = 0; j < theta_size; ++j) {
379 for (size_t k = 0; k < t_size; ++k) {
380 Real Y_T = compute_Thomson_Y(shock.B(i, j, k), shock.t_comv(i, j, k), shock.rad.eps_e, shock.rad.eps_B,
