284template <typename Electrons, typename Photons>
285typename ICPhoton<Electrons, Photons>::GridParams ICPhoton<Electrons, Photons>::compute_grid_params() const {
286 GridParams params;
287
288 constexpr Real eps_tail = 1e-2; // target suppression level for single-cutoff tail
289 constexpr Real safety_margin = 2.0; // retain headroom beyond estimated tail
290 constexpr Real min_tail_factor = 5.0;
291
292 const Real tail_factor = std::max(-std::log(eps_tail), min_tail_factor);
293
294 params.gamma_min = std::min(electrons.gamma_m, electrons.gamma_c) / 3;
295 params.gamma_max = std::max(electrons.gamma_M * tail_factor, params.gamma_min);
296 params.gamma_breaks = {electrons.gamma_a, electrons.gamma_m, electrons.gamma_c, electrons.gamma_M};
297
298 params.nu_min = std::min(photons.nu_a, photons.nu_m) / 10;
299 params.nu_max = std::max(photons.nu_M * tail_factor, params.nu_min);
300
301 params.nu_breaks = {photons.nu_a, photons.nu_m, photons.nu_c, photons.nu_M};
302
303 params.nu_IC_min = 4 * IC_x0 * params.nu_min * params.gamma_min * params.gamma_min;
304
305 const Real nu_ic_base = 4 * IC_x0 * photons.nu_M * electrons.gamma_M * electrons.gamma_M;
306 const Real nu_ic_single_cut = std::max(nu_ic_base * tail_factor * tail_factor, nu_ic_base * tail_factor);
307 params.nu_IC_max = nu_ic_single_cut * safety_margin;
308
309 compute_break_weight(params);
310
311 return params;
312}

314template <typename Electrons, typename Photons>
315void ICPhoton<Electrons, Photons>::compute_break_weight(GridParams& params) const {
316 params.nu_break_weight.fill(0);
317 params.gamma_break_weight.fill(0);
318 params.nu_IC_break_weight.fill(0);
319 params.nu_IC_breaks.fill(std::numeric_limits<Real>::quiet_NaN());
320
321 const std::array<Real, 4> nu_seed_breaks = params.nu_breaks;
322 const std::array<Real, 4> gamma_breaks = params.gamma_breaks;
323
324 static_assert(std::tuple_size_v<decltype(GridParams::nu_IC_breaks)> == nu_seed_breaks.size() * gamma_breaks.size());
325
326 std::array<Real, 4> dN_e_break{};
327 for (size_t j = 0; j < gamma_breaks.size(); ++j) {
328 dN_e_break[j] = electrons.compute_column_den(gamma_breaks[j]);
329 params.gamma_break_weight[j] = dN_e_break[j] / (gamma_breaks[j] * gamma_breaks[j]);
330 }
331
332 size_t k = 0;
333 for (size_t i = 0; i < nu_seed_breaks.size(); ++i) {
334 const Real nu_seed = nu_seed_breaks[i];
335 const Real I_seed = photons.compute_I_nu(nu_seed);
336 params.nu_break_weight[i] = I_seed / (nu_seed * nu_seed);
337
338 for (size_t j = 0; j < gamma_breaks.size(); ++j) {
339 const Real gamma_b = gamma_breaks[j];
340 const Real nu_IC = 2 * IC_x0 * nu_seed * gamma_b * gamma_b;
341 const Real dN_e = dN_e_break[j];
342 const Real KN_correction = KN ? compton_correction(gamma_b * nu_seed) : Real(1);
343
344 params.nu_IC_break_weight[k] = dN_e * I_seed * KN_correction;
345 params.nu_IC_breaks[k] = nu_IC;
346 ++k;
347 }
348 }
349}

351template <typename Electrons, typename Photons>
352void ICPhoton<Electrons, Photons>::initialize_grids(GridParams const& params, Array& gamma, Array& dgamma,
353 Array& nu_seed, Array& dnu_seed, Array& nu_IC) {
354 constexpr Real gamma_grid_per_order{5};
355 constexpr Real nu_grid_per_order{3.5};
356 constexpr Real ic_grid_per_order{2};
357
358 adaptive_grid_with_breaks(std::log2(params.nu_IC_min), std::log2(params.nu_IC_max),
359 std::span<const Real>(params.nu_IC_breaks),
360 std::span<const Real>(params.nu_IC_break_weight), ic_grid_per_order, nu_IC, 1, 2, 3);
361 log2_nu_IC = xt::log2(nu_IC);
362
363 adaptive_grid_with_breaks(std::log2(params.nu_min), std::log2(params.nu_max),
364 std::span<const Real>(params.nu_breaks), std::span<const Real>(params.nu_break_weight),
365 nu_grid_per_order, nu_seed, 1, 0.5, 3);
366 compute_bin_widths(nu_seed, dnu_seed);
367
368 log2space(std::log2(params.gamma_min), std::log2(params.gamma_max), gamma_grid_per_order, gamma);
369 compute_trapezoidal_weights(gamma, dgamma);
370}

372template <typename Electrons, typename Photons>
373void ICPhoton<Electrons, Photons>::sample_distributions(Array const& gamma, Array const& dgamma, Array const& nu_seed,
374 Array& dN_e_boost, Array& I_nu_seed) {
375 dN_e_boost = Array::from_shape({gamma.size()});
376 const size_t g_size = gamma.size();
377 for (size_t i = 0; i < g_size; ++i) {
378 const Real gamma_i = gamma(i);
379 dN_e_boost(i) = electrons.compute_column_den(gamma_i) * dgamma(i) / (gamma_i * gamma_i);
380 }
381
382 I_nu_seed = Array::from_shape({nu_seed.size()});
383 const size_t nu_size = nu_seed.size();
384 for (size_t j = 0; j < nu_size; ++j) {
385 I_nu_seed(j) = photons.compute_I_nu(nu_seed(j));
386 }
387}

389template <typename Electrons, typename Photons>
390void ICPhoton<Electrons, Photons>::build_cdf_thomson(Array const& I_nu_seed, Array const& nu_seed,
391 Array const& dnu_seed, Array& fv_buf, Array& cdf_buf) {
392 const int nu_last = static_cast<int>(nu_seed.size()) - 1;
393
394 fv_buf(nu_last) = I_nu_seed(nu_last) / (nu_seed(nu_last) * nu_seed(nu_last));
395 cdf_buf(nu_last) = 0;
396 for (int j = nu_last - 1; j >= 0; --j) {
397 fv_buf(j) = I_nu_seed(j) / (nu_seed(j) * nu_seed(j));
398 cdf_buf(j) = cdf_buf(j + 1) + 0.5 * (fv_buf(j) + fv_buf(j + 1)) * dnu_seed(j);
399 }
400}

402template <typename Electrons, typename Photons>
403void ICPhoton<Electrons, Photons>::build_cdf_KN(Real gamma_i, Array const& I_nu_seed, Array const& nu_seed,
404 Array const& dnu_seed, Array& fv_buf, Array& cdf_buf) {
405 const int nu_last = static_cast<int>(nu_seed.size()) - 1;
406 Real const* I_nu_ptr = I_nu_seed.data();
407 Real const* nu_ptr = nu_seed.data();
408 Real const* dnu_ptr = dnu_seed.data();
409 Real* fv_p = fv_buf.data();
410 Real* cdf_p = cdf_buf.data();
411
412 fv_p[nu_last] =
413 I_nu_ptr[nu_last] / (nu_ptr[nu_last] * nu_ptr[nu_last]) * compton_correction(gamma_i * nu_ptr[nu_last]);
414 cdf_p[nu_last] = 0;
415 for (int j = nu_last - 1; j >= 0; --j) {
416 fv_p[j] = I_nu_ptr[j] / (nu_ptr[j] * nu_ptr[j]) * compton_correction(gamma_i * nu_ptr[j]);
417 cdf_p[j] = cdf_p[j + 1] + 0.5 * (fv_p[j] + fv_p[j + 1]) * dnu_ptr[j];
418 }

421template <typename Electrons, typename Photons>
422void ICPhoton<Electrons, Photons>::accumulate_IC(Real dN_e_boost, Real gamma_i2, Array const& nu_IC,
423 Array const& nu_seed, Array const& dnu_seed, Array const& fv_buf,
424 Array const& cdf_buf, Array& I_buf, size_t spec_size) {
425 Real const* nu_IC_ptr = nu_IC.data();
426 Real const* nu_ptr = nu_seed.data();
427 Real const* dnu_ptr = dnu_seed.data();
428 Real const* fv_p = fv_buf.data();
429 Real const* cdf_p = cdf_buf.data();
430 Real* I_p = I_buf.data();
431 const int nu_last = static_cast<int>(nu_seed.size()) - 1;
432 const Real inv_gi2 = inv_4x0 / gamma_i2;
433
434 if (cdf_p[0] <= 0)
435 return;
436
437 // Plateau: nu_seed < nu_seed_min, all seed photons overshoot, CDF = cdf_p[0]
438 const Real plateau = dN_e_boost * cdf_p[0];
439 size_t k = 0;
440 while (k < spec_size && nu_IC_ptr[k] * inv_gi2 < nu_ptr[0]) {
441 I_p[k++] += plateau;
442 }
443
444 for (int j = 0; j < nu_last && k < spec_size; ++j) {
445 const Real f_lo = fv_p[j];
446 const Real f_hi = fv_p[j + 1];
447 const Real cdf_hi = cdf_p[j + 1];
448 const Real dnu = dnu_ptr[j];
449 const Real nu_lo = nu_ptr[j];
450 const Real inv_dnu = 1.0 / dnu;
451
452 size_t k_hi = k;
453 while (k_hi < spec_size && nu_IC_ptr[k_hi] * inv_gi2 < nu_ptr[j + 1]) {
454 ++k_hi;
455 }
456
457 for (size_t kk = k; kk < k_hi; ++kk) {
458 const Real frac = (nu_IC_ptr[kk] * inv_gi2 - nu_lo) * inv_dnu;
459 const Real rem = 1.0 - frac;
460 const Real f_seed = f_lo * rem + f_hi * frac;
461 I_p[kk] += dN_e_boost * (cdf_hi + 0.5 * (f_seed + f_hi) * rem * dnu);
462 }
463
464 k = k_hi;
465 }
466}

468template <typename Electrons, typename Photons>
469void ICPhoton<Electrons, Photons>::compute_IC_spectrum(Array const& gamma, Array const& dN_e_boost,
470 Array const& nu_seed, Array const& dnu_seed, Array const& nu_IC,
471 Array const& I_nu_seed) {
472 const size_t spec_size = log2_nu_IC.size();
473 const size_t nu_size = nu_seed.size();
474 const int gamma_size = static_cast<int>(std::min(gamma.size(), dN_e_boost.size()));
475
476 log2_I_nu_IC = Array::from_shape({spec_size});
477
478 static thread_local Array I_buf, cdf_buf, fv_buf;
479 I_buf.resize({spec_size});
480 cdf_buf.resize({nu_size});
481 fv_buf.resize({nu_size});
482 I_buf.fill(0);
483
484 if (KN) {
485 for (int i = 0; i < gamma_size; ++i) {
486 if (dN_e_boost(i) <= 0)
487 continue;
488 const Real gamma_i = gamma(i);
489 build_cdf_KN(gamma_i, I_nu_seed, nu_seed, dnu_seed, fv_buf, cdf_buf);
490 accumulate_IC(dN_e_boost(i), gamma_i * gamma_i, nu_IC, nu_seed, dnu_seed, fv_buf, cdf_buf, I_buf,
491 spec_size);
492 }
493 } else {
494 build_cdf_thomson(I_nu_seed, nu_seed, dnu_seed, fv_buf, cdf_buf);
495 for (int i = 0; i < gamma_size; ++i) {
496 if (dN_e_boost(i) <= 0)
497 continue;
498 accumulate_IC(dN_e_boost(i), gamma(i) * gamma(i), nu_IC, nu_seed, dnu_seed, fv_buf, cdf_buf, I_buf,
499 spec_size);
500 }
501 }
502
503 const Real log2_scale = fast_log2(0.25 * con::sigmaT);
504 interp_slope = Array::from_shape({spec_size - 1});
505 for (size_t i = 0; i < spec_size; ++i) {
506 log2_I_nu_IC(i) = fast_log2(I_buf[i]) + log2_nu_IC(i) + log2_scale;
507 if (i > 0) {
508 const Real dl = log2_nu_IC(i) - log2_nu_IC(i - 1);
509 interp_slope(i - 1) = (dl != 0) ? (log2_I_nu_IC(i) - log2_I_nu_IC(i - 1)) / dl : 0;
510 }
511 }
512}

519template <typename Electrons, typename Photons>
520Real ICPhoton<Electrons, Photons>::compute_log2_I_nu(Real log2_nu) {
521 if (!generated) {
522 generate_spectrum();
523 }
524
525 const size_t n = log2_nu_IC.size();
526 if (n < 2 || log2_nu > log2_nu_IC(n - 1)) {
527 return -con::inf;
528 }
529
530 size_t idx = 0;
531 while (idx + 2 < n && log2_nu_IC(idx + 1) <= log2_nu) {
532 ++idx;
533 }
534
535 return log2_I_nu_IC(idx) + (log2_nu - log2_nu_IC(idx)) * interp_slope(idx);
536}

538template <typename Electrons, typename Photons>
539ICPhotonGrid<Electrons, Photons> generate_IC_photons(ElectronGrid<Electrons> const& electrons,
540 PhotonGrid<Photons> const& photons, bool KN,
541 Shock const& shock) noexcept {
542 size_t phi_size = electrons.shape()[0];
543 size_t theta_size = electrons.shape()[1];
544 size_t t_size = electrons.shape()[2];
545
546 ICPhotonGrid<Electrons, Photons> IC_ph({phi_size, theta_size, t_size});
547
548 const size_t phi_compute = (shock.symmetry != Symmetry::structured) ? 1 : phi_size;
549
550 for (size_t i = 0; i < phi_compute; ++i) {
551 for (size_t j : shock.theta_reps) {
552 for (size_t k = 0; k < t_size; ++k) {

610template <typename Electrons, typename Photons, typename Updater>
611void IC_cooling(ElectronGrid<Electrons>& electrons, PhotonGrid<Photons>& photons, Shock const& shock,
612 Updater&& update_gamma_c) {
613 const size_t phi_size = electrons.shape()[0];
614 const size_t t_size = electrons.shape()[2];
615
616 const size_t phi_compute = (shock.symmetry != Symmetry::structured) ? 1 : phi_size;
617
618 for (size_t i = 0; i < phi_compute; ++i) {
619 for (size_t j : shock.theta_reps) {
620 const size_t k_inj = shock.injection_idx(i, j);
621
622 for (size_t k = 0; k < t_size; ++k) {
623 const Real t_com = shock.t_comv(i, j, k);
624 const Real B = shock.B(i, j, k);
625
626 auto& elec = electrons(i, j, k);
627 auto& Ys = elec.Ys;
628 const Real p = elec.p;
629 const Real gamma_c_last = electrons(i, j, k > 0 ? k - 1 : 0).gamma_c;
630
631 update_gamma_c(elec.gamma_c, Ys, shock.rad, B, t_com, elec.gamma_m, gamma_c_last);
632
633 update_gamma_M(elec.gamma_M, Ys, p, B);
634
635 if (k >= k_inj) {
636 auto const& inj = electrons(i, j, k_inj - 1);
637 const Real dt_comv = t_com - shock.t_comv(i, j, k_inj - 1);
638 elec.gamma_c = cool_after_crossing(inj.gamma_c, inj.gamma_m, elec.gamma_m, t_com, B, 0);
639
640 elec.gamma_M = cool_after_crossing(inj.gamma_M, inj.gamma_m, elec.gamma_m, dt_comv, B, 0);
641 }
642
643 const Real I_nu_peak = compute_syn_I_peak(B, p, elec.column_den);
644 elec.gamma_a = compute_syn_gamma_a(B, I_nu_peak, elec.gamma_m, elec.gamma_c, elec.gamma_M, p);
645 elec.regime = determine_regime(elec.gamma_a, elec.gamma_c, elec.gamma_m);
646 elec.Y_c = Ys.gamma_spectrum(elec.gamma_c);
647 }
648 }
649 }
650 broadcast_symmetry(electrons, shock);
651 generate_syn_photons(photons, shock, electrons);
652}
