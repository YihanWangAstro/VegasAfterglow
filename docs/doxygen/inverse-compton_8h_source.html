40/**
41 * <!-- ************************************************************************************** -->
42 * @struct IntegratorGrid
43 * @brief Defines a grid for numerical integration in log-space. Stack struct to avoid memory allocation.
44 * @details Given minimum and maximum values for nu and gamma, it computes logarithmically spaced bins (nu_edge and
45 * gamma_edge) and then determines center values (nu and gamma) from those bins.
46 * <!-- ************************************************************************************** -->
47 */
48struct IntegratorGrid {
49 template <std::size_t N>
50 using StackArray = xt::xtensor_fixed<Real, xt::xshape<N>>;
51
52 template <std::size_t N, std::size_t M>
53 using StackMesh = xt::xtensor_fixed<Real, xt::xshape<N, M>>;
54 /**
55 * @brief Constructor: Initializes the grid with given nu and gamma boundaries.
56 * @param nu_min Minimum nu-value
57 * @param nu_max Maximum nu-value
58 * @param gamma_min Minimum gamma-value
59 * @param gamma_max Maximum gamma-value
60 */
61 IntegratorGrid(Real nu_min, Real nu_max, Real gamma_min, Real gamma_max) {
62 // Generate logarithmically spaced bin edges for nu.
63 nu_edge = xt::logspace(std::log10(nu_min), std::log10(nu_max), num + 1);
64 // Generate logarithmically spaced bin edges for gamma.
65 gamma_edge = xt::logspace(std::log10(gamma_min), std::log10(gamma_max), num + 1);
66 boundary_to_center(nu_edge, nu); // Compute center values for nu.
67 boundary_to_center(gamma_edge, gamma); // Compute center values for gamma.
68 }
69
70 static constexpr size_t num{128}; ///< Number of bins.
71 StackArray<num + 1> nu_edge{0}; ///< Bin edges for nu.
72 StackArray<num + 1> gamma_edge{0}; ///< Bin edges for gamma.
73 StackArray<num> nu{0}; ///< Center values for nu.
74 StackArray<num> gamma{0}; ///< Center values for gamma.
75 StackArray<num> I_nu_syn{0}; ///< Synchrotron intensity at each nu center.
76 StackArray<num> column_den{0}; ///< Number density at each gamma center.
77 StackMesh<num, num> I0{{{0}}}; ///< 2D array to store computed intermediate values.
78};

80struct IntegratorGridDynamic {
81 /**
82 * @brief Constructor: Initializes the grid with given nu and gamma boundaries.
83 * @param nu_min Minimum nu-value
84 * @param nu_max Maximum nu-value
85 * @param gamma_min Minimum gamma-value
86 * @param gamma_max Maximum gamma-value
87 */
88 IntegratorGridDynamic(Real nu_min, Real nu_max, Real gamma_min, Real gamma_max) {
89 Real log_nu_min = std::log10(nu_min);
90 Real log_nu_max = std::log10(nu_max);
91 num = static_cast<size_t>(std::ceil((log_nu_max - log_nu_min) * ppd));
92 // Generate logarithmically spaced bin edges for nu.
93 nu_edge = xt::logspace(log_nu_min, log_nu_max, num + 1);
94 // Generate logarithmically spaced bin edges for gamma.
95 gamma_edge = xt::logspace(std::log10(gamma_min), std::log10(gamma_max), num + 1);
96 nu = xt::zeros<Real>({num});
97 gamma = xt::zeros<Real>({num});
98 boundary_to_center(nu_edge, nu); // Compute center values for nu.
99 boundary_to_center(gamma_edge, gamma); // Compute center values for gamma.
100 I_nu_syn = xt::zeros<Real>({num});
101 column_den = xt::zeros<Real>({num});
102 I0 = xt::zeros<Real>({num, num});
103 }
104
105 constexpr static Real ppd = 5;
106 size_t num;
107 Array nu_edge; ///< Bin edges for nu.
108 Array gamma_edge; ///< Bin edges for gamma.
109 Array nu; ///< Center values for nu.
110 Array gamma; ///< Center values for gamma.
111 Array I_nu_syn; ///< Synchrotron intensity at each nu center.
112 Array column_den; ///< Number density at each gamma center.
113 MeshGrid I0; ///< 2D array to store computed intermediate values.
114};

273template <typename Electrons, typename Photons>
274void ICPhoton::compute_IC_spectrum(Electrons const& electrons, Photons const& photons, bool KN) noexcept {
275 // Get integration boundaries
276 auto [nu0_min, nu0_max, gamma_min, gamma_max] = get_integration_bounds(electrons, photons);
277
278 // Construct an integration grid in nu0 and gamma
279 IntegratorGrid grid(nu0_min, nu0_max, gamma_min, gamma_max);
280
281 fill_input_spectrum(grid, electrons, photons);
282
283 // Compute the differential contributions
284 fill_integration_grid(grid, KN);
285
286 // Perform final integration
287 integrate_IC_spectrum(grid);
288
289 remove_zero_tail();
290
291 // Convert to log space for interpolation
292 log2_I_nu_ = xt::log2(I_nu_IC_);
293 log2_nu_IC_ = xt::log2(nu_IC_);
294}

296template <typename Electrons, typename Photons>
297std::tuple<Real, Real, Real, Real> ICPhoton::get_integration_bounds(Electrons const& electrons,
298 Photons const& photons) noexcept {
299 Real nu0_min = min(photons.nu_m, photons.nu_c, photons.nu_a) / 1e5;
300 Real nu0_max = photons.nu_M * 5;
301
302 Real gamma_min = min(electrons.gamma_m, electrons.gamma_c, electrons.gamma_a);
303 Real gamma_max = electrons.gamma_M * 5;
304
305 return std::make_tuple(nu0_min, nu0_max, gamma_min, gamma_max);
306}
