143/**
144 * <!-- ************************************************************************************** -->
145 * @struct IntegratorGrid
146 * @brief Defines a grid for numerical integration in log-space. Stack struct to avoid memory allocation.
147 * @details Given minimum and maximum values for nu and gamma, it computes logarithmically spaced bins (nu_edge and
148 * gamma_edge) and then determines center values (nu and gamma) from those bins.
149 * <!-- ************************************************************************************** -->
150 */
151struct IntegratorGrid {
152 /**
153 * @brief Constructor: Initializes the grid with given nu and gamma boundaries.
154 * @param nu_min Minimum nu-value
155 * @param nu_max Maximum nu-value
156 * @param gamma_min Minimum gamma-value
157 * @param gamma_max Maximum gamma-value
158 */
159 IntegratorGrid(Real nu_min, Real nu_max, Real gamma_min, Real gamma_max) {
160 // Generate logarithmically spaced bin edges for nu.
161 nu_edge = xt::logspace(std::log10(nu_min), std::log10(nu_max), num + 1);
162 // Generate logarithmically spaced bin edges for gamma.
163 gamma_edge = xt::logspace(std::log10(gamma_min), std::log10(gamma_max), num + 1);
164 boundary_to_center(nu_edge, nu); // Compute center values for nu.
165 boundary_to_center(gamma_edge, gamma); // Compute center values for gamma.
166 }
167
168 static constexpr size_t num{128}; ///< Number of bins.
169 StackArray<num + 1> nu_edge{0}; ///< Bin edges for nu.
170 StackArray<num + 1> gamma_edge{0}; ///< Bin edges for gamma.
171 StackArray<num> nu{0}; ///< Center values for nu.
172 StackArray<num> gamma{0}; ///< Center values for gamma.
173 StackArray<num> P_nu{0}; ///< Specificpower at each nu center.
174 StackArray<num> column_num_den{0}; ///< column Number density at each gamma center.
175 StackMesh<num, num> I0{{{0}}}; ///< 2D array to store computed intermediate values.
176};

343template <typename Electrons, typename Photons>
344void ICPhoton::compute_IC_spectrum(Electrons const& electrons, Photons const& photons, bool KN) noexcept {
345 // Get integration boundaries
346 auto [nu0_min, nu0_max, gamma_min, gamma_max] = get_integration_bounds(electrons, photons);
347
348 // Construct an integration grid in nu0 and gamma
349 // IntegratorGrid grid(nu0_min, nu0_max, gamma_min, gamma_max);
350 IntegratorGrid grid(nu0_min, nu0_max, gamma_min, gamma_max);
351
352 fill_input_spectrum(grid, electrons, photons);
353
354 // Compute the differential contributions
355 fill_integration_grid(grid, KN);
356
357 // Perform final integration
358 integrate_IC_spectrum(grid);
359
360 remove_zero_tail();
361
362 // Convert to log space for interpolation
363 log2_P_nu_ = xt::log2(P_nu_IC_);
364 log2_nu_IC_ = xt::log2(nu_IC_);
365}

367template <typename Electrons, typename Photons>
368std::tuple<Real, Real, Real, Real> ICPhoton::get_integration_bounds(Electrons const& electrons,
369 Photons const& photons) noexcept {
370 Real nu0_min = min(photons.nu_m, photons.nu_c, photons.nu_a) / 1e5;
371 Real nu0_max = photons.nu_M * 5;
372
373 Real gamma_min = min(electrons.gamma_m, electrons.gamma_c, electrons.gamma_a);
374 Real gamma_max = electrons.gamma_M * 5;
375
376 return std::make_tuple(nu0_min, nu0_max, gamma_min, gamma_max);
377}

381 // For each bin in nu0, compute the synchrotron intensity and column number density

383 grid.P_nu(i) = photons.compute_P_nu(grid.nu(i));
