137/**
138 * <!-- ************************************************************************************** -->
139 * @struct IntegratorGrid
140 * @brief Defines a grid for numerical integration in log-space. Stack struct to avoid memory allocation.
141 * @details Given minimum and maximum values for nu and gamma, it computes logarithmically spaced bins (nu_edge and
142 * gamma_edge) and then determines center values (nu and gamma) from those bins.
143 * <!-- ************************************************************************************** -->
144 */
145struct IntegratorGrid {
146 template <std::size_t N>
147 using StackArray = xt::xtensor_fixed<Real, xt::xshape<N>>;
148
149 template <std::size_t N, std::size_t M>
150 using StackMesh = xt::xtensor_fixed<Real, xt::xshape<N, M>>;
151 /**
152 * @brief Constructor: Initializes the grid with given nu and gamma boundaries.
153 * @param nu_min Minimum nu-value
154 * @param nu_max Maximum nu-value
155 * @param gamma_min Minimum gamma-value
156 * @param gamma_max Maximum gamma-value
157 */
158 IntegratorGrid(Real nu_min, Real nu_max, Real gamma_min, Real gamma_max) {
159 // Generate logarithmically spaced bin edges for nu.
160 nu_edge = xt::logspace(std::log10(nu_min), std::log10(nu_max), num + 1);
161 // Generate logarithmically spaced bin edges for gamma.
162 gamma_edge = xt::logspace(std::log10(gamma_min), std::log10(gamma_max), num + 1);
163 boundary_to_center(nu_edge, nu); // Compute center values for nu.
164 boundary_to_center(gamma_edge, gamma); // Compute center values for gamma.
165 }
166
167 static constexpr size_t num{128}; ///< Number of bins.
168 StackArray<num + 1> nu_edge{0}; ///< Bin edges for nu.
169 StackArray<num + 1> gamma_edge{0}; ///< Bin edges for gamma.
170 StackArray<num> nu{0}; ///< Center values for nu.
171 StackArray<num> gamma{0}; ///< Center values for gamma.
172 StackArray<num> I_nu_syn{0}; ///< Synchrotron intensity at each nu center.
173 StackArray<num> column_den{0}; ///< Number density at each gamma center.
174 StackMesh<num, num> I0{{{0}}}; ///< 2D array to store computed intermediate values.
175};

177struct IntegratorGridDynamic {
178 /**
179 * @brief Constructor: Initializes the grid with given nu and gamma boundaries.
180 * @param nu_min Minimum nu-value
181 * @param nu_max Maximum nu-value
182 * @param gamma_min Minimum gamma-value
183 * @param gamma_max Maximum gamma-value
184 */
185 IntegratorGridDynamic(Real nu_min, Real nu_max, Real gamma_min, Real gamma_max) {
186 Real log_nu_min = std::log10(nu_min);
187 Real log_nu_max = std::log10(nu_max);
188 num = static_cast<size_t>(std::ceil((log_nu_max - log_nu_min) * ppd));
189 // Generate logarithmically spaced bin edges for nu.
190 nu_edge = xt::logspace(log_nu_min, log_nu_max, num + 1);
191 // Generate logarithmically spaced bin edges for gamma.
192 gamma_edge = xt::logspace(std::log10(gamma_min), std::log10(gamma_max), num + 1);
193 nu = xt::zeros<Real>({num});
194 gamma = xt::zeros<Real>({num});
195 boundary_to_center(nu_edge, nu); // Compute center values for nu.
196 boundary_to_center(gamma_edge, gamma); // Compute center values for gamma.
197 I_nu_syn = xt::zeros<Real>({num});
198 column_den = xt::zeros<Real>({num});
199 I0 = xt::zeros<Real>({num, num});
200 }
201
202 constexpr static Real ppd = 5;
203 size_t num;
204 Array nu_edge; ///< Bin edges for nu.
205 Array gamma_edge; ///< Bin edges for gamma.

378template <typename Electrons, typename Photons>
379void ICPhoton::compute_IC_spectrum(Electrons const& electrons, Photons const& photons, bool KN) noexcept {
380 // Get integration boundaries
381 auto [nu0_min, nu0_max, gamma_min, gamma_max] = get_integration_bounds(electrons, photons);
382
383 // Construct an integration grid in nu0 and gamma
384 IntegratorGrid grid(nu0_min, nu0_max, gamma_min, gamma_max);
385
386 fill_input_spectrum(grid, electrons, photons);
387
388 // Compute the differential contributions
389 fill_integration_grid(grid, KN);
390
391 // Perform final integration
392 integrate_IC_spectrum(grid);
393
394 remove_zero_tail();
395
396 // Convert to log space for interpolation
397 log2_I_nu_ = xt::log2(I_nu_IC_);
398 log2_nu_IC_ = xt::log2(nu_IC_);
399}

401template <typename Electrons, typename Photons>
402std::tuple<Real, Real, Real, Real> ICPhoton::get_integration_bounds(Electrons const& electrons,
403 Photons const& photons) noexcept {
404 Real nu0_min = min(photons.nu_m, photons.nu_c, photons.nu_a) / 1e5;
405 Real nu0_max = photons.nu_M * 5;
406
407 Real gamma_min = min(electrons.gamma_m, electrons.gamma_c, electrons.gamma_a);
408 Real gamma_max = electrons.gamma_M * 5;
409
410 return std::make_tuple(nu0_min, nu0_max, gamma_min, gamma_max);
411}
