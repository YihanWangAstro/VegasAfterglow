288template <typename Electrons, typename Photons>
289typename ICPhoton<Electrons, Photons>::GridParams ICPhoton<Electrons, Photons>::compute_grid_params() const {
290 GridParams params;
291
292 params.gamma_min = electrons.gamma_m;
293 params.gamma_max = electrons.gamma_M * 30;
294 params.gamma_size =
295 static_cast<size_t>(std::max(std::log10(params.gamma_max / params.gamma_min), 3.) * gamma_grid_per_order);
296
297 params.nu_min = std::min(photons.nu_a, photons.nu_m) / 10;
298 params.nu_max = photons.nu_M * 30;
299 params.nu_size = static_cast<size_t>(std::max(std::log10(params.nu_max / params.nu_min), 3.) * nu_grid_per_order);
300
301 params.nu_IC_min = 4 * IC_x0 * params.nu_min * params.gamma_min * params.gamma_min;
302 params.nu_IC_max = 4 * IC_x0 * params.nu_max * params.gamma_max * params.gamma_max;
303 params.spectrum_resol = static_cast<size_t>(std::max(10 * std::log10(params.nu_IC_max / (params.nu_IC_min)), 30.));
304
305 return params;
306}

317template <typename Electrons, typename Photons>
318void ICPhoton<Electrons, Photons>::preprocess_distributions(GridParams const& params, Array& gamma, Array& dgamma,
319 Array& nu_sync, Array& dnu_sync, Array& dN_e,
320 Array& I_nu_sync) const {
321
322 logspace_boundary_center(std::log2(params.nu_min), std::log2(params.nu_max), params.nu_size, nu_sync, dnu_sync);
323 logspace_boundary_center(std::log2(params.gamma_min), std::log2(params.gamma_max), params.gamma_size, gamma,
324 dgamma);
325
326 dN_e = Array::from_shape({params.gamma_size});
327 for (size_t i = 0; i < params.gamma_size; ++i) {
328 dN_e(i) = electrons.compute_column_den(gamma(i)) * dgamma(i);
329 }
330
331 I_nu_sync = Array::from_shape({params.nu_size});
332 for (size_t j = 0; j < params.nu_size; ++j) {
333 I_nu_sync(j) = photons.compute_I_nu(nu_sync(j));
334 }

337template <typename Electrons, typename Photons>
338void ICPhoton<Electrons, Photons>::compute_IC_spectrum(GridParams const& params, Array const& gamma,
339 Array const& nu_sync, Array const& dN_e, Array const& I_nu_sync,
340 Array const& dnu_sync) {
341 Array nu_IC = xt::exp2(log2_nu_IC);
342 Array I_nu_IC = xt::zeros<Real>({params.spectrum_resol});
343
344 //const auto sigma =
345 // KN ? +[](Real nu_comv) { return compton_cross_section(nu_comv); } : +[](Real) { return con::sigmaT; };
346
347 if (KN) { //separate the KN and non-KN case to ensure inline
348 for (size_t i = params.gamma_size; i-- > 0;) {
349 const Real gamma_i = gamma(i);
350 const Real Ndgamma = dN_e(i);
351 const Real down_scatter = 1 / (4 * IC_x0 * gamma_i * gamma_i);
352
353 const Real nu_comv_max = gamma_i * nu_sync.back();
354 Real row_integral =
355 I_nu_sync.back() * dnu_sync.back() * compton_cross_section(nu_comv_max) / (nu_comv_max * nu_comv_max);
356 Real slope = 0;
357 Real nu_sync_j = 0;
358
359 int k = static_cast<int>(params.spectrum_resol) - 1;
360 for (; k >= 0; --k) {
361 if (nu_IC(k) * down_scatter < nu_sync.back()) {
362 break;
363 }
364 }
365
366 for (size_t j = params.nu_size - 1; j-- > 0 && k >= 0;) {
367 nu_sync_j = nu_sync(j);
368 const Real nu_comv = gamma_i * nu_sync_j;
369 const Real inv = 1 / nu_comv;
370
371 slope = I_nu_sync(j) * compton_cross_section(nu_comv) * inv * inv;
372 row_integral += slope * dnu_sync(j);
373
374 for (; k >= 0; --k) {
375 if (const Real nu_seed = nu_IC(k) * down_scatter; nu_seed >= nu_sync_j) {
376 I_nu_IC(k) += Ndgamma * (row_integral + slope * (nu_sync_j - nu_seed));
377 } else {
378 break;
379 }
380 }
381 }
382
383 // Handle the remaining low-frequency tail
384 for (; k >= 0; --k) {
385 const Real nu_seed = nu_IC(k) * down_scatter;
386 I_nu_IC(k) += Ndgamma * (row_integral + slope * (nu_sync_j - nu_seed));
387 }
388 }
389 log2_I_nu_IC = xt::log2(I_nu_IC * nu_IC * 0.25);
390 } else {
391 Array factor = I_nu_sync / (nu_sync * nu_sync);
392 for (size_t i = params.gamma_size; i-- > 0;) {
393 const Real gamma_i = gamma(i);
394 const Real gamma_i2 = gamma_i * gamma_i;
395 const Real Ndgamma = dN_e(i);
396 const Real down_scatter = 1 / (4 * IC_x0 * gamma_i2);
397 const Real nu_comv_max = gamma_i * nu_sync.back();
398
399 Real row_integral = I_nu_sync.back() * dnu_sync.back() / (nu_comv_max * nu_comv_max);
400 Real slope = 0;
401 Real nu_sync_j = 0;
402
403 int k = static_cast<int>(params.spectrum_resol) - 1;
404 for (; k >= 0; --k) {
405 if (nu_IC(k) * down_scatter < nu_sync.back()) {
406 break;
407 }
408 }
409
410 for (size_t j = params.nu_size - 1; j-- > 0 && k >= 0;) {
411 nu_sync_j = nu_sync(j);
412
413 slope = factor(j) / gamma_i2;
414 row_integral += slope * dnu_sync(j);
415
416 for (; k >= 0; --k) {
417 if (const Real nu_seed = nu_IC(k) * down_scatter; nu_seed >= nu_sync_j) {
418 I_nu_IC(k) += Ndgamma * (row_integral + slope * (nu_sync_j - nu_seed));
419 } else {
420 break;
421 }
422 }
423 }
424
425 // Handle the remaining low-frequency tail
426 for (; k >= 0; --k) {
427 const Real nu_seed = nu_IC(k) * down_scatter;
428 I_nu_IC(k) += Ndgamma * (row_integral + slope * (nu_sync_j - nu_seed));
429 }
430 }
431 log2_I_nu_IC = xt::log2(I_nu_IC * nu_IC * 0.25 * con::sigmaT);
432 }
433
434 for (size_t i = 0; i < params.spectrum_resol - 1; ++i) {
435 interp_slope(i) = (log2_I_nu_IC(i + 1) - log2_I_nu_IC(i)) * inv_dlog2_nu;
436 }
437}

444template <typename Electrons, typename Photons>
445Real ICPhoton<Electrons, Photons>::compute_log2_I_nu(Real log2_nu) {
446 if (generated == false) {
447 generate_spectrum();
448 }
449
450 size_t idx = 0;
451 if (const Real off_set = (log2_nu - log2_nu_IC(0)); off_set >= 0) {
452 idx = static_cast<size_t>(std::floor(off_set * inv_dlog2_nu));
453 }
454
455 if (idx >= interp_slope.size()) {
456 return -con::inf;
457 } else [[likely]] {
458 return log2_I_nu_IC(idx) + (log2_nu - log2_nu_IC(idx)) * interp_slope(idx);
459 }

500template <typename Electrons, typename Photons, typename Updater>
501void IC_cooling(ElectronGrid<Electrons>& electrons, PhotonGrid<Photons>& photons, Shock const& shock,
502 Updater&& update_gamma_c) {
503 const size_t phi_size = electrons.shape()[0];
504 const size_t theta_size = electrons.shape()[1];
505 const size_t t_size = electrons.shape()[2];
506
507 for (size_t i = 0; i < phi_size; ++i) {
508 for (size_t j = 0; j < theta_size; ++j) {
509 const size_t k_inj = shock.injection_idx(i, j);
510
511 for (size_t k = 0; k < t_size; ++k) {
512 if (shock.required(i, j, k) == 0)
513 continue;
514
515 const Real t_com = shock.t_comv(i, j, k);
516 const Real B = shock.B(i, j, k);
517
518 auto& elec = electrons(i, j, k);
519 auto& Ys = elec.Ys;
520 const Real p = elec.p;
521
522 update_gamma_c(elec.gamma_c, Ys, shock.rad, B, t_com, elec.gamma_m);
523
524 update_gamma_M(elec.gamma_M, Ys, p, B);
525
526 if (k >= k_inj) {
527 const auto& inj = electrons(i, j, k_inj);
528 elec.gamma_c = inj.gamma_c * elec.gamma_m / inj.gamma_m;
529 elec.gamma_M = elec.gamma_c;
530 }
531
532 const Real I_nu_peak = compute_syn_I_peak(B, p, elec.column_den);
533 elec.gamma_a = compute_syn_gamma_a(B, I_nu_peak, elec.gamma_m, elec.gamma_c, elec.gamma_M, p);
534 elec.regime = determine_regime(elec.gamma_a, elec.gamma_c, elec.gamma_m);
535 elec.Y_c = Ys.evaluate_at_gamma(elec.gamma_c, p);
536 }
537 }
538 }
539 generate_syn_photons(photons, shock, electrons);
