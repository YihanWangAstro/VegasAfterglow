39/**
40 * <!-- ************************************************************************************** -->
41 * @struct IntegratorGrid
42 * @brief Defines a grid for numerical integration in log-space.
43 * @details Given minimum and maximum values for x and y, it computes logarithmically spaced bins (x_bin and y_bin)
44 * and then determines center values (x and y) from those bins.
45 * <!-- ************************************************************************************** -->
46 */
47struct IntegratorGrid {
48 /**
49 * @brief Constructor: Initializes the grid with given x and y boundaries.
50 * @param x_min Minimum x-value
51 * @param x_max Maximum x-value
52 * @param y_min Minimum y-value
53 * @param y_max Maximum y-value
54 */
55 IntegratorGrid(Real x_min, Real x_max, Real y_min, Real y_max)
56 : x_min(x_min), x_max(x_max), y_min(y_min), y_max(y_max) {
57 logspace(std::log10(x_min), std::log10(x_max), x_bin); // Generate logarithmically spaced bin edges for x.
58 logspace(std::log10(y_min), std::log10(y_max), y_bin); // Generate logarithmically spaced bin edges for y.
59 boundary_to_center(x_bin, x); // Compute center values for x.
60 boundary_to_center(y_bin, y); // Compute center values for y.
61 }
62
63 Real x_min; ///< Minimum x-value.
64 Real x_max; ///< Maximum x-value.
65 Real y_min; ///< Minimum y-value.
66 Real y_max; ///< Maximum y-value.
67 static constexpr size_t num{80}; ///< Number of bins.
68 std::array<Real, num + 1> x_bin{0}; ///< Bin edges for x.
69 std::array<Real, num + 1> y_bin{0}; ///< Bin edges for y.
70 std::array<Real, num> x{0}; ///< Center values for x.
71 std::array<Real, num> y{0}; ///< Center values for y.
72 std::array<Real, num> j_syn{0}; ///< Synchrotron intensity at each x center.
73 std::array<Real, num> ns{0}; ///< Number density at each y center.
74 std::array<std::array<Real, num>, num> I0{{{0}}}; ///< 2D array to store computed intermediate values.
75};

111 /**
112 * <!-- ************************************************************************************** -->
113 * @brief Generates the IC photon spectrum from the given electron and photon data.
114 * @tparam Electrons Type of the electron distribution
115 * @tparam Photons Type of the photon distribution
116 * @param e The electron distribution
117 * @param ph The photon distribution
118 * @details This template member function uses the properties of the electrons (e) and synchrotron photons (ph) to:
119 * - Determine minimum electron Lorentz factor and minimum synchrotron frequency.
120 * - Define integration limits for the synchrotron frequency (nu0) and electron Lorentz factor (gamma).
121 * - Fill in an IntegratorGrid with computed synchrotron intensity and electron column density.
122 * - Compute a 2D array I0 representing differential contributions.
123 * - Finally, integrate over the grid to populate the IC photon spectrum (j_nu_).
124 * <!-- ************************************************************************************** -->
125 */
126 template <typename Electrons, typename Photons>
127 void gen(Electrons const& e, Photons const& ph) {
128 // Real gamma_min = min(e.gamma_m, e.gamma_c, e.gamma_a);
129 Real nu_ph_min = min(ph.nu_m, ph.nu_c, ph.nu_a);
130
131 Real nu0_max = ph.nu_M * 10;
132 Real nu0_min = nu_ph_min / 1e5;
133
134 Real gamma_min = std::min(std::min(e.gamma_m, e.gamma_c), e.gamma_a);
135 Real gamma_max = e.gamma_M * 10;
136
137 // Construct an integration grid in nu0 and gamma.
138 IntegratorGrid grid(nu0_min, nu0_max, gamma_min, gamma_max);
139
140 // For each bin in nu0, compute the synchrotron intensity.
141 for (size_t i = 0; i < grid.num; i++) {
142 grid.j_syn[i] = ph.compute_I_nu(grid.x[i]);
143 grid.ns[i] = e.compute_column_num_den(grid.y[i]);
144 }
145
146 // For each (nu0, gamma) pair, compute differential contributions and fill in I0.
147 for (size_t i = 0; i < grid.num; ++i) {
148 Real nu0_ = grid.x[i];
149 Real dnu = grid.x_bin[i + 1] - grid.x_bin[i];
150 for (size_t j = 0; j < grid.num; ++j) {
151 Real gamma_ = grid.y[j];
152 Real dgamma = grid.y_bin[j + 1] - grid.y_bin[j];
153 Real dS = std::fabs(dnu * dgamma);
154 Real f = 4 * gamma_ * gamma_ * nu0_ * nu0_;
155 grid.I0[i][j] = grid.ns[j] * grid.j_syn[i] * dS / f * compton_sigma(nu0_ / gamma_);
156 }
157 }
158
159 // Compute integration limits for the IC spectrum.
160 Real nu_min = 4 * gamma_min * gamma_min * nu0_min;
161 Real nu_max = 4 * gamma_max * gamma_max * nu0_max;
162
163 // Generate the IC frequency grid and allocate the output array.
164 nu_IC_ = xt::logspace(std::log10(nu_min), std::log10(nu_max), spectrum_resol);
165 j_nu_ = xt::zeros<Real>({spectrum_resol});
166
167 // Integrate over the grid to compute the final IC photon spectrum.
168 for (size_t k = 0; k < nu_IC_.size(); ++k) {
169 for (size_t i = 0; i < grid.num; ++i) {
170 Real nu0_ = grid.x[i];
171 for (size_t j = 0; j < grid.num; ++j) {
172 Real gamma_ = grid.y[j];
173 if (nu0_ <= nu_IC_[k] && nu_IC_[k] <= 4 * gamma_ * gamma_ * nu0_ * IC_x0) {
174 j_nu_[k] += grid.I0[i][j] * nu_IC_[k];
175 }
176 }
177 }
178 }
179 };
180
181 private:
182 Array j_nu_; ///< IC photon spectrum intensity array.
183 Array nu_IC_; ///< Frequency grid for the IC photon spectrum.
184};
