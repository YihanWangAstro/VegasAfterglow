241template <typename Electrons, typename Photons>
242ICPhoton<Electrons, Photons>::ICPhoton(Electrons const& electrons, Photons const& photons, bool KN) noexcept
243 : electrons(electrons), photons(photons), KN(KN) {
244 constexpr Real IC_nu_min = 1e6 * unit::Hz;
245
246 Real gamma_min = std::min(electrons.gamma_m, electrons.gamma_c);
247 Real gamma_max = electrons.gamma_M * 10;
248 size_t gamma_size = static_cast<size_t>(std::log10(gamma_max / gamma_min) * gamma_grid_per_order);
249
250 Real nu_min = std::min(photons.nu_a, photons.nu_m) / 10;
251 Real nu_max = photons.nu_M * 10;
252 size_t nu_size = static_cast<size_t>(std::log10(nu_max / nu_min) * nu_grid_per_order);
253
254 Array nu0_boundary = xt::logspace(std::log10(nu_min), std::log10(nu_max), nu_size + 1);
255 Array gamma_boundary = xt::logspace(std::log10(gamma_min), std::log10(gamma_max), gamma_size + 1);
256
257 nu0 = Array({nu_size}, 0);
258 dnu0 = Array({nu_size}, 0);
259 I_nu = Array({nu_size}, -1);
260 gamma = Array({gamma_size}, 0);
261 dgamma = Array({gamma_size}, 0);
262 column_den = Array({gamma_size}, -1);
263 IC_tab = MeshGrid({gamma_size, nu_size}, -1);
264
265 for (size_t i = 0; i < nu_size; ++i) {
266 nu0(i) = std::sqrt(nu0_boundary(i) * nu0_boundary(i + 1)); // centre value of nu0
267 dnu0(i) = std::fabs(nu0_boundary(i + 1) - nu0_boundary(i)); // width of nu0 bin
268 }
269
270 for (size_t i = 0; i < gamma_size; ++i) {
271 gamma(i) = std::sqrt(gamma_boundary(i) * gamma_boundary(i + 1)); // centre value of gamma
272 dgamma(i) = std::fabs(gamma_boundary(i + 1) - gamma_boundary(i)); // width of gamma bin
273 }
274}

276template <typename Electrons, typename Photons>
277Real ICPhoton<Electrons, Photons>::compute_I_nu(Real nu) {
278 /// A constant used in the IC photon calculation, defined as √2/3.
279 Real IC_I_nu = 0;
280
281 size_t gamma_size = gamma.size();
282 size_t nu_size = nu0.size();
283
284 for (int i = gamma_size - 1; i >= 0; --i) {
285 Real gamma_i = gamma(i);
286 if (column_den(i) < 0) {
287 column_den(i) = electrons.compute_column_den(gamma_i);
288 }
289 int j = nu_size - 1;
290 for (; j >= 0; --j) {
291 Real nu0_j = nu0(j);
292 if (I_nu(j) < 0) {
293 I_nu(j) = photons.compute_I_nu(nu0_j);
294 }
295 if (IC_tab(i, j) < 0) { // integral at (gamma(i), nu(j)) has not been evaluated
296 Real nu_comv = gamma_i * nu0_j;
297 Real sigma = con::sigmaT;
298 if (KN) {
299 sigma = compton_cross_section(nu_comv);
300 }
301 Real grid_value = column_den(i) * I_nu(j) * sigma / (4 * nu_comv * nu_comv) * dnu0(j) * dgamma(i);
302 if (j != nu_size - 1) {
303 IC_tab(i, j) = IC_tab(i, j + 1) + grid_value;
304 } else {
305 IC_tab(i, j) = grid_value;
306 }
307 }
308
309 if (4 * gamma_i * gamma_i * IC_x0 * nu0_j < nu) {
310 if (j != nu_size - 1) {
311 // IC_I_nu += IC_tab(i, j + 1);
312 IC_I_nu += IC_tab(i, j + 1) + (IC_tab(i, j) - IC_tab(i, j + 1)) / (nu0(j + 1) - nu0(j)) *
313 (nu0(j + 1) - nu / (4 * gamma_i * gamma_i * IC_x0));
314 }
315 break;
316 }
317 }
318 if (j < 0) {
319 IC_I_nu += IC_tab(i, 0) + (IC_tab(i, 0) - IC_tab(i, 1)) / (nu0(1) - nu0(0)) *
320 (nu0(0) - nu / (4 * gamma_i * gamma_i * IC_x0));
321 }
322 }
323
324 return IC_I_nu * nu;
325}

354template <typename Electrons>
355Real compute_Thomson_Y(Real B, Real t_com, Real eps_e, Real eps_B, Electrons const& e) {
356 Real eta_e = eta_rad(e.gamma_m, e.gamma_c, e.p);
357 Real b = eta_e * eps_e / eps_B;
358 Real Y0 = (std::sqrt(1 + 4 * b) - 1) / 2;
359 Real Y1 = 2 * Y0;
360 for (; std::fabs((Y1 - Y0) / Y0) > 1e-4;) {
361 Y1 = Y0;
362 Real gamma_c = compute_gamma_c(t_com, B, e.Ys, e.p);
363 eta_e = eta_rad(e.gamma_m, gamma_c, e.p);
364 b = eta_e * eps_e / eps_B;
365 Y0 = (std::sqrt(1 + 4 * b) - 1) / 2;
366 }
367 return Y0;
368}
