57void write_npz(std::string const& filename, SynPhotonGrid const& ph) {
58 auto shape = ph.shape();
59 xt::xarray<Real> nu_a = xt::zeros<Real>({shape[0], shape[1], shape[2]});
60 xt::xarray<Real> nu_m = xt::zeros<Real>({shape[0], shape[1], shape[2]});
61 xt::xarray<Real> nu_c = xt::zeros<Real>({shape[0], shape[1], shape[2]});
62 xt::xarray<Real> nu_M = xt::zeros<Real>({shape[0], shape[1], shape[2]});
63 xt::xarray<Real> I_nu_peak = xt::zeros<Real>({shape[0], shape[1], shape[2]});
64 xt::xarray<Real> regime = xt::zeros<Real>({shape[0], shape[1], shape[2]});
65
66 for (size_t i = 0; i < shape[0]; ++i)
67 for (size_t j = 0; j < shape[1]; ++j)
68 for (size_t k = 0; k < shape[2]; ++k) {
69 nu_a(i, j, k) = ph(i, j, k).nu_a / unit::Hz;
70 nu_m(i, j, k) = ph(i, j, k).nu_m / unit::Hz;
71 nu_c(i, j, k) = ph(i, j, k).nu_c / unit::Hz;
72 nu_M(i, j, k) = ph(i, j, k).nu_M / unit::Hz;
73 I_nu_peak(i, j, k) = ph(i, j, k).I_nu_max / (unit::erg / unit::cm2 / unit::sec / unit::Hz);
74 regime(i, j, k) = ph(i, j, k).regime;
75 }
76
77 xt::dump_npz(filename + ".npz", "nu_a", nu_a, false, false);
78 xt::dump_npz(filename + ".npz", "nu_m", nu_m, false, true);
79 xt::dump_npz(filename + ".npz", "nu_c", nu_c, false, true);
80 xt::dump_npz(filename + ".npz", "nu_Max", nu_M, false, true);
81 xt::dump_npz(filename + ".npz", "I_nu_peak", I_nu_peak, false, true);
82 xt::dump_npz(filename + ".npz", "regime", regime, false, true);
83}
