126 /**
127 * <!-- ************************************************************************************** -->
128 * @brief Creates a stellar wind medium profile
129 * @param A_star Wind parameter in standard units
130 * @param n_ism Number density of the ISM [cm^-3]
131 * @param n0 Number density of inner region [cm^-3]
132 * @return functions for density calculation
133 * @details Converts A_star to proper units (A_star * 5e11 g/cm) and returns functions that compute
134 * density = A/r² and mass = A*r, representing a steady-state stellar wind where density
135 * falls off as 1/r²
136 * <!-- ************************************************************************************** -->
137 */
138 inline auto wind(Real A_star, Real n_ism = 0, Real n0 = con::inf) {
139 // Convert A_star to proper units: A_star * 5e11 g/cm
140 Real A = A_star * 5e11 * unit::g / unit::cm;
141 Real rho_ism = n_ism * con::mp;
142 Real r02 = A / (n0 * con::mp);
143
144 // Return a function that computes density = A/r^2
145 // This represents a steady-state stellar wind where density falls off as 1/r^2
146 return [=](Real phi, Real theta, Real r) noexcept { return A / (r02 + r * r) + rho_ism; };
147 }
148} // namespace evn
