80/**
81 * <!-- ************************************************************************************** -->
82 * @class Wind
83 * @brief Implements a stellar wind medium with density proportional to 1/r².
84 * @details Provides methods to compute density at any position.
85 * The wind is characterized by the wind parameter A_star.
86 * <!-- ************************************************************************************** -->
87 */
88class Wind {
89 public:
90 /**
91 * <!-- ************************************************************************************** -->
92 * @brief Constructor: Initialize with wind parameter A_star (in standard units)
93 * @param A_star Wind density parameter in standard units
94 * @param n_ism number density of ISM at large radii
95 * @param n0 number density of wind at small radii
96 * <!-- ************************************************************************************** -->
97 */
98 explicit Wind(Real A_star, Real n_ism = 0, Real n0 = con::inf) noexcept
99 : A(A_star * 5e11 * unit::g / unit::cm), rho_ism(n_ism * con::mp), r02(A / (n0 * 1.3 * con::mp)) {}
100
101 /**
102 * <!-- ************************************************************************************** -->
103 * @brief Return density at given positions (proportional to 1/r²)
104 * @param phi Azimuthal angle (unused)
105 * @param theta Polar angle (unused)
106 * @param r Radial distance
107 * @return Density value at radius r (= A/r²)
108 * <!-- ************************************************************************************** -->
109 */
110
111 [[nodiscard]] inline Real rho(Real /*phi*/, Real /*theta*/, Real r) const noexcept {
112 return A / (r02 + r * r) + rho_ism;
113 }
114
115 /// Enclosed mass per solid angle up to radius r.
116 [[nodiscard]] inline Real mass(Real r) const noexcept {
117 Real m = rho_ism * r * r * r / 3.0;
118 if (A != 0) {
119 if (r02 > 0) {
120 const Real a = std::sqrt(r02);
121 m += A * (r - a * std::atan(r / a));
122 } else {
123 m += A * r;
124 }
125 }
126 return m;
127 }
128
129 /// Expose parameters for analytic integrals in hot dynamics setup paths.
130 [[nodiscard]] inline Real A_param() const noexcept { return A; }
131 [[nodiscard]] inline Real rho_ism_param() const noexcept { return rho_ism; }
132 [[nodiscard]] inline Real r02_param() const noexcept { return r02; }
133
134 bool isotropic{true}; ///< Flag indicating if the medium is isotropic within computational domain.
135
136 private:
137 Real A{0}; ///< Wind density parameter in physical units
138 Real rho_ism{0}; ///< ISM density floor
139 Real r02{0}; ///< Radius where ISM transitions to
140};

172 /**
173 * <!-- ************************************************************************************** -->
174 * @brief Creates a stellar wind medium profile
175 * @param A_star Wind parameter in standard units
176 * @param n_ism Number density of the ISM [cm^-3]
177 * @param n0 Number density of inner region [cm^-3]
178 * @param k power-law index
179 * @return functions for density calculation
180 * @details Converts A_star to proper units (A_star * 5e11 g/cm) and returns functions that compute
181 * density = A/r² and mass = A*r, representing a steady-state stellar wind where density
182 * falls off as 1/r²
183 * <!-- ************************************************************************************** -->
184 */
185 inline auto wind(Real A_star, Real n_ism = 0, Real n0 = con::inf, Real k = 2) {
186 // Convert A_star to proper units: A_star * 5e11 g/cm
187 constexpr Real r0 = 1e17 * unit::cm; // reference radius
188 const Real A = A_star * 5e11 * unit::g / unit::cm * std::pow(r0, k - 2);
189 const Real rho_ism = n_ism * con::mp;
190 const Real r0k = A / (n0 * 1.3 * con::mp);
191
192 // Return a function that computes density = A/r^k
193 // This represents a steady-state stellar wind where density falls off as 1/r^k
194 return [=](Real /*phi*/, Real /*theta*/, Real r) noexcept { return A / (r0k + std::pow(r, k)) + rho_ism; };
195 }
196} // namespace evn
