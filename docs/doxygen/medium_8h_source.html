137 /**
138 * <!-- ************************************************************************************** -->
139 * @brief Creates a stellar wind medium profile
140 * @param A_star Wind parameter in standard units
141 * @param n_ism Number density of the ISM [cm^-3]
142 * @param n0 Number density of inner region [cm^-3]
143 * @param k power-law index
144 * @return functions for density calculation
145 * @details Converts A_star to proper units (A_star * 5e11 g/cm) and returns functions that compute
146 * density = A/r² and mass = A*r, representing a steady-state stellar wind where density
147 * falls off as 1/r²
148 * <!-- ************************************************************************************** -->
149 */
150 inline auto wind(Real A_star, Real n_ism = 0, Real n0 = con::inf, Real k = 2) {
151 // Convert A_star to proper units: A_star * 5e11 g/cm
152 constexpr Real r0 = 1e17 * unit::cm; // reference radius
153 const Real A = A_star * 5e11 * unit::g / unit::cm * std::pow(r0, k - 2);
154 const Real rho_ism = n_ism * con::mp;
155 const Real r0k = A / (n0 * con::mp);
156
157 // Return a function that computes density = A/r^k
158 // This represents a steady-state stellar wind where density falls off as 1/r^k
159 return [=](Real phi, Real theta, Real r) noexcept { return A / (r0k + std::pow(r, k)) + rho_ism; };
160 }
161} // namespace evn
