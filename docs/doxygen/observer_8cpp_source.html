31void Observer::calc_t_obs(Coord const& coord, Shock const& shock) {
32 const Real cos_obs = std::cos(coord.theta_view);
33 const Real sin_obs = std::sin(coord.theta_view);
34
35 static thread_local MeshGrid3d cos_theta, sin_theta;
36 if (cos_theta.shape() != shock.theta.shape()) {
37 // cos_theta.resize(shock.theta.shape());
38 // sin_theta.resize(shock.theta.shape());
39 cos_theta = MeshGrid3d::from_shape(shock.theta.shape());
40 sin_theta = MeshGrid3d::from_shape(shock.theta.shape());
41 }
42
43 const size_t shock_phi_size = shock.theta.shape(0);
44
45 if (jet_spreading_) {
46 for (size_t i = 0; i < shock_phi_size; ++i) {
47 for (size_t j = 0; j < theta_grid; ++j) {
48 for (size_t k = 0; k < t_grid; ++k) {
49 cos_theta(i, j, k) = std::cos(shock.theta(i, j, k));
50 sin_theta(i, j, k) = std::sin(shock.theta(i, j, k));
51 }
52 }
53 }
54 } else {
55 // theta is constant along k for non-spreading jets, compute once per (i,j)
56 for (size_t i = 0; i < shock_phi_size; ++i) {
57 for (size_t j = 0; j < theta_grid; ++j) {
58 const Real ct = std::cos(shock.theta(i, j, 0));
59 const Real st = std::sin(shock.theta(i, j, 0));
60 for (size_t k = 0; k < t_grid; ++k) {
61 cos_theta(i, j, k) = ct;
62 sin_theta(i, j, k) = st;
63 }
64 }
65 }
66 }
67
68 for (size_t i = 0; i < eff_phi_grid; ++i) {
69 const Real cos_phi = std::cos(coord.phi[i] - coord.phi_view);
70 const size_t i_eff = i * jet_3d;
71
72 for (size_t j = 0; j < theta_grid; ++j) {
73 for (size_t k = 0; k < t_grid; ++k) {
74 const Real gamma_ = shock.Gamma(i_eff, j, k);
75 const Real r = shock.r(i_eff, j, k);
76 const Real t_eng_ = coord.t(i_eff, j, k);
77 const Real cos_v = sin_theta(i_eff, j, k) * cos_phi * sin_obs + cos_theta(i_eff, j, k) * cos_obs;
78 const Real t_val = (t_eng_ + (1 - cos_v) * r / con::c) * one_plus_z;
79
80 lg2_doppler(i, j, k) = -fast_log2(gamma_ - std::sqrt(gamma_ * gamma_ - 1) * cos_v);
81 time(i, j, k) = t_val;
82 lg2_t(i, j, k) = std::log2(t_val);
83 }
84 }
85 }
86}

88void Observer::calc_solid_angle(Coord const& coord, Shock const& shock) {
89 Array dphi({eff_phi_grid}, 0);
90
91 if (eff_phi_grid == 1) {
92 dphi(0) = 2 * con::pi;
93 } else {
94 const size_t last = eff_phi_grid - 1;
95 for (size_t i = 0; i < eff_phi_grid; ++i) {
96 dphi(i) = 0.5 * (coord.phi(std::min(i + 1, last)) - coord.phi(i > 0 ? i - 1 : size_t(0)));
97 }
98 }
99
100 // precompute the dcos to avoid recomputing in axisymmetric jet
101 static thread_local MeshGrid3d dcos;
102 if (dcos.shape() != shock.theta.shape()) {
103 // dcos.resize(shock.theta.shape());
104 dcos = MeshGrid3d::from_shape(shock.theta.shape());
105 }
106
107 const size_t last = theta_grid - 1;
108 const size_t shock_phi_size = shock.theta.shape(0);
109
110 // Interpolate theta of neighbor cell (i, j_nb) at engine time t_target
111 auto interp_theta = [&](size_t i, size_t j_nb, Real t_target, size_t& k_hint) -> Real {
112 while (k_hint + 1 < t_grid && coord.t(i, j_nb, k_hint + 1) < t_target)
113 k_hint++;
114 if (k_hint + 1 >= t_grid)
115 return shock.theta(i, j_nb, t_grid - 1);
116 const Real w =
117 (t_target - coord.t(i, j_nb, k_hint)) / (coord.t(i, j_nb, k_hint + 1) - coord.t(i, j_nb, k_hint));
118 return shock.theta(i, j_nb, k_hint) + w * (shock.theta(i, j_nb, k_hint + 1) - shock.theta(i, j_nb, k_hint));
119 };
120
121 for (size_t i = 0; i < shock_phi_size; ++i) {
122 for (size_t j = 0; j < theta_grid; ++j) {
123 const size_t j_p1 = (j == last) ? last : (j + 1);
124
125 if (jet_spreading_) [[unlikely]] {
126 size_t k_hint_lo = 0, k_hint_hi = 0;
127 for (size_t k = 0; k < t_grid; ++k) {
128 const Real t_target = coord.t(i, j, k);
129 const Real th_lo =
130 (j == 0) ? 0.0 : 0.5 * (shock.theta(i, j, k) + interp_theta(i, j - 1, t_target, k_hint_lo));
131 const Real th_hi = (j == last)
132 ? shock.theta(i, j, k)
133 : 0.5 * (shock.theta(i, j, k) + interp_theta(i, j_p1, t_target, k_hint_hi));
134 dcos(i, j, k) = std::cos(th_hi) - std::cos(th_lo);
135 }
136 } else {
137 const Real th_lo = (j == 0) ? 0.0 : 0.5 * (shock.theta(i, j, 0) + shock.theta(i, j - 1, 0));
138 const Real th_hi = 0.5 * (shock.theta(i, j, 0) + shock.theta(i, j_p1, 0));
139 const Real val = std::cos(th_hi) - std::cos(th_lo);
140 for (size_t k = 0; k < t_grid; ++k) {
141 dcos(i, j, k) = val;
142 }
143 }
144 }
145 }
146
147 for (size_t i = 0; i < eff_phi_grid; ++i) {
148 const size_t i_eff = i * jet_3d;
149 for (size_t j = 0; j < theta_grid; ++j) {
150 for (size_t k = 0; k < t_grid; ++k) {
151 const Real dOmega = std::fabs(dcos(i_eff, j, k) * dphi(i));
152 const Real r = shock.r(i_eff, j, k);
153 lg2_geom_factor(i, j, k) = fast_log2(dOmega * r * r) + 3 * lg2_doppler(i, j, k);
154 }
155 }
156 }
157}

159void Observer::update_required(MaskGrid& required, Array const& t_obs) {
160 xt::noalias(required) = 0;
161 const size_t t_obs_size = t_obs.size();
162
163 // Loop over effective phi and theta grid points.
164 for (size_t i = 0; i < eff_phi_grid; i++) {
165 const size_t i_eff = i * jet_3d;
166
167 for (size_t j = 0; j < theta_grid; j++) {
168 // Skip observation times that are below the grid's start time
169 size_t t_idx = 0;
170 iterate_to(time(i, j, 0), t_obs, t_idx);
171
172 // find the grid points that are required for the interpolation.
173 for (size_t k = 0; k < t_grid - 1 && t_idx < t_obs_size; k++) {
174 const Real t_lo = time(i, j, k);
175 const Real t_hi = time(i, j, k + 1);
176
177 if (t_lo < t_obs(t_idx) && t_obs(t_idx) <= t_hi) {
178 required(i_eff, j, k) = 1;
179 required(i_eff, j, k + 1) = 1;
180 }
181
182 iterate_to(t_hi, t_obs, t_idx);
183 }
184 }
185 }

192void Observer::build_time_grid(Coord const& coord, Shock const& shock, Real luminosity_dist, Real redshift) {
193 auto [phi_size, theta_size, t_size] = shock.shape();
194
195 // Determine if the jet is 3D (more than one phi value)
196 jet_3d = static_cast<size_t>((phi_size > 1));
197
198 // Set the effective phi grid size based on the observation angle and jet dimensionality.
199 if (coord.theta_view == 0 && jet_3d == 0) {
200 this->eff_phi_grid = 1; // optimize for on-axis observer
201 } else {
202 this->eff_phi_grid = coord.phi.size();
203 }
204
205 this->theta_grid = theta_size;
206 this->t_grid = t_size;
207 this->lumi_dist = luminosity_dist;
208 this->one_plus_z = 1 + redshift;
209
210 // Detect jet spreading: theta varies along the k (time) axis
211 this->jet_spreading_ = false;
212 if (t_size > 1) {
213 for (size_t i = 0; i < phi_size && !jet_spreading_; ++i) {
214 for (size_t j = 0; j < theta_size && !jet_spreading_; ++j) {
215 if (shock.theta(i, j, 0) != shock.theta(i, j, t_size - 1)) {
216 jet_spreading_ = true;
217 }
218 }
219 }
220 }
221
222 time = MeshGrid3d::from_shape({eff_phi_grid, theta_size, t_size});
223 lg2_t = MeshGrid3d::from_shape({eff_phi_grid, theta_size, t_size});
224 lg2_doppler = MeshGrid3d::from_shape({eff_phi_grid, theta_size, t_size});
225 lg2_geom_factor = MeshGrid3d::from_shape({eff_phi_grid, theta_size, t_size});
226
227 // Calculate the solid angle grid and observation time grid.
228 calc_t_obs(coord, shock);
229}

235xt::xtensor<size_t, 3> Observer::compute_k_indices(Array const& t_obs) const noexcept {
236 const size_t t_obs_len = t_obs.size();
237 xt::xtensor<size_t, 3> k_indices({eff_phi_grid, theta_grid, t_obs_len}, SIZE_MAX);
238
239 for (size_t i = 0; i < eff_phi_grid; i++) {
240 for (size_t j = 0; j < theta_grid; j++) {
241 // Skip observation times that are below the grid's start time
242 size_t t_idx = 0;
243 iterate_to(time(i, j, 0), t_obs, t_idx);
244
245 // Find the k index for each observation time
246 for (size_t k = 0; k < t_grid - 1 && t_idx < t_obs_len; k++) {
247 const Real t_hi = time(i, j, k + 1);
248 // All observation times in [time(i,j,k), time(i,j,k+1)) map to k
249 while (t_idx < t_obs_len && t_obs(t_idx) <= t_hi) {
250 k_indices(i, j, t_idx) = k;
251 t_idx++;
252 }
253 }
254 }
255 }
256
257 return k_indices;
258}
