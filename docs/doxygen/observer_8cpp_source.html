22 Array dphi({eff_phi_grid}, 0);
23
24 if (eff_phi_grid == 1) {

26 } else {
27 int last = eff_phi_grid - 1;
28 for (int i = 0; i < eff_phi_grid; ++i) {

43 size_t j_m1 = (j == 0) ? 0 : (j - 1);
44 size_t j_p1 = (j == last) ? last : (j + 1);
45 for (size_t k = 0; k < t_grid; ++k) {
46 Real theta_lo = 0.5 * (shock.theta(i, j, k) + shock.theta(i, j_m1, k));
47 Real theta_hi = 0.5 * (shock.theta(i, j, k) + shock.theta(i, j_p1, k));
48 dcos(i, j, k) = std::cos(theta_hi) - std::cos(theta_lo);
49 }
50 }
51 }
52
53 for (size_t i = 0; i < eff_phi_grid; ++i) {
54 size_t i_eff = i * jet_3d;
55 for (size_t j = 0; j < theta_grid; ++j) {
56 for (size_t k = 0; k < t_grid; ++k) {
57 // if (shock.required(i_eff, j, k) == 0) {
58 // continue;
59 // } // maybe remove this inner branch harm to vectorization
60 Real dOmega = std::fabs(dcos(i_eff, j, k) * dphi(i));
61 Real r = shock.r(i_eff, j, k);

68void Observer::calc_t_obs(Coord const& coord, Shock const& shock) {
69 Real cos_obs = std::cos(coord.theta_view);
70 Real sin_obs = std::sin(coord.theta_view);
71
72 static thread_local MeshGrid3d cos_theta, sin_theta;
73 if (cos_theta.shape() != shock.theta.shape()) {
74 cos_theta.resize(shock.theta.shape());
75 sin_theta.resize(shock.theta.shape());
76 }
77
78 size_t shock_phi_size = shock.theta.shape(0);
79 for (size_t i = 0; i < shock_phi_size; ++i) {
80 for (size_t j = 0; j < theta_grid; ++j) {
81 for (size_t k = 0; k < t_grid; ++k) {
82 cos_theta(i, j, k) = std::cos(shock.theta(i, j, k));
83 sin_theta(i, j, k) = std::sin(shock.theta(i, j, k));
84 }
85 }
86 }
87
88 for (size_t i = 0; i < eff_phi_grid; ++i) {
89 Real cos_phi = std::cos(coord.phi[i]);
90 size_t i_eff = i * jet_3d;
91 for (size_t j = 0; j < theta_grid; ++j) {
92 // Compute the cosine of the angle between the local velocity vector and the observer's line of sight.
93 for (size_t k = 0; k < t_grid; ++k) {
94 Real gamma_ = shock.Gamma(i_eff, j, k);
95 Real r = shock.r(i_eff, j, k);
96 Real t_eng_ = coord.t(i_eff, j, k);
97 Real cos_v = sin_theta(i_eff, j, k) * cos_phi * sin_obs + cos_theta(i_eff, j, k) * cos_obs;
98 // Compute the Doppler factor: D = 1 / [Gamma * (1 - beta * cos_v)]
99 lg2_doppler(i, j, k) = -std::log2((gamma_ - std::sqrt(gamma_ * gamma_ - 1) * cos_v));
100 // Compute the observed time: t_obs = [t_eng + (1 - cos_v) * r / c] * (1 + z)
101 time(i, j, k) = (t_eng_ + (1 - cos_v) * r / con::c) * one_plus_z;
102 }
103 }
104 }
105 lg2_t = xt::log2(time);
106}

108void Observer::update_required(MaskGrid& required, Array const& t_obs) {
109 size_t t_obs_size = t_obs.size();
110 Array log2_t_obs = xt::log2(t_obs);
111 // Loop over effective phi and theta grid points.
112 for (size_t i = 0; i < eff_phi_grid; i++) {
113 size_t i_eff = i * jet_3d;
114 for (size_t j = 0; j < theta_grid; j++) {
115 // Skip observation times that are below the grid's start time
116 size_t t_idx = 0;
117 iterate_to(lg2_t(i, j, 0), log2_t_obs, t_idx);
118 // find the grid points that are required for the interpolation.
119 for (size_t k = 0; k < t_grid - 1 && t_idx < t_obs_size; k++) {
120 Real const t_lo = lg2_t(i, j, k);
121 Real const t_hi = lg2_t(i, j, k + 1);
122
123 if (t_lo < log2_t_obs(t_idx) && log2_t_obs(t_idx) <= t_hi) {
124 required(i_eff, j, k) = 1;
125 required(i_eff, j, k + 1) = 1;
126 }
127 iterate_to(t_hi, log2_t_obs, t_idx);
128 }
129 }
130 }

133void Observer::build_time_grid(Coord const& coord, Shock const& shock, Real luminosity_dist, Real redshift) {
134 auto [phi_size, theta_size, t_size] = shock.shape();
135
136 // Determine if the jet is 3D (more than one phi value)
137 jet_3d = static_cast<size_t>((phi_size > 1));
138 // Set effective phi grid size based on the observation angle and jet dimensionality.
139 if (coord.theta_view == 0 && jet_3d == 0) {
140 this->eff_phi_grid = 1; // optimize for on-axis observer
141 } else {
142 this->eff_phi_grid = coord.phi.size();
143 }
144 this->theta_grid = theta_size;
145 this->t_grid = t_size;
146 this->lumi_dist = luminosity_dist;
147 this->one_plus_z = 1 + redshift;
148 this->lg2_one_plus_z = std::log2(one_plus_z);
149
150 time.resize({eff_phi_grid, theta_size, t_size});
151 lg2_t.resize({eff_phi_grid, theta_size, t_size});
152 lg2_doppler.resize({eff_phi_grid, theta_size, t_size});
153 lg2_surface.resize({eff_phi_grid, theta_size, t_size});
154
155 // Calculate the solid angle grid and observation time grid.
156 calc_t_obs(coord, shock);
157}
