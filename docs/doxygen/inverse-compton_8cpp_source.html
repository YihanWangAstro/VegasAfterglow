18InverseComptonY::InverseComptonY(Real gamma_m, Real gamma_c, Real p, Real B, Real Y_T, bool is_KN) noexcept {
19 const Real nu_m = compute_syn_freq(gamma_m, B);
20 this->gamma_m_hat = std::max(con::me * con::c2 / con::h / nu_m, 1.0);
21
22 this->gamma_self = fast_pow(gamma_m_hat * gamma_m * gamma_m, 1.0 / 3.0);
23
24 this->gamma_self3 = gamma_self * gamma_self * gamma_self;
25
26 this->B_ = B;
27 this->gamma_m_ = gamma_m;
28 this->p_ = p;
29 if (is_KN) {
30 update_cooling_breaks(gamma_c, Y_T);
31 } else {
32 this->gamma_c_hat = compute_gamma_hat(gamma_c);
33 this->Y_T = Y_T;
34 regime = 0;
35 build_segments();
36 }
37}

119void InverseComptonY::build_segments() noexcept {
120 switch (regime) {
121 case 0: // Thomson
122 segments_.first_segment(Y_T, 1.0, 0.0);
123 break;
124
125 case 1: // Slow Cooling
126 segments_.first_segment(Y_T, 1.0, 0.0);
127 segments_.add_segment(gamma_c_hat, 0.5 * (p_ - 3.0));
128 segments_.add_segment(gamma_m_hat, -4.0 / 3.0);
129 break;
130
131 case 2: // Fast Cooling, Weak KN
132 segments_.first_segment(Y_T, 1.0, 0.0);
133 segments_.add_segment(gamma_m_hat, -0.5);
134 segments_.add_segment(gamma_c_hat, -4.0 / 3.0);
135 break;
136
137 case 3: // Fast Cooling, Strong KN (gamma_m_hat <= gamma0 < gamma_self)
138 {
139 Real gamma0_hat = compute_gamma_hat(gamma0);
140 Real gamma_m_hat_hat = compute_gamma_hat(gamma_m_hat);
141
142 segments_.first_segment(Y_T, 1.0, 0.0);
143 segments_.add_segment(gamma_m_hat, -0.5);
144 segments_.add_segment(gamma0_hat, -0.75);
145 segments_.add_segment(gamma_m_hat_hat, -0.5);
146 break;
147 }
148 case 4: // Fast Cooling, Strong KN (gamma_self <= gamma0 < gamma_m)
149 {
150 Real gamma0_hat = compute_gamma_hat(gamma0);
151 Real gamma0_hat_hat = compute_gamma_hat(gamma0_hat);
152 Real gamma_m_hat_hat = compute_gamma_hat(gamma_m_hat);
153
154 segments_.first_segment(Y_T, 1.0, 0.0);
155 segments_.add_segment(gamma_m_hat, -0.5);
156 segments_.add_segment(gamma0_hat, -1.0);
157 segments_.add_segment(gamma0_hat_hat, -0.75);
158 segments_.add_segment(gamma_m_hat_hat, -0.5);
159 break;
160 }
161 case 5: // Fast Cooling, Strong KN (gamma_m < gamma0)
162 {
163 Real gamma0_hat = compute_gamma_hat(gamma0);
164 Real gamma_m_hat_hat = compute_gamma_hat(gamma_m_hat);
165 Real gamma0_hat_hat = compute_gamma_hat(gamma0_hat);
166
167 segments_.first_segment(Y_T, 1.0, 0.0);
168 segments_.add_segment(gamma0_hat, 0.5 * (p_ - 3.0));
169 segments_.add_segment(gamma_m_hat, -1.0);
170 segments_.add_segment(gamma_m_hat_hat, 0.25 * (p_ - 5.0));
171 segments_.add_segment(gamma0_hat_hat, -0.5);
172 break;
173 }
174 }

258 // Returns dimensionless KN cross-section ratio: sigma/sigmaT.
259 inline Real compton_cross_section_ratio_from_x(Real x) {
260 /*if (x <= 1) {
261 return con::sigmaT;
262 } else {
263 return 0;
264 }*/
265
266 if (x < 1e-2) {
267 return 1 - 2 * x;
268 } else if (x > 1e2) {
269 return 3. / 8 * (log(2 * x) + 0.5) / x;
270 } else {
271 const Real l = std::log1p(2.0 * x); // log(1+2x)
272 const Real invx = 1.0 / x;
273 const Real invx2 = invx * invx;
274 const Real term1 = 1.0 + 2.0 * x;
275 const Real invt1 = 1.0 / term1;
276 const Real invt1_2 = invt1 * invt1;
277
278 // ((1+x)/x^3) * (2x(1+x)/(1+2x) - log(1+2x)) + log(1+2x)/(2x) - (1+3x)/(1+2x)^2
279 const Real a = (1.0 + x) * invx2 * invx; // (1+x)/x^3
280 const Real b = 2.0 * x * (1.0 + x) * invt1 - l; // bracket
281 const Real c = 0.5 * l * invx; // log_term/(2x)
282 const Real d = (1.0 + 3.0 * x) * invt1_2; // (1+3x)/(1+2x)^2
283
284 return 0.75 * (a * b + c - d);
285 }
286 }

288 struct ComptonSigmaLUT {
289 static constexpr size_t n = 128;
290 static constexpr Real x_min = 1e-2;
291 static constexpr Real x_max = 1e2;
292 static constexpr Real lg2_x_min = -6.6438561897747247;
293 static constexpr Real lg2_x_max = 6.6438561897747247;
294
295 std::array<Real, n> ratio{};
296 Real inv_step{0};
297
298 ComptonSigmaLUT() {
299 const Real step = (lg2_x_max - lg2_x_min) / static_cast<Real>(n - 1);
300 inv_step = 1.0 / step;
301 for (size_t i = 0; i < n; ++i) {
302 const Real lg2_x = lg2_x_min + step * static_cast<Real>(i);
303 ratio[i] = compton_cross_section_ratio_from_x(std::exp2(lg2_x));
304 }
305 }
306
307 [[nodiscard]] inline Real eval_mid_x(Real x) const noexcept {
308 const Real lg2_x = fast_log2(x);
309 const Real pos = (lg2_x - lg2_x_min) * inv_step;
310 if (pos <= 0) {
311 return ratio.front();
312 }
313 if (pos >= static_cast<Real>(n - 1)) {
314 return ratio.back();
315 }
316 const size_t idx = static_cast<size_t>(pos);
317 const Real frac = pos - static_cast<Real>(idx);
318 return ratio[idx] + (ratio[idx + 1] - ratio[idx]) * frac;
319 }
320 };
321} // namespace

323Real compton_correction(Real nu) {
324 const Real x = con::h / (con::me * con::c2) * nu;
325 if (!(x > 0)) {
326 return 0;
327 }
328 if (x <= ComptonSigmaLUT::x_min || x >= ComptonSigmaLUT::x_max) {
329 return compton_cross_section_ratio_from_x(x);
330 }
