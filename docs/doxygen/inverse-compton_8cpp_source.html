18InverseComptonY::InverseComptonY(Real nu_m, Real nu_c, Real B, Real Y_T) noexcept {
19 gamma_hat_m = con::me * con::c2 / con::h / nu_m; // Compute minimum characteristic Lorentz factor
20 gamma_hat_c = con::me * con::c2 / con::h / nu_c; // Compute cooling characteristic Lorentz factor
21 this->Y_T = Y_T; // Set the Thomson Y parameter
22 nu_hat_m = compute_syn_freq(gamma_hat_m, B); // Compute corresponding synchrotron frequency for gamma_hat_m
23 nu_hat_c = compute_syn_freq(gamma_hat_c, B); // Compute corresponding synchrotron frequency for gamma_hat_c
24
25 if (nu_hat_m <= nu_hat_c) {
26 regime = 1; // fast IC cooling regime
27 } else {
28 regime = 2; // slow IC cooling regime
29 }
30}

46Real InverseComptonY::compute_val_at_gamma(Real gamma, Real p) const {
47 switch (regime) {
48 case 3:
49 return Y_T; // In regime 3, simply return Y_T
50 break;
51 case 1:
52 if (gamma <= gamma_hat_m) {
53 return Y_T; // For gamma below gamma_hat_m, no modification
54 } else if (gamma <= gamma_hat_c) {
55 return Y_T / std::sqrt(gamma / gamma_hat_m); // Intermediate regime scaling
56 } else
57 return Y_T * pow43(gamma_hat_c / gamma) / std::sqrt(gamma_hat_c / gamma_hat_m); // High gamma scaling
58
59 break;
60 case 2:
61 if (gamma <= gamma_hat_c) {
62 return Y_T; // For gamma below gamma_hat_c, no modification
63 } else if (gamma <= gamma_hat_m) {
64 return Y_T * fast_pow(gamma / gamma_hat_c, (p - 3) / 2); // Scaling in intermediate regime
65 } else
66 return Y_T * pow43(gamma_hat_m / gamma) *
67 fast_pow(gamma_hat_m / gamma_hat_c, (p - 3) / 2); // High gamma scaling
68
69 break;
70 default:
71 return 0;
72 break;
73 }
74}

76Real InverseComptonY::compute_val_at_nu(Real nu, Real p) const {
77 switch (regime) {
78 case 3:
79 return Y_T; // In regime 3, simply return Y_T
80 break;
81 case 1:
82 if (nu <= nu_hat_m) {
83 return Y_T; // For frequencies below nu_hat_m, no modification
84 } else if (nu <= nu_hat_c) {
85 return Y_T * std::sqrt(std::sqrt(nu_hat_m / nu)); // Intermediate frequency scaling
86 } else
87 return Y_T * pow23(nu_hat_c / nu) * std::sqrt(std::sqrt(nu_hat_m / nu)); // High frequency scaling
88
89 break;
90 case 2:
91 if (nu <= nu_hat_c) {
92 return Y_T; // For frequencies below nu_hat_c, no modification
93 } else if (nu <= nu_hat_m) {
94 return Y_T * fast_pow(nu / nu_hat_c, (p - 3) / 4); // Intermediate frequency scaling
95 } else
96 return Y_T * pow23(nu_hat_m / nu) *
97 fast_pow(nu_hat_m / nu_hat_c, (p - 3) / 4); // High frequency scaling
98
99 break;
100 default:
101 return 0;
102 break;
103 }

116void ICPhoton::fill_integration_grid(IntegratorGrid& grid, bool KN) noexcept {
117 // For each (nu0, gamma) pair, compute differential contributions and fill in I0
118
119 for (size_t i = 0; i < grid.num; ++i) {
120 Real dnu = grid.nu_edge(i + 1) - grid.nu_edge(i);
121 for (size_t j = 0; j < grid.num; ++j) {
122 Real dgamma = grid.gamma_edge(j + 1) - grid.gamma_edge(j);
123 Real dS = std::fabs(dnu * dgamma);
124 Real gamma_nu = grid.nu(i) * grid.gamma(j);
125 Real factor = 4 * gamma_nu * gamma_nu;
126
127 // Store cross-section based on KN regime
128 Real cross_section = KN ? compton_cross_section(gamma_nu) : con::sigmaT;
129
130 grid.I0(i, j) = grid.column_num_den(j) * grid.I_nu_syn(i) * dS / factor * cross_section;
131 }
132 }
133}

135void ICPhoton::integrate_IC_spectrum(IntegratorGrid const& grid) noexcept {
136 // Calculate output frequency range
137 Real gamma_min = grid.gamma.front();
138 Real gamma_max = grid.gamma.back();
139 Real nu0_min = grid.nu.front();
140 Real nu0_max = grid.nu.back();
141
142 Real nu_min = 4 * gamma_min * gamma_min * nu0_min * IC_x0 * 0.999; // 0.999 to avoid the nu_IC_>max(max_freq)
143 Real nu_max = 4 * gamma_max * gamma_max * nu0_max * IC_x0 * 0.999;
144
145 nu_IC_ = xt::logspace(std::log10(nu_min), std::log10(nu_max), spectrum_resol);
146
147 // Integrate over the grid to compute the final IC photon spectrum//TODO: Optimize this
148 for (size_t i = 0; i < grid.num; ++i) {
149 for (size_t j = 0; j < grid.num; ++j) {
150 Real max_freq = 4 * IC_x0 * grid.gamma(j) * grid.gamma(j) * grid.nu(i);
151 for (size_t k = 0; k < spectrum_resol; ++k) {
152 if (nu_IC_(k) <= max_freq) {
153 I_nu_IC_(k) += grid.I0(i, j);
154 } else {
155 break;
156 }
157 }
158 }
159 }
160
161 I_nu_IC_ *= nu_IC_;
162}

176/**
177 * <!-- ************************************************************************************** -->
178 * @internal
179 * @brief Computes the radiative efficiency parameter () given minimum electron Lorentz factors.
180 * @details If gamma_c is less than gamma_m, it returns 1; otherwise, it returns (gamma_c/gamma_m)^(2-p).
181 * @param gamma_m Minimum electron Lorentz factor
182 * @param gamma_c Cooling electron Lorentz factor
183 * @param p Electron distribution power-law index
184 * @return The radiative efficiency parameter
185 * <!-- ************************************************************************************** -->
186 */
187inline Real eta_rad(Real gamma_m, Real gamma_c, Real p) {
188 return gamma_c < gamma_m ? 1 : fast_pow(gamma_c / gamma_m, (2 - p));
189}

191/**
192 * <!-- ************************************************************************************** -->
193 * @internal
194 * @brief Computes the effective Compton Y parameter in the Thomson regime.
195 * @details Iteratively solves for Y until convergence using the relation:
196 * Y0 = (sqrt(1+4b) - 1)/2, where b = ( * eps_e / eps_B).
197 * The electron cooling parameters are updated during each iteration.
198 * @param B Magnetic field strength
199 * @param t_com Comoving time
200 * @param eps_e Electron energy fraction
201 * @param eps_B Magnetic energy fraction
202 * @param e Synchrotron electron properties
203 * @return The effective Thomson Y parameter
204 * <!-- ************************************************************************************** -->
205 */
206Real compute_Thomson_Y(Real B, Real t_com, Real eps_e, Real eps_B, SynElectrons const& e) {
207 Real eta_e = eta_rad(e.gamma_m, e.gamma_c, e.p);
208 Real b = eta_e * eps_e / eps_B;
209 Real Y0 = (std::sqrt(1 + 4 * b) - 1) / 2;
210 Real Y1 = 2 * Y0;
211 for (; std::fabs((Y1 - Y0) / Y0) > 1e-4;) {
212 Y1 = Y0;
213 Real gamma_c = compute_gamma_c(t_com, B, e.Ys, e.p);
214 eta_e = eta_rad(e.gamma_m, gamma_c, e.p);
215 b = eta_e * eps_e / eps_B;
216 Y0 = (std::sqrt(1 + 4 * b) - 1) / 2;
217 }
218 return Y0;
219}

225Real ICPhoton::compute_log2_I_nu(Real log2_nu) const noexcept {
226 Real dlog2_nu = log2_nu_IC_(1) - log2_nu_IC_(0);
227 size_t idx = static_cast<size_t>((log2_nu - log2_nu_IC_(0)) / dlog2_nu + 1);
228 if (idx < 1) {
229 idx = 1;
230 } else if (idx > log2_nu_IC_.size() - 1) {
231 idx = log2_nu_IC_.size() - 1;
232 }
233
234 Real slope = (log2_I_nu_(idx) - log2_I_nu_(idx - 1)) / dlog2_nu;
235
236 return log2_I_nu_(idx - 1) + slope * (log2_nu - log2_nu_IC_(idx - 1));
237}

239Real compton_cross_section(Real nu) {
240 Real x = con::h / (con::me * con::c2) * nu;
241 /*if (x <= 1) {
242 return con::sigmaT;
243 } else {
244 return 0;
245 }*/
246
247 if (x < 1e-2) {
248 return con::sigmaT * (1 - 2 * x);
249 } else if (x > 1e2) {
250 return 3. / 8 * con::sigmaT * (log(2 * x) + 0.5) / x;
251 } else {
252 Real log_term = log(1 + 2 * x);
253 Real term1 = 1 + 2 * x;
254 return 0.75 * con::sigmaT *
255 ((1 + x) / (x * x * x) * (2 * x * (1 + x) / term1 - log_term) + log_term / (2 * x) -
256 (1 + 3 * x) / (term1 * term1));
257 }
258}
