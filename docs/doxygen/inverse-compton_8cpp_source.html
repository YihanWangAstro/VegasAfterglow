14InverseComptonY::InverseComptonY(Real nu_m, Real nu_c, Real B, Real Y_T) noexcept {
15 gamma_hat_m = con::me * con::c2 / con::h / nu_m; // Compute minimum characteristic Lorentz factor
16 gamma_hat_c = con::me * con::c2 / con::h / nu_c; // Compute cooling characteristic Lorentz factor
17 this->Y_T = Y_T; // Set the Thomson Y parameter
18 nu_hat_m = compute_syn_freq(gamma_hat_m, B); // Compute the corresponding synchrotron frequency for gamma_hat_m
19 nu_hat_c = compute_syn_freq(gamma_hat_c, B); // Compute the corresponding synchrotron frequency for gamma_hat_c
20
21 if (nu_hat_m <= nu_hat_c) {
22 regime = 1; // fast IC cooling regime
23 } else {
24 regime = 2; // slow IC cooling regime
25 }
26}

42Real InverseComptonY::compute_val_at_gamma(Real gamma, Real p) const {
43 switch (regime) {
44 case 3:
45 return Y_T; // In regime 3, simply return Y_T
46 break;
47 case 1:
48 if (gamma <= gamma_hat_m) {
49 return Y_T; // For gamma below gamma_hat_m, no modification
50 } else if (gamma <= gamma_hat_c) {
51 return Y_T / std::sqrt(gamma / gamma_hat_m); // Intermediate regime scaling
52 } else {
53 return Y_T * pow43(gamma_hat_c / gamma) * std::sqrt(gamma_hat_m / gamma_hat_c); // High gamma scaling
54 }
55 break;
56 case 2:
57 if (gamma <= gamma_hat_c) {
58 return Y_T; // For gamma below gamma_hat_c, no modification
59 } else if (gamma <= gamma_hat_m) {
60 return Y_T * fast_pow(gamma / gamma_hat_c, (p - 3) / 2); // Scaling in intermediate regime
61 } else {
62 return Y_T * pow43(gamma_hat_m / gamma) *
63 fast_pow(gamma_hat_m / gamma_hat_c, (p - 3) / 2); // High gamma scaling
64 }
65 break;
66 default:
67 return 0;
68 break;
69 }

78 if (nu <= nu_hat_m) {
79 return Y_T; // For frequencies below nu_hat_m, no modification

114Real compton_cross_section(Real nu) {
115 const Real x = con::h / (con::me * con::c2) * nu;
116 /*if (x <= 1) {
117 return con::sigmaT;
118 } else {
119 return 0;
120 }*/
121
122 if (x < 1e-2) {
123 return con::sigmaT * (1 - 2 * x);
124 } else if (x > 1e2) {
125 return 3. / 8 * con::sigmaT * (log(2 * x) + 0.5) / x;
126 } else {
127 const Real l = std::log1p(2.0 * x); // log(1+2x)
128 const Real invx = 1.0 / x;
129 const Real invx2 = invx * invx;
130 const Real term1 = 1.0 + 2.0 * x;
131 const Real invt1 = 1.0 / term1;
132 const Real invt1_2 = invt1 * invt1;
133
134 // ((1+x)/x^3) * (2x(1+x)/(1+2x) - log(1+2x)) + log(1+2x)/(2x) - (1+3x)/(1+2x)^2
135 const Real a = (1.0 + x) * invx2 * invx; // (1+x)/x^3
136 const Real b = 2.0 * x * (1.0 + x) * invt1 - l; // bracket
137 const Real c = 0.5 * l * invx; // log_term/(2x)
138 const Real d = (1.0 + 3.0 * x) * invt1_2; // (1+3x)/(1+2x)^2
139
140 return 0.75 * con::sigmaT * (a * b + c - d);
141 }
