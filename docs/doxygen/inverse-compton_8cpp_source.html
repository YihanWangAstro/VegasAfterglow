17/**
18 * <!-- ************************************************************************************** -->
19 * @internal
20 * @brief Computes the radiative efficiency parameter (ηₑ) given minimum electron Lorentz factors.
21 * @details If gamma_c is less than gamma_m, it returns 1; otherwise, it returns (gamma_c/gamma_m)^(2-p).
22 * @param gamma_m Minimum electron Lorentz factor
23 * @param gamma_c Cooling electron Lorentz factor
24 * @param p Electron distribution power-law index
25 * @return The radiative efficiency parameter
26 * <!-- ************************************************************************************** -->
27 */
28inline Real eta_rad(Real gamma_m, Real gamma_c, Real p) {
29 return gamma_c < gamma_m ? 1 : std::pow(gamma_c / gamma_m, (2 - p));
30}

32/**
33 * <!-- ************************************************************************************** -->
34 * @internal
35 * @brief Computes the effective Compton Y parameter in the Thomson regime.
36 * @details Iteratively solves for Y until convergence using the relation:
37 * Y0 = (sqrt(1+4b) - 1)/2, where b = (ηₑ * eps_e / eps_B).
38 * The electron cooling parameters are updated during each iteration.
39 * @param B Magnetic field strength
40 * @param t_com Comoving time
41 * @param eps_e Electron energy fraction
42 * @param eps_B Magnetic energy fraction
43 * @param e Synchrotron electron properties
44 * @return The effective Thomson Y parameter
45 * <!-- ************************************************************************************** -->
46 */
47Real effectiveYThomson(Real B, Real t_com, Real eps_e, Real eps_B, SynElectrons const& e) {
48 Real eta_e = eta_rad(e.gamma_m, e.gamma_c, e.p);
49 Real b = eta_e * eps_e / eps_B;
50 Real Y0 = (std::sqrt(1 + 4 * b) - 1) / 2;
51 Real Y1 = 2 * Y0;
52 for (; std::fabs((Y1 - Y0) / Y0) > 1e-5;) {
53 Y1 = Y0;
54 Real gamma_c = compute_gamma_c(t_com, B, e.Ys, e.p);
55 eta_e = eta_rad(e.gamma_m, gamma_c, e.p);
56 b = eta_e * eps_e / eps_B;
57 Y0 = (std::sqrt(1 + 4 * b) - 1) / 2;
58 }
59 return Y0;
60}
