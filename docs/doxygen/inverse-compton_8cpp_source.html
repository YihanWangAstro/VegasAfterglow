19InverseComptonY::InverseComptonY(Real gamma_m, Real gamma_c, Real B, Real Y_T, Real eps_e_on_eps_B) noexcept {
20 constexpr Real B_QED = 4 * con::pi * con::me * con::me * con::c2 * con::c / (3 * con::e * con::h);
21 const Real nu_m = compute_syn_freq(gamma_m, B);
22 const Real nu_c = compute_syn_freq(gamma_c, B);
23 gamma_m_hat = con::me * con::c2 / con::h / nu_m;
24 gamma_c_hat = con::me * con::c2 / con::h / nu_c;
25
26 this->Y_T = Y_T; // Set the Thomson Y parameter
27 nu_m_hat = compute_syn_freq(gamma_m_hat, B);
28 nu_c_hat = compute_syn_freq(gamma_c_hat, B);
29 if (gamma_m <= gamma_c) { // slow IC cooling regime
30 regime = 1;
31 } else { //fast IC cooling regime
32 if (gamma_m <= gamma_m_hat) { //weak KN effects
33 regime = 2;
34 } else { //strong KN effects
35 gamma_self = std::cbrt(B_QED / B);
36 nu_self = compute_syn_freq(gamma_self, B);
37
38 gamma0 = std::sqrt(eps_e_on_eps_B * gamma_m * gamma_m_hat);
39 nu0 = compute_syn_freq(gamma0, B);
40
41 if (gamma0 > gamma_m) {
42 regime = 3;
43 //.../
44 } else if (gamma0 > gamma_self) {
45 regime = 4;
46 //../
47 } else {
48 regime = 5;
49 }
50 }
51 }
52}

72Real InverseComptonY::evaluate_at_gamma(Real gamma, Real p) const {
73 switch (regime) {
74 case 0:
75 return Y_T; // Thomas only, no KN effects
76 break;
77 case 1:
78 if (gamma <= gamma_c_hat) {
79 return Y_T; // For gamma below gamma_hat_c, no modification
80 } else if (gamma <= gamma_m_hat) {
81 return Y_T * fast_pow(gamma / gamma_c_hat, (p - 3) / 2); // Scaling in intermediate regime
82 } else {
83 return Y_T * pow43(gamma_m_hat / gamma) *
84 fast_pow(gamma_m_hat / gamma_c_hat, (p - 3) / 2); // High gamma scaling
85 }
86 break;
87 case 2:
88 if (gamma <= gamma_m_hat) {
89 return Y_T; // For gamma below gamma_hat_m, no modification
90 } else if (gamma <= gamma_c_hat) {
91 return Y_T / std::sqrt(gamma / gamma_m_hat); // Intermediate regime scaling
92 } else {
93 return Y_T * pow43(gamma_c_hat / gamma) * std::sqrt(gamma_m_hat / gamma_c_hat); // High gamma scaling
94 }
95 break;
96
97 default:
98 return 0;
99 break;
100 }
101}

103Real InverseComptonY::evaluate_at_nu(Real nu, Real p) const {
104 switch (regime) {
105 case 0:
106 return Y_T; // Thomas only, no KN effects
107 break;
108 case 1:
109 if (nu <= nu_c_hat) {
110 return Y_T; // For frequencies below nu_hat_c, no modification
111 } else if (nu <= nu_m_hat) {
112 return Y_T * fast_pow(nu / nu_c_hat, (p - 3) / 4); // Intermediate frequency scaling
113 } else {
114 return Y_T * pow23(nu_m_hat / nu) *
115 fast_pow(nu_m_hat / nu_c_hat, (p - 3) / 4); // High-frequency scaling
116 }
117 break;
118 case 2:
119 if (nu <= nu_m_hat) {
120 return Y_T; // For frequencies below nu_hat_m, no modification
121 } else if (nu <= nu_c_hat) {
122 return Y_T * std::sqrt(std::sqrt(nu_m_hat / nu)); // Intermediate frequency scaling
123 } else {
124 return Y_T * pow23(nu_c_hat / nu) * std::sqrt(std::sqrt(nu_m_hat / nu_c_hat)); // High-frequency scaling
125 }
126 break;
127
128 default:
129 return 0;
130 break;
131 }
132}

138void update_gamma_c_Thomson(Real& gamma_c, InverseComptonY& Ys, RadParams const& rad, Real B, Real t_com,
139 Real gamma_m) {
140 Real eta_e = eta_rad(gamma_m, gamma_c, rad.p);
141 Real b = eta_e * rad.eps_e / rad.eps_B;
142 Real Y_T = (std::sqrt(1 + 4 * b) - 1) / 2;
143
144 Real gamma_c_new = compute_gamma_c(t_com, B, Y_T);
145 while (std::fabs((gamma_c_new - gamma_c) / gamma_c) > 1e-3) {
146 gamma_c = gamma_c_new;
147 eta_e = eta_rad(gamma_m, gamma_c, rad.p);
148 b = eta_e * rad.eps_e / rad.eps_B;
149 Y_T = (std::sqrt(1 + 4 * b) - 1) / 2;
150 gamma_c_new = compute_gamma_c(t_com, B, Y_T);
151 }
152 gamma_c = gamma_c_new;
153 Ys = InverseComptonY(Y_T);
154}

176void update_gamma_M(Real& gamma_M, InverseComptonY const& Ys, Real p, Real B) {
177 if (B == 0) {
178 gamma_M = std::numeric_limits<Real>::infinity();
179 return;
180 }
181
182 Real Y_M = Ys.evaluate_at_gamma(gamma_M, p);
183 Real gamma_M_new = compute_syn_gamma_M(B, Y_M, p);
184
185 while (std::fabs((gamma_M - gamma_M_new) / gamma_M_new) > 1e-3) {
186 gamma_M = gamma_M_new;
187 Y_M = Ys.evaluate_at_gamma(gamma_M, p);
188 gamma_M_new = compute_syn_gamma_M(B, Y_M, p);
189 }

196Real compton_cross_section(Real nu) {
197 const Real x = con::h / (con::me * con::c2) * nu;
198 /*if (x <= 1) {
199 return con::sigmaT;
200 } else {
201 return 0;
202 }*/
203
204 if (x < 1e-2) {
205 return con::sigmaT * (1 - 2 * x);
206 } else if (x > 1e2) {
207 return 3. / 8 * con::sigmaT * (log(2 * x) + 0.5) / x;
208 } else {
209 const Real l = std::log1p(2.0 * x); // log(1+2x)
210 const Real invx = 1.0 / x;
211 const Real invx2 = invx * invx;
212 const Real term1 = 1.0 + 2.0 * x;
213 const Real invt1 = 1.0 / term1;
214 const Real invt1_2 = invt1 * invt1;
215
216 // ((1+x)/x^3) * (2x(1+x)/(1+2x) - log(1+2x)) + log(1+2x)/(2x) - (1+3x)/(1+2x)^2
217 const Real a = (1.0 + x) * invx2 * invx; // (1+x)/x^3
218 const Real b = 2.0 * x * (1.0 + x) * invt1 - l; // bracket
219 const Real c = 0.5 * l * invx; // log_term/(2x)
220 const Real d = (1.0 + 3.0 * x) * invt1_2; // (1+3x)/(1+2x)^2
221
222 return 0.75 * con::sigmaT * (a * b + c - d);
223 }
