386template <typename ICPhotonGrid>
387void MultiBandModel::generate_photons(Params const& param, double t_min, double t_max, Observer& obs,
388 SynPhotonGrid& fwd_photons, SynPhotonGrid& rvs_photons,
389 ICPhotonGrid& fwd_IC_photons, ICPhotonGrid& rvs_IC_photons) {
390 Real theta_v = param.theta_v;
391 Real theta_w = param.theta_w;
392 RadParams rad;
393 rad.p = param.p;
394 rad.eps_e = param.eps_e;
395 rad.eps_B = param.eps_B;
396 rad.xi_e = param.xi_e;
397
398 Real lumi_dist = config.lumi_dist * unit::cm;
399 Real z = config.z;
400
401 Medium medium = select_medium(param);
402 Ejecta jet = select_jet(param);
403
404 Real t_resol = config.t_resol;
405 Real theta_resol = config.theta_resol;
406 Real phi_resol = config.phi_resol;
407
408 Array t_eval = xt::linspace<Real>(t_min, t_max, 5);
409
410 auto coord = auto_grid(jet, t_eval, theta_w, theta_v, z, phi_resol, theta_resol, t_resol);
411
412 if (config.rvs_shock == false) {
413 auto shock = generate_fwd_shock(coord, medium, jet, rad, config.rtol);
414
415 obs.observe(coord, shock, lumi_dist, z);
416
417 auto electrons = generate_syn_electrons(shock);
418 fwd_photons = generate_syn_photons(shock, electrons);
419
420 if (config.ssc_cooling) {
421 if (config.kn) {
422 KN_cooling(electrons, fwd_photons, shock);
423 } else {
424 Thomson_cooling(electrons, fwd_photons, shock);
425 }
426 }
427
428 if (config.fwd_ssc) {
429 fwd_IC_photons = generate_IC_photons(electrons, fwd_photons, config.kn);
430 }
431
432 } else {
433 RadParams rad_rvs;
434
435 rad_rvs.p = param.p_r;
436 rad_rvs.eps_e = param.eps_e_r;
437 rad_rvs.eps_B = param.eps_B_r;
438 rad_rvs.xi_e = param.xi_e_r;
439
440 auto [f_shock, r_shock] = generate_shock_pair(coord, medium, jet, rad, rad_rvs, config.rtol);
441
442 obs.observe(coord, f_shock, lumi_dist, z);
443
444 auto f_electrons = generate_syn_electrons(f_shock);
445 fwd_photons = generate_syn_photons(f_shock, f_electrons);
446
447 auto r_electrons = generate_syn_electrons(r_shock);
448 rvs_photons = generate_syn_photons(r_shock, r_electrons);
449
450 if (config.ssc_cooling) {
451 if (config.kn) {
452 KN_cooling(f_electrons, fwd_photons, f_shock);
453 KN_cooling(r_electrons, rvs_photons, r_shock);
454 } else {
455 Thomson_cooling(f_electrons, fwd_photons, f_shock);
456 Thomson_cooling(r_electrons, rvs_photons, r_shock);
457 }
458 }
459
460 if (config.fwd_ssc) {
461 fwd_IC_photons = generate_IC_photons(f_electrons, fwd_photons, config.kn);
462 }
463
464 if (config.rvs_ssc) {
465 rvs_IC_photons = generate_IC_photons(r_electrons, rvs_photons, config.kn);
466 }
467 }
