107template <typename View>
108void MultiBandModel::build_system(Params const& param, Array const& t_eval, Array const& nu_eval, View& F_nu) {
109 Real eps_iso = param.E_iso * unit::erg / (4 * con::pi);
110 Real Gamma0 = param.Gamma0;
111 Real theta_c = param.theta_c;
112 Real theta_v = param.theta_v;
113 Real theta_w = param.theta_w;
114 Real eps_iso_w = param.E_iso_w * unit::erg / (4 * con::pi);
115 Real Gamma0_w = param.Gamma0_w;
116 RadParams rad;
117
118 rad.p = param.p;
119 rad.eps_e = param.eps_e;
120 rad.eps_B = param.eps_B;
121 rad.xi_e = param.xi_e;
122
123 Real lumi_dist = config.lumi_dist * unit::cm;
124 Real z = config.z;
125
126 // create model
127 Medium medium;
128 if (config.medium == "ism") {
129 medium.rho = evn::ISM(param.n_ism / unit::cm3);
130 } else if (config.medium == "wind") {
131 medium.rho = evn::wind(param.A_star, param.n_ism / unit::cm3, param.n0 / unit::cm3);
132 } else {
133 std::cerr << "Error: Unknown medium type" << std::endl;
134 }
135
136 Ejecta jet;
137 jet.T0 = param.duration * unit::sec;
138 if (config.jet == "tophat") {
139 jet.eps_k = math::tophat(theta_c, eps_iso);
140 jet.Gamma0 = math::tophat(theta_c, Gamma0);
141 } else if (config.jet == "gaussian") {
142 jet.eps_k = math::gaussian(theta_c, eps_iso);
143 jet.Gamma0 = math::gaussian_plus_one(theta_c, Gamma0 - 1);
144 } else if (config.jet == "powerlaw") {
145 jet.eps_k = math::powerlaw(theta_c, eps_iso, param.k_e);
146 jet.Gamma0 = math::powerlaw_plus_one(theta_c, Gamma0 - 1, param.k_g);
147 } else if (config.jet == "twocomponent") {
148 jet.eps_k = math::two_component(theta_c, theta_w, eps_iso, eps_iso_w);
149 jet.Gamma0 = math::two_component_plus_one(theta_c, theta_w, Gamma0 - 1, Gamma0_w - 1);
150 } else if (config.jet == "steppowerlaw") {
151 jet.eps_k = math::step_powerlaw(theta_c, eps_iso, eps_iso_w, param.k_e);
152 jet.Gamma0 = math::step_powerlaw_plus_one(theta_c, Gamma0 - 1, Gamma0_w - 1, param.k_g);
153 } else {
154 std::cerr << "Error: Unknown jet type" << std::endl;
155 }
156
157 if (config.magnetar == true) {
158 jet.deps_dt =
159 math::magnetar_injection(param.t0 * unit::sec, param.q, param.L0 * unit::erg / unit::sec, theta_c);
160 }
161
162 Real t_resol = config.t_resol;
163 Real theta_resol = config.theta_resol;
164 Real phi_resol = config.phi_resol;
165
166 auto coord = auto_grid(jet, t_eval, theta_w, theta_v, z, phi_resol, theta_resol, t_resol);
167
168 if (config.rvs_shock == false) {
169 auto shock = generate_fwd_shock(coord, medium, jet, rad, config.rtol);
170
171 Observer obs;
172 // obs.observe_at(t_eval, coord, shock, lumi_dist, z);
173 obs.observe(coord, shock, lumi_dist, z);
174
175 auto electrons = generate_syn_electrons(shock);
176 auto photons = generate_syn_photons(shock, electrons);
177
178 if (config.IC_cooling) {
179 if (config.KN) {
180 KN_cooling(electrons, photons, shock);
181 } else {
182 Thomson_cooling(electrons, photons, shock);
183 }
184 }
185
186 F_nu = obs.specific_flux_series(t_eval, nu_eval, photons);
187
188 if (config.fwd_SSC) {
189 auto IC_photon = generate_IC_photons(electrons, photons, config.KN);
190 auto IC_F_nu = obs.specific_flux_series(t_eval, nu_eval, IC_photon);
191 F_nu += IC_F_nu;
192 }
193
194 } else {
195 RadParams rad_rvs;
196
197 rad_rvs.p = param.p_r;
198 rad_rvs.eps_e = param.eps_e_r;
199 rad_rvs.eps_B = param.eps_B_r;
200 rad_rvs.xi_e = param.xi_e_r;
201
202 auto [f_shock, r_shock] = generate_shock_pair(coord, medium, jet, rad, rad_rvs, config.rtol);
203
204 Observer obs;
205 // obs.observe_at(t_eval, coord, shock, lumi_dist, z);
206 obs.observe(coord, f_shock, lumi_dist, z);
207
208 auto f_electrons = generate_syn_electrons(f_shock);
209 auto f_photons = generate_syn_photons(f_shock, f_electrons);
210
211 auto r_electrons = generate_syn_electrons(r_shock);
212 auto r_photons = generate_syn_photons(r_shock, r_electrons);
213
214 if (config.IC_cooling) {
215 if (config.KN) {
216 KN_cooling(f_electrons, f_photons, f_shock);
217 KN_cooling(r_electrons, r_photons, r_shock);
218 } else {
219 Thomson_cooling(f_electrons, f_photons, f_shock);
220 Thomson_cooling(r_electrons, r_photons, r_shock);
221 }
222 }
223
224 F_nu = obs.specific_flux_series(t_eval, nu_eval, f_photons);
225 auto F_nu_rvs = obs.specific_flux_series(t_eval, nu_eval, r_photons);
226 F_nu += F_nu_rvs;
227
228 if (config.fwd_SSC) {
229 auto IC_photon_f = generate_IC_photons(f_electrons, f_photons, config.KN);
230 auto IC_F_nu_fwd = obs.specific_flux_series(t_eval, nu_eval, IC_photon_f);
231 F_nu += IC_F_nu_fwd;
232 }
233
234 if (config.rvs_SSC) {
235 auto IC_photon_r = generate_IC_photons(r_electrons, r_photons, config.KN);
236 auto IC_F_nu_rvs = obs.specific_flux_series(t_eval, nu_eval, IC_photon_r);
237 F_nu += IC_F_nu_rvs;
238 }
239 }
