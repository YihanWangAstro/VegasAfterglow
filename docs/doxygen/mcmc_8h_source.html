24 * @struct FluxData
25 * @brief Container for observational flux data and model predictions for chi-squared analysis.
26 * @details This structure organizes time-series flux measurements with their uncertainties,

41 * <!-- ************************************************************************************** -->
42 * @brief Calculate chi-squared goodness-of-fit statistic for this data set.
43 * @details Computes the weighted chi-squared statistic comparing observed and model flux
44 * densities, accounting for observational uncertainties and optional weights.
45 * This metric quantifies how well the theoretical model reproduces the observations.
46 * @return double Chi-squared value for goodness-of-fit assessment
47 * <!-- ************************************************************************************** -->
48 */
49 [[nodiscard]] double estimate_chi2() const;

176 * @struct Params
177 * @brief Complete parameter set for gamma-ray burst afterglow model configuration.

354/**
355 * <!-- ************************************************************************************** -->
356 * @struct BatchWorkspace
357 * @brief Thread-local workspace for batch chi-squared computation.
358 * @details Holds pre-allocated mutable arrays for model outputs, avoiding repeated allocations
359 * in the hot loop. Each thread gets its own workspace while sharing read-only
360 * observation data via const references.
361 * <!-- ************************************************************************************** -->
362 */
363struct BatchWorkspace {
364 Array model_fluxes; ///< Pre-allocated array for model flux predictions
365 std::vector<Array> flux_data_models; ///< Pre-allocated arrays for flux_data Fv_model
366 bool initialized{false}; ///< Whether workspace has been sized
367
368 /// Initialize workspace arrays to match observation data dimensions
369 void initialize(size_t tuple_size, std::vector<size_t> const& flux_data_sizes) {
370 if (tuple_size > 0) {
371 model_fluxes = Array::from_shape({tuple_size});
372 }
373 flux_data_models.resize(flux_data_sizes.size());
374 for (size_t i = 0; i < flux_data_sizes.size(); ++i) {
375 flux_data_models[i] = Array::from_shape({flux_data_sizes[i]});
376 }
377 initialized = true;
378 }

536template <typename ICPhotonGrid>
537void MultiBandModel::generate_photons(Params const& param, double t_min, double t_max, Observer& obs,
538 SynPhotonGrid& fwd_photons, SynPhotonGrid& rvs_photons,
539 ICPhotonGrid& fwd_IC_photons, ICPhotonGrid& rvs_IC_photons) {
540 RadParams rad{param.eps_e, param.eps_B, param.p, param.xi_e};
541
542 Real lumi_dist = config.lumi_dist * unit::cm;
543 Real z = config.z;
544
545 auto jet = select_jet(param);
546 auto medium = select_medium(param);
547
548 Array t_eval = xt::linspace<Real>(t_min, t_max, 5);
549 GridConfig gcfg{
550 param.theta_w, param.theta_v, z, config.phi_resol, config.theta_resol, config.t_resol, true, 32, 0.4};
551
552 if (!config.rvs_shock) {
553 auto [coord, shock] = solve_fwd_shock(jet, medium, t_eval, gcfg, rad, config.rtol);
554
555 obs.observe(coord, shock, lumi_dist, z);
556
557 auto electrons = generate_syn_electrons(shock);
558 fwd_photons = generate_syn_photons(shock, electrons);
559
560 if (config.ssc_cooling) {
561 if (config.kn) {
562 KN_cooling(electrons, fwd_photons, shock);
563 } else {
564 Thomson_cooling(electrons, fwd_photons, shock);
565 }
566 }
567
568 if (config.fwd_ssc) {
569 fwd_IC_photons = generate_IC_photons(electrons, fwd_photons, config.kn, shock);
570 }
571
572 } else {
573 RadParams rad_rvs{param.eps_e_r, param.eps_B_r, param.p_r, param.xi_e_r};
574
575 gcfg.min_t_pts = 36;
576 gcfg.min_t_frac = 0.5;
577 auto [coord, f_shock, r_shock] = solve_shock_pair(jet, medium, t_eval, gcfg, rad, rad_rvs, config.rtol);
578
579 obs.observe(coord, f_shock, lumi_dist, z);
580
581 auto f_electrons = generate_syn_electrons(f_shock);
582 fwd_photons = generate_syn_photons(f_shock, f_electrons);
583
584 auto r_electrons = generate_syn_electrons(r_shock);
585 rvs_photons = generate_syn_photons(r_shock, r_electrons);
586
587 if (config.ssc_cooling) {
588 if (config.kn) {
