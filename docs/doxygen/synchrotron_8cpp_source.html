18Real SynElectrons::compute_N_gamma(Real gamma) const {
19 Real exp_cut = gamma < gamma_M ? 1. : fast_exp2(-gamma / gamma_M);
20
21 if (gamma <= gamma_c) { // Below cooling Lorentz factor: direct scaling
22 return exp_cut * N_e * compute_spectrum(gamma);
23 } else {
24 return exp_cut * N_e * compute_spectrum(gamma) * (1 + Y_c) /
25 (1 + InverseComptonY::compute_Y_tilt_at_gamma(Ys, gamma, p));
26 }
27}

41 * @internal

43 * @param a First value
44 * @param b Middle value
45 * @param c Last value
46 * @return True if a ≤ b ≤ c, false otherwise
47 * <!-- ************************************************************************************** -->

51/**
52 * <!-- ************************************************************************************** -->
53 * @internal
54 * @brief Determines the spectral regime (1-6) based on the ordering of characteristic Lorentz factors.
55 * @details Classifies the regime based on the ordering of absorption (a), cooling (c),
56 * and minimum (m) Lorentz factors.
57 * @param a Absorption Lorentz factor
58 * @param c Cooling Lorentz factor
59 * @param m Minimum Lorentz factor
60 * @return Regime number (1-6) or 0 if no valid regime is found
61 * <!-- ************************************************************************************** -->
62 */
63size_t determine_regime(Real a, Real c, Real m) {
64 if (order(a, m, c)) {
65 return 1;
66 } else if (order(m, a, c)) {
67 return 2;
68 } else if (order(a, c, m)) {
69 return 3;
70 } else if (order(c, a, m)) {
71 return 4;
72 } else if (order(m, c, a)) {
73 return 5;
74 } else if (order(c, m, a)) {
75 return 6;
76 } else
77 return 0;
78}

80Real SynElectrons::compute_spectrum(Real gamma) const {
81 switch (regime) {
82 case 1: // same as case 2
83 case 2:
84 if (gamma <= gamma_m) {
85 return 0; // Below minimum Lorentz factor, spectrum is zero
86 } else if (gamma <= gamma_c) {
87 return (p - 1) * fast_pow(gamma / gamma_m, -p) /
88 gamma_m; // Power-law spectrum between gamma_m and gamma_c
89 } else
90 return (p - 1) * fast_pow(gamma / gamma_m, -p) * gamma_c / (gamma * gamma_m);
91 // Above cooling Lorentz factor: exponential cutoff applied
92
93 break;
94 case 3:
95 if (gamma <= gamma_c) {
96 return 0; // Below cooling Lorentz factor, spectrum is zero
97 } else if (gamma <= gamma_m) {
98 return gamma_c / (gamma * gamma); // Intermediate regime scaling
99 } else
100 return gamma_c / (gamma * gamma_m) * fast_pow(gamma / gamma_m, -p);
101 // Above minimum Lorentz factor: power-law with exponential cutoff
102
103 break;
104 case 4: // Gao, Lei, Wu and Zhang 2013 Eq 18
105 if (gamma <= gamma_a) {
106 return 3 * gamma * gamma / (gamma_a * gamma_a * gamma_a); // Rising part of the spectrum
107 } else if (gamma <= gamma_m) {
108 return gamma_c / (gamma * gamma); // Transition region
109 } else
110 return gamma_c / (gamma * gamma_m) * fast_pow(gamma / gamma_m, -p);
111 // High energy tail with exponential cutoff
112
113 break;
114 case 5: // Gao, Lei, Wu and Zhang 2013 Eq 19
115 if (gamma <= gamma_a) {
116 return 3 * gamma * gamma / (gamma_a * gamma_a * gamma_a); // Rising part of the spectrum
117 } else
118 return (p - 1) * gamma_c / (gamma * gamma_m) * fast_pow(gamma / gamma_m, -p);
119 // Power-law decay with exponential cutoff
120
121 break;
122 case 6: // Gao, Lei, Wu and Zhang 2013 Eq 20
123 if (gamma <= gamma_a) {
124 return 3 * gamma * gamma / (gamma_a * gamma_a * gamma_a); // Rising part of the spectrum
125 } else
126 return fast_pow(gamma_m, p - 1) * gamma_c * fast_pow(gamma, -(p + 1));
127 // Steeper decay in this regime
128
129 break;
130 default:
131 return 0;
132 }

135Real SynPhotons::compute_P_nu(Real nu) const {
136 Real exp_cut = nu < nu_M ? 1. : fast_exp2(-nu / nu_M);
137
138 if (nu <= nu_c) { // Below cooling frequency, simple scaling
139 return exp_cut * P_nu_max * compute_spectrum(nu);
140 } else {
141 return exp_cut * P_nu_max * compute_spectrum(nu) * (1 + elec->Y_c) /
142 (1 + InverseComptonY::compute_Y_tilt_at_nu(elec->Ys, nu, elec->p));
143 }

146Real SynPhotons::compute_log2_P_nu(Real log2_nu) const {
147 Real exp_cut = log2_nu < log2_nu_M ? 0. : fast_exp2(log2_nu) / nu_M;
148
149 if (log2_nu <= log2_nu_c) { // Below cooling frequency, simple scaling
150 return log2_P_nu_max + compute_log2_spectrum(log2_nu) - exp_cut;
151 } else {
152 Real cooling_factor =
153 (1 + elec->Y_c) / (1 + InverseComptonY::compute_Y_tilt_at_nu(elec->Ys, std::exp2(log2_nu), elec->p));
154 return log2_P_nu_max + compute_log2_spectrum(log2_nu) + fast_log2(cooling_factor) - exp_cut;
155 }

158void SynPhotons::update_constant() {
159 // Update constants based on spectral parameters
160 Real p = elec->p;
161 if (elec->regime == 1) {
162 // a_m_1_3 = std::cbrt(nu_a / nu_m); // (nu_a / nu_m)^(1/3)
163 // c_m_mpa1_2 = fastPow(nu_c / nu_m, (-p + 1) / 2); // (nu_c / nu_m)^((-p+1)/2)
164 C1_ = std::cbrt(nu_a / nu_m);
165 C2_ = fast_pow(nu_c / nu_m, (-p + 1) / 2);
166
167 log2_C1_ = fast_log2(nu_a / nu_m) / 3 - 2 * fast_log2(nu_a);
168 log2_C2_ = -fast_log2(nu_m) / 3;
169 log2_C3_ = (p - 1) / 2 * fast_log2(nu_m);
170 log2_C4_ = (p - 1) / 2 * fast_log2(nu_m / nu_c) + p / 2 * fast_log2(nu_c);
171 } else if (elec->regime == 2) {
172 // m_a_pa4_2 = fastPow(nu_m / nu_a, (p + 4) / 2); // (nu_m / nu_a)^((p+4)/2)
173 // a_m_mpa1_2 = fastPow(nu_a / nu_m, (-p + 1) / 2); // (nu_a / nu_m)^((-p+1)/2)
174 // c_m_mpa1_2 = fastPow(nu_c / nu_m, (-p + 1) / 2); // (nu_c / nu_m)^((-p+1)/2)
175 C1_ = fast_pow(nu_m / nu_a, (p + 4) / 2);
176 C2_ = fast_pow(nu_a / nu_m, (-p + 1) / 2);
177 C3_ = fast_pow(nu_c / nu_m, (-p + 1) / 2);
178
179 log2_C1_ = (p + 4) / 2 * fast_log2(nu_m / nu_a) - 2 * fast_log2(nu_m);
180 log2_C2_ = (p - 1) / 2 * fast_log2(nu_m / nu_a) - 2.5 * fast_log2(nu_a);
181 log2_C3_ = (p - 1) / 2 * fast_log2(nu_m);
182 log2_C4_ = (p - 1) / 2 * fast_log2(nu_m / nu_c) + p / 2 * fast_log2(nu_c);
183 } else if (elec->regime == 3) {
184 // a_c_1_3 = std::cbrt(nu_a / nu_c); // (nu_a / nu_c)^(1/3)
185 // c_m_1_2 = std::sqrt(nu_c / nu_m); // (nu_c / nu_m)^(1/2)
186 C1_ = std::cbrt(nu_a / nu_c);
187 C2_ = std::sqrt(nu_c / nu_m);
188
189 log2_C1_ = fast_log2(nu_a / nu_c) / 3 - 2 * fast_log2(nu_a);
190 log2_C2_ = -fast_log2(nu_c) / 3;
191 log2_C3_ = fast_log2(nu_c) / 2;
192 log2_C4_ = fast_log2(nu_c / nu_m) / 2 + p / 2 * fast_log2(nu_m);
193 } else if (elec->regime == 4) {
194 // a_m_1_2 = std::sqrt(nu_a / nu_m); // (nu_a / nu_m)^(1/2)
195 // R4 = std::sqrt(nu_c / nu_a) / 3; // (nu_c / nu_a)^(1/2) / 3; // R4: scaling factor for regime 4
196 C1_ = std::sqrt(nu_a / nu_m);
197 C2_ = std::sqrt(nu_c / nu_a) / 3;
198
199 log2_C1_ = -2 * fast_log2(nu_a);
200 log2_C2_ = fast_log2(C2_) + fast_log2(nu_a) / 2;
201 log2_C3_ = fast_log2(C2_) + fast_log2(nu_a / nu_m) / 2 + p / 2 * fast_log2(nu_m);
202 } else if (elec->regime == 5 || elec->regime == 6) {
203 // R4 = std::sqrt(nu_c / nu_a) / 3; // (nu_c / nu_a)^(1/2) / 3; // R4: scaling factor for
204 // regime 4 R6 = R4 * fastPow(nu_m / nu_a, (p - 1) / 2); // R6: scaling factor for regime 6
205 C1_ = std::sqrt(nu_c / nu_a) / 3;
206 C2_ = C1_ * fast_pow(nu_m / nu_a, (p - 1) / 2);
207
208 log2_C1_ = -2 * fast_log2(nu_a);
209 log2_C2_ = fast_log2((p - 1) * C2_) + p / 2 * fast_log2(nu_a);
210 log2_C3_ = fast_log2(C2_) + p / 2 * fast_log2(nu_a);
211 }
212 // R5 = (p - 1) * R6; // R5 scales R6 (commented out)
213}

215Real SynPhotons::compute_spectrum(Real nu) const {
216 Real p = elec->p;
217 switch (elec->regime) {
218 case 1:
219 if (nu <= nu_a) {
220 return C1_ * (nu / nu_a) * (nu / nu_a);
221 }
222 if (nu <= nu_m) {
223 return std::cbrt(nu / nu_m);
224 }
225 if (nu <= nu_c) {
226 return fast_pow(nu / nu_m, -(p - 1) / 2);
227 }
228
229 return C2_ * fast_pow(nu / nu_c, -p / 2);
230
231 break;
232 case 2:
233 if (nu <= nu_m) {
234 return C1_ * (nu / nu_m) * (nu / nu_m);
235 }
236 if (nu <= nu_a) {
237 return C2_ * pow52(nu / nu_a); // Using pow52 for (nu / nu_a)^(5/2)
238 }
239 if (nu <= nu_c) {
240 return fast_pow(nu / nu_m, -(p - 1) / 2);
241 }
242
243 return C3_ * fast_pow(nu / nu_c, -p / 2);
244
245 break;
246 case 3:
247 if (nu <= nu_a) {
248 return C1_ * (nu / nu_a) * (nu / nu_a);
249 }
250 if (nu <= nu_c) {
251 return std::cbrt(nu / nu_c);
252 }
253 if (nu <= nu_m) {
254 return std::sqrt(nu_c / nu);
255 }
256 return C2_ * fast_pow(nu / nu_m, -p / 2);
257
258 break;
259 case 4:
260 if (nu <= nu_a) {
261 return (nu / nu_a) * (nu / nu_a);
262 }
263 if (nu <= nu_m) {
264 return C2_ * std::sqrt(nu_a / nu);
265 }
266 return C2_ * C1_ * fast_pow(nu / nu_m, -p / 2);
267
268 break;
269 case 5:
270 if (nu <= nu_a) {
271 return (nu / nu_a) * (nu / nu_a);
272 }
273 return (p - 1) * C2_ * fast_pow(nu / nu_a, -p / 2);
274
275 break;
276 case 6:
277 if (nu <= nu_a) {
278 return (nu / nu_a) * (nu / nu_a);
279 }
280 return C2_ * fast_pow(nu / nu_a, -p / 2);
281
282 break;
283
284 default:
285 return 0;
286 break;
287 }

290Real SynPhotons::compute_log2_spectrum(Real log2_nu) const {
291 Real p = elec->p;
292 switch (elec->regime) {
293 case 1:
294 if (log2_nu <= log2_nu_a) {
295 return log2_C1_ + 2. * log2_nu;
296 }
297 if (log2_nu <= log2_nu_m) {
298 return log2_C2_ + log2_nu / 3.;
299 }
300 if (log2_nu <= log2_nu_c) {
301 return log2_C3_ - (p - 1.) / 2. * log2_nu;
302 }
303 return log2_C4_ - p / 2. * log2_nu;
304
305 break;
306 case 2:
307 if (log2_nu <= log2_nu_m) {
308 return log2_C1_ + 2. * log2_nu;
309 }
310 if (log2_nu <= log2_nu_a) {
311 return log2_C2_ + 2.5 * log2_nu;
312 }
313 if (log2_nu <= log2_nu_c) {
314 return log2_C3_ - (p - 1.) / 2. * log2_nu;
315 }
316
317 return log2_C4_ - p / 2. * log2_nu;
318
319 break;
320 case 3:
321 if (log2_nu <= log2_nu_a) {
322 return log2_C1_ + 2. * log2_nu;
323 }
324 if (log2_nu <= log2_nu_c) {
325 return log2_C2_ + log2_nu / 3.;
326 }
327 if (log2_nu <= log2_nu_m) {
328 return log2_C3_ - log2_nu / 2.;
329 }
330
331 return log2_C4_ - p / 2. * log2_nu;
332
333 break;
334 case 4:
335 if (log2_nu <= log2_nu_a) {
336 return log2_C1_ + 2. * log2_nu;
337 }
338 if (log2_nu <= log2_nu_m) {
339 return log2_C2_ - log2_nu / 2.;
340 }
341
342 return log2_C3_ - p / 2. * log2_nu;
343
344 break;
345 case 5:
346 if (log2_nu <= log2_nu_a) {
347 return log2_C1_ + 2. * log2_nu;
348 }
349
350 return log2_C2_ - p / 2. * log2_nu;
351
352 break;
353 case 6:
354 if (log2_nu <= log2_nu_a) {
355 return log2_C1_ + 2. * log2_nu;
356 }
357
358 return log2_C3_ - p / 2. * log2_nu;
359
360 break;
361
362 default:
363 return -con::inf;
364 break;
365 }

369 * <!-- ************************************************************************************** -->
370 * @internal
371 * @brief Calculates the peak synchrotron power per electron in the comoving frame.
372 * @details Based on magnetic field strength B and power-law index p of the electron distribution.
373 * @param B Magnetic field strength
374 * @param p Power-law index of electron distribution
375 * @return Peak synchrotron power per electron
376 * <!-- ************************************************************************************** -->
377 */
378Real compute_single_elec_P_nu_max(Real B, Real p) { return B * (1.73205080757 * con::e3 / (con::me * con::c2)); }

380/**
381 * <!-- ************************************************************************************** -->
382 * @internal
383 * @brief Calculates the peak synchrotron intensity for a given column number density.
384 * @details Uses the peak synchrotron power and column number density.
385 * @param B Magnetic field strength
386 * @param p Power-law index of electron distribution
387 * @param column_den Column number density
388 * @return Peak synchrotron intensity
389 * <!-- ************************************************************************************** -->

391Real compute_syn_I_peak(Real B, Real p, Real column_den) {
392 return compute_single_elec_P_nu_max(B, p) * column_den / (4 * con::pi);

395/**
396 * <!-- ************************************************************************************** -->
397 * @internal
398 * @brief Calculates the characteristic synchrotron frequency for electrons with a given Lorentz factor.
399 * @details Uses the standard synchrotron formula.
400 * @param gamma Electron Lorentz factor
401 * @param B Magnetic field strength
402 * @return Characteristic synchrotron frequency
403 * <!-- ************************************************************************************** -->
404 */
405Real compute_syn_freq(Real gamma, Real B) {
406 Real nu = 3 * con::e / (4 * con::pi * con::me * con::c) * B * gamma * gamma;
407 return nu;
408}

411 * <!-- ************************************************************************************** -->
412 * @internal
413 * @brief Calculates the electron Lorentz factor corresponding to a synchrotron frequency.
414 * @details Inverse of the compute_syn_freq function.

425/**
426 * <!-- ************************************************************************************** -->
427 * @internal
428 * @brief Calculates the maximum electron Lorentz factor for synchrotron emission.
429 * @details Uses an iterative approach to account for inverse Compton cooling effects.
430 * @param B Magnetic field strength
431 * @param Ys InverseComptonY object
432 * @param p Spectral index of electron distribution
433 * @return Maximum electron Lorentz factor
434 * <!-- ************************************************************************************** -->
435 */
436Real compute_syn_gamma_M(Real B, InverseComptonY const& Ys, Real p) {
437 if (B == 0) {
438 return std::numeric_limits<Real>::infinity();
439 }
440 Real Y0 = InverseComptonY::compute_Y_Thompson(Ys);
441 Real gamma_M = std::sqrt(6 * con::pi * con::e / con::sigmaT / (B * (1 + Y0)));
442 Real Y1 = InverseComptonY::compute_Y_tilt_at_gamma(Ys, gamma_M, p);
443
444 for (; std::fabs((Y1 - Y0) / Y0) > 1e-3;) {
445 gamma_M = std::sqrt(6 * con::pi * con::e / con::sigmaT / (B * (1 + Y1)));
446 Y0 = Y1;
447 Y1 = InverseComptonY::compute_Y_tilt_at_gamma(Ys, gamma_M, p);
448 }
449
450 return gamma_M;
451}

453/**
454 * <!-- ************************************************************************************** -->
455 * @internal
456 * @brief Calculates the minimum electron Lorentz factor for synchrotron emission.
457 * @details Accounts for different power-law indices with special handling for the p=2 case.
458 * Uses the fraction of shock energy given to electrons (eps_e) and electron fraction (xi).
459 * @param Gamma_rel Bulk Lorentz factor of the shock
460 * @param gamma_M Maximum electron Lorentz factor
461 * @param eps_e Fraction of shock energy given to electrons
462 * @param p Power-law index of electron distribution
463 * @param xi Fraction of electrons accelerated
464 * @return Minimum electron Lorentz factor
465 * <!-- ************************************************************************************** -->
466 */
467Real compute_syn_gamma_m(Real Gamma_rel, Real gamma_M, Real eps_e, Real p, Real xi) {
468 Real gamma_ave_minus_1 = eps_e * (Gamma_rel - 1) * (con::mp / con::me) / xi;
469 Real gamma_m_minus_1 = 1;
470 if (p > 2) {
471 gamma_m_minus_1 = (p - 2) / (p - 1) * gamma_ave_minus_1;
472 } else if (p < 2) {
473 gamma_m_minus_1 = std::pow((2 - p) / (p - 1) * gamma_ave_minus_1 * std::pow(gamma_M, p - 2), 1 / (p - 1));
474 } else {
475 gamma_m_minus_1 = root_bisect(
476 [=](Real x) -> Real {
477 return (x * std::log(gamma_M) - (x + 1) * std::log(x) - gamma_ave_minus_1 - std::log(gamma_M));
478 },
479 0, gamma_M);
480 }
481 return gamma_m_minus_1 + 1;
482}

484Real compute_gamma_c(Real t_comv, Real B, InverseComptonY const& Ys, Real p) {
485 // t_com = (6*pi*gamma*me*c^2) /(gamma^2*beta^2*sigma_T*c*B^2*(1 + Y_tilt))
486
487 Real Y0 = InverseComptonY::compute_Y_Thompson(Ys);
488 Real gamma_bar = (6 * con::pi * con::me * con::c / con::sigmaT) / (B * B * (1 + Y0) * t_comv);
489 Real gamma_c = (gamma_bar + std::sqrt(gamma_bar * gamma_bar + 4)) / 2; // correction on newtonian regime
490
491 Real Y1 = InverseComptonY::compute_Y_tilt_at_gamma(Ys, gamma_c, p);
492 for (; std::fabs((Y1 - Y0) / Y0) > 1e-3;) { // iterate for IC cooling
493 gamma_bar = (6 * con::pi * con::me * con::c / con::sigmaT) / (B * B * (1 + Y1) * t_comv);
494 gamma_c = (gamma_bar + std::sqrt(gamma_bar * gamma_bar + 4)) / 2;
495 Y0 = Y1;
496 Y1 = InverseComptonY::compute_Y_tilt_at_gamma(Ys, gamma_c, p);
497 }
498
499 return gamma_c;
500}

502/**
503 * <!-- ************************************************************************************** -->
504 * @internal
505 * @brief Calculates the self-absorption Lorentz factor by equating synchrotron emission to blackbody.
506 * @details Uses the peak intensity and shock parameters to determine where absorption becomes important.
507 * Handles both weak and strong absorption regimes.
508 * @param Gamma_rel Bulk Lorentz factor of the shock
509 * @param B Magnetic field strength
510 * @param I_syn_peak Peak synchrotron intensity
511 * @param gamma_m Minimum electron Lorentz factor
512 * @param gamma_c Cooling electron Lorentz factor
513 * @param gamma_M Maximum electron Lorentz factor
514 * @param p Power-law index of electron distribution
515 * @return Self-absorption Lorentz factor
516 * <!-- ************************************************************************************** -->
517 */
518Real compute_syn_gamma_a(Real Gamma_rel, Real B, Real I_syn_peak, Real gamma_m, Real gamma_c, Real gamma_M, Real p) {
519 Real gamma_peak = std::min(gamma_m, gamma_c);
520 Real nu_peak = compute_syn_freq(gamma_peak, B);
521
522 Real kT = (gamma_peak - 1) * (con::me * con::c2) / 3;
523 // 2kT(nu_a/c)^2 = I_peak*(nu_a/nu_peak)^(1/3) // first assume nu_a is in the 1/3 segment
524 Real nu_a = fast_pow(I_syn_peak * con::c2 / (std::cbrt(nu_peak) * 2 * kT), 0.6);
525
526 if (nu_a > nu_peak) { // nu_a is not in the 1/3 segment
527 constexpr Real coef = 3 * con::e / (4 * con::pi * con::me * con::c);
528 if (gamma_c > gamma_m) { // then assume nu_a is in the -(p-1)/2 segment
529 Real nu_m = compute_syn_freq(gamma_m, B);
530 nu_a = fast_pow(I_syn_peak * con::c2 / (2 * kT) * fast_pow(nu_m, p / 2), 2 / (p + 4));
531 Real nu_c = compute_syn_freq(gamma_c, B);
532 if (nu_a > nu_c) { // nu_a is not in the -(p-1)/2 segment, strong absorption
533 Real C = 1.5 * I_syn_peak / (con::me * coef * coef * B * B);
534 Real gamma_a = root_bisect([C](Real x) -> Real { return x * x * x * x * x - x - C; }, gamma_c, gamma_M);
535 return gamma_a;
536 }
537 } else { // strong absorption
538 Real C = 1.5 * I_syn_peak / (con::me * coef * coef * B * B);
539 Real gamma_a = root_bisect([C](Real x) -> Real { return x * x * x * x * x - x - C; }, gamma_c, gamma_M);
540 return gamma_a;
541 }
542 }

555Real cyclotron_correction(Real gamma_m, Real p) {
556 Real f = (gamma_m - 1) / gamma_m;
557 if (p > 3) {
558 f = fast_pow(f, (p - 1) / 2);
559 }
560 return f;

573void update_electrons_4Y(SynElectronGrid& electrons, Shock const& shock) {
574 auto [phi_size, theta_size, t_size] = shock.shape();
575
576 for (size_t i = 0; i < phi_size; ++i) {
577 for (size_t j = 0; j < theta_size; ++j) {
578 size_t k_inj = shock.injection_idx(i, j);
579 for (size_t k = 0; k < t_size; ++k) {
580 if (shock.required(i, j, k) == 0) {
581 continue;
582 }
583 Real Gamma_rel = shock.Gamma_rel(i, j, k);
584 Real t_com = shock.t_comv(i, j, k);
585 Real B = shock.B(i, j, k);
586 Real r = shock.r(i, j, k);
587 Real p = electrons(i, j, k).p;
588 auto& Ys = electrons(i, j, k).Ys;
589 auto& elec = electrons(i, j, k);
590
591 elec.gamma_M = compute_syn_gamma_M(B, Ys, p); // Update maximum electron Lorentz factor
592 if (k <= k_inj) {
593 elec.gamma_c = compute_gamma_c(t_com, B, Ys, p); // Update cooling electron Lorentz factor
594 } else { // no shocked electron injection, just adiabatic cooling
595 elec.gamma_c = electrons(i, j, k_inj).gamma_c * elec.gamma_m / electrons(i, j, k_inj).gamma_m;
596 elec.gamma_M = elec.gamma_c;
597 }
598 Real I_nu_peak = compute_syn_I_peak(B, p, elec.column_den);
599 elec.gamma_a =
600 compute_syn_gamma_a(Gamma_rel, B, I_nu_peak, elec.gamma_m, elec.gamma_c, elec.gamma_M, p);
601 elec.regime = determine_regime(elec.gamma_a, elec.gamma_c, elec.gamma_m);
602 elec.Y_c = InverseComptonY::compute_Y_tilt_at_gamma(Ys, elec.gamma_c, p);
603 }
604 }

618void generate_syn_electrons(SynElectronGrid& electrons, Shock const& shock) {
619 auto [phi_size, theta_size, t_size] = shock.shape();
620
621 RadParams rad = shock.rad;
622
623 electrons.resize({phi_size, theta_size, t_size});
624
625 for (size_t i = 0; i < phi_size; ++i) {
626 for (size_t j = 0; j < theta_size; ++j) {
627 size_t k_inj = shock.injection_idx(i, j);
628 for (size_t k = 0; k < t_size; ++k) {
629 if (shock.required(i, j, k) == 0) {
630 continue;
631 }
632 Real Gamma_rel = shock.Gamma_rel(i, j, k);
633 Real t_com = shock.t_comv(i, j, k);
634 Real B = shock.B(i, j, k);
635 Real r = shock.r(i, j, k);
636
637 auto& elec = electrons(i, j, k);
638
639 elec.gamma_M = compute_syn_gamma_M(B, elec.Ys, rad.p);
640 elec.gamma_m = compute_syn_gamma_m(Gamma_rel, elec.gamma_M, rad.eps_e, rad.p, rad.xi_e);
641
642 // Fraction of synchrotron electrons; the rest are cyclotron
643 Real f_syn = cyclotron_correction(elec.gamma_m, rad.p);
644
645 elec.N_e = (rad.p - 1) * shock.proton_num(i, j, k) * rad.xi_e * f_syn;
646 elec.column_den = elec.N_e / (r * r);
647 Real I_nu_peak = compute_syn_I_peak(B, rad.p, elec.column_den);
648
649 if (k <= k_inj) {
650 elec.gamma_c = compute_gamma_c(t_com, B, electrons(i, j, k).Ys, rad.p);
651 } else { // no shocked electron injection, just adiabatic cooling
652 elec.gamma_c = electrons(i, j, k_inj).gamma_c * elec.gamma_m / electrons(i, j, k_inj).gamma_m;
653 elec.gamma_M = elec.gamma_c; // no injection, so cooling gamma is cutoff gamma
654 }
655
656 elec.gamma_a =
657 compute_syn_gamma_a(Gamma_rel, B, I_nu_peak, elec.gamma_m, elec.gamma_c, elec.gamma_M, rad.p);
658 elec.regime = determine_regime(elec.gamma_a, elec.gamma_c, elec.gamma_m);
659
660 elec.p = rad.p;
661 }
662 }
663 }
