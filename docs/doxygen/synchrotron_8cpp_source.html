20/**
21 * <!-- ************************************************************************************** -->
22 * @internal
23 * @brief Helper function that checks if three values are in non-decreasing order.
24 * @param a First value
25 * @param b Middle value
26 * @param c Last value
27 * @return True if a ≤ b ≤ c, false otherwise
28 * <!-- ************************************************************************************** -->
29 */
30inline bool order(Real a, Real b, Real c) {
31 return a <= b && b <= c;
32};

34/**
35 * <!-- ************************************************************************************** -->
36 * @internal
37 * @brief Determines the spectral regime (1-6) based on the ordering of characteristic Lorentz factors.
38 * @details Classifies the regime based on the ordering of absorption (a), cooling (c),
39 * and minimum (m) Lorentz factors.
40 * @param a Absorption Lorentz factor
41 * @param c Cooling Lorentz factor
42 * @param m Minimum Lorentz factor
43 * @return Regime number (1-6) or 0 if no valid regime is found
44 * <!-- ************************************************************************************** -->
45 */
46size_t determine_regime(Real a, Real c, Real m) {
47 if (order(a, m, c)) {
48 return 1;
49 } else if (order(m, a, c)) {
50 return 2;
51 } else if (order(a, c, m)) {
52 return 3;
53 } else if (order(c, a, m)) {
54 return 4;
55 } else if (order(m, c, a)) {
56 return 5;
57 } else if (order(c, m, a)) {
58 return 6;
59 } else
60 return 0;
61}

67/**
68 * <!-- ************************************************************************************** -->
69 * @internal
70 * @brief Calculates the peak synchrotron power per electron in the comoving frame.
71 * @details Based on magnetic field strength B and power-law index p of the electron distribution.
72 * @param B Magnetic field strength
73 * @param p Power-law index of electron distribution
74 * @return Peak synchrotron power per electron
75 * <!-- ************************************************************************************** -->
76 */
77Real compute_single_elec_P_nu_max(Real B, Real p) {
78 constexpr Real sin_angle_ave = con::pi / 4;
79 constexpr Real Fx_max = 0.92; // Bing's book 5.5
80 return B * (sin_angle_ave * Fx_max * 1.73205080757 * con::e3 / (con::me * con::c2));
81}

83/**
84 * <!-- ************************************************************************************** -->
85 * @internal
86 * @brief Calculates the peak synchrotron intensity for a given column number density.
87 * @details Uses the peak synchrotron power and column number density.
88 * @param B Magnetic field strength
89 * @param p Power-law index of electron distribution
90 * @param column_den Column number density
91 * @return Peak synchrotron intensity
92 * <!-- ************************************************************************************** -->
93 */
94Real compute_syn_I_peak(Real B, Real p, Real column_den) {
95 return compute_single_elec_P_nu_max(B, p) * column_den / (4 * con::pi);
96}

98/**
99 * <!-- ************************************************************************************** -->
100 * @internal
101 * @brief Calculates the characteristic synchrotron frequency for electrons with a given Lorentz factor.
102 * @details Uses the standard synchrotron formula.
103 * @param gamma Electron Lorentz factor
104 * @param B Magnetic field strength
105 * @return Characteristic synchrotron frequency
106 * <!-- ************************************************************************************** -->
107 */
108Real compute_syn_freq(Real gamma, Real B) {
109 return 3 * con::e / (4 * con::pi * con::me * con::c) * B * gamma * gamma;
110}
111
112/**
113 * <!-- ************************************************************************************** -->
114 * @internal
115 * @brief Calculates the electron Lorentz factor corresponding to a synchrotron frequency.
116 * @details Inverse of the compute_syn_freq function.
117 * @param nu Synchrotron frequency
118 * @param B Magnetic field strength
119 * @return Corresponding electron Lorentz factor
120 * <!-- ************************************************************************************** -->
121 */
122Real compute_syn_gamma(Real nu, Real B) {
123 return std::sqrt((4 * con::pi * con::me * con::c / (3 * con::e)) * (nu / B));
124}

130/**
131 * <!-- ************************************************************************************** -->
132 * @internal
133 * @brief Calculates the maximum electron Lorentz factor for synchrotron emission.
134 * @details Uses an iterative approach to account for inverse Compton cooling effects.
135 * @param B Magnetic field strength
136 * @param Ys InverseComptonY object
137 * @param p Spectral index of electron distribution
138 * @return Maximum electron Lorentz factor
139 * <!-- ************************************************************************************** -->
140 */
141Real compute_syn_gamma_M(Real B, Real Y, Real p) {
142 if (B == 0) {
143 return std::numeric_limits<Real>::infinity();
144 }
145 return std::sqrt(6 * con::pi * con::e / con::sigmaT / (B * (1 + Y)));
146}

148/**
149 * <!-- ************************************************************************************** -->
150 * @internal
151 * @brief Calculates the minimum electron Lorentz factor for synchrotron emission.
152 * @details Accounts for different power-law indices with special handling for the p=2 case.
153 * Uses the fraction of shock energy given to electrons (eps_e) and electron fraction (xi).
154 * @param Gamma_th Downstream thermal Lorentz factor
155 * @param gamma_M Maximum electron Lorentz factor
156 * @param eps_e Fraction of shock energy given to electrons
157 * @param p Power-law index of electron distribution
158 * @param xi Fraction of electrons accelerated
159 * @return Minimum electron Lorentz factor
160 * <!-- ************************************************************************************** -->
161 */
162Real compute_syn_gamma_m(Real Gamma_th, Real gamma_M, Real eps_e, Real p, Real xi) {
163 const Real gamma_ave_minus_1 = eps_e * (Gamma_th - 1) * (con::mp / con::me) / xi;
164 Real gamma_m_minus_1 = 1;
165 if (p > 2) {
166 gamma_m_minus_1 = (p - 2) / (p - 1) * gamma_ave_minus_1;
167 } else if (p < 2) {
168 gamma_m_minus_1 = std::pow((2 - p) / (p - 1) * gamma_ave_minus_1 * std::pow(gamma_M, p - 2), 1 / (p - 1));
169 } else {
170 gamma_m_minus_1 = root_bisect(
171 [=](Real x) -> Real {
172 return (x * std::log(gamma_M) - (x + 1) * std::log(x) - gamma_ave_minus_1 - std::log(gamma_M));
173 },
174 0, gamma_M);
175 }
176 return gamma_m_minus_1 + 1;
177}

179Real compute_gamma_c(Real t_comv, Real B, Real Y) {
180 constexpr Real ad_cooling = 1;
181 //-sqrt(Gamma * Gamma - 1) * con::c* t_comv / r; // adiabatic cooling
182
183 const Real gamma_bar = (6 * con::pi * con::me * con::c / con::sigmaT) / (B * B * (1 + Y) * t_comv) * ad_cooling;
184 const Real gamma_c = (gamma_bar + std::sqrt(gamma_bar * gamma_bar + 4)) / 2; // correction on newtonian regime
185
186 return gamma_c;
187}

189/**
190 * <!-- ************************************************************************************** -->
191 * @internal
192 * @brief Calculates the self-absorption Lorentz factor by equating synchrotron emission to blackbody.
193 * @details Uses the peak intensity and shock parameters to determine where absorption becomes important.
194 * Handles both weak and strong absorption regimes.
195 * @param B Magnetic field strength
196 * @param I_syn_peak Peak synchrotron intensity
197 * @param gamma_m Minimum electron Lorentz factor
198 * @param gamma_c Cooling electron Lorentz factor
199 * @param gamma_M Maximum electron Lorentz factor
200 * @param p Power-law index of electron distribution
201 * @return Self-absorption Lorentz factor
202 * <!-- ************************************************************************************** -->
203 */
204Real compute_syn_gamma_a(Real B, Real I_syn_peak, Real gamma_m, Real gamma_c, Real gamma_M, Real p) {
205 const Real gamma_peak = std::min(gamma_m, gamma_c);
206 const Real nu_peak = compute_syn_freq(gamma_peak, B);
207
208 const Real kT = (gamma_peak - 1) * (con::me * con::c2) / 3;
209 // 2kT(nu_a/c)^2 = I_peak*(nu_a/nu_peak)^(1/3) // first assume nu_a is in the 1/3 segment
210 Real nu_a = fast_pow(I_syn_peak * con::c2 / (std::cbrt(nu_peak) * 2 * kT), 0.6);
211
212#ifdef SELF_ABSORPTION_HEATING
213 if (nu_a > nu_peak) { // nu_a is not in the 1/3 segment
214 constexpr Real coef = 3 * con::e / (4 * con::pi * con::me * con::c);
215 if (gamma_c > gamma_m) { // first assume nu_a is in the -(p-1)/2 segment, 2kT(nu_a/nu_m)^2.5 nu_m^2/c^2
216 Real nu_m = compute_syn_freq(gamma_m, B);
217 nu_a = fast_pow(I_syn_peak * con::c2 / (2 * kT) * fast_pow(nu_m, p / 2), 2 / (p + 4));
218 Real nu_c = compute_syn_freq(gamma_c, B);
219 if (nu_a > nu_c) { // nu_a is not in the -(p-1)/2 segment, strong absorption
220 Real C = 1.5 * I_syn_peak / (con::me * pow52(coef * B) * std::sqrt(nu_m));
221 Real gamma_a =
222 root_bisect([C](Real x) -> Real { return x * x * x * x * x * x - x - C; }, gamma_c, gamma_M);
223 return gamma_a;
224 }
225 } else { // strong absorption
226 Real nu_m = compute_syn_freq(gamma_m, B);
227 Real C = 1.5 * I_syn_peak / (con::me * pow52(coef * B) * std::sqrt(nu_m));
228 Real gamma_a = root_bisect([C](Real x) -> Real { return x * x * x * x * x * x - x - C; }, gamma_c, gamma_M);
229 return gamma_a;
230 }
231 }
232#else
233 if (nu_a > nu_peak) { // nu_a is not in the 1/3 segment
234 if (gamma_c > gamma_m) { // first assume nu_a is in the -(p-1)/2 segment
235 const Real nu_m = compute_syn_freq(gamma_m, B);
236 nu_a = fast_pow(I_syn_peak * con::c2 / (2 * kT) * fast_pow(nu_m, p / 2), 2 / (p + 4));
237 const Real nu_c = compute_syn_freq(gamma_c, B);
238 if (nu_a > nu_c) { // nu_a is not in the -(p-1)/2 but -p/2 segment
239 nu_a = fast_pow(I_syn_peak * con::c2 / (2 * kT) * std::sqrt(nu_c) * fast_pow(nu_m, p / 2), 2 / (p + 5));
240 }
241 } else { //first assume nu_a is in the -1/2 segment
242 const Real nu_c = compute_syn_freq(gamma_c, B);
243 nu_a = fast_pow(I_syn_peak * con::c2 / (2 * kT) * std::sqrt(nu_c), 0.4);
244 const Real nu_m = compute_syn_freq(gamma_m, B);
245 if (nu_a > nu_m) { // nu_a is not in the -1/2 segment but -p/2 segment
246 nu_a = fast_pow(I_syn_peak * con::c2 / (2 * kT) * std::sqrt(nu_c) * fast_pow(nu_m, p / 2), 2 / (p + 5));
247 }
248 }
249 }
250
251#endif
252 return compute_syn_gamma(nu_a, B) + 1;
253}

276Real cyclotron_correction(Real gamma_m, Real p) {
277 Real f = (gamma_m - 1) / gamma_m;
278 if (p > 3) {
279 f = fast_pow(f, (p - 1) / 2);

288Real SynElectrons::compute_spectrum(Real gamma) const {
289 switch (regime) {
290 case 1: // same as case 2
291 case 2:
292 if (gamma <= gamma_m) {
293 return 0; // Below minimum Lorentz factor, spectrum is zero
294 } else if (gamma <= gamma_c) {
295 return (p - 1) * fast_pow(gamma / gamma_m, -p) /
296 gamma_m; // Power-law spectrum between gamma_m and gamma_c
297 } else
298 return (p - 1) * fast_pow(gamma / gamma_m, -p) * gamma_c / (gamma * gamma_m);
299 // Above the cooling Lorentz factor: exponential cutoff applied
300
301 break;
302 case 3:
303 if (gamma <= gamma_c) {
304 return 0; // Below cooling Lorentz factor, the spectrum is zero
305 } else if (gamma <= gamma_m) {
306 return gamma_c / (gamma * gamma); // Intermediate regime scaling
307 } else
308 return gamma_c / (gamma * gamma_m) * fast_pow(gamma / gamma_m, -p);
309 // Above minimum Lorentz factor: power-law with exponential cutoff
310
311 break;
312
313#ifdef SELF_ABSORPTION_HEATING
314 case 4: // Gao, Lei, Wu and Zhang 2013 Eq 18
315 if (gamma <= gamma_a) {
316 return 3 * gamma * gamma / (gamma_a * gamma_a * gamma_a); // thermal part
317 } else if (gamma <= gamma_m) {
318 return gamma_c / (gamma * gamma); // Transition region
319 } else
320 return gamma_c / (gamma * gamma_m) * fast_pow(gamma / gamma_m, -p);
321 // High-energy tail with exponential cutoff
322
323 break;
324 case 5: // Gao, Lei, Wu and Zhang 2013 Eq 19
325 if (gamma <= gamma_a) {
326 return 3 * gamma * gamma / (gamma_a * gamma_a * gamma_a); // thermal part
327 } else
328 return (p - 1) * gamma_c / (gamma * gamma_m) * fast_pow(gamma / gamma_m, -p);
329
330 break;
331 case 6: // Gao, Lei, Wu and Zhang 2013 Eq 20
332 if (gamma <= gamma_a) {
333 return 3 * gamma * gamma / (gamma_a * gamma_a * gamma_a); // thermal part
334 } else
335 return fast_pow(gamma_m, p - 1) * gamma_c * fast_pow(gamma, -(p + 1));
336
337 break;
338#else
339 case 4:
340 if (gamma <= gamma_c) {
341 return 0; // Below cooling Lorentz factor, the spectrum is zero
342 } else if (gamma <= gamma_m) {
343 return gamma_c / (gamma * gamma); // Intermediate regime scaling
344 } else
345 return gamma_c / (gamma * gamma_m) * fast_pow(gamma / gamma_m, -p);
346 // High-energy tail with exponential cutoff
347
348 break;
349 case 5:
350 if (gamma <= gamma_m) {
351 return 0; // Below minimum Lorentz factor, spectrum is zero
352 } else if (gamma <= gamma_c) {
353 return (p - 1) * fast_pow(gamma / gamma_m, -p) /
354 gamma_m; // Power-law spectrum between gamma_m and gamma_c
355 } else
356 return (p - 1) * fast_pow(gamma / gamma_m, -p) * gamma_c / (gamma * gamma_m);
357 break;
358 case 6:
359 if (gamma <= gamma_c) {
360 return 0; // Below cooling Lorentz factor, the spectrum is zero
361 } else if (gamma <= gamma_m) {
362 return gamma_c / (gamma * gamma); // Intermediate regime scaling
363 } else
364 return gamma_c / (gamma * gamma_m) * fast_pow(gamma / gamma_m, -p);
365
366 break;
367#endif
368 default:
369 return 0;
370 }
371}

373Real SynElectrons::compute_N_gamma(Real gamma) const {
374 if (gamma <= gamma_c) { // Below the cooling Lorentz factor: direct scaling
375 return N_e * compute_spectrum(gamma);
376 } else {
377 return fast_exp2((gamma_c - gamma) / gamma_M) * N_e * compute_spectrum(gamma) * (1 + Y_c) /
378 (1 + Ys.evaluate_at_gamma(gamma, p));
379 }
380}

388 }

395Real SynPhotons::compute_spectrum(Real nu) const {
396 switch (regime) {
397 case 1:
398 if (nu <= nu_a) {
399 return C1_ * (nu / nu_a) * (nu / nu_a);
400 }
401 if (nu <= nu_m) {
402 return std::cbrt(nu / nu_m);
403 }
404 if (nu <= nu_c) {
405 return fast_pow(nu / nu_m, -(p - 1) / 2);
406 }
407
408 return C2_ * fast_pow(nu / nu_c, -p / 2);
409
410 break;
411 case 2:
412 if (nu <= nu_m) {
413 return C1_ * (nu / nu_m) * (nu / nu_m);
414 }
415 if (nu <= nu_a) {
416 return C2_ * pow52(nu / nu_a); // Using pow52 for (nu / nu_a)^(5/2)
417 }
418 if (nu <= nu_c) {
419 return fast_pow(nu / nu_m, -(p - 1) / 2);
420 }
421
422 return C3_ * fast_pow(nu / nu_c, -p / 2);
423
424 break;
425 case 3:
426 if (nu <= nu_a) {
427 return C1_ * (nu / nu_a) * (nu / nu_a);
428 }
429 if (nu <= nu_c) {
430 return std::cbrt(nu / nu_c);
431 }
432 if (nu <= nu_m) {
433 return std::sqrt(nu_c / nu);
434 }
435 return C2_ * fast_pow(nu / nu_m, -p / 2);
436
437 break;
438#ifdef SELF_ABSORPTION_HEATING
439 case 4:
440 if (nu <= nu_a) {
441 return (nu / nu_a) * (nu / nu_a);
442 }
443 if (nu <= nu_m) {
444 return C2_ * std::sqrt(nu_a / nu);
445 }
446 return C2_ * C1_ * fast_pow(nu / nu_m, -p / 2);
447
448 break;
449 case 5:
450 case 6:
451 if (nu < nu_m) {
452 return C1_ * (nu / nu_a) * (nu / nu_a);
453 }
454 if (nu <= nu_a) {
455 return pow52(nu / nu_a);
456 }
457 return C2_ * fast_pow(nu / nu_a, -p / 2);
458
459 break;
460
461#else
462 case 4:
463 if (nu <= nu_a) {
464 return 3 * C2_ * (nu / nu_a) * (nu / nu_a);
465 }
466 if (nu <= nu_m) {
467 return 3 * C2_ * std::sqrt(nu_a / nu);
468 }
469 return 3 * C2_ * C1_ * fast_pow(nu / nu_m, -p / 2);
470
471 break;
472 case 5:
473 case 6:
474 if (nu <= nu_m) {
475 return C3_ * C2_ * C1_ * (nu / nu_a) * (nu / nu_a);
476 }
477 if (nu <= nu_a) {
478 return C3_ * C2_ * pow52(nu / nu_a);
479 }
480 return C3_ * C2_ * fast_pow(nu / nu_a, -p / 2);
481
482 break;
483
484#endif
485 default:
486 return 0;
487 break;
488 }

491Real SynPhotons::compute_log2_spectrum(Real log2_nu) const {
492 constexpr Real log2_3 = 1.5849625007; // log2(3)
493 switch (regime) {
494 case 1:
495 if (log2_nu <= log2_nu_a) {
496 return log2_C1_ + 2. * log2_nu;
497 }
498 if (log2_nu <= log2_nu_m) {
499 return log2_C2_ + log2_nu / 3.;
500 }
501 if (log2_nu <= log2_nu_c) {
502 return log2_C3_ - (p - 1.) / 2. * log2_nu;
503 }
504 return log2_C4_ - p / 2. * log2_nu;
505
506 break;
507 case 2:
508 if (log2_nu <= log2_nu_m) {
509 return log2_C1_ + 2. * log2_nu;
510 }
511 if (log2_nu <= log2_nu_a) {
512 return log2_C2_ + 2.5 * log2_nu;
513 }
514 if (log2_nu <= log2_nu_c) {
515 return log2_C3_ - (p - 1.) / 2. * log2_nu;
516 }
517
518 return log2_C4_ - p / 2. * log2_nu;
519
520 break;
521 case 3:
522 if (log2_nu <= log2_nu_a) {
523 return log2_C1_ + 2. * log2_nu;
524 }
525 if (log2_nu <= log2_nu_c) {
526 return log2_C2_ + log2_nu / 3.;
527 }
528 if (log2_nu <= log2_nu_m) {
529 return log2_C3_ - log2_nu / 2.;
530 }
531
532 return log2_C4_ - p / 2. * log2_nu;
533
534 break;
535#ifdef SELF_ABSORPTION_HEATING
536 case 4:
537 if (log2_nu <= log2_nu_a) {
538 return log2_C1_ + 2. * log2_nu;
539 }
540 if (log2_nu <= log2_nu_m) {
541 return log2_C2_ - log2_nu / 2.;
542 }
543
544 return log2_C3_ - p / 2. * log2_nu;
545
546 break;
547 case 5:
548 case 6:
549 if (log2_nu <= log2_nu_m) {
550 return 0.5 * log2_nu_m + log2_C1_ + 2. * log2_nu;
551 }
552
553 if (log2_nu <= log2_nu_a) {
554 return log2_C1_ + 2.5 * log2_nu;
555 }
556
557 return log2_C2_ - p / 2. * log2_nu;
558
559 break;
560#else
561 case 4:
562
563 if (log2_nu <= log2_nu_a) {
564 return log2_3 + log2_C4_ + log2_C1_ + 2. * log2_nu;
565 }
566 if (log2_nu <= log2_nu_m) {
567 return log2_3 + log2_C2_ - log2_nu / 2.;
568 }
569
570 return log2_3 + log2_C3_ - p / 2. * log2_nu;
571
572 break;
573 case 5:
574 case 6:
575 if (log2_nu <= log2_nu_m) {
576 return log2_C3_ + log2_C4_ + 0.5 * log2_nu_m + log2_C1_ + 2. * log2_nu;
577 }
578 if (log2_nu <= log2_nu_a) {
579 return log2_C3_ + log2_C4_ + log2_C1_ + 2.5 * log2_nu;
580 }
581
582 return log2_C3_ + log2_C2_ - p / 2. * log2_nu;
583
584 break;
585#endif
586 default:
587 return -con::inf;
588 break;
589 }

592void SynPhotons::update_constant() {
593 // Update constants based on spectral parameters
594 if (regime == 1) {
595 // a_m_1_3 = std::cbrt(nu_a / nu_m); // (nu_a / nu_m)^(1/3)
596 // c_m_mpa1_2 = fastPow(nu_c / nu_m, (-p + 1) / 2); // (nu_c / nu_m)^((-p+1)/2)
597 C1_ = std::cbrt(nu_a / nu_m);
598 C2_ = fast_pow(nu_c / nu_m, (-p + 1) / 2);
599
600 log2_C1_ = (log2_nu_a - log2_nu_m) / 3 - 2 * log2_nu_a;
601 log2_C2_ = -log2_nu_m / 3;
602 log2_C3_ = (p - 1) / 2 * log2_nu_m;
603 log2_C4_ = (p - 1) / 2 * (log2_nu_m - log2_nu_c) + p / 2 * log2_nu_c;
604 } else if (regime == 2) {
605 // m_a_pa4_2 = fastPow(nu_m / nu_a, (p + 4) / 2); // (nu_m / nu_a)^((p+4)/2)
606 // a_m_mpa1_2 = fastPow(nu_a / nu_m, (-p + 1) / 2); // (nu_a / nu_m)^((-p+1)/2)
607 // c_m_mpa1_2 = fastPow(nu_c / nu_m, (-p + 1) / 2); // (nu_c / nu_m)^((-p+1)/2)
608 C1_ = fast_pow(nu_m / nu_a, (p + 4) / 2);
609 C2_ = fast_pow(nu_a / nu_m, (-p + 1) / 2);
610 C3_ = fast_pow(nu_c / nu_m, (-p + 1) / 2);
611
612 log2_C1_ = (p + 4) / 2 * (log2_nu_m - log2_nu_a) - 2 * log2_nu_m;
613 log2_C2_ = (p - 1) / 2 * (log2_nu_m - log2_nu_a) - 2.5 * log2_nu_a;
614 log2_C3_ = (p - 1) / 2 * log2_nu_m;
615 log2_C4_ = (p - 1) / 2 * (log2_nu_m - log2_nu_c) + p / 2 * log2_nu_c;
616 } else if (regime == 3) {
617 // a_c_1_3 = std::cbrt(nu_a / nu_c); // (nu_a / nu_c)^(1/3)
618 // c_m_1_2 = std::sqrt(nu_c / nu_m); // (nu_c / nu_m)^(1/2)
619 C1_ = std::cbrt(nu_a / nu_c);
620 C2_ = std::sqrt(nu_c / nu_m);
621
622 log2_C1_ = (log2_nu_a - log2_nu_c) / 3 - 2 * log2_nu_a;
623 log2_C2_ = -log2_nu_c / 3;
624 log2_C3_ = log2_nu_c / 2;
625 log2_C4_ = (log2_nu_c - log2_nu_m) / 2 + p / 2 * log2_nu_m;
626 } else if (regime == 4) {
627 C1_ = std::sqrt(nu_a / nu_m);
628 C3_ = 3;
629 C2_ = std::sqrt(nu_c / nu_a) / C3_;
630
631 log2_C4_ = fast_log2(C2_);
632
633 log2_C1_ = -2 * log2_nu_a;
634 log2_C2_ = log2_C4_ + log2_nu_a / 2;
635 log2_C3_ = log2_C4_ + (log2_nu_a - log2_nu_m) / 2 + p / 2 * log2_nu_m;
636
637 } else if (regime == 5) {
638 C1_ = std::sqrt(nu_m / nu_a);
639 C3_ = 3 / (p - 1);
640 C2_ = std::sqrt(nu_c / nu_a) * fast_pow(nu_m / nu_a, (p - 1) / 2) / C3_;
641
642 log2_C4_ = fast_log2(C2_);
643
644 log2_C1_ = -2.5 * log2_nu_a;
645 log2_C2_ = log2_C4_ + p / 2 * log2_nu_a;
646
647 log2_C3_ = fast_log2(C3_);
648 } else if (regime == 6) {
649 C1_ = std::sqrt(nu_m / nu_a);
650 C3_ = 3;
651 C2_ = std::sqrt(nu_c / nu_a) * fast_pow(nu_m / nu_a, (p - 1) / 2) / C3_;
652
653 log2_C4_ = fast_log2(C2_);
654
655 log2_C1_ = -2.5 * log2_nu_a;
656 log2_C2_ = log2_C4_ + p / 2 * log2_nu_a;
657
658 log2_C3_ = 1.5849625007; // log2(3)
659 }
660}

662Real SynPhotons::compute_I_nu(Real nu) const {
663 if (nu <= nu_c) { // Below cooling frequency, simple scaling
664 return I_nu_max * compute_spectrum(nu);
665 } else {
666 return fast_exp2((nu_c - nu) / nu_M) * I_nu_max * compute_spectrum(nu) * (1 + Y_c) /
667 (1 + Ys.evaluate_at_nu(nu, p));
668 }
669}

671Real SynPhotons::compute_log2_I_nu(Real log2_nu) const {
672 if (log2_nu <= log2_nu_c) { // Below cooling frequency, simple scaling
673 return log2_I_nu_max + compute_log2_spectrum(log2_nu);
674 } else {
675 const Real cooling_factor = (1 + Y_c) / (1 + Ys.evaluate_at_nu(std::exp2(log2_nu), p));
676 return log2_I_nu_max + compute_log2_spectrum(log2_nu) + fast_log2(cooling_factor) +
677 (nu_c - fast_exp2(log2_nu)) / nu_M;
678 }

695void generate_syn_electrons(SynElectronGrid& electrons, Shock const& shock) {
696 auto [phi_size, theta_size, t_size] = shock.shape();
697
698 const RadParams rad = shock.rad;
699
700 electrons.resize({phi_size, theta_size, t_size});
701
702 for (size_t i = 0; i < phi_size; ++i) {
703 for (size_t j = 0; j < theta_size; ++j) {
704 const size_t k_inj = shock.injection_idx(i, j);
705 for (size_t k = 0; k < t_size; ++k) {
706 if (shock.required(i, j, k) == 0) {
707 continue;
708 }
709 const Real t_com = shock.t_comv(i, j, k);
710 const Real B = shock.B(i, j, k);
711 const Real r = shock.r(i, j, k);
712 // Real Gamma = shock.Gamma(i, j, k);
713 const Real Gamma_th = shock.Gamma_th(i, j, k);
714
715 auto& elec = electrons(i, j, k);
716
717 elec.gamma_M = compute_syn_gamma_M(B, 0., rad.p);
718 elec.gamma_m = compute_syn_gamma_m(Gamma_th, elec.gamma_M, rad.eps_e, rad.p, rad.xi_e);
719
720 // Fraction of synchrotron electrons; the rest are cyclotron
721 const Real f_syn = cyclotron_correction(elec.gamma_m, rad.p);
722
723 elec.N_e = shock.N_p(i, j, k) * rad.xi_e * f_syn;
724 elec.column_den = elec.N_e / (r * r);
725 const Real I_nu_peak = compute_syn_I_peak(B, rad.p, elec.column_den);
726
727 elec.gamma_c = compute_gamma_c(t_com, B, 0.);
728
729 // no new shocked electrons, the cooling Lorentz factor is the truncation Lorentz factor
730 if (k >= k_inj) {
731 elec.gamma_c = electrons(i, j, k_inj).gamma_c * elec.gamma_m / electrons(i, j, k_inj).gamma_m;
732 elec.gamma_M = elec.gamma_c;
733 }
734
735 elec.gamma_a = compute_syn_gamma_a(B, I_nu_peak, elec.gamma_m, elec.gamma_c, elec.gamma_M, rad.p);
736 elec.regime = determine_regime(elec.gamma_a, elec.gamma_c, elec.gamma_m);
737 elec.p = rad.p;
738 }
739 }
740 }
