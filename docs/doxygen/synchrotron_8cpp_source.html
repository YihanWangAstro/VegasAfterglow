18InverseComptonY::InverseComptonY(Real nu_m, Real nu_c, Real B, Real Y_T) noexcept {
19 gamma_hat_m = con::me * con::c2 / con::h / nu_m; // Compute minimum characteristic Lorentz factor
20 gamma_hat_c = con::me * con::c2 / con::h / nu_c; // Compute cooling characteristic Lorentz factor
21 this->Y_T = Y_T; // Set the Thomson Y parameter
22 nu_hat_m = compute_syn_freq(gamma_hat_m, B); // Compute corresponding synchrotron frequency for gamma_hat_m
23 nu_hat_c = compute_syn_freq(gamma_hat_c, B); // Compute corresponding synchrotron frequency for gamma_hat_c
24
25 if (nu_hat_m <= nu_hat_c) {
26 regime = 1; // fast IC cooling regime
27 } else {
28 regime = 2; // slow IC cooling regime
29 }
30}

46Real InverseComptonY::compute_val_at_gamma(Real gamma, Real p) const {
47 switch (regime) {
48 case 3:
49 return Y_T; // In regime 3, simply return Y_T
50 break;
51 case 1:
52 if (gamma <= gamma_hat_m) {
53 return Y_T; // For gamma below gamma_hat_m, no modification
54 } else if (gamma <= gamma_hat_c) {
55 return Y_T / std::sqrt(gamma / gamma_hat_m); // Intermediate regime scaling
56 } else
57 return Y_T * pow43(gamma_hat_c / gamma) / std::sqrt(gamma_hat_c / gamma_hat_m); // High gamma scaling
58
59 break;
60 case 2:
61 if (gamma <= gamma_hat_c) {
62 return Y_T; // For gamma below gamma_hat_c, no modification
63 } else if (gamma <= gamma_hat_m) {
64 return Y_T * fast_pow(gamma / gamma_hat_c, (p - 3) / 2); // Scaling in intermediate regime
65 } else
66 return Y_T * pow43(gamma_hat_m / gamma) *
67 fast_pow(gamma_hat_m / gamma_hat_c, (p - 3) / 2); // High gamma scaling
68
69 break;
70 default:
71 return 0;
72 break;
73 }

76Real InverseComptonY::compute_val_at_nu(Real nu, Real p) const {
77 switch (regime) {
78 case 3:
79 return Y_T; // In regime 3, simply return Y_T
80 break;
81 case 1:
82 if (nu <= nu_hat_m) {
83 return Y_T; // For frequencies below nu_hat_m, no modification
84 } else if (nu <= nu_hat_c) {
85 return Y_T * std::sqrt(std::sqrt(nu_hat_m / nu)); // Intermediate frequency scaling
86 } else
87 return Y_T * pow23(nu_hat_c / nu) * std::sqrt(std::sqrt(nu_hat_m / nu)); // High frequency scaling
88
89 break;
90 case 2:
91 if (nu <= nu_hat_c) {
92 return Y_T; // For frequencies below nu_hat_c, no modification
93 } else if (nu <= nu_hat_m) {
94 return Y_T * fast_pow(nu / nu_hat_c, (p - 3) / 4); // Intermediate frequency scaling
95 } else
96 return Y_T * pow23(nu_hat_m / nu) *
97 fast_pow(nu_hat_m / nu_hat_c, (p - 3) / 4); // High frequency scaling
98
99 break;
100 default:
101 return 0;
102 break;
103 }

126 * <!-- ************************************************************************************** -->
127 * @internal
128 * @brief Helper function that checks if three values are in non-decreasing order.
129 * @param a First value
130 * @param b Middle value
131 * @param c Last value
132 * @return True if a ≤ b ≤ c, false otherwise
133 * <!-- ************************************************************************************** -->

135inline bool order(Real a, Real b, Real c) { return a <= b && b <= c; };

137/**
138 * <!-- ************************************************************************************** -->
139 * @internal
140 * @brief Determines the spectral regime (1-6) based on the ordering of characteristic Lorentz factors.
141 * @details Classifies the regime based on the ordering of absorption (a), cooling (c),
142 * and minimum (m) Lorentz factors.
143 * @param a Absorption Lorentz factor
144 * @param c Cooling Lorentz factor
145 * @param m Minimum Lorentz factor
146 * @return Regime number (1-6) or 0 if no valid regime is found
147 * <!-- ************************************************************************************** -->
148 */
149size_t determine_regime(Real a, Real c, Real m) {
150 if (order(a, m, c)) {
151 return 1;
152 } else if (order(m, a, c)) {
153 return 2;
154 } else if (order(a, c, m)) {
155 return 3;
156 } else if (order(c, a, m)) {
157 return 4;
158 } else if (order(m, c, a)) {
159 return 5;
160 } else if (order(c, m, a)) {
161 return 6;
162 } else
163 return 0;
164}

166Real SynElectrons::compute_gamma_spectrum(Real gamma) const {
167 switch (regime) {
168 case 1: // same as case 2
169 case 2:
170 if (gamma <= gamma_m) {
171 return 0; // Below minimum Lorentz factor, spectrum is zero
172 } else if (gamma <= gamma_c) {
173 return (p - 1) * fast_pow(gamma / gamma_m, -p) /
174 gamma_m; // Power-law spectrum between gamma_m and gamma_c
175 } else
176 return (p - 1) * fast_pow(gamma / gamma_m, -p) * gamma_c / (gamma * gamma_m) *
177 fast_exp(-gamma / gamma_M);
178 // Above cooling Lorentz factor: exponential cutoff applied
179
180 break;
181 case 3:
182 if (gamma <= gamma_c) {
183 return 0; // Below cooling Lorentz factor, spectrum is zero
184 } else if (gamma <= gamma_m) {
185 return gamma_c / (gamma * gamma); // Intermediate regime scaling
186 } else
187 return gamma_c / (gamma * gamma_m) * fast_pow(gamma / gamma_m, -p) * fast_exp(-gamma / gamma_M);
188 // Above minimum Lorentz factor: power-law with exponential cutoff
189
190 break;
191 case 4: // Gao, Lei, Wu and Zhang 2013 Eq 18
192 if (gamma <= gamma_a) {
193 return 3 * gamma * gamma / (gamma_a * gamma_a * gamma_a); // Rising part of the spectrum
194 } else if (gamma <= gamma_m) {
195 return gamma_c / (gamma * gamma); // Transition region
196 } else
197 return gamma_c / (gamma * gamma_m) * fast_pow(gamma / gamma_m, -p) * fast_exp(-gamma / gamma_M);
198 // High energy tail with exponential cutoff
199
200 break;
201 case 5: // Gao, Lei, Wu and Zhang 2013 Eq 19
202 if (gamma <= gamma_a) {
203 return 3 * gamma * gamma / (gamma_a * gamma_a * gamma_a); // Rising part of the spectrum
204 } else
205 return (p - 1) * gamma_c / (gamma * gamma_m) * fast_pow(gamma / gamma_m, -p) *
206 fast_exp(-gamma / gamma_M);
207 // Power-law decay with exponential cutoff
208
209 break;
210 case 6: // Gao, Lei, Wu and Zhang 2013 Eq 20
211 if (gamma <= gamma_a) {
212 return 3 * gamma * gamma / (gamma_a * gamma_a * gamma_a); // Rising part of the spectrum
213 } else
214 return fast_pow(gamma_m, p - 1) * gamma_c * fast_pow(gamma, -(p + 1)) * fast_exp(-gamma / gamma_M);
215 // Steeper decay in this regime
216
217 break;
218 default:
219 return 0;
220 }
221}

241void SynPhotons::update_constant() {
242 // Update constants based on spectral parameters
243 Real p = elec->p;
244 if (elec->regime == 1) {
245 // a_m_1_3 = std::cbrt(nu_a / nu_m); // (nu_a / nu_m)^(1/3)
246 // c_m_mpa1_2 = fastPow(nu_c / nu_m, (-p + 1) / 2); // (nu_c / nu_m)^((-p+1)/2)
247 C1_ = std::cbrt(nu_a / nu_m);
248 C2_ = fast_pow(nu_c / nu_m, (-p + 1) / 2);
249
250 log2_C1_ = fast_log2(nu_a / nu_m) / 3 - 2 * fast_log2(nu_a);
251 log2_C2_ = -fast_log2(nu_m) / 3;
252 log2_C3_ = (p - 1) / 2 * fast_log2(nu_m);
253 log2_C4_ = (p - 1) / 2 * fast_log2(nu_m / nu_c) + p / 2 * fast_log2(nu_c);
254 } else if (elec->regime == 2) {
255 // m_a_pa4_2 = fastPow(nu_m / nu_a, (p + 4) / 2); // (nu_m / nu_a)^((p+4)/2)
256 // a_m_mpa1_2 = fastPow(nu_a / nu_m, (-p + 1) / 2); // (nu_a / nu_m)^((-p+1)/2)
257 // c_m_mpa1_2 = fastPow(nu_c / nu_m, (-p + 1) / 2); // (nu_c / nu_m)^((-p+1)/2)
258 C1_ = fast_pow(nu_m / nu_a, (p + 4) / 2);
259 C2_ = fast_pow(nu_a / nu_m, (-p + 1) / 2);
260 C3_ = fast_pow(nu_c / nu_m, (-p + 1) / 2);
261
262 log2_C1_ = (p + 4) / 2 * fast_log2(nu_m / nu_a) - 2 * fast_log2(nu_m);
263 log2_C2_ = (p - 1) / 2 * fast_log2(nu_m / nu_a) - 2.5 * fast_log2(nu_a);
264 log2_C3_ = (p - 1) / 2 * fast_log2(nu_m);
265 log2_C4_ = (p - 1) / 2 * fast_log2(nu_m / nu_c) + p / 2 * fast_log2(nu_c);
266 } else if (elec->regime == 3) {
267 // a_c_1_3 = std::cbrt(nu_a / nu_c); // (nu_a / nu_c)^(1/3)
268 // c_m_1_2 = std::sqrt(nu_c / nu_m); // (nu_c / nu_m)^(1/2)
269 C1_ = std::cbrt(nu_a / nu_c);
270 C2_ = std::sqrt(nu_c / nu_m);
271
272 log2_C1_ = fast_log2(nu_a / nu_c) / 3 - 2 * fast_log2(nu_a);
273 log2_C2_ = -fast_log2(nu_c) / 3;
274 log2_C3_ = fast_log2(nu_c) / 2;
275 log2_C4_ = fast_log2(nu_c / nu_m) / 2 + p / 2 * fast_log2(nu_m);
276 } else if (elec->regime == 4) {
277 // a_m_1_2 = std::sqrt(nu_a / nu_m); // (nu_a / nu_m)^(1/2)
278 // R4 = std::sqrt(nu_c / nu_a) / 3; // (nu_c / nu_a)^(1/2) / 3; // R4: scaling factor for regime 4
279 C1_ = std::sqrt(nu_a / nu_m);
280 C2_ = std::sqrt(nu_c / nu_a) / 3;
281
282 log2_C1_ = -2 * fast_log2(nu_a);
283 log2_C2_ = fast_log2(C2_) + fast_log2(nu_a) / 2;
284 log2_C3_ = fast_log2(C2_) + fast_log2(nu_a / nu_m) / 2 + p / 2 * fast_log2(nu_m);
285 } else if (elec->regime == 5 || elec->regime == 6) {
286 // R4 = std::sqrt(nu_c / nu_a) / 3; // (nu_c / nu_a)^(1/2) / 3; // R4: scaling factor for
287 // regime 4 R6 = R4 * fastPow(nu_m / nu_a, (p - 1) / 2); // R6: scaling factor for regime 6
288 C1_ = std::sqrt(nu_c / nu_a) / 3;
289 C2_ = C1_ * fast_pow(nu_m / nu_a, (p - 1) / 2);
290
291 log2_C1_ = -2 * fast_log2(nu_a);
292 log2_C2_ = fast_log2((p - 1) * C2_) + p / 2 * fast_log2(nu_a);
293 log2_C3_ = fast_log2(C2_) + p / 2 * fast_log2(nu_a);
294 }
295 // R5 = (p - 1) * R6; // R5 scales R6 (commented out)
296}

298Real SynPhotons::compute_spectrum(Real nu) const {
299 Real p = elec->p;
300 switch (elec->regime) {
301 case 1:
302 if (nu <= nu_a) {
303 return C1_ * (nu / nu_a) * (nu / nu_a);
304 }
305 if (nu <= nu_m) {
306 return std::cbrt(nu / nu_m);
307 }
308 if (nu <= nu_c) {
309 return fast_pow(nu / nu_m, -(p - 1) / 2);
310 }
311
312 return C2_ * fast_pow(nu / nu_c, -p / 2) * fast_exp2(-nu / nu_M);
313
314 break;
315 case 2:
316 if (nu <= nu_m) {
317 return C1_ * (nu / nu_m) * (nu / nu_m);
318 }
319 if (nu <= nu_a) {
320 return C2_ * pow52(nu / nu_a); // Using pow52 for (nu / nu_a)^(5/2)
321 }
322 if (nu <= nu_c) {
323 return fast_pow(nu / nu_m, -(p - 1) / 2);
324 }
325
326 return C3_ * fast_pow(nu / nu_c, -p / 2) * fast_exp2(-nu / nu_M);
327
328 break;
329 case 3:
330 if (nu <= nu_a) {
331 return C1_ * (nu / nu_a) * (nu / nu_a);
332 }
333 if (nu <= nu_c) {
334 return std::cbrt(nu / nu_c);
335 }
336 if (nu <= nu_m) {
337 return std::sqrt(nu_c / nu);
338 }
339 return C2_ * fast_pow(nu / nu_m, -p / 2) * fast_exp2(-nu / nu_M);
340
341 break;
342 case 4:
343 if (nu <= nu_a) {
344 return (nu / nu_a) * (nu / nu_a);
345 }
346 if (nu <= nu_m) {
347 return C2_ * std::sqrt(nu_a / nu);
348 }
349 return C2_ * C1_ * fast_pow(nu / nu_m, -p / 2) * fast_exp2(-nu / nu_M);
350
351 break;
352 case 5:
353 if (nu <= nu_a) {
354 return (nu / nu_a) * (nu / nu_a);
355 }
356 return (p - 1) * C2_ * fast_pow(nu / nu_a, -p / 2) * fast_exp2(-nu / nu_M);
357
358 break;
359 case 6:
360 if (nu <= nu_a) {
361 return (nu / nu_a) * (nu / nu_a);
362 }
363 return C2_ * fast_pow(nu / nu_a, -p / 2) * fast_exp2(-nu / nu_M);
364
365 break;
366
367 default:
368 return 0;
369 break;
370 }
371}

374 Real p = elec->p;
375 switch (elec->regime) {
376 case 1:
377 if (log2_nu <= log2_nu_a) {
378 return log2_C1_ + 2 * log2_nu;
379 }
380 if (log2_nu <= log2_nu_m) {
381 return log2_C2_ + log2_nu / 3;
382 }
383 if (log2_nu <= log2_nu_c) {
384 return log2_C3_ - (p - 1) / 2 * log2_nu;

386 return log2_C4_ - p / 2 * log2_nu - fast_exp2(log2_nu) / nu_M;
387
388 break;
389 case 2:
390 if (log2_nu <= log2_nu_m) {
391 return log2_C1_ + 2 * log2_nu;

394 return log2_C2_ + 2.5 * log2_nu;

396 if (log2_nu <= log2_nu_c) {

398 }
399
400 return log2_C4_ - p / 2 * log2_nu - fast_exp2(log2_nu) / nu_M;
401
402 break;
403 case 3:

405 return log2_C1_ + 2 * log2_nu;

452 * <!-- ************************************************************************************** -->
453 * @internal
454 * @brief Calculates the peak synchrotron power per electron in the comoving frame.
455 * @details Based on magnetic field strength B and power-law index p of the electron distribution.
456 * @param B Magnetic field strength
457 * @param p Power-law index of electron distribution
458 * @return Peak synchrotron power per electron
459 * <!-- ************************************************************************************** -->
460 */
461Real compute_syn_peak_power(Real B, Real p) {
462 constexpr double sqrt3_half = 1.73205080757 / 2;
463 return (p - 1) * B * (sqrt3_half * con::e3 / (con::me * con::c2));
464}

466/**
467 * <!-- ************************************************************************************** -->
468 * @internal
469 * @brief Calculates the characteristic synchrotron frequency for electrons with a given Lorentz factor.
470 * @details Uses the standard synchrotron formula.
471 * @param gamma Electron Lorentz factor
472 * @param B Magnetic field strength
473 * @return Characteristic synchrotron frequency
474 * <!-- ************************************************************************************** -->
475 */
476Real compute_syn_freq(Real gamma, Real B) {
477 Real nu = 3 * con::e / (4 * con::pi * con::me * con::c) * B * gamma * gamma;
478 return nu;
479}

481/**
482 * <!-- ************************************************************************************** -->
483 * @internal
484 * @brief Calculates the electron Lorentz factor corresponding to a synchrotron frequency.
485 * @details Inverse of the compute_syn_freq function.
486 * @param nu Synchrotron frequency
487 * @param B Magnetic field strength
488 * @return Corresponding electron Lorentz factor
489 * <!-- ************************************************************************************** -->
490 */
491Real compute_syn_gamma(Real nu, Real B) {
492 Real gamma = std::sqrt((4 * con::pi * con::me * con::c / (3 * con::e)) * (nu / B));
493 return gamma;
494}

496/**
497 * <!-- ************************************************************************************** -->
498 * @internal
499 * @brief Calculates the maximum electron Lorentz factor for synchrotron emission.
500 * @details Uses an iterative approach to account for inverse Compton cooling effects.
501 * @param B Magnetic field strength
502 * @param Ys InverseComptonY object
503 * @param p Spectral index of electron distribution
504 * @return Maximum electron Lorentz factor
505 * <!-- ************************************************************************************** -->
506 */
507Real compute_syn_gamma_M(Real B, InverseComptonY const& Ys, Real p) {
508 if (B == 0) {
509 return std::numeric_limits<Real>::infinity();
510 }
511 Real Y0 = InverseComptonY::compute_Y_Thompson(Ys);
512 Real gamma_M = std::sqrt(6 * con::pi * con::e / con::sigmaT / (B * (1 + Y0)));
513 Real Y1 = InverseComptonY::compute_Y_tilt_at_gamma(Ys, gamma_M, p);
514
515 for (; std::fabs((Y1 - Y0) / Y0) > 1e-5;) {
516 gamma_M = std::sqrt(6 * con::pi * con::e / con::sigmaT / (B * (1 + Y1)));
517 Y0 = Y1;
518 Y1 = InverseComptonY::compute_Y_tilt_at_gamma(Ys, gamma_M, p);
519 }
520
521 return gamma_M;
522}

524/**
525 * <!-- ************************************************************************************** -->
526 * @internal
527 * @brief Calculates the minimum electron Lorentz factor for synchrotron emission.
528 * @details Accounts for different power-law indices with special handling for the p=2 case.
529 * Uses the fraction of shock energy given to electrons (eps_e) and electron fraction (xi).
530 * @param Gamma_rel Bulk Lorentz factor of the shock
531 * @param gamma_M Maximum electron Lorentz factor
532 * @param eps_e Fraction of shock energy given to electrons
533 * @param p Power-law index of electron distribution
534 * @param xi Fraction of electrons accelerated
535 * @return Minimum electron Lorentz factor
536 * <!-- ************************************************************************************** -->
537 */
538Real compute_syn_gamma_m(Real Gamma_rel, Real gamma_M, Real eps_e, Real p, Real xi) {
539 Real gamma_bar_minus_1 = eps_e * (Gamma_rel - 1) * (con::mp / con::me) / xi;
540 Real gamma_m_minus_1 = 1;
541 if (p > 2) {
542 gamma_m_minus_1 = (p - 2) / (p - 1) * gamma_bar_minus_1;
543 } else if (p < 2) {
544 gamma_m_minus_1 = std::pow((2 - p) / (p - 1) * gamma_bar_minus_1 * std::pow(gamma_M, p - 2), 1 / (p - 1));
545 } else {
546 gamma_m_minus_1 = root_bisect(
547 [=](Real x) -> Real {
548 return (x * std::log(gamma_M) - (x + 1) * std::log(x) - gamma_bar_minus_1 - std::log(gamma_M));
549 },
550 0, gamma_M);
551 }
552 return gamma_m_minus_1 + 1;
553}

555Real compute_gamma_c(Real t_com, Real B, InverseComptonY const& Ys, Real p) {
556 // t_com = (6*pi*gamma*me*c^2) /(gamma^2*beta^2*sigma_T*c*B^2*(1 + Y_tilt))
557 // Real gamma_c = 6 * con::pi * con::me * con::c / (con::sigmaT * B * B * (1 + Y_tilt) * t_com);
558
559 Real Y0 = InverseComptonY::compute_Y_Thompson(Ys);
560 Real gamma_bar = (6 * con::pi * con::me * con::c / con::sigmaT) / (B * B * (1 + Y0) * t_com);
561 Real gamma_c = (gamma_bar + std::sqrt(gamma_bar * gamma_bar + 4)) / 2;
562 Real Y1 = InverseComptonY::compute_Y_tilt_at_gamma(Ys, gamma_c, p);
563
564 for (; std::fabs((Y1 - Y0) / Y0) > 1e-3;) {
565 gamma_bar = (6 * con::pi * con::me * con::c / con::sigmaT) / (B * B * (1 + Y1) * t_com);
566 gamma_c = (gamma_bar + std::sqrt(gamma_bar * gamma_bar + 4)) / 2;
567 Y0 = Y1;
568 Y1 = InverseComptonY::compute_Y_tilt_at_gamma(Ys, gamma_c, p);
569 }
570
571 return gamma_c;
572}

574/**
575 * <!-- ************************************************************************************** -->
576 * @internal
577 * @brief Calculates the self-absorption Lorentz factor by equating synchrotron emission to blackbody.
578 * @details Uses the peak intensity and shock parameters to determine where absorption becomes important.
579 * Handles both weak and strong absorption regimes.
580 * @param Gamma_rel Bulk Lorentz factor of the shock
581 * @param B Magnetic field strength
582 * @param I_syn_peak Peak synchrotron intensity
583 * @param gamma_m Minimum electron Lorentz factor
584 * @param gamma_c Cooling electron Lorentz factor
585 * @param p Power-law index of electron distribution
586 * @return Self-absorption Lorentz factor
587 * <!-- ************************************************************************************** -->
588 */
589Real compute_syn_gamma_a(Real Gamma_rel, Real B, Real I_syn_peak, Real gamma_m, Real gamma_c, Real p) {
590 Real gamma_peak = std::min(gamma_m, gamma_c);
591 Real nu_peak = compute_syn_freq(gamma_peak, B);
592
593 Real kT = (gamma_peak - 1) * (con::me * con::c2) * 2. / 3;
594 // 2kT(nu_a/c)^2 = I_peak*(nu_a/nu_peak)^(1/3) // first assume nu_a is in the 1/3 segment
595 Real nu_a = fast_pow(I_syn_peak * con::c2 / (std::cbrt(nu_peak) * 2 * kT), 0.6);
596
597 // strong absorption
598 if (nu_a > nu_peak) { // nu_a is not in the 1/3 segment
599 Real nu_c = compute_syn_freq(gamma_c, B);
600 Real factor = I_syn_peak / (4. / 3 * con::me * std::sqrt((4 * con::pi * con::me * con::c / (3 * con::e)) / B));
601 if (nu_a < nu_c) { // medium absorption, nu_a is in the -(p-1)/2 segment
602 Real nu_m = compute_syn_freq(gamma_m, B);
603 nu_a = fast_pow(factor, 2 / (p + 4)) * fast_pow(nu_m, (p - 1) / (p + 4));
604 } else { // strong absorption, electron pile-up, nu_a reaches I_syn_peak
605 nu_a = fast_pow(factor, 0.4);
606 }
607 }
608 return compute_syn_gamma(nu_a, B) + 1;
609}

612 Real gamma_peak = std::min(gamma_m, gamma_c);
613 if (gamma_a > gamma_c) {
614 return gamma_a;
615 } else {
616 return gamma_peak;
617 }

630void update_electrons_4Y(SynElectronGrid& electrons, Shock const& shock) {
631 auto [phi_size, theta_size, t_size] = shock.shape();
632
633 for (size_t i = 0; i < phi_size; ++i) {
634 for (size_t j = 0; j < theta_size; ++j) {
635 size_t k_inj = shock.injection_idx(i, j);
636 for (size_t k = 0; k < t_size; ++k) {
637 if (shock.required(i, j, k) == 0) {
638 continue;
639 }
640 Real Gamma_rel = shock.Gamma_rel(i, j, k);
641 Real t_com = shock.t_comv(i, j, k);
642 Real B = shock.B(i, j, k);
643 Real p = electrons(i, j, k).p;
644 auto& Ys = electrons(i, j, k).Ys;
645 auto& elec = electrons(i, j, k);
646
647 elec.gamma_M = compute_syn_gamma_M(B, Ys, p); // Update maximum electron Lorentz factor
648 if (k <= k_inj) {
649 elec.gamma_c = compute_gamma_c(t_com, B, Ys, p); // Update cooling electron Lorentz factor
650 } else { // no shocked electron injection, just adiabatic cooling
651 elec.gamma_c = electrons(i, j, k_inj).gamma_c * elec.gamma_m / electrons(i, j, k_inj).gamma_m;
652 elec.gamma_M = elec.gamma_c;
653 }
654 elec.gamma_a = compute_syn_gamma_a(Gamma_rel, B, elec.I_nu_peak, elec.gamma_m, elec.gamma_c, p);
655 elec.regime = determine_regime(elec.gamma_a, elec.gamma_c, elec.gamma_m);
656 elec.Y_c = InverseComptonY::compute_Y_tilt_at_gamma(Ys, elec.gamma_c, p);
657 }
658 }

662SynElectronGrid generate_syn_electrons(Shock const& shock, Real p, Real xi) {
663 auto [phi_size, theta_size, t_size] = shock.shape();
664
665 SynElectronGrid electrons({phi_size, theta_size, t_size});
666
667 constexpr Real gamma_cyclotron = 3;
668
669 for (size_t i = 0; i < phi_size; ++i) {
670 for (size_t j = 0; j < theta_size; ++j) {
671 size_t k_inj = shock.injection_idx(i, j);
672 for (size_t k = 0; k < t_size; ++k) {
673 if (shock.required(i, j, k) == 0) {
674 continue;
675 }
676 Real Gamma_rel = shock.Gamma_rel(i, j, k);
677 Real t_com = shock.t_comv(i, j, k);
678 Real B = shock.B(i, j, k);
679 Real Sigma = shock.column_num_den(i, j, k);
680
681 auto& elec = electrons(i, j, k);
682
683 elec.gamma_M = compute_syn_gamma_M(B, elec.Ys, p);
684 elec.gamma_m = compute_syn_gamma_m(Gamma_rel, elec.gamma_M, shock.eps_e, p, xi);
685 // Fraction of synchrotron electrons; the rest are cyclotron
686 Real f = 1.;
687 if (1 < elec.gamma_m && elec.gamma_m < gamma_cyclotron) {
688 f = std::min(fast_pow((gamma_cyclotron - 1) / (elec.gamma_m - 1), 1 - p), 1_r);
689 elec.gamma_m = gamma_cyclotron;
690 }
691 elec.column_num_den = Sigma * f * xi;
692 elec.I_nu_peak = compute_syn_peak_power(B, p) * elec.column_num_den / (4 * con::pi);
693 if (k <= k_inj) {
694 elec.gamma_c = compute_gamma_c(t_com, B, elec.Ys, p);
695 } else { // no shocked electron injection, just adiabatic cooling
696 elec.gamma_c = electrons(i, j, k_inj).gamma_c * elec.gamma_m / electrons(i, j, k_inj).gamma_m;
697 elec.gamma_M = elec.gamma_c;
698 }
699 elec.gamma_a = compute_syn_gamma_a(Gamma_rel, B, elec.I_nu_peak, elec.gamma_m, elec.gamma_c, p);
700 elec.regime = determine_regime(elec.gamma_a, elec.gamma_c, elec.gamma_m);
701 elec.p = p;
702 }
703 }
704 }
705 return electrons;
