34/**
35 * <!-- ************************************************************************************** -->
36 * @internal
37 * @brief Determines the spectral regime (1-6) based on the ordering of characteristic Lorentz factors.
38 * @details Classifies the regime based on the ordering of absorption (a), cooling (c),
39 * and minimum (m) Lorentz factors.
40 * @param a Absorption Lorentz factor
41 * @param c Cooling Lorentz factor
42 * @param m Minimum Lorentz factor
43 * @return Regime number (1-6) or 0 if no valid regime is found
44 * <!-- ************************************************************************************** -->
45 */
46size_t determine_regime(Real a, Real c, Real m) {
47 if (order(a, m, c)) {
48 return 1;
49 } else if (order(m, a, c)) {
50 return 2;
51 } else if (order(a, c, m)) {
52 return 3;
53 } else if (order(c, a, m)) {
54 return 4;
55 } else if (order(m, c, a)) {
56 return 5;
57 } else if (order(c, m, a)) {
58 return 6;
59 } else
60 return 0;
61}

67/**
68 * <!-- ************************************************************************************** -->
69 * @internal
70 * @brief Calculates the peak synchrotron power per electron in the comoving frame.
71 * @details Based on magnetic field strength B and power-law index p of the electron distribution.
72 * @param B Magnetic field strength
73 * @param p Power-law index of electron distribution
74 * @return Peak synchrotron power per electron
75 * <!-- ************************************************************************************** -->
76 */
77Real compute_single_elec_P_nu_max(Real B, Real p) {
78 constexpr Real sin_angle_ave = con::pi / 4;
79 constexpr Real Fx_max = 0.92; // Bing's book 5.5

98/**
99 * <!-- ************************************************************************************** -->
100 * @internal
101 * @brief Calculates the characteristic synchrotron frequency for electrons with a given Lorentz factor.
102 * @details Uses the standard synchrotron formula.
103 * @param gamma Electron Lorentz factor
104 * @param B Magnetic field strength
105 * @return Characteristic synchrotron frequency
106 * <!-- ************************************************************************************** -->

108Real compute_syn_freq(Real gamma, Real B) {

130/**
131 * <!-- ************************************************************************************** -->
132 * @internal
133 * @brief Calculates the maximum electron Lorentz factor for synchrotron emission.
134 * @details Uses an iterative approach to account for inverse Compton cooling effects.
135 * @param B Magnetic field strength
136 * @param Ys InverseComptonY object
137 * @param p Spectral index of electron distribution
138 * @return Maximum electron Lorentz factor
139 * <!-- ************************************************************************************** -->
140 */
141Real compute_syn_gamma_M(Real B, Real Y, Real p) {
142 if (B == 0) {
143 return std::numeric_limits<Real>::infinity();
144 }
145 return std::sqrt(6 * con::pi * con::e / con::sigmaT / (B * (1 + Y)));
146}

148/**
149 * <!-- ************************************************************************************** -->
150 * @internal
151 * @brief Calculates the minimum electron Lorentz factor for synchrotron emission.
152 * @details Accounts for different power-law indices with special handling for the p=2 case.
153 * Uses the fraction of shock energy given to electrons (eps_e) and electron fraction (xi).
154 * @param Gamma_th Downstream thermal Lorentz factor
155 * @param gamma_M Maximum electron Lorentz factor
156 * @param eps_e Fraction of shock energy given to electrons
157 * @param p Power-law index of electron distribution
158 * @param xi Fraction of electrons accelerated
159 * @return Minimum electron Lorentz factor
160 * <!-- ************************************************************************************** -->
161 */
162Real compute_syn_gamma_m(Real Gamma_th, Real gamma_M, Real eps_e, Real p, Real xi) {
163 const Real gamma_ave_minus_1 = eps_e * (Gamma_th - 1) * (con::mp / con::me) / xi;
164 Real gamma_m_minus_1 = 1;
165 if (p > 2) {
166 gamma_m_minus_1 = (p - 2) / (p - 1) * gamma_ave_minus_1;
167 } else if (p < 2) {
168 gamma_m_minus_1 = std::pow((2 - p) / (p - 1) * gamma_ave_minus_1 * std::pow(gamma_M, p - 2), 1 / (p - 1));
169 } else {
170 gamma_m_minus_1 = root_bisect(
171 [=](Real x) -> Real {
172 return (x * std::log(gamma_M) - (x + 1) * std::log(x) - gamma_ave_minus_1 - std::log(gamma_M));
173 },
174 0, gamma_M);
175 }
176 return gamma_m_minus_1 + 1;
177}

179Real compute_gamma_c(Real t_comv, Real B, Real Y) {
180 constexpr Real ad_cooling = 1;
181 //-sqrt(Gamma * Gamma - 1) * con::c* t_comv / r; // adiabatic cooling
182
183 const Real gamma_bar = (6 * con::pi * con::me * con::c / con::sigmaT) / (B * B * (1 + Y) * t_comv) * ad_cooling;
184 const Real gamma_c = (gamma_bar + std::sqrt(gamma_bar * gamma_bar + 4)) / 2; // correction on newtonian regime
185
186 return gamma_c;
187}

196/**
197 * <!-- ************************************************************************************** -->
198 * @internal
199 * @brief Calculates the self-absorption Lorentz factor by equating synchrotron emission to blackbody.
200 * @details Uses the peak intensity and shock parameters to determine where absorption becomes important.
201 * Handles both weak and strong absorption regimes.
202 * @param B Magnetic field strength
203 * @param I_syn_peak Peak synchrotron intensity
204 * @param gamma_m Minimum electron Lorentz factor
205 * @param gamma_c Cooling electron Lorentz factor
206 * @param gamma_M Maximum electron Lorentz factor
207 * @param p Power-law index of electron distribution
208 * @return Self-absorption Lorentz factor
209 * <!-- ************************************************************************************** -->
210 */
211Real compute_syn_gamma_a(Real B, Real I_syn_peak, Real gamma_m, Real gamma_c, Real gamma_M, Real p) {
212 const Real gamma_peak = std::min(gamma_m, gamma_c);
213 const Real nu_peak = compute_syn_freq(gamma_peak, B);
214
215 const Real kT = (gamma_peak - 1) * (con::me * con::c2) / 3;
216 // 2kT(nu_a/c)^2 = I_peak*(nu_a/nu_peak)^(1/3) // first assume nu_a is in the 1/3 segment
217 Real nu_a = fast_pow(I_syn_peak * con::c2 / (std::cbrt(nu_peak) * 2 * kT), 0.6);
218
219 if (nu_a > nu_peak) { // nu_a is not in the 1/3 segment
220 if (gamma_c > gamma_m) { // first assume nu_a is in the -(p-1)/2 segment
221 const Real nu_m = compute_syn_freq(gamma_m, B);
222 nu_a = fast_pow(I_syn_peak * con::c2 / (2 * kT) * fast_pow(nu_m, p / 2), 2 / (p + 4));
223 const Real nu_c = compute_syn_freq(gamma_c, B);
224 if (nu_a > nu_c) { // nu_a is not in the -(p-1)/2 but -p/2 segment
225 nu_a = fast_pow(I_syn_peak * con::c2 / (2 * kT) * std::sqrt(nu_c) * fast_pow(nu_m, p / 2), 2 / (p + 5));
226 }
227 } else { //first assume nu_a is in the -1/2 segment
228 const Real nu_c = compute_syn_freq(gamma_c, B);
229 nu_a = fast_pow(I_syn_peak * con::c2 / (2 * kT) * std::sqrt(nu_c), 0.4);
230 const Real nu_m = compute_syn_freq(gamma_m, B);
231 if (nu_a > nu_m) { // nu_a is not in the -1/2 segment but -p/2 segment
232 nu_a = fast_pow(I_syn_peak * con::c2 / (2 * kT) * std::sqrt(nu_c) * fast_pow(nu_m, p / 2), 2 / (p + 5));
233 }
234 }
235 }
236 return compute_syn_gamma(nu_a, B) + 1;
237}

261 Real f = (gamma_m - 1) / gamma_m;

272Real SynElectrons::compute_spectrum(Real gamma) const {
273 constexpr Real s = 1.0; // smoothing sharpness (larger = sharper transition)
274
275 switch (regime) {
276 case 1: // slow cooling: gamma_m < gamma_c
277 case 2:
278 case 5:
279
280 // (p-1)/gamma_m * (gamma/gamma_m)^-p * smooth_break_at_gamma_c
281 return (p - 1) * fast_exp(-gamma / gamma_M - gamma_m / gamma) * fast_pow(gamma / gamma_m, -p) / gamma_m *
282 fast_pow(1.0 + fast_pow(gamma / gamma_c, s), -1.0 / s);
283
284 case 3: // fast cooling: gamma_c < gamma_m
285 case 4:
286 case 6:
287 // gamma_c/gamma^2 * smooth_break_at_gamma_m
288 return fast_exp(-gamma / gamma_M - gamma_c / gamma) * gamma_c / (gamma * gamma) *
289 fast_pow(1.0 + fast_pow(gamma / gamma_m, s * (p - 1)), -1.0 / s);
290
291 default:
292 return 0;
293 }
294}

326void generate_syn_electrons(SynElectronGrid& electrons, Shock const& shock) {
327 auto [phi_size, theta_size, t_size] = shock.shape();
328
329 const RadParams rad = shock.rad;
330
331 electrons.resize({phi_size, theta_size, t_size});
332
333 const size_t phi_compute = (shock.symmetry != Symmetry::structured) ? 1 : phi_size;
334
335 for (size_t i = 0; i < phi_compute; ++i) {
336 for (size_t j : shock.theta_reps) {
337 const size_t k_inj = shock.injection_idx(i, j);
338 for (size_t k = 0; k < t_size; ++k) {
339 const Real t_com = shock.t_comv(i, j, k);
340 const Real B = shock.B(i, j, k);
341 const Real r = shock.r(i, j, k);
342 // Real Gamma = shock.Gamma(i, j, k);
343 const Real Gamma_th = shock.Gamma_th(i, j, k);
344
345 auto& elec = electrons(i, j, k);
346
347 elec.gamma_M = compute_syn_gamma_M(B, 0., rad.p);
348 elec.gamma_m = compute_syn_gamma_m(Gamma_th, elec.gamma_M, rad.eps_e, rad.p, rad.xi_e);
349
350 // Fraction of synchrotron electrons; the rest are cyclotron
351 const Real f_syn = cyclotron_correction(elec.gamma_m, rad.p);
352
353 elec.N_e = shock.N_p(i, j, k) * rad.xi_e * f_syn;
354 elec.column_den = elec.N_e / (r * r);
355 const Real I_nu_peak = compute_syn_I_peak(B, rad.p, elec.column_den);
356
357 // no new shocked electrons, cool the relic population from crossing
358 if (k >= k_inj) {
359 auto const& inj = electrons(i, j, k_inj - 1);
360 const Real dt_comv = t_com - shock.t_comv(i, j, k_inj - 1);
361 elec.gamma_c = cool_after_crossing(inj.gamma_c, inj.gamma_m, elec.gamma_m, t_com, B, 0);
362
363 elec.gamma_M = cool_after_crossing(inj.gamma_M, inj.gamma_m, elec.gamma_m, dt_comv, B, 0);
364 } else {
365 elec.gamma_c = compute_gamma_c(t_com, B, 0.);
366 }
367
368 elec.gamma_a = compute_syn_gamma_a(B, I_nu_peak, elec.gamma_m, elec.gamma_c, elec.gamma_M, rad.p);
369 elec.regime = determine_regime(elec.gamma_a, elec.gamma_c, elec.gamma_m);
370 elec.p = rad.p;
371 }
372 }
373 }
374
375 broadcast_symmetry(electrons, shock);
376}
