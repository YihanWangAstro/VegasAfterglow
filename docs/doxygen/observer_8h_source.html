266template <typename... PhotonGrid>
267bool Observer::set_boundaries(InterpState& state, size_t i, size_t j, size_t k, Real lg2_nu_obs,
268 PhotonGrid const&... photons) noexcept {
269 Real lg2_t_ratio = lg2_t(i, j, k + 1) - lg2_t(i, j, k);
270
271 size_t eff_i = i * jet_3d;
272 Real lg2_S_ratio = lg2_surface(i, j, k + 1) - lg2_surface(i, j, k);
273
274 // continuing from previous boundary, shift the high boundary to lower.
275 // Calling .I_nu()/.log_I_nu() could be expensive.
276 if (state.last_hi != 0 && k == state.last_hi) {
277 state.lg2_I_lo = state.lg2_I_hi;
278 } else {
279 Real lg2_nu_lo = lg2_one_plus_z + lg2_nu_obs - lg2_doppler(i, j, k);
280 state.lg2_I_lo = (photons(eff_i, j, k).compute_log2_I_nu(lg2_nu_lo) + ...);
281 }
282
283 Real lg2_nu_hi = lg2_one_plus_z + lg2_nu_obs - lg2_doppler(i, j, k + 1);
284 state.lg2_I_hi = (photons(eff_i, j, k + 1).compute_log2_I_nu(lg2_nu_hi) + ...);
285
286 state.slope = (state.lg2_I_hi - state.lg2_I_lo + lg2_S_ratio) / lg2_t_ratio;
287
288 if (!std::isfinite(state.slope)) {
289 return false;
290 }
291
292 state.last_hi = k + 1;
293 return true;
294}

296template <typename... PhotonGrid>
297MeshGrid Observer::specific_flux(Array const& t_obs, Array const& nu_obs, PhotonGrid const&... photons) {
298 size_t t_obs_len = t_obs.size();
299 size_t nu_len = nu_obs.size();
300
301 Array lg2_t_obs = xt::log2(t_obs);
302 Array lg2_nu = xt::log2(nu_obs);
303
304 MeshGrid F_nu({nu_len, t_obs_len}, 0);
305
306 InterpState state;
307 for (size_t l = 0; l < nu_len; l++) {
308 // Loop over effective phi and theta grid points.
309 for (size_t i = 0; i < eff_phi_grid; i++) {
310 for (size_t j = 0; j < theta_grid; j++) {
311 // Skip observation times that are below the grid's start time
312 size_t t_idx = 0;
313 iterate_to(lg2_t(i, j, 0), lg2_t_obs, t_idx);
314
315 for (size_t k = 0; k < t_grid - 1 && t_idx < t_obs_len; k++) {
316 Real const lg2_t_hi = lg2_t(i, j, k + 1);
317 if (lg2_t_hi < lg2_t_obs(t_idx)) {
318 continue;
319 } else {
320 if (set_boundaries(state, i, j, k, lg2_nu[l], photons...)) {
321 for (; t_idx < t_obs_len && lg2_t_obs(t_idx) <= lg2_t_hi; t_idx++) {
322 F_nu(l, t_idx) += interpolate(state, i, j, k, lg2_t_obs(t_idx));
323 }
324 } else {
325 iterate_to(lg2_t_hi, lg2_t_obs, t_idx);
326 }
327 }
328 }
329 }
330 }
331 }
332
333 // Normalize the flux by the factor (1+z)/(lumi_dist^2).
334 Real const coef = one_plus_z / (lumi_dist * lumi_dist);
335 F_nu *= coef;
336
337 return F_nu;
338}

355template <typename... PhotonGrid>
356Array Observer::flux(Array const& t_obs, Array const& band_freq, PhotonGrid const&... photons) {
357 Array nu_obs = boundary_to_center_log(band_freq);
358 MeshGrid F_nu = specific_flux(t_obs, nu_obs, photons...);
359 Array flux({t_obs.size()}, 0);
360 for (size_t i = 0; i < nu_obs.size(); ++i) {
361 Real dnu = band_freq(i + 1) - band_freq(i);
362 for (size_t j = 0; j < flux.size(); ++j) {
363 flux(j) += dnu * F_nu(i, j);
364 }
365 }
366 return flux;
