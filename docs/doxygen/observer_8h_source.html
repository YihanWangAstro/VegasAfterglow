280template <typename... PhotonGrid>
281bool Observer::set_boundaries(InterpState& state, size_t i, size_t j, size_t k, Real lg2_nu_obs,
282 PhotonGrid&... photons) noexcept {
283 if (state.last_hi == k + 1 && state.last_lg2_nu == lg2_nu_obs) {
284 if (!std::isfinite(state.slope)) {
285 return false;
286 } else {
287 return true;
288 }
289 }
290
291 Real lg2_t_ratio = lg2_t(i, j, k + 1) - lg2_t(i, j, k);
292
293 size_t eff_i = i * jet_3d;
294
295 // continuing from previous boundary, shift the high boundary to lower.
296 // Calling .I_nu()/.log_I_nu() could be expensive.
297 if (state.last_hi != 0 && k == state.last_hi && lg2_nu_obs == state.last_lg2_nu) {
298 state.lg2_I_nu_lo = state.lg2_I_nu_hi;
299 } else {
300 Real lg2_nu_lo = lg2_one_plus_z + lg2_nu_obs - lg2_doppler(i, j, k);
301 state.lg2_I_nu_lo = 3 * lg2_doppler(i, j, k) + (photons(eff_i, j, k).compute_log2_I_nu(lg2_nu_lo) + ...) +
302 lg2_emission_area(i, j, k);
303 }
304
305 Real lg2_nu_hi = lg2_one_plus_z + lg2_nu_obs - lg2_doppler(i, j, k + 1);
306 state.lg2_I_nu_hi = 3 * lg2_doppler(i, j, k + 1) + (photons(eff_i, j, k + 1).compute_log2_I_nu(lg2_nu_hi) + ...) +
307 lg2_emission_area(i, j, k + 1);
308
309 state.slope = (state.lg2_I_nu_hi - state.lg2_I_nu_lo) / lg2_t_ratio;
310
311 if (!std::isfinite(state.slope)) {
312 return false;
313 }
314
315 state.last_hi = k + 1;
316 state.last_lg2_nu = lg2_nu_obs;
317 return true;
318}

320template <typename... PhotonGrid>
321MeshGrid Observer::specific_flux(Array const& t_obs, Array const& nu_obs, PhotonGrid&... photons) {
322 size_t t_obs_len = t_obs.size();
323 size_t nu_len = nu_obs.size();
324
325 Array lg2_t_obs = xt::log2(t_obs);
326 Array lg2_nu = xt::log2(nu_obs);
327
328 MeshGrid F_nu({nu_len, t_obs_len}, 0);
329
330 for (size_t l = 0; l < nu_len; l++) {
331 // Loop over effective phi and theta grid points.
332 for (size_t i = 0; i < eff_phi_grid; i++) {
333 for (size_t j = 0; j < theta_grid; j++) {
334 // Skip observation times that are below the grid's start time
335 size_t t_idx = 0;
336 iterate_to(lg2_t(i, j, 0), lg2_t_obs, t_idx);
337
338 InterpState state;
339 for (size_t k = 0; k < t_grid - 1 && t_idx < t_obs_len; k++) {
340 Real lg2_t_hi = lg2_t(i, j, k + 1);
341 if (lg2_t_hi < lg2_t_obs(t_idx)) {
342 continue;
343 } else {
344 if (set_boundaries(state, i, j, k, lg2_nu[l], photons...)) {
345 for (; t_idx < t_obs_len && lg2_t_obs(t_idx) <= lg2_t_hi; t_idx++) {
346 F_nu(l, t_idx) += interpolate(state, i, j, k, lg2_t_obs(t_idx));
347 }
348 } else {
349 iterate_to(lg2_t_hi, lg2_t_obs, t_idx);
350 }
351 }
352 }
353 }
354 }
355 }
356
357 // Normalize the flux by the factor (1+z)/(lumi_dist^2).
358 Real const coef = one_plus_z / (lumi_dist * lumi_dist);
359 F_nu *= coef;
360
361 return F_nu;
362}

364template <typename... PhotonGrid>
365Array Observer::specific_flux_series(Array const& t_obs, Array const& nu_obs, PhotonGrid&... photons) {
366 size_t t_obs_len = t_obs.size();
367 size_t nu_len = nu_obs.size();
368
369 if (nu_len != t_obs_len) {
370 std::cout << "nu_obs and t_obs must have the same length" << std::endl;
371 }
372
373 Array lg2_t_obs = xt::log2(t_obs);
374 Array lg2_nu = xt::log2(nu_obs);
375
376 Array F_nu = xt::zeros<Real>({t_obs_len});
377
378 for (size_t i = 0; i < eff_phi_grid; i++) {
379 for (size_t j = 0; j < theta_grid; j++) {
380 size_t t_idx = 0;
381 iterate_to(lg2_t(i, j, 0), lg2_t_obs, t_idx);
382
383 size_t k = 0;
384 InterpState state;
385 while (t_idx < t_obs_len && k < t_grid - 1) {
386 Real lg2_t_hi = lg2_t(i, j, k + 1);
387 if (lg2_t_hi < lg2_t_obs(t_idx)) {
388 k++;
389 continue;
390 } else {
391 if (set_boundaries(state, i, j, k, lg2_nu[t_idx], photons...)) {
392 F_nu(t_idx) += interpolate(state, i, j, k, lg2_t_obs(t_idx));
393 }
394 t_idx++;
395 }
396 }
397 }
398 }
399
400 // Normalize the flux by the factor (1+z)/(lumi_dist^2).
401 Real const coef = one_plus_z / (lumi_dist * lumi_dist);
402 F_nu *= coef;
403
404 return F_nu;
405}

422template <typename... PhotonGrid>
423Array Observer::flux(Array const& t_obs, Array const& band_freq, PhotonGrid&... photons) {
424 Array nu_obs = boundary_to_center_log(band_freq);
425 MeshGrid F_nu = specific_flux(t_obs, nu_obs, photons...);
426 Array flux({t_obs.size()}, 0);
427 for (size_t i = 0; i < nu_obs.size(); ++i) {
428 Real dnu = band_freq(i + 1) - band_freq(i);
429 for (size_t j = 0; j < flux.size(); ++j) {
430 flux(j) += dnu * F_nu(i, j);
431 }
432 }
433 return flux;
