316template <typename PhotonGrid>
317bool Observer::set_boundaries(InterpState& state, size_t eff_i, size_t i, size_t j, size_t k, Real lg2_nu_src,
318 PhotonGrid& photons) noexcept {
319 if (state.last_hi == k + 1 && state.last_lg2_nu == lg2_nu_src) {
320 return std::isfinite(state.slope);
321 }
322
323 const Real lg2_t_ratio = lg2_t(i, j, k + 1) - lg2_t(i, j, k);
324
325 // Reuse previous hi boundary as new lo when advancing sequentially
326 if (state.last_hi != 0 && k == state.last_hi && lg2_nu_src == state.last_lg2_nu) {
327 state.lg2_L_nu_lo = state.lg2_L_nu_hi;
328 } else {
329 Real lg2_nu_lo = lg2_nu_src - lg2_doppler(i, j, k);
330 state.lg2_L_nu_lo = photons(eff_i, j, k).compute_log2_I_nu(lg2_nu_lo) + lg2_geom_factor(i, j, k);
331 }
332
333 Real lg2_nu_hi = lg2_nu_src - lg2_doppler(i, j, k + 1);
334 state.lg2_L_nu_hi = photons(eff_i, j, k + 1).compute_log2_I_nu(lg2_nu_hi) + lg2_geom_factor(i, j, k + 1);
335
336 state.slope = (state.lg2_L_nu_hi - state.lg2_L_nu_lo) / lg2_t_ratio;
337
338 if (!std::isfinite(state.slope)) {
339 return false;
340 }
341
342 state.last_hi = k + 1;
343 state.last_lg2_nu = lg2_nu_src;
344 return true;
345}

347template <typename PhotonGrid>
348MeshGrid Observer::specific_flux(Array const& t_obs, Array const& nu_obs, PhotonGrid& photons) {
349 const size_t t_obs_len = t_obs.size();
350 const size_t nu_len = nu_obs.size();
351
352 const Array lg2_t_obs = xt::log2(t_obs);
353 const Array lg2_nu_src = xt::log2(nu_obs) + std::log2(one_plus_z);
354
355 MeshGrid F_nu({nu_len, t_obs_len}, 0);
356
357 const Real* lg2t = lg2_t_obs.data();
358 const Real* lg2nu = lg2_nu_src.data();
359 Real* fnu_ptr = F_nu.data();
360
361 // Precomputed log2-luminosity at each time boundary × frequency. [k * nu_len + l]
362 Array boundary_vals({t_grid * nu_len}, 0);
363 Array slope_buf({nu_len}, 0);
364 Array lo_buf({nu_len}, 0);
365 Real* slope = slope_buf.data();
366 Real* lo = lo_buf.data();
367
368 for (size_t i = 0; i < eff_phi_grid; i++) {
369 const size_t eff_i = i * jet_3d;
370 for (size_t j = 0; j < theta_grid; j++) {
371 // Precompute boundary values for all time grid points.
372 for (size_t k = 0; k < t_grid; k++) {
373 const Real lg2_dop = lg2_doppler(i, j, k);
374 const Real lg2_geom = lg2_geom_factor(i, j, k);
375 Real* vals = boundary_vals.data() + k * nu_len;
376 for (size_t l = 0; l < nu_len; l++) {
377 vals[l] = photons(eff_i, j, k).compute_log2_I_nu(lg2nu[l] - lg2_dop) + lg2_geom;
378 }
379 }
380
381 size_t t_idx = 0;
382 iterate_to(lg2_t(i, j, 0), lg2_t_obs, t_idx);
383
384 for (size_t k = 0; k < t_grid - 1 && t_idx < t_obs_len; k++) {
385 if (lg2_t(i, j, k + 1) < lg2_t_obs(t_idx))
386 continue;
387
388 const size_t idx_start = t_idx;
389 iterate_to(lg2_t(i, j, k + 1), lg2_t_obs, t_idx);
390
391 const Real* lo_row = boundary_vals.data() + k * nu_len;
392 const Real* hi_row = boundary_vals.data() + (k + 1) * nu_len;
393 const Real t_lo_val = lg2_t(i, j, k);
394 const Real inv_t_ratio = 1.0 / (lg2_t(i, j, k + 1) - t_lo_val);
395
396 for (size_t l = 0; l < nu_len; l++) {
397 const Real s = (hi_row[l] - lo_row[l]) * inv_t_ratio;
398 const bool ok = std::isfinite(s);
399 slope[l] = ok ? s : 0.0;
400 lo[l] = ok ? lo_row[l] : -con::inf;
401 }
402 for (size_t idx = idx_start; idx < t_idx; idx++) {
403 const Real dlg2_t = lg2t[idx] - t_lo_val;
404 for (size_t l = 0; l < nu_len; l++) {
405 fnu_ptr[l * t_obs_len + idx] += fast_exp2(lo[l] + dlg2_t * slope[l]);
406 }
407 }
408 }
409 }
410 }
411
412 F_nu *= one_plus_z / (lumi_dist * lumi_dist);
413
414 return F_nu;
415}

417template <typename PhotonGrid>
418Array Observer::specific_flux_series(Array const& t_obs, Array const& nu_obs, PhotonGrid& photons) {
419 const size_t t_obs_len = t_obs.size();
420
421 if (nu_obs.size() != t_obs_len) {
422 throw std::invalid_argument("specific_flux_series requires nu_obs and t_obs to have the same length");
423 }
424
425 const Array lg2_t_obs = xt::log2(t_obs);
426 const Array lg2_nu_src = xt::log2(nu_obs) + std::log2(one_plus_z);
427
428 Array F_nu = xt::zeros<Real>({t_obs_len});
429
430 const Real* lg2t = lg2_t_obs.data();
431 const Real* lg2nu = lg2_nu_src.data();
432 Real* fnu = F_nu.data();
433
434 for (size_t i = 0; i < eff_phi_grid; i++) {
435 const size_t eff_i = i * jet_3d;
436 for (size_t j = 0; j < theta_grid; j++) {
437 size_t idx = 0;
438 iterate_to(lg2_t(i, j, 0), lg2_t_obs, idx);
439 InterpState state;
440
441 for (size_t k = 0; idx < t_obs_len && k < t_grid - 1;) {
442 if (lg2_t(i, j, k + 1) < lg2t[idx]) {
443 k++;
444 } else {
445 if (set_boundaries(state, eff_i, i, j, k, lg2nu[idx], photons)) [[likely]] {
446 fnu[idx] += loglog_interpolate(state, lg2t[idx], lg2_t(i, j, k));
447 }
448 idx++;
449 }
450 }
451 }
452 }
453
454 // Normalize the flux by the factor (1+z)/(lumi_dist^2).
455 F_nu *= one_plus_z / (lumi_dist * lumi_dist);
456
457 return F_nu;
458}

475template <typename PhotonGrid>
476Array Observer::flux(Array const& t_obs, Array const& band_freq, PhotonGrid& photons) {
477 MeshGrid F_nu = specific_flux(t_obs, band_freq, photons);
478 Array weights;
479 compute_boole_weights(band_freq, weights);
480 Array flux({t_obs.size()}, 0);
481 for (size_t i = 0; i < band_freq.size(); ++i) {
482 for (size_t j = 0; j < flux.size(); ++j) {
483 flux(j) += F_nu(i, j) * weights(i);
484 }
485 }
486 return flux;
487}

493template <typename PhotonGrid>
494void Observer::build_splat_grid(SplatGrid& grid, Coord const& coord, Shock const& shock, Real t_obs, Real nu_obs,
495 PhotonGrid& photons, std::vector<Real> const& cos_phi_tab,
496 std::vector<Real> const& sin_phi_tab) {
497 const Real lg2_t_target = fast_log2(t_obs);
498 const Real lg2_nu_src = fast_log2(nu_obs * one_plus_z);
499 const Real inv_d_A = one_plus_z * one_plus_z / lumi_dist;
500 const Real cos_obs = std::cos(coord.theta_view);
501 const Real sin_obs = std::sin(coord.theta_view);
502 const bool is_axisym = (eff_phi_grid == 1);
503 const size_t copies_per_phys = grid.n_phi / eff_phi_grid;
504
505 for (size_t i_phys = 0; i_phys < eff_phi_grid; ++i_phys) {
506 const size_t eff_i = i_phys * jet_3d;
507 const Real cp_fixed = is_axisym ? 0 : std::cos(coord.phi[i_phys] - coord.phi_view);
508 const Real sp_fixed = is_axisym ? 0 : std::sin(coord.phi[i_phys] - coord.phi_view);
509
510 for (size_t j = 0; j < theta_grid; ++j) {
511 if (lg2_t_target < lg2_t(i_phys, j, 0) || lg2_t_target >= lg2_t(i_phys, j, t_grid - 1)) {
512 continue;
513 }
514
515 // Find time bracket
516 size_t k = 0;
517 while (k + 2 < t_grid && lg2_t(i_phys, j, k + 1) < lg2_t_target) {
518 ++k;
519 }
520
521 // Interpolate luminosity
522 InterpState state_L;
523 if (!set_boundaries(state_L, eff_i, i_phys, j, k, lg2_nu_src, photons)) {
524 continue;
525 }
526 const Real L_val = loglog_interpolate(state_L, lg2_t_target, lg2_t(i_phys, j, k));
527 if (!std::isfinite(L_val) || L_val <= 0) {
528 continue;
529 }
530
531 // Interpolate position
532 const Real w = (lg2_t_target - lg2_t(i_phys, j, k)) / (lg2_t(i_phys, j, k + 1) - lg2_t(i_phys, j, k));
533 const Real r_ang = (shock.r(eff_i, j, k) + w * (shock.r(eff_i, j, k + 1) - shock.r(eff_i, j, k))) * inv_d_A;
534 const Real theta_val = jet_spreading_ ? shock.theta(eff_i, j, k) +
535 w * (shock.theta(eff_i, j, k + 1) - shock.theta(eff_i, j, k))
536 : shock.theta(eff_i, j, 0);
537 const Real ct = std::cos(theta_val);
538 const Real st = std::sin(theta_val);
539 const Real L_per_copy = L_val / copies_per_phys;
540
541 // Project onto sky plane for each azimuthal copy
542 for (size_t c = 0; c < copies_per_phys; ++c) {
543 const Real cp = is_axisym ? cos_phi_tab[c] : cp_fixed;
544 const Real sp = is_axisym ? sin_phi_tab[c] : sp_fixed;
545 const Real x_ang = r_ang * (st * cp * cos_obs - ct * sin_obs);
546 const Real y_ang = r_ang * st * sp;
547
548 const size_t i_splat = i_phys * copies_per_phys + c;
549 grid.x(i_splat, j) = x_ang;
550 grid.y(i_splat, j) = y_ang;
551 grid.L(i_splat, j) = L_per_copy;

566SkyImageResult Observer::sky_image(Coord const& coord, Shock const& shock, Array const& t_obs, Real nu_obs,
567 PhotonGrid& photons, size_t npixel, Real pix_size) {
568 const size_t n_frames = t_obs.size();
569 SkyImageResult result;
570 result.image = MeshGrid3d({n_frames, npixel, npixel}, 0);
571
572 if (npixel == 0 || t_grid < 2) {
573 result.extent = {0, 0, 0, 0};
574 return result;
575 }
576
577 // Compute extent and pixel_solid_angle once (depends only on pix_size and npixel)
578 const Real half_fov = 0.5 * pix_size * static_cast<Real>(npixel);
579 result.extent = {-half_fov, half_fov, -half_fov, half_fov};
580 result.pixel_solid_angle = pix_size * pix_size;
581
582 // 2D Gaussian splatting
583 const size_t n_splat_phi = (eff_phi_grid == 1) ? 4 * theta_grid : eff_phi_grid;
584 SplatGrid grid(n_splat_phi, theta_grid);
585
586 const size_t copies_per_phys = n_splat_phi / eff_phi_grid;
587 std::vector<Real> cos_phi_tab, sin_phi_tab;
588 if (eff_phi_grid == 1) {
589 cos_phi_tab.resize(copies_per_phys);
590 sin_phi_tab.resize(copies_per_phys);
591 for (size_t c = 0; c < copies_per_phys; ++c) {
592 const Real phi = 2.0 * con::pi * c / n_splat_phi;
593 cos_phi_tab[c] = std::cos(phi);
594 sin_phi_tab[c] = std::sin(phi);
595 }
596 }
597
598 const bool is_axisym = (jet_3d == 0);
599 const bool azimuthal_avg = is_axisym && (coord.theta_view == 0);
600 const bool mirror_y = is_axisym && !azimuthal_avg;
601
602 for (size_t f = 0; f < n_frames; ++f) {
603 grid.reset();
604 build_splat_grid(grid, coord, shock, t_obs(f), nu_obs, photons, cos_phi_tab, sin_phi_tab);
605 grid.render(result.image, f, pix_size, azimuthal_avg, mirror_y);
606 }
607 return result;
