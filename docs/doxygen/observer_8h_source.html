291template <typename PhotonGrid>
292bool Observer::set_boundaries(InterpState& state, size_t eff_i, size_t i, size_t j, size_t k, Real lg2_nu_src,
293 PhotonGrid& photons) noexcept {
294 if (state.last_hi == k + 1 && state.last_lg2_nu == lg2_nu_src) {
295 if (!std::isfinite(state.slope)) {
296 return false;
297 } else {
298 return true;
299 }
300 }
301
302 const Real lg2_t_ratio = lg2_t(i, j, k + 1) - lg2_t(i, j, k);
303
304 // continuing from the previous boundary, shift the high boundary to lower.
305 // Calling .I_nu()/.log_I_nu() could be expensive.
306 if (state.last_hi != 0 && k == state.last_hi && lg2_nu_src == state.last_lg2_nu) {
307 state.lg2_L_nu_lo = state.lg2_L_nu_hi;
308 } else {
309 Real lg2_nu_lo = lg2_nu_src - lg2_doppler(i, j, k);
310 state.lg2_L_nu_lo = photons(eff_i, j, k).compute_log2_I_nu(lg2_nu_lo) + lg2_geom_factor(i, j, k);
311 }
312
313 Real lg2_nu_hi = lg2_nu_src - lg2_doppler(i, j, k + 1);
314 state.lg2_L_nu_hi = photons(eff_i, j, k + 1).compute_log2_I_nu(lg2_nu_hi) + lg2_geom_factor(i, j, k + 1);
315
316 state.slope = (state.lg2_L_nu_hi - state.lg2_L_nu_lo) / lg2_t_ratio;
317
318 if (!std::isfinite(state.slope)) {
319 return false;
320 }
321
322 state.last_hi = k + 1;
323 state.last_lg2_nu = lg2_nu_src;
324 return true;
325}

327template <typename PhotonGrid>
328MeshGrid Observer::specific_flux(Array const& t_obs, Array const& nu_obs, PhotonGrid& photons) {
329 const size_t t_obs_len = t_obs.size();
330 const size_t nu_len = nu_obs.size();
331
332 const Array lg2_t_obs = xt::log2(t_obs);
333 const Array lg2_nu_src = xt::log2(nu_obs) + std::log2(one_plus_z);
334
335 MeshGrid F_nu({nu_len, t_obs_len}, 0);
336
337 for (size_t i = 0; i < eff_phi_grid; i++) {
338 size_t eff_i = i * jet_3d;
339 for (size_t j = 0; j < theta_grid; j++) {
340 // Skip observation times that are below the grid's start time
341 size_t t_idx = 0;
342 iterate_to(lg2_t(i, j, 0), lg2_t_obs, t_idx);
343
344 InterpState state;
345 for (size_t k = 0; k < t_grid - 1 && t_idx < t_obs_len; k++) {
346 const Real t_lo = lg2_t(i, j, k);
347 const Real t_hi = lg2_t(i, j, k + 1);
348
349 if (t_hi < lg2_t_obs(t_idx))
350 continue;
351
352 const size_t idx_start = t_idx;
353 iterate_to(t_hi, lg2_t_obs, t_idx);
354 const size_t idx_end = t_idx;
355
356 for (size_t l = 0; l < nu_len; l++) {
357 if (set_boundaries(state, eff_i, i, j, k, lg2_nu_src[l], photons)) [[likely]] {
358 for (size_t idx = idx_start; idx < idx_end; idx++) {
359 F_nu(l, idx) += loglog_interpolate(state, lg2_t_obs(idx), t_lo);
360 }
361 }
362 }
363 }
364 }
365 }
366
367 F_nu *= one_plus_z / (lumi_dist * lumi_dist);
368
369 return F_nu;
370}

372template <typename PhotonGrid>
373Array Observer::specific_flux_series(Array const& t_obs, Array const& nu_obs, PhotonGrid& photons) {
374 const size_t t_obs_len = t_obs.size();
375
376 if (nu_obs.size() != t_obs_len) {
377 std::cout << "nu_obs and t_obs must have the same length" << std::endl;
378 }
379
380 const Array lg2_t_obs = xt::log2(t_obs);
381 const Array lg2_nu_src = xt::log2(nu_obs) + std::log2(one_plus_z);
382
383 Array F_nu = xt::zeros<Real>({t_obs_len});
384
385 for (size_t i = 0; i < eff_phi_grid; i++) {
386 const size_t eff_i = i * jet_3d;
387 for (size_t j = 0; j < theta_grid; j++) {
388 // Skip observation times below the grid's start time
389 size_t idx = 0;
390 iterate_to(lg2_t(i, j, 0), lg2_t_obs, idx);
391
392 InterpState state;
393 for (size_t k = 0; idx < t_obs_len && k < t_grid - 1;) {
394 if (lg2_t(i, j, k + 1) < lg2_t_obs(idx)) {
395 k++;
396 } else {
397 if (set_boundaries(state, eff_i, i, j, k, lg2_nu_src(idx), photons)) [[likely]] {
398 F_nu(idx) += loglog_interpolate(state, lg2_t_obs(idx), lg2_t(i, j, k));
399 }
400 idx++;
401 }
402 }
403 }
404 }
405
406 // Normalize the flux by the factor (1+z)/(lumi_dist^2).
407 F_nu *= one_plus_z / (lumi_dist * lumi_dist);
408
409 return F_nu;
410}

427template <typename PhotonGrid>
428Array Observer::flux(Array const& t_obs, Array const& band_freq, PhotonGrid& photons) {
429 Array nu_obs = boundary_to_center_log(band_freq);
430 MeshGrid F_nu = specific_flux(t_obs, nu_obs, photons);
431 Array flux({t_obs.size()}, 0);
432 for (size_t i = 0; i < nu_obs.size(); ++i) {
433 const Real dnu = band_freq(i + 1) - band_freq(i);
434 for (size_t j = 0; j < flux.size(); ++j) {
435 flux(j) += dnu * F_nu(i, j);
436 }
437 }
438 return flux;
