163template <typename Arr1, typename Arr2>
164void boundary_to_center(Arr1 const& boundary, Arr2& center) {
165 if (boundary.size() < 2 || center.size() + 1 != boundary.size()) {
166 return;
167 }
168 for (size_t i = 0; i < center.size(); ++i) {
169 center[i] = 0.5 * (boundary[i] + boundary[i + 1]);
170 }
171}

173template <typename Arr1, typename Arr2>
174void boundary_to_center_log(Arr1 const& boundary, Arr2& center) {
175 if (boundary.size() < 2 || center.size() + 1 != boundary.size()) {
176 return;
177 }
178 for (size_t i = 0; i < center.size(); ++i) {
179 center[i] = std::sqrt(boundary[i] * boundary[i + 1]);
180 }

183template <typename Arr = Array>
184void logspace_center(Real lg2_min, Real lg2_max, size_t size, Arr& center) {
185 center = Arr::from_shape({size});
186 if (size == 0) {
187 return;
188 }
189
190 const Real dlg2 = (lg2_max - lg2_min) / static_cast<Real>(size);
191 const Real r = std::exp2(dlg2);
192 const Real s = std::sqrt(r);
193 Real left = std::exp2(lg2_min);
194
195 for (std::size_t i = 0; i < size; ++i) {
196 center(i) = left * s;
197 left *= r;
198 }
199}

201template <typename Arr = Array>
202void log2space(Real lg2_min, Real lg2_max, Real grid_per_order, Arr& array) {
203 const Real decades = (lg2_max - lg2_min) / log2_10;
204 const size_t n = std::max<size_t>(2, static_cast<size_t>(std::ceil(decades * grid_per_order)));
205
206 array = xt::logspace<Real>(lg2_min, lg2_max, n + 1, 2.0);
207}

209template <typename Arr>
210void logspace_boundary_center(Real lg2_min, Real lg2_max, size_t size, Arr& center, Array& bin_width) {
211 center = Arr::from_shape({size});
212 bin_width = Array::from_shape({size});
213 if (size == 0) {
214 return;
215 }
216
217 const Real dlg2 = (lg2_max - lg2_min) / static_cast<Real>(size);
218 const Real r = std::exp2(dlg2);
219 const Real s = std::sqrt(r);
220 const Real w = r - 1.;
221
222 Real left = std::exp2(lg2_min);
223
224 for (std::size_t i = 0; i < size; ++i) {
225 center(i) = left * s;
226 bin_width(i) = left * w;
227 left *= r;
228 }
