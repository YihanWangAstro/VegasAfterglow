192template <typename Ejecta>
193Coord auto_grid(Ejecta const& jet, Array const& t_obs, Real theta_cut, Real theta_view, Real z, Real phi_resol,
194 Real theta_resol, Real t_resol, bool is_axisymmetric, Real phi_view) {
195 // constexpr size_t min_grid_size = 24;
196 Coord coord;
197 coord.theta_view = theta_view;
198
199 size_t phi_num = std::max<size_t>(static_cast<size_t>(360 * phi_resol), 1);
200 coord.phi = xt::linspace(0., 2 * con::pi, phi_num); // Generate phi grid linearly spaced.
201
202 Real jet_edge = find_jet_edge(jet, con::Gamma_cut); // Determine the jet edge angle.
203 Real theta_min = 1e-6;
204 Real theta_max = std::min(jet_edge, theta_cut);
205 size_t theta_num = std::max<size_t>(static_cast<size_t>((theta_max - theta_min) * 180 / con::pi * theta_resol), 32);
206 coord.theta = xt::linspace(theta_min, theta_max, theta_num); //
207
208 Real t_max = *std::max_element(t_obs.begin(), t_obs.end()); // Maximum observation time.
209 Real t_min = *std::min_element(t_obs.begin(), t_obs.end()); // Minimum observation time.
210 size_t t_num = std::max<size_t>(static_cast<size_t>(std::log10(t_max / t_min) * t_resol), 24);
211
212 size_t phi_size_needed = is_axisymmetric ? 1 : phi_num;
213 coord.t = xt::zeros<Real>({phi_size_needed, theta_num, t_num});
214 for (size_t i = 0; i < phi_size_needed; ++i) {
215 for (size_t j = 0; j < theta_num; ++j) {
216 Real b = gamma_to_beta(jet.Gamma0(coord.phi(i), coord.theta(j)));
217 // Real theta_max = coord.theta(j) + theta_view;
218 Real theta_max = coord.theta.back() + theta_view;
219 Real t_start = 0.99 * t_min * (1 - b) / (1 - std::cos(theta_max) * b) / (1 + z);
220 Real t_end = 1.01 * t_max / (1 + z);
221 xt::view(coord.t, i, j, xt::all()) = xt::logspace(std::log10(t_start), std::log10(t_end), t_num);
222 }
223 }
224
225 return coord; // Construct coordinate object.
226}
