226template <typename Arr>
227void logspace_boundary_center(Real lg2_min, Real lg2_max, size_t size, Arr& center, Array& bin_width) {
228 center = Arr::from_shape({size});
229 bin_width = Array::from_shape({size});
230 if (size == 0) return;
231
232 const Real dlg2 = (lg2_max - lg2_min) / static_cast<Real>(size);
233
234 const Real r = std::exp2(dlg2);
235 const Real s = std::sqrt(r);
236 const Real w = r - 1.;
237
238 Real left = std::exp2(lg2_min);
239
240 for (std::size_t i = 0; i < size; ++i) {
241 center(i) = left * s;
242 bin_width(i) = left * w;
243 left *= r;
244 }
245}

247template <typename Ejecta>
248Real find_jet_edge(Ejecta const& jet, Real gamma_cut, Real phi_resol, Real theta_resol, bool is_axisymmetric) {
249 // binary search for the edge of the jet
250 if (jet.Gamma0(0, con::pi / 2) >= gamma_cut) {
251 return con::pi / 2; // If the Lorentz factor at pi/2 is above the cut, the jet extends to pi/2.
252 }
253 Real low = 0;
254 Real hi = con::pi / 2;
255 Real eps = 1e-9;
256 for (; hi - low > eps;) {
257 Real mid = 0.5 * (low + hi);
258 if (jet.Gamma0(0, mid) > gamma_cut) {
259 low = mid;
260 } else {
261 hi = mid;
262 }
263 }
264
265 // grid based search for the edge of the jet
266 size_t phi_num = std::max<size_t>(static_cast<size_t>(360. * phi_resol), 1);
267 phi_num = is_axisymmetric ? 1 : phi_num;
268 size_t theta_num = std::max<size_t>(static_cast<size_t>(90. * theta_resol), 32);
269 auto phi = xt::linspace(0., 2 * con::pi, phi_num);
270 auto theta = xt::linspace(0., con::pi / 2, theta_num);
271
272 Real theta_edge = con::pi / 2;
273
274 for (size_t i = 0; i < phi_num; ++i) {
275 for (int j = theta_num - 1; j >= 0; --j) {
276 if (jet.Gamma0(phi[i], theta[j]) >= gamma_cut) {
277 theta_edge = theta[j];
278 break; // Found the edge for this phi, no need to check lower theta values
279 } else {
280 theta_edge = theta[j];
281 }
282 }
283 }
284
285 return std::max(theta_edge, low);
286}

288template <typename Ejecta, typename Medium>
289Real jet_spreading_edge(Ejecta const& jet, Medium const& medium, Real phi, Real theta_min, Real theta_max, Real t0) {
290 Real step = (theta_max - theta_min) / 256;
291 Real theta_s = theta_min;
292 Real dp_min = 0;
293
294 for (Real theta = theta_min; theta <= theta_max; theta += step) {
295 // Real G = jet.Gamma0(phi, theta);
296 // Real beta0 = gamma_to_beta(G);
297 // Real r0 = beta0 * con::c * t0 / (1 - beta0);
298 // Real rho = medium.rho(phi, theta, 0);
299 Real th_lo = std::max(theta - step, theta_min);
300 Real th_hi = std::min(theta + step, theta_max);
301 Real dG = (jet.Gamma0(phi, th_hi) - jet.Gamma0(phi, th_lo)) / (th_hi - th_lo);
302 // Real drho = (medium.rho(phi, th_hi, r0) - medium.rho(phi, th_lo, r0)) / (th_hi - th_lo);
303 Real dp = dG; //(2 * G - 1) * rho * dG + (G - 1) * G * drho;
304
305 if (dp < dp_min) {
306 dp_min = dp;
307 theta_s = theta;
308 }
309 }
310 if (dp_min == 0) {
311 theta_s = theta_max;
312 }
313
314 return theta_s;
315}

317template <typename Func>
318Array inverse_CFD_sampling(Func&& pdf, Real min, Real max, size_t num) {
319 using namespace boost::numeric::odeint;
320 constexpr Real rtol = 1e-6;
321 constexpr size_t sample_num = 200;
322 Array x_i = xt::linspace(min, max, sample_num);
323 Array CDF_i = xt::zeros<Real>({sample_num});
324
325 auto stepper = make_dense_output(rtol, rtol, runge_kutta_dopri5<Real>());
326 stepper.initialize(0, min, (max - min) / 1e3);
327
328 for (size_t k = 1; stepper.current_time() <= max;) {
329 stepper.do_step(pdf);
330 while (k < x_i.size() && stepper.current_time() > x_i(k)) {
331 stepper.calc_state(x_i(k), CDF_i(k));
332 ++k;
333 }
334 }
335
336 Array CDF_out = xt::linspace(CDF_i.front(), CDF_i.back(), num);
337 Array x_out = xt::zeros<Real>({num});
338
339 for (size_t k = 0; k < num; ++k) {
340 for (size_t j = 0; j < sample_num; ++j) {
341 if (CDF_out(k) <= CDF_i(j)) {
342 if (j == 0) {
343 x_out(k) = x_i(j);
344 } else {
345 Real slope = (x_i(j) - x_i(j - 1)) / (CDF_i(j) - CDF_i(j - 1));
346 x_out(k) = x_i(j - 1) + slope * (CDF_out(k) - CDF_i(j - 1));
347 }
348 break;
349 }
350 }
351 }
352 return x_out;
353}

355template <typename Ejecta>
356Array adaptive_theta_grid(Ejecta const& jet, Real theta_min, Real theta_max, size_t theta_num, Real theta_v) {
357 auto eqn = [=, &jet](Real const& cdf, Real& pdf, Real theta) {
358 Real Gamma = jet.Gamma0(0, theta);
359 Real beta = std::sqrt(std::fabs(Gamma * Gamma - 1)) / Gamma;
360 // Real D = 1 / (Gamma * (1 - beta * std::cos(theta - theta_v)));
361 Real a = (1 - beta) / (1 - beta * std::cos(theta - theta_v));
362 pdf = a * Gamma * std::sqrt((Gamma - 1) * Gamma) * std::sin(theta);
363 // pdf = D * std::sin(theta);
364 };
365
366 return inverse_CFD_sampling(eqn, theta_min, theta_max, theta_num);
367}

369template <typename Ejecta>
370Array adaptive_phi_grid(Ejecta const& jet, size_t phi_num, Real theta_v, Real theta_max, bool is_axisymmetric) {
371 if ((theta_v == 0 && is_axisymmetric) || theta_v > theta_max) {
372 return xt::linspace(0., 2 * con::pi, phi_num);
373 } else {
374 Real cos2 = std::cos(theta_v);
375 cos2 = cos2 * cos2;
376 Real sin2 = 1 - cos2;
377
378 auto eqn = [=, &jet](Real const& cdf, Real& pdf, Real phi) {
379 Real Gamma = jet.Gamma0(phi, theta_v);
380 Real beta = std::sqrt(std::fabs(Gamma * Gamma - 1)) / Gamma;
381 // Real D = 1 / (Gamma * (1 - beta * (std::cos(phi) * sin2 + cos2)));
382 Real a = (1 - beta) / (1 - beta * (std::cos(phi) * sin2 + cos2));
383 pdf = a * Gamma * std::sqrt((Gamma - 1) * Gamma);
384 // pdf = D * std::sqrt((Gamma - 1) * Gamma);
385 };
386
387 return inverse_CFD_sampling(eqn, 0, 2 * con::pi, phi_num);
388 }

391template <typename Arr>
392Array merge_grids(Arr const& arr1, Arr const& arr2) {
393 std::vector<Real> result;
394 result.reserve(arr1.size() + arr2.size());
395
396 size_t i = 0, j = 0;
397 auto add_unique = [&](Real val) {
398 if (result.empty() || result.back() != val) result.push_back(val);
399 };
400
401 while (i < arr1.size() && j < arr2.size()) {
402 if (arr1[i] <= arr2[j]) {
403 add_unique(arr1[i++]);
404 if (arr1[i - 1] == arr2[j]) j++;
405 } else {
406 add_unique(arr2[j++]);
407 }
408 }
409 while (i < arr1.size()) add_unique(arr1[i++]);
410 while (j < arr2.size()) add_unique(arr2[j++]);
411
412 return xt::adapt(result);
413}

415template <typename Ejecta>
416Coord auto_grid(Ejecta const& jet, Array const& t_obs, Real theta_cut, Real theta_view, Real z, Real phi_resol,
417 Real theta_resol, Real t_resol, bool is_axisymmetric, Real phi_view) {
418 Coord coord;
419 coord.theta_view = theta_view;
420
421 Real jet_edge = find_jet_edge(jet, con::Gamma_cut, phi_resol, theta_resol, is_axisymmetric);
422 Real theta_min = 1e-6;
423 Real theta_max = std::min(jet_edge, theta_cut);
424
425 size_t theta_num = std::max<size_t>(static_cast<size_t>((theta_max - theta_min) * 180 / con::pi * theta_resol), 56);
426 size_t uniform_theta_num = static_cast<size_t>(theta_num * 0.3);
427 size_t adaptive_theta_num = theta_num - uniform_theta_num;
428
429 Array uniform_theta = xt::linspace(theta_min, theta_max, uniform_theta_num);
430 Array adaptive_theta = adaptive_theta_grid(jet, theta_min, theta_max, adaptive_theta_num, theta_view);
431 coord.theta = merge_grids(uniform_theta, adaptive_theta);
432
433 // coord.theta = uniform_theta;
434
435 size_t phi_num = std::max<size_t>(static_cast<size_t>(360 * phi_resol), 1);
436 size_t uniform_phi_num = static_cast<size_t>(phi_num * 0.3);
437 size_t adaptive_phi_num = phi_num - uniform_phi_num;
438
439 Array uniform_phi = xt::linspace(0., 2 * con::pi, uniform_phi_num);
440 Array adaptive_phi = adaptive_phi_grid(jet, adaptive_phi_num, theta_view, theta_max, is_axisymmetric);
441 coord.phi = merge_grids(uniform_phi, adaptive_phi);
442 // coord.phi = uniform_phi;
443
444 Real t_max = *std::max_element(t_obs.begin(), t_obs.end());
445 Real t_min = *std::min_element(t_obs.begin(), t_obs.end());
446 size_t t_num = std::max<size_t>(static_cast<size_t>(std::log10(t_max / t_min) * t_resol), 24);
447
448 size_t phi_size_needed = is_axisymmetric ? 1 : phi_num;
449 coord.t = xt::zeros<Real>({phi_size_needed, theta_num, t_num});
450 for (size_t i = 0; i < phi_size_needed; ++i) {
451 for (size_t j = 0; j < theta_num; ++j) {
452 Real b = gamma_to_beta(jet.Gamma0(coord.phi(i), coord.theta(j)));
453 // Real theta_max = coord.theta(j) + theta_view;
454 Real theta_max = coord.theta.back() + theta_view;
455 Real t_start =
456 std::max<Real>(0.99 * t_min * (1 - b) / (1 - std::cos(theta_max) * b) / (1 + z), 1e-2 * unit::sec);
457 Real t_end = 1.01 * t_max / (1 + z);
458 xt::view(coord.t, i, j, xt::all()) = xt::logspace(std::log10(t_start), std::log10(t_end), t_num);
459 }
460 }
461
462 return coord; // Construct coordinate object.
