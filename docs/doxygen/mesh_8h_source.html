149//========================================================================================================
150// template function implementation
151//========================================================================================================
152template <typename Arr1, typename Arr2>
153void boundary_to_center(Arr1 const& boundary, Arr2& center) {
154 for (size_t i = 0; i < center.size(); ++i) {
155 center[i] = 0.5 * (boundary[i] + boundary[i + 1]);
156 }
157}

159template <typename Arr1, typename Arr2>
160void boundary_to_center_log(Arr1 const& boundary, Arr2& center) {
161 for (size_t i = 0; i < center.size(); ++i) {
162 center[i] = std::sqrt(boundary[i] * boundary[i + 1]);
163 }
