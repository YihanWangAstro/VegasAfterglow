201void boundary_to_center(Arr1 const& boundary, Arr2& center) {
202 for (size_t i = 0; i < center.size(); ++i) {
203 center[i] = 0.5 * (boundary[i] + boundary[i + 1]);
204 }

208void boundary_to_center_log(Arr1 const& boundary, Arr2& center) {

210 center[i] = std::sqrt(boundary[i] * boundary[i + 1]);

214template <typename Arr>
215void logspace_boundary_center(Real lg2_min, Real lg2_max, size_t size, Arr& center, Array& bin_width) {
216 center = Arr::from_shape({size});
217 bin_width = Array::from_shape({size});
218 if (size == 0)
219 return;
220
221 const Real dlg2 = (lg2_max - lg2_min) / static_cast<Real>(size);
222
223 const Real r = std::exp2(dlg2);
224 const Real s = std::sqrt(r);
225 const Real w = r - 1.;
226
227 Real left = std::exp2(lg2_min);
228
229 for (std::size_t i = 0; i < size; ++i) {
230 center(i) = left * s;
231 bin_width(i) = left * w;
232 left *= r;
233 }
234}

236template <typename Ejecta>
237std::vector<Real> find_jet_edges(Ejecta const& jet, Real gamma_cut, [[maybe_unused]] bool is_axisymmetric) {
238 // Find all edges in the active jet region (Gamma0 >= gamma_cut).
239 // Sharp edges: large fractional jumps (> 50%) between adjacent scan points.
240 // Smooth jets: the outermost theta where Gamma0 crosses gamma_cut.
241 constexpr size_t n_scan = 512;
242 constexpr Real eps = defaults::solver::binary_search_eps;
243 const Real theta_lo = defaults::grid::theta_min;
244 const Real theta_hi = con::pi / 2;
245
246 if (jet.Gamma0(0, theta_hi) >= gamma_cut) {
247 return {theta_hi};
248 }
249
250 std::vector<Real> edges;
251 Real prev_th = theta_lo;
252 Real prev_G = jet.Gamma0(0, theta_lo);
253
254 for (size_t j = 1; j < n_scan; ++j) {
255 const Real cur_th = theta_lo + (theta_hi - theta_lo) * static_cast<Real>(j) / (n_scan - 1);
256 const Real cur_G = jet.Gamma0(0, cur_th);
257
258 if (prev_G >= gamma_cut || cur_G >= gamma_cut) {
259 const Real dG = std::abs(cur_G - prev_G);
260 const Real scale = std::max(prev_G - 1, cur_G - 1);
261 if (scale > 0 && dG > 0.5 * scale) {
262 Real lo = prev_th, hi = cur_th;
263 while (hi - lo > eps) {
264 const Real mid = 0.5 * (lo + hi);
265 const Real G_mid = jet.Gamma0(0, mid);
266 if (std::abs(G_mid - prev_G) < std::abs(G_mid - cur_G)) {
267 lo = mid;
268 } else {
269 hi = mid;
270 }
271 }
272 edges.push_back(lo);
273 }
274 }
275 prev_th = cur_th;
276 prev_G = cur_G;
277 }
278
279 // Smooth jet: no sharp edges found scan downward from pi/2 to find outermost Gamma0 >= gamma_cut
280 if (edges.empty()) {
281 const Real step = (theta_hi - theta_lo) / n_scan;
282 for (Real th = theta_hi; th >= theta_lo; th -= step) {
283 if (jet.Gamma0(0, th) >= gamma_cut) {
284 edges.push_back(th);
285 break;
286 }
287 }
288 }
289
290 return edges;
291}

293template <typename Ejecta, typename Medium>
294Real jet_spreading_edge(Ejecta const& jet, Medium const& medium, Real phi, Real theta_min, Real theta_max, Real t0) {
295 const Real step = (theta_max - theta_min) / 256;
296 Real theta_s = theta_min;
297 Real dp_min = 0;
298
299 for (Real theta = theta_min; theta <= theta_max; theta += step) {
300 // Real G = jet.Gamma0(phi, theta);
301 // Real beta0 = physics::relativistic::gamma_to_beta(G);
302 // Real r0 = beta0 * con::c * t0 / (1 - beta0);
303 // Real rho = medium.rho(phi, theta, 0);
304 Real th_lo = std::max(theta - step, theta_min);
305 Real th_hi = std::min(theta + step, theta_max);
306 const Real dG = (jet.Gamma0(phi, th_hi) - jet.Gamma0(phi, th_lo)) / (th_hi - th_lo);
307 // Real drho = (medium.rho(phi, th_hi, r0) - medium.rho(phi, th_lo, r0)) / (th_hi - th_lo);
308 const Real dp = dG; //(2 * G - 1) * rho * dG + (G - 1) * G * drho;
309
310 if (dp < dp_min) {
311 dp_min = dp;
312 theta_s = theta;
313 }
314 }
315 if (dp_min == 0) {
316 theta_s = theta_max;
317 }
318
319 return theta_s;
320}

322template <typename Func>
323Array inverse_CFD_sampling(Func&& pdf, Real min, Real max, size_t num) {
324 using namespace boost::numeric::odeint;
325 constexpr Real rtol = defaults::solver::ode_rtol;
326 constexpr size_t sample_num = defaults::sampling::theta_samples;
327 Array x_i = xt::linspace(min, max, sample_num);
328 Array CDF_i = xt::zeros<Real>({sample_num});
329
330 auto stepper = make_dense_output(rtol, rtol, runge_kutta_dopri5<Real>());
331 stepper.initialize(0, min, (max - min) / 1e3);
332
333 for (size_t k = 1; stepper.current_time() <= max;) {
334 stepper.do_step(pdf);
335 while (k < x_i.size() && stepper.current_time() > x_i(k)) {
336 stepper.calc_state(x_i(k), CDF_i(k));
337 ++k;
338 }
339 }
340
341 Array CDF_out = xt::linspace(CDF_i.front(), CDF_i.back(), num);
342 Array x_out = xt::zeros<Real>({num});
343
344 for (size_t k = 0; k < num; ++k) {
345 for (size_t j = 0; j < sample_num; ++j) {
346 if (CDF_out(k) <= CDF_i(j)) {
347 if (j == 0) {
348 x_out(k) = x_i(j);
349 } else {
350 Real slope = (x_i(j) - x_i(j - 1)) / (CDF_i(j) - CDF_i(j - 1));
351 x_out(k) = x_i(j - 1) + slope * (CDF_out(k) - CDF_i(j - 1));
352 }
353 break;
354 }
355 }
356 }
357 return x_out;
358}

360template <typename Ejecta>
361Array adaptive_theta_grid(Ejecta const& jet, Real theta_min, Real theta_max, size_t theta_num, Real theta_v) {
362 auto eqn = [=, &jet](Real const& cdf, Real& pdf, Real theta) {
363 const Real Gamma = jet.Gamma0(0, theta);
364 const Real beta = std::sqrt(std::fabs(Gamma * Gamma - 1)) / Gamma;
365 // Real D = 1 / (Gamma * (1 - beta * std::cos(theta - theta_v)));
366 const Real a = (1 - beta) / (1 - beta * std::cos(theta - theta_v));
367 pdf = a * Gamma * std::sqrt((Gamma - 1) * Gamma) * std::sin(theta);
368 // pdf = D * std::sin(theta);
369 };
370
371 return inverse_CFD_sampling(eqn, theta_min, theta_max, theta_num);
372}

374template <typename Ejecta>
375Array adaptive_phi_grid(Ejecta const& jet, size_t phi_num, Real theta_v, Real theta_max, bool is_axisymmetric) {
376 if ((theta_v == 0 && is_axisymmetric) || theta_v > theta_max) {
377 return xt::linspace(0., 2 * con::pi, phi_num);
378 } else {
379 const Real cos_tv = std::cos(theta_v);
380 const Real sin_tv = std::sin(theta_v);
381
382 // Sample a few theta values to find the peak Doppler weight at each phi
383 constexpr size_t n_theta_sample = 8;
384 const Real theta_min = defaults::grid::theta_min;
385
386 auto eqn = [=, &jet](Real const& cdf, Real& pdf, Real phi) {
387 const Real cos_phi = std::cos(phi);
388 Real max_weight = 0;
389 for (size_t it = 0; it < n_theta_sample; ++it) {
390 const Real theta = theta_min + (theta_max - theta_min) * static_cast<Real>(it) / (n_theta_sample - 1);
391 const Real Gamma = jet.Gamma0(phi, theta);
392 const Real beta = std::sqrt(std::fabs(Gamma * Gamma - 1)) / Gamma;
393 // cos(angle to LOS) = cos(theta)cos(theta_v) + sin(theta)sin(theta_v)cos(phi)
394 const Real cos_alpha = std::cos(theta) * cos_tv + std::sin(theta) * sin_tv * cos_phi;
395 const Real a = (1 - beta) / (1 - beta * cos_alpha);
396 const Real weight = a * Gamma * std::sqrt((Gamma - 1) * Gamma) * std::sin(theta);
397 max_weight = std::max(max_weight, weight);
398 }
399 pdf = max_weight;
400 };
401
402 return inverse_CFD_sampling(eqn, 0, 2 * con::pi, phi_num);
403 }

406template <typename Arr>
407Array merge_grids(Arr const& arr1, Arr const& arr2) {
408 std::vector<Real> result;
409 result.reserve(arr1.size() + arr2.size());
410
411 size_t i = 0, j = 0;
412 auto add_unique = [&](Real val) {
413 if (result.empty() || result.back() != val)
414 result.push_back(val);
415 };
416
417 while (i < arr1.size() && j < arr2.size()) {
418 if (arr1[i] <= arr2[j]) {
419 add_unique(arr1[i++]);
420 if (arr1[i - 1] == arr2[j])
421 j++;
422 } else {
423 add_unique(arr2[j++]);
424 }
425 }
426 while (i < arr1.size())
427 add_unique(arr1[i++]);
428 while (j < arr2.size())
429 add_unique(arr2[j++]);
430
431 return xt::adapt(result);
432}

434/**
435 * @brief Estimates deceleration time for a given angle from ejecta and medium properties.
436 * @details Integrates swept mass outward until m_swept = m_jet/Gamma0. Returns lab-frame time.
437 */
438template <typename Ejecta, typename Medium>
439Real estimate_t_dec(Ejecta const& jet, Medium const& medium, Real phi, Real theta) {
440 const Real gamma = jet.Gamma0(phi, theta);
441 if (gamma <= 1) {
442 return con::inf;
443 }
444 const Real beta = physics::relativistic::gamma_to_beta(gamma);
445 const Real m_jet = jet.eps_k(phi, theta) / (gamma * con::c2);
446 const Real target = m_jet / gamma;
447
448 constexpr size_t N = 256;
449 const Real u_min = std::log(1e-3);
450 const Real u_max = u_min + 40 * std::log(10.0);
451 const Real du = (u_max - u_min) / N;
452
453 Real mass = 0;
454 Real r_prev = std::exp(u_min);
455 Real f_prev = medium.rho(phi, theta, r_prev) * r_prev * r_prev;
456
457 for (size_t i = 1; i <= N; ++i) {
458 const Real r_i = std::exp(u_min + i * du);
459 const Real f_i = medium.rho(phi, theta, r_i) * r_i * r_i;
460 const Real dr = r_i - r_prev;
461 mass += 0.5 * (f_prev + f_i) * dr;
462
463 if (mass >= target) {
464 const Real r_dec = r_prev + (target - (mass - 0.5 * (f_prev + f_i) * dr)) / f_i;
465 return r_dec * (1 - beta) / (beta * con::c);
466 }
467 f_prev = f_i;
468 r_prev = r_i;
469 }
470 return std::exp(u_max) * (1 - beta) / (beta * con::c);
471}

473/**
474 * @brief Create a refined logarithmic time grid with higher density around a specified time.
475 *
476 * Constructs a three-segment logspace grid:
477 * - Segment 1: [t_start, t_lo] with base resolution
478 * - Segment 2: [t_lo, t_hi] with higher resolution (around t_refine)
479 * - Segment 3: [t_hi, t_end] with base resolution
480 *
481 * Falls back to simple logspace if refinement is not applicable.
482 *
483 * @param t_start Start time of the grid
484 * @param t_end End time of the grid
485 * @param t_refine Time around which to refine (e.g., deceleration time)
486 * @param t_num Total number of grid points
487 * @param t_resol Resolution parameter (points per decade)
488 * @param refine_lo_factor Lower bound factor for refinement region (t_lo = refine_lo_factor * t_refine)
489 * @param refine_hi_factor Upper bound factor for refinement region (t_hi = refine_hi_factor * t_refine)
490 * @return Array of time grid points
491 */
492inline Array refined_time_grid(Real t_start, Real t_end, Real t_refine, size_t t_num, Real t_resol,
493 Real refine_lo_factor = 0.3, Real refine_hi_factor = 10.0) {
494 Array result = xt::zeros<Real>({t_num});
495
496 const Real log_s = std::log10(t_start);
497 const Real log_e = std::log10(t_end);
498
499 if (t_refine <= t_start || t_refine >= t_end) {
500 return xt::logspace(log_s, log_e, t_num);
501 }
502
503 const Real t_lo = std::max(refine_lo_factor * t_refine, t_start);
504 const Real t_hi = std::min(refine_hi_factor * t_refine, t_end);
505 const Real log_lo = std::log10(t_lo);
506 const Real log_hi = std::log10(t_hi);
507 const Real log_before = log_lo - log_s;
508 const Real log_after = log_e - log_hi;
509
510 size_t n1 = (log_before > 0) ? std::max<size_t>(static_cast<size_t>(t_resol * log_before), 2) : 0;
511 size_t n3 = (log_after > 0) ? std::max<size_t>(static_cast<size_t>(t_resol * log_after), 2) : 0;
512 size_t shared = (n1 > 0 ? 1 : 0) + (n3 > 0 ? 1 : 0);
513 size_t n2_raw = (n1 + n3 + 2 <= t_num + shared) ? (t_num + shared - n1 - n3) : 0;
514
515 if (n2_raw < 2) {
516 return xt::logspace(log_s, log_e, t_num);
517 }
518
519 size_t n2 = n2_raw;
520 size_t idx = 0;
521
522 // Segment 1: [t_start, t_lo]
523 if (n1 > 0) {
524 auto seg1 = xt::logspace(log_s, log_lo, n1);
525 for (size_t k = 0; k < n1; ++k)
526 result(idx++) = seg1(k);
527 }
528
529 // Segment 2: [t_lo, t_hi] - skip first point if it overlaps with seg1
530 {
531 auto seg2 = xt::logspace(log_lo, log_hi, n2);
532 size_t k0 = (n1 > 0) ? 1 : 0;
533 for (size_t k = k0; k < n2; ++k)
534 result(idx++) = seg2(k);
535 }
536
537 // Segment 3: [t_hi, t_end] - skip first point (overlaps with seg2)
538 if (n3 > 0) {
539 auto seg3 = xt::logspace(log_hi, log_e, n3);
540 for (size_t k = 1; k < n3; ++k)
541 result(idx++) = seg3(k);
542 }
543
544 return result;
545}

547template <typename Ejecta, typename Medium>
548Coord auto_grid(Ejecta const& jet, Medium const& medium, Array const& t_obs, Real theta_cut, Real theta_view, Real z,
549 Real phi_resol, Real theta_resol, Real t_resol, bool is_axisymmetric, Real phi_view,
550 size_t min_theta_num, Real fwd_ratio) {
551 Coord coord;
552 coord.theta_view = theta_view;
553
554 const auto jet_edges = find_jet_edges(jet, con::Gamma_cut, is_axisymmetric);
555 const Real jet_edge = jet_edges.back(); // outermost edge
556 Real theta_min = defaults::grid::theta_min;
557 Real theta_max = std::min(jet_edge, theta_cut);
558
559 size_t theta_num =
560 std::max<size_t>(static_cast<size_t>((theta_max - theta_min) * 180 / con::pi * theta_resol), min_theta_num);
561 const size_t uniform_theta_num = static_cast<size_t>(static_cast<Real>(theta_num) * fwd_ratio);
562 size_t adaptive_theta_num = theta_num - uniform_theta_num;
563
564 const Array uniform_theta = xt::linspace(theta_min, theta_max, uniform_theta_num);
565 const Array adaptive_theta = adaptive_theta_grid(jet, theta_min, theta_max, adaptive_theta_num, theta_view);
566 const Array anchor_theta = xt::adapt(jet_edges);
567 coord.theta = merge_grids(merge_grids(uniform_theta, adaptive_theta), anchor_theta);
568

571
572 coord.phi = adaptive_phi_grid(jet, phi_num, theta_view, theta_max, is_axisymmetric);
573
574 const Real t_max = *std::ranges::max_element(t_obs);
575 const Real t_min = *std::ranges::min_element(t_obs);
576
577 // Compute base time grid size + extra points for refinement around t_dec
578 const size_t base_t_num = std::max<size_t>(static_cast<size_t>(std::log10(t_max / t_min) * t_resol), 24);
579 constexpr Real refine_ratio = 2.0;
580 constexpr Real refine_lo_factor = 0.3;
581 constexpr Real refine_hi_factor = 10.0;
582 const size_t refine_extra =
583 static_cast<size_t>((refine_ratio - 1.0) * t_resol * std::log10(refine_hi_factor / refine_lo_factor));
584 const size_t t_num = base_t_num + refine_extra;
585
586 const size_t theta_size = coord.theta.size();
587 const size_t phi_size_needed = is_axisymmetric ? 1 : coord.phi.size();
588 const Real theta_v_max = coord.theta.back() + theta_view;
589
590 coord.t = xt::zeros<Real>({phi_size_needed, theta_size, t_num});
591 for (size_t i = 0; i < phi_size_needed; ++i) {
592 for (size_t j = 0; j < theta_size; ++j) {
593 const Real b = physics::relativistic::gamma_to_beta(jet.Gamma0(coord.phi(i), coord.theta(j)));
594 const Real t_start =
595 std::max<Real>(0.99 * t_min * (1 - b) / (1 - std::cos(theta_v_max) * b) / (1 + z), 1e-2 * unit::sec);
596 const Real t_end = 1.01 * t_max / (1 + z);
597 const Real t_dec = estimate_t_dec(jet, medium, coord.phi(i), coord.theta(j));
598
599 xt::view(coord.t, i, j, xt::all()) =
600 refined_time_grid(t_start, t_end, t_dec, t_num, t_resol, refine_lo_factor, refine_hi_factor);
601 }
602 }
