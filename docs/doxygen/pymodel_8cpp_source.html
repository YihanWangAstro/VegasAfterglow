12Ejecta PyTophatJet(Real theta_c, Real E_iso, Real Gamma0, bool spreading, Real T0, std::optional<PyMagnetar> magnetar) {
13 Ejecta jet;
14 jet.eps_k = math::tophat(theta_c, E_iso * unit::erg / (4 * con::pi));
15 jet.Gamma0 = math::tophat(theta_c, Gamma0);
16 jet.spreading = spreading;
17 jet.T0 = T0 * unit::sec;
18
19 if (magnetar) {
20 jet.deps_dt = [=](Real phi, Real theta, Real t) {
21 if (theta <= theta_c) {
22 Real tt = 1 + t / magnetar->t_0;
23 return magnetar->L_0 / (tt * tt);
24 } else {
25 return 0.;
26 }
27 };
28 }
29
30 return jet;
31}

89void PyModel::single_shock_emission(Shock const& shock, Coord const& coord, Array const& t_obs, Array const& nu_obs,
90 Observer& obs, PyRadiation rad, FluxDict& flux_dict, std::string suffix) {
91 obs.observe(coord, shock, obs_setup.lumi_dist, obs_setup.z);
92
93 auto syn_e = generate_syn_electrons(shock);
94
95 auto syn_ph = generate_syn_photons(shock, syn_e);
96
97 if (rad.IC_cooling) {
98 if (rad.KN) {
99 KN_cooling(syn_e, syn_ph, shock);
100 } else {
101 Thomson_cooling(syn_e, syn_ph, shock);
102 }
103 }
104
105 if (rad.SSC) {
106 auto IC_ph = generate_IC_photons(syn_e, syn_ph, rad.KN);
107
108 flux_dict["IC" + suffix] = obs.specific_flux(t_obs, nu_obs, IC_ph) / unit::flux_den_cgs;
109 }
110
111 flux_dict["syn" + suffix] = obs.specific_flux(t_obs, nu_obs, syn_ph) / unit::flux_den_cgs;
112}

114auto PyModel::compute_specific_flux(Array const& t_obs, Array const& nu_obs) -> FluxDict {
115 Coord coord = auto_grid(jet, t_obs, this->theta_w, obs_setup.theta_obs, obs_setup.z, phi_resol, theta_resol,
116 t_resol, axisymmetric);
117
118 FluxDict flux_dict;
119
120 Observer observer;
121
122 if (!rvs_rad_opt) {
123 auto fwd_shock = generate_fwd_shock(coord, medium, jet, fwd_rad.rad, rtol);
124
125 single_shock_emission(fwd_shock, coord, t_obs, nu_obs, observer, fwd_rad, flux_dict, "");
