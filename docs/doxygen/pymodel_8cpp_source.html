122void convert_unit(Ejecta& jet, Medium& medium) {
123 auto eps_k_cgs = jet.eps_k;
124 jet.eps_k = [=](Real phi, Real theta) { return eps_k_cgs(phi, theta) * (unit::erg / (4 * con::pi)); };
125
126 auto deps_dt_cgs = jet.deps_dt;
127 jet.deps_dt = [=](Real phi, Real theta, Real t) {
128 return deps_dt_cgs(phi, theta, t / unit::sec) * (unit::erg / (4 * con::pi * unit::sec));
129 };
130
131 auto dm_dt_cgs = jet.dm_dt;
132 jet.dm_dt = [=](Real phi, Real theta, Real t) {
133 return dm_dt_cgs(phi, theta, t / unit::sec) * (unit::g / (4 * con::pi * unit::sec));
134 };
135
136 jet.T0 *= unit::sec;
137
138 auto rho_cgs = medium.rho; // number density from python side
139 medium.rho = [=](Real phi, Real theta, Real r) {
140 return rho_cgs(phi, theta, r / unit::cm) * (unit::g / unit::cm3); // convert to density
141 };

154template <typename ElectronGrid>
155void save_electron_details(ElectronGrid const& electrons, PyShock& details) {
156 auto shape = electrons.shape();
157
158 details.gamma_m = xt::zeros<Real>({shape[0], shape[1], shape[2]});
159 details.gamma_c = xt::zeros<Real>({shape[0], shape[1], shape[2]});
160 details.gamma_a = xt::zeros<Real>({shape[0], shape[1], shape[2]});
161 details.gamma_M = xt::zeros<Real>({shape[0], shape[1], shape[2]});
162 details.N_e = xt::zeros<Real>({shape[0], shape[1], shape[2]});
163
164 for (size_t i = 0; i < shape[0]; ++i) {
165 for (size_t j = 0; j < shape[1]; ++j) {
166 for (size_t k = 0; k < shape[2]; ++k) {
167 details.gamma_a(i, j, k) = electrons(i, j, k).gamma_a;
168 details.gamma_m(i, j, k) = electrons(i, j, k).gamma_m;
169 details.gamma_c(i, j, k) = electrons(i, j, k).gamma_c;
170 details.gamma_M(i, j, k) = electrons(i, j, k).gamma_M;
171 details.N_e(i, j, k) = electrons(i, j, k).N_e;
172 }
173 }
174 }

226
227 PyDetails details;

310auto PyModel::generate_exposure_sampling(PyArray const& t, PyArray const& nu, PyArray const& expo_time,
311 size_t num_points) -> ExposureSampling {
312 size_t total_points = t.size() * num_points;
313 Array t_obs = Array::from_shape({total_points});
314 Array nu_obs = Array::from_shape({total_points});
315 std::vector<size_t> idx(total_points);
316
317 // Generate time-frequency samples within each exposure window
318 for (size_t i = 0, j = 0; i < t.size() && j < total_points; ++i) {
319 Real t_start = t(i);
320 Real dt = expo_time(i) / (num_points - 1);
321
322 for (size_t k = 0; k < num_points && j < total_points; ++k, ++j) {
323 t_obs(j) = t_start + k * dt;
324 nu_obs(j) = nu(i);
325 idx[j] = i;
326 }
327 }
328
329 std::vector<size_t> sort_indices(total_points);
330 std::iota(sort_indices.begin(), sort_indices.end(), 0);
331 std::sort(sort_indices.begin(), sort_indices.end(), [&t_obs](size_t i, size_t j) { return t_obs(i) < t_obs(j); });
332
333 Array t_obs_sorted = Array::from_shape({total_points});
334 Array nu_obs_sorted = Array::from_shape({total_points});
335 std::vector<size_t> idx_sorted(idx.size());
336
337 for (size_t i = 0; i < sort_indices.size(); ++i) {
338 size_t orig_idx = sort_indices[i];
339 t_obs_sorted(i) = t_obs(orig_idx);
340 nu_obs_sorted(i) = nu_obs(orig_idx);
341 idx_sorted[i] = idx[orig_idx];
342 }
343
344 t_obs_sorted *= unit::sec;
345 nu_obs_sorted *= unit::Hz;
346
347 return {std::move(t_obs_sorted), std::move(nu_obs_sorted), std::move(idx_sorted)};
348}

350void PyModel::average_exposure_flux(PyFlux& result, std::vector<size_t> const& idx_sorted, size_t original_size,
351 size_t num_points) {
352 auto average_component = [&](XTArray& component) {
353 if (component.size() > 0) {
354 Array summed = xt::zeros<Real>({original_size});
355 for (size_t j = 0; j < component.size(); j++) {
356 size_t orig_time_idx = idx_sorted[j];
357 summed(orig_time_idx) += component(j);
358 }
359 summed /= static_cast<Real>(num_points);
360 component = std::move(summed);
361 }
362 };
363
364 average_component(result.fwd.sync);
365 average_component(result.fwd.ssc);
366 average_component(result.rvs.sync);
367 average_component(result.rvs.ssc);
368}

370auto PyModel::flux_density_exposures(PyArray const& t, PyArray const& nu, PyArray const& expo_time, size_t num_points)
371 -> PyFlux {
372 AFTERGLOW_REQUIRE(t.size() == nu.size() && t.size() == expo_time.size(),
373 "time, frequency, and exposure time arrays must have the same size");
374 AFTERGLOW_REQUIRE(num_points >= 2, "num_points must be at least 2 to sample within each exposure time");
375
376 auto sampling = generate_exposure_sampling(t, nu, expo_time, num_points);
377
378 auto flux_func = [](Observer& obs, Array const& t, Array const& nu, auto& photons) -> XTArray {
379 return obs.specific_flux_series(t, nu, photons) / unit::flux_den_cgs;
380 };
381
382 auto result = compute_emission(sampling.t_obs_sorted, sampling.nu_obs_sorted, flux_func);
383
384 average_exposure_flux(result, sampling.idx_sorted, t.size(), num_points);
385
386 result.calc_total();
387 return result;
388}
