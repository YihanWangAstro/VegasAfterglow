88Medium PyWind(Real A_star, Real n_ism, Real n0, Real k) {
89 Medium medium;
90
91 constexpr Real r0 = 1e17; // reference radius
92 const Real A = A_star * 5e11 * std::pow(r0, k - 2);
93
94 if (k >= 0) {
95 const Real rho_ism = n_ism * 1.67e-24;
96 const Real r0k = A / (n0 * 1.67e-24);
97 medium.rho = [=](Real phi, Real theta, Real r) { return A / (r0k + std::pow(r, k)) + rho_ism; };
98 } else {
99 const Real rho0 = n0 * 1.67e-24;
100 const Real r_ism = A / (n_ism * 1.67e-24);
101 medium.rho = [=](Real phi, Real theta, Real r) { return A / (r_ism + std::pow(r, k)) + rho0; };
102 }
103 return medium;
104}

106void convert_unit(Ejecta& jet, Medium& medium) {
107 const auto eps_k_cgs = jet.eps_k;
108 jet.eps_k = [=](Real phi, Real theta) { return eps_k_cgs(phi, theta) * (unit::erg / (4 * con::pi)); };
109
110 const auto deps_dt_cgs = jet.deps_dt;
111 jet.deps_dt = [=](Real phi, Real theta, Real t) {
112 return deps_dt_cgs(phi, theta, t / unit::sec) * (unit::erg / (4 * con::pi * unit::sec));
113 };
114
115 const auto dm_dt_cgs = jet.dm_dt;
116 jet.dm_dt = [=](Real phi, Real theta, Real t) {
117 return dm_dt_cgs(phi, theta, t / unit::sec) * (unit::g / (4 * con::pi * unit::sec));
118 };
119
120 jet.T0 *= unit::sec;
121
122 const auto rho_cgs = medium.rho; // number density from python side
123 medium.rho = [=](Real phi, Real theta, Real r) {
124 return rho_cgs(phi, theta, r / unit::cm) * (unit::g / unit::cm3); // convert to density
125 };

139void save_electron_details(ElectronGrid const& electrons, PyShock& details) {
140 auto shape = electrons.shape();
141
142 details.gamma_m = xt::zeros<Real>({shape[0], shape[1], shape[2]});
143 details.gamma_c = xt::zeros<Real>({shape[0], shape[1], shape[2]});
144 details.gamma_a = xt::zeros<Real>({shape[0], shape[1], shape[2]});
145 details.gamma_M = xt::zeros<Real>({shape[0], shape[1], shape[2]});
146 details.gamma_m_hat = xt::zeros<Real>({shape[0], shape[1], shape[2]});
147 details.gamma_c_hat = xt::zeros<Real>({shape[0], shape[1], shape[2]});
148 details.N_e = xt::zeros<Real>({shape[0], shape[1], shape[2]});
149
150 for (size_t i = 0; i < shape[0]; ++i) {
151 for (size_t j = 0; j < shape[1]; ++j) {
152 for (size_t k = 0; k < shape[2]; ++k) {
153 details.gamma_a(i, j, k) = electrons(i, j, k).gamma_a;
154 details.gamma_m(i, j, k) = electrons(i, j, k).gamma_m;
155 details.gamma_c(i, j, k) = electrons(i, j, k).gamma_c;
156 details.gamma_M(i, j, k) = electrons(i, j, k).gamma_M;
157 details.gamma_m_hat(i, j, k) = electrons(i, j, k).Ys.gamma_m_hat;
158 details.gamma_c_hat(i, j, k) = electrons(i, j, k).Ys.gamma_c_hat;
159 details.N_e(i, j, k) = electrons(i, j, k).N_e;
160 }
161 }

198 obs.observe(coord, shock, obs_setup.lumi_dist, obs_setup.z);

222
223 PyDetails details;

249 single_evo_details(rvs_shock, coord, observer, rvs_rad, details.rvs);

306auto PyModel::generate_exposure_sampling(PyArray const& t, PyArray const& nu, PyArray const& expo_time,
307 size_t num_points) -> ExposureSampling {
308 const size_t total_points = t.size() * num_points;
309 Array t_obs = Array::from_shape({total_points});
310 Array nu_obs = Array::from_shape({total_points});
311 std::vector<size_t> idx(total_points);
312
313 // Generate time-frequency samples within each exposure window
314 for (size_t i = 0, j = 0; i < t.size() && j < total_points; ++i) {
315 const Real t_start = t(i);
316 const Real dt = expo_time(i) / static_cast<Real>(num_points - 1);
317
318 for (size_t k = 0; k < num_points && j < total_points; ++k, ++j) {
319 t_obs(j) = t_start + k * dt;
320 nu_obs(j) = nu(i);
321 idx[j] = i;
322 }
323 }
324
325 std::vector<size_t> sort_indices(total_points);
326 std::iota(sort_indices.begin(), sort_indices.end(), 0);
327 std::ranges::sort(sort_indices, [&t_obs](size_t i, size_t j) { return t_obs(i) < t_obs(j); });
328
329 Array t_obs_sorted = Array::from_shape({total_points});
330 Array nu_obs_sorted = Array::from_shape({total_points});
331 std::vector<size_t> idx_sorted(idx.size());
332
333 for (size_t i = 0; i < sort_indices.size(); ++i) {
334 const size_t orig_idx = sort_indices[i];
335 t_obs_sorted(i) = t_obs(orig_idx);
336 nu_obs_sorted(i) = nu_obs(orig_idx);
337 idx_sorted[i] = idx[orig_idx];
338 }
339
340 t_obs_sorted *= unit::sec;
341 nu_obs_sorted *= unit::Hz;
342
343 return {std::move(t_obs_sorted), std::move(nu_obs_sorted), std::move(idx_sorted)};
344}

346void PyModel::average_exposure_flux(PyFlux& result, std::vector<size_t> const& idx_sorted, size_t original_size,
347 size_t num_points) {
348 auto average_component = [&](XTArray& component) {
349 if (component.size() > 0) {
350 Array summed = xt::zeros<Real>({original_size});
351 for (size_t j = 0; j < component.size(); j++) {
352 const size_t orig_time_idx = idx_sorted[j];
353 summed(orig_time_idx) += component(j);
354 }
355 summed /= static_cast<Real>(num_points);
356 component = std::move(summed);
357 }
358 };
359
360 average_component(result.fwd.sync);
361 average_component(result.fwd.ssc);
362 average_component(result.rvs.sync);
363 average_component(result.rvs.ssc);
364}

366auto PyModel::flux_density_exposures(PyArray const& t, PyArray const& nu, PyArray const& expo_time, size_t num_points)
367 -> PyFlux {
368 AFTERGLOW_REQUIRE(t.size() == nu.size() && t.size() == expo_time.size(),
369 "time, frequency, and exposure time arrays must have the same size");
370 AFTERGLOW_REQUIRE(num_points >= 2, "num_points must be at least 2 to sample within each exposure time");
371
372 const auto [t_obs_sorted, nu_obs_sorted, idx_sorted] = generate_exposure_sampling(t, nu, expo_time, num_points);
373
374 auto flux_func = [](Observer& obs, Array const& time, Array const& freq, auto& photons) -> XTArray {
375 return obs.specific_flux_series(time, freq, photons) / unit::flux_den_cgs;
376 };
377
378 auto result = compute_emission(t_obs_sorted, nu_obs_sorted, flux_func);
379
380 average_exposure_flux(result, idx_sorted, t.size(), num_points);
381
382 result.calc_total();
383 return result;
