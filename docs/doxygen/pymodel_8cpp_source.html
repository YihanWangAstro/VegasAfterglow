15Ejecta PyTophatJet(Real theta_c, Real E_iso, Real Gamma0, bool spreading, Real duration,
16 std::optional<PyMagnetar> magnetar) {
17 Ejecta jet;
18 jet.eps_k = [=](Real phi, Real theta) { return theta < theta_c ? E_iso : 0.; };
19 jet.Gamma0 = [=](Real phi, Real theta) { return theta < theta_c ? Gamma0 : 1.; };
20 jet.spreading = spreading;
21 jet.T0 = duration;
22
23 if (magnetar) {
24 jet.deps_dt = [=](Real phi, Real theta, Real t) {
25 if (theta <= theta_c) {
26 Real tt = 1 + t / magnetar->t_0;
27 return magnetar->L_0 * std::pow(tt, -magnetar->q);
28 } else {
29 return 0.;
30 }
31 };
32 }
33
34 return jet;
35}

37Ejecta PyGaussianJet(Real theta_c, Real E_iso, Real Gamma0, bool spreading, Real duration,
38 std::optional<PyMagnetar> magnetar) {
39 Ejecta jet;
40 Real norm = -1 / (2 * theta_c * theta_c);
41 jet.eps_k = [=](Real phi, Real theta) { return E_iso * fast_exp(norm * theta * theta); };
42 jet.Gamma0 = [=](Real phi, Real theta) { return (Gamma0 - 1) * fast_exp(norm * theta * theta) + 1; };
43 jet.spreading = spreading;
44 jet.T0 = duration;
45
46 if (magnetar) {
47 jet.deps_dt = [=](Real phi, Real theta, Real t) {
48 if (theta <= theta_c) {
49 Real tt = 1 + t / magnetar->t_0;
50 return magnetar->L_0 * std::pow(tt, -magnetar->q);
51 } else {
52 return 0.;
53 }
54 };
55 }
56
57 return jet;

82Ejecta PyTwoComponentJet(Real theta_n, Real E_iso_n, Real Gamma0_n, Real theta_w, Real E_iso_w, Real Gamma0_w,
83 bool spreading, Real duration, std::optional<PyMagnetar> magnetar) {
84 Ejecta jet;
85 jet.eps_k = [=](Real phi, Real theta) {
86 if (theta <= theta_n) {
87 return E_iso_n;
88 } else if (theta <= theta_w) {
89 return E_iso_w;
90 } else {
91 return 0.;
92 }
93 };
94
95 jet.Gamma0 = [=](Real phi, Real theta) {
96 if (theta <= theta_n) {
97 return Gamma0_n;
98 } else if (theta <= theta_w) {
99 return Gamma0_w;
100 } else {
101 return 1.0; // Default Lorentz factor outside the jet
102 }
103 };
104
105 jet.spreading = spreading;
106 jet.T0 = duration;
107
108 if (magnetar) {
109 jet.deps_dt = [=](Real phi, Real theta, Real t) {
110 if (theta <= theta_w) {
111 Real tt = 1 + t / magnetar->t_0;
112 return magnetar->L_0 * std::pow(tt, -magnetar->q);
113 } else {
114 return 0.;
115 }
116 };
117 }
118
119 return jet;

139void convert_unit(Ejecta& jet, Medium& medium) {
140 auto eps_k_cgs = jet.eps_k;
141 jet.eps_k = [=](Real phi, Real theta) { return eps_k_cgs(phi, theta) * (unit::erg / (4 * con::pi)); };
142
143 auto deps_dt_cgs = jet.deps_dt;
144 jet.deps_dt = [=](Real phi, Real theta, Real t) {
145 return deps_dt_cgs(phi, theta, t / unit::sec) * (unit::erg / (4 * con::pi * unit::sec));
146 };
147
148 auto dm_dt_cgs = jet.dm_dt;
149 jet.dm_dt = [=](Real phi, Real theta, Real t) {
150 return dm_dt_cgs(phi, theta, t / unit::sec) * (unit::g / (4 * con::pi * unit::sec));
151 };
152
153 jet.T0 *= unit::sec;
154
155 auto rho_cgs = medium.rho;
156 medium.rho = [=](Real phi, Real theta, Real r) {
157 return rho_cgs(phi, theta, r / unit::cm) * (unit::g / unit::cm3);
158 };
159
160 auto mass_cgs = medium.mass;
161 medium.mass = [=](Real phi, Real theta, Real r) { return mass_cgs(phi, theta, r / unit::cm) * unit::g; };
162}

164void PyModel::single_shock_emission(Shock const& shock, Coord const& coord, Array const& t_obs, Array const& nu_obs,
165 Observer& obs, PyRadiation rad, ArrayDict& flux_dict, std::string suffix,
166 bool serilized) {
167 obs.observe(coord, shock, obs_setup.lumi_dist, obs_setup.z);
168
169 auto syn_e = generate_syn_electrons(shock);
170
171 auto syn_ph = generate_syn_photons(shock, syn_e);
172
173 if (rad.IC_cooling) {
174 if (rad.KN) {
175 KN_cooling(syn_e, syn_ph, shock);
176 } else {
177 Thomson_cooling(syn_e, syn_ph, shock);
178 }
179 }
180
181 if (rad.SSC) {
182 auto IC_ph = generate_IC_photons(syn_e, syn_ph, rad.KN);
183
184 if (serilized) {
185 flux_dict["IC" + suffix] = obs.specific_flux_series(t_obs, nu_obs, IC_ph) / unit::flux_den_cgs;
186 } else {
187 flux_dict["IC" + suffix] = obs.specific_flux(t_obs, nu_obs, IC_ph) / unit::flux_den_cgs;
188 }
189 }
190
191 if (serilized) {
192 flux_dict["syn" + suffix] = obs.specific_flux_series(t_obs, nu_obs, syn_ph) / unit::flux_den_cgs;
193 } else {
194 flux_dict["syn" + suffix] = obs.specific_flux(t_obs, nu_obs, syn_ph) / unit::flux_den_cgs;
195 }

198auto PyModel::compute_specific_flux(Array const& t_obs, Array const& nu_obs, bool serilized) -> ArrayDict {
199 Coord coord = auto_grid(jet, t_obs, this->theta_w, obs_setup.theta_obs, obs_setup.z, phi_resol, theta_resol,
200 t_resol, axisymmetric);
201
202 ArrayDict flux_dict;
203
204 Observer observer;
205
206 if (!rvs_rad_opt) {
207 auto fwd_shock = generate_fwd_shock(coord, medium, jet, fwd_rad.rad, rtol);
208
209 single_shock_emission(fwd_shock, coord, t_obs, nu_obs, observer, fwd_rad, flux_dict, "", serilized);

212 } else {

215

217
218 single_shock_emission(rvs_shock, coord, t_obs, nu_obs, observer, rvs_rad, flux_dict, "_rvs", serilized);
219

349auto PyModel::specific_flux_series(PyArray const& t, PyArray const& nu) -> ArrayDict {
350 Array t_obs = t * unit::sec;
351 Array nu_obs = nu * unit::Hz;
352 bool serilized = true;
353
354 if (t_obs.size() != nu_obs.size()) {
355 throw std::invalid_argument(
356 "time and frequency arrays must have the same size\n"
357 "If you intend to get matrix-like output, use the generic `specific_flux` instead");
358 }
359
360 // Create sorted indices to handle random order
361 std::vector<size_t> sorted_indices(t_obs.size());
362 std::iota(sorted_indices.begin(), sorted_indices.end(), 0);
363 std::sort(sorted_indices.begin(), sorted_indices.end(),
364 [&t_obs](size_t i, size_t j) { return t_obs(i) < t_obs(j); });
365
366 // Create sorted arrays
367 Array t_sorted = xt::zeros<Real>({t_obs.size()});
368 Array nu_sorted = xt::zeros<Real>({nu_obs.size()});
369 for (size_t i = 0; i < sorted_indices.size(); ++i) {
370 t_sorted(i) = t_obs(sorted_indices[i]);
371 nu_sorted(i) = nu_obs(sorted_indices[i]);
372 }
373
374 // Compute flux with sorted arrays
375 ArrayDict sorted_flux_dict = compute_specific_flux(t_sorted, nu_sorted, serilized);
376
377 // Reorder results back to original order
378 ArrayDict flux_dict;
379 for (auto const& [key, sorted_flux] : sorted_flux_dict) {
380 Array reordered_flux = xt::zeros<Real>({sorted_flux.size()});
381 for (size_t i = 0; i < sorted_indices.size(); ++i) {
382 reordered_flux(sorted_indices[i]) = sorted_flux(i);
383 }
384 flux_dict[key] = reordered_flux;
385 }
386
387 return flux_dict;
388}
