129void convert_unit_jet(JetVariant& jet) {
130 std::visit(
131 [](auto& j) {
132 if constexpr (std::is_same_v<std::decay_t<decltype(j)>, Ejecta>) {
133 const auto eps_k_cgs = j.eps_k;
134 j.eps_k = [=](Real phi, Real theta) { return eps_k_cgs(phi, theta) * (unit::erg / (4 * con::pi)); };
135
136 const auto deps_dt_cgs = j.deps_dt;
137 j.deps_dt = [=](Real phi, Real theta, Real t) {
138 return deps_dt_cgs(phi, theta, t / unit::sec) * (unit::erg / (4 * con::pi * unit::sec));
139 };
140
141 const auto dm_dt_cgs = j.dm_dt;
142 j.dm_dt = [=](Real phi, Real theta, Real t) {
143 return dm_dt_cgs(phi, theta, t / unit::sec) * (unit::g / (4 * con::pi * unit::sec));
144 };
145
146 j.T0 *= unit::sec;
147 }
148 // TophatJet, GaussianJet, PowerLawJet are already constructed in internal units
149 },
150 jet);
151}

235void PyModel::single_evo_details(Shock const& shock, Coord const& coord, Observer& obs, PyRadiation const& rad,
236 PyShock& details) const {
237 obs.observe(coord, shock, obs_setup.lumi_dist, obs_setup.z);
238
239 details.t_obs = obs.time / unit::sec;
240 details.Doppler = xt::exp2(obs.lg2_doppler);
241
242 auto syn_e = generate_syn_electrons(shock, coord);
243
244 auto syn_ph = generate_syn_photons(shock, syn_e, coord);
245
246 if (rad.ssc) {
247 if (rad.kn) {
248 KN_cooling(syn_e, syn_ph, shock, coord, obs_setup.z);
249 } else {
250 Thomson_cooling(syn_e, syn_ph, shock, coord, obs_setup.z);
251 }
252 } else if (rad.rad.cmb_cooling) {
253 CMB_cooling(syn_e, syn_ph, shock, coord, obs_setup.z);
254 }
255 save_electron_details(syn_e, details);
256 save_photon_details(syn_ph, details, shock);
257
258 // Store photon grids for per-cell spectrum evaluation
259 details.syn_photons_ = syn_ph;
260 details.has_syn_spectrum_ = true;
261
262 if (rad.ssc) {
263 details.ic_photons_ = generate_IC_photons(syn_e, syn_ph, rad.kn, coord);
264 details.has_ssc_spectrum_ = true;
265 }

268auto PyModel::details(Real t_min, Real t_max) const -> PyDetails {
269 const Array t_obs = xt::logspace(std::log10(t_min * unit::sec), std::log10(t_max * unit::sec), 10);
270
271 PyDetails details;
272 Observer observer;
273
274 if (!rvs_rad_opt) {
275 auto [coord, fwd_shock] = solve_fwd_shock(jet_, medium_, t_obs, grid_config(32, 0.4), fwd_rad.rad, rtol);
276
277 details.phi = coord.phi;
278 details.theta = coord.theta;
279 details.t_src = coord.t / unit::sec;
280 save_shock_details(fwd_shock, details.fwd);
281 single_evo_details(fwd_shock, coord, observer, fwd_rad, details.fwd);
282 } else {
283 auto [coord, fwd_shock, rvs_shock] =
284 solve_shock_pair(jet_, medium_, t_obs, grid_config(36, 0.5), fwd_rad.rad, rvs_rad_opt->rad, rtol);
285
286 details.phi = coord.phi;
287 details.theta = coord.theta;
288 details.t_src = coord.t / unit::sec;
289 save_shock_details(fwd_shock, details.fwd);
290 save_shock_details(rvs_shock, details.rvs);
291 single_evo_details(fwd_shock, coord, observer, fwd_rad, details.fwd);
292 single_evo_details(rvs_shock, coord, observer, *rvs_rad_opt, details.rvs);
293 }
294 return details;

348auto PyModel::generate_exposure_sampling(PyArray const& t, PyArray const& nu, PyArray const& expo_time,
349 size_t num_points) -> ExposureSampling {
350 const size_t total_points = t.size() * num_points;
351 Array t_obs = Array::from_shape({total_points});
352 Array nu_obs = Array::from_shape({total_points});
353 std::vector<size_t> idx(total_points);
354
355 // Generate time-frequency samples within each exposure window
356 for (size_t i = 0, j = 0; i < t.size() && j < total_points; ++i) {
357 const Real t_start = t(i);
358 const Real dt = expo_time(i) / static_cast<Real>(num_points - 1);
359
360 for (size_t k = 0; k < num_points && j < total_points; ++k, ++j) {
361 t_obs(j) = t_start + k * dt;
362 nu_obs(j) = nu(i);
363 idx[j] = i;
364 }
365 }
366
367 std::vector<size_t> sort_indices(total_points);
368 std::iota(sort_indices.begin(), sort_indices.end(), 0);
369 std::ranges::sort(sort_indices, [&t_obs](size_t i, size_t j) { return t_obs(i) < t_obs(j); });
370
371 Array t_obs_sorted = Array::from_shape({total_points});
372 Array nu_obs_sorted = Array::from_shape({total_points});
373 std::vector<size_t> idx_sorted(idx.size());
374
375 for (size_t i = 0; i < sort_indices.size(); ++i) {
376 const size_t orig_idx = sort_indices[i];
377 t_obs_sorted(i) = t_obs(orig_idx);
378 nu_obs_sorted(i) = nu_obs(orig_idx);
379 idx_sorted[i] = idx[orig_idx];
380 }
381
382 t_obs_sorted *= unit::sec;
383 nu_obs_sorted *= unit::Hz;
384
385 return {std::move(t_obs_sorted), std::move(nu_obs_sorted), std::move(idx_sorted)};
386}

388void PyModel::average_exposure_flux(PyFlux& result, std::vector<size_t> const& idx_sorted, size_t original_size,
389 size_t num_points) {
390 auto average_component = [&](XTArray& component) {
391 if (component.size() > 0) {
392 Array summed = xt::zeros<Real>({original_size});
393 for (size_t j = 0; j < component.size(); j++) {
394 const size_t orig_time_idx = idx_sorted[j];
395 summed(orig_time_idx) += component(j);
396 }
397 summed /= static_cast<Real>(num_points);
398 component = std::move(summed);
399 }
400 };
401
402 average_component(result.fwd.sync);
403 average_component(result.fwd.ssc);
404 average_component(result.rvs.sync);
405 average_component(result.rvs.ssc);
406}

408auto PyModel::flux_density_exposures(PyArray const& t, PyArray const& nu, PyArray const& expo_time, size_t num_points)
409 -> PyFlux {
410 AFTERGLOW_REQUIRE(t.size() == nu.size() && t.size() == expo_time.size(),

412 AFTERGLOW_REQUIRE(num_points >= 2, "num_points must be at least 2 to sample within each exposure time");

418 };
419
420 auto result = compute_emission(t_obs_sorted, nu_obs_sorted, flux_func);
421
422 average_exposure_flux(result, idx_sorted, t.size(), num_points);

424 result.calc_total();
425 return result;
