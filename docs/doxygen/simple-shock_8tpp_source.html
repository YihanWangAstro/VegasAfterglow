11template <typename Ejecta, typename Medium>
12SimpleShockEqn<Ejecta, Medium>::SimpleShockEqn(Medium const& medium, Ejecta const& ejecta, Real phi, Real theta,
13 RadParams const& rad_params, Real theta_s)
14 : medium(medium),

16 phi(phi),
17 theta0(theta),
18 rad(rad_params),

27template <typename Ejecta, typename Medium>
28void SimpleShockEqn<Ejecta, Medium>::operator()(State const& state, State& diff, Real t) const noexcept {
29 const Real beta = physics::relativistic::gamma_to_beta(state.Gamma);
30
31 diff.r = compute_dr_dt(beta);
32 diff.t_comv = compute_dt_dt_comv(state.Gamma, beta);
33
34 if (ejecta.spreading && state.theta < 0.5 * con::pi) {
35 diff.theta = compute_dtheta_dt(theta_s, state.theta, diff.r, state.r, state.Gamma);
36 } else {
37 diff.theta = 0;
38 }
39
40 if constexpr (State::mass_inject) {
41 diff.m_jet = ejecta.dm_dt(phi, theta0, t);
42 }
43
44 if constexpr (State::energy_inject) {
45 diff.eps_jet = ejecta.deps_dt(phi, theta0, t);
46 }
47
48 Real rho = medium.rho(phi, state.theta, state.r);
49 diff.m2 = state.r * state.r * rho * diff.r;
50
51 const Real e_th = (state.Gamma - 1) * 4 * state.Gamma * rho * con::c2;
52 const Real eps_rad = compute_radiative_efficiency(state.t_comv, state.Gamma, e_th, rad);
53
54 diff.Gamma = dGamma_dt(eps_rad, state, diff);
55}

57template <typename Ejecta, typename Medium>
58Real SimpleShockEqn<Ejecta, Medium>::dGamma_dt(Real eps_rad, State const& state, State const& diff) const noexcept {
59 Real m_swept = state.m2;
60 Real m_jet = this->m_jet0;
61 Real dm_dt_swept = diff.m2;
62
63 if (ejecta.spreading) {
64 const Real f_spread = (1 - std::cos(state.theta)) / dOmega0;
65 dm_dt_swept = dm_dt_swept * f_spread + m_swept / dOmega0 * std::sin(state.theta) * diff.theta;
66 m_swept *= f_spread;
67 }
68
69 Real a1 = (1 - state.Gamma * state.Gamma) * dm_dt_swept;
70
71 if constexpr (State::energy_inject) {
72 a1 += diff.eps_jet / con::c2;
73 }
74
75 if constexpr (State::mass_inject) {
76 a1 -= state.Gamma * diff.m_jet;
77 m_jet = state.m_jet;
78 }
79
80 return a1 / (m_jet + eps_rad * m_swept + 2 * (1 - eps_rad) * state.Gamma * m_swept);
81}

83template <typename Ejecta, typename Medium>
84void SimpleShockEqn<Ejecta, Medium>::set_init_state(State& state, Real t0) const noexcept {
85 state.Gamma = ejecta.Gamma0(phi, theta0);
86
87 const Real beta0 = physics::relativistic::gamma_to_beta(state.Gamma);
88 state.r = beta0 * con::c * t0 / (1 - beta0);
89
90 state.t_comv = state.r / std::sqrt(state.Gamma * state.Gamma - 1) / con::c;
91
92 state.m2 = enclosed_mass([&](Real r_) { return medium.rho(phi, theta0, r_); }, state.r);
93
94 state.theta = theta0;
95
96 if constexpr (State::energy_inject) {
97 state.eps_jet = ejecta.eps_k(phi, theta0);
98 }
99
100 if constexpr (State::mass_inject) {
101 state.m_jet = m_jet0;
102 }
