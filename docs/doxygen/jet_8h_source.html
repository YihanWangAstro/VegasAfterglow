120 * <!-- ************************************************************************************** -->
121 * @class GaussianJet
122 * @brief Implements a Gaussian jet profile where properties follow a Gaussian distribution with angle.
123 * @details This class provides a smooth model for GRB jets, characterized by core angle theta_c,
124 * isotropic equivalent energy E_iso, and initial Lorentz factor Gamma0 at the center.
125 * <!-- ************************************************************************************** -->
126 */
127class GaussianJet {
128 public:
129 /**
130 * <!-- ************************************************************************************** -->
131 * @brief Constructor: Initialize with core angle, isotropic energy, and initial Lorentz factor
132 * @param theta_c Core angle of the jet
133 * @param E_iso Isotropic equivalent energy
134 * @param Gamma0 Initial Lorentz factor
135 * @param T0 Duration of the ejecta
136 * @param spreading Flag indicating if the ejecta spreads laterally during evolution
137 * <!-- ************************************************************************************** -->
138 */
139 GaussianJet(Real theta_c, Real E_iso, Real Gamma0, bool spreading = false, Real T0 = 1 * unit::sec) noexcept
140 : norm_(-1 / (2 * theta_c * theta_c)),
141 eps_k_(E_iso / (4 * con::pi)),
142 Gamma0_(Gamma0),
143 T0(T0),
144 spreading(spreading) {}
145
146 /**
147 * <!-- ************************************************************************************** -->
148 * @brief Energy per solid angle as a function of phi and theta, with Gaussian falloff
149 * @param phi Azimuthal angle (unused)
150 * @param theta Polar angle
151 * @return Energy per solid angle with Gaussian angular dependence
152 * <!-- ************************************************************************************** -->
153 */
154 inline Real eps_k(Real phi, Real theta) const noexcept { return eps_k_ * fast_exp(theta * theta * norm_); }
155
156 /**
157 * <!-- ************************************************************************************** -->
158 * @brief Initial Lorentz factor as a function of phi and theta, with Gaussian falloff
159 * @param phi Azimuthal angle (unused)
160 * @param theta Polar angle
161 * @return Lorentz factor with Gaussian angular dependence
162 * <!-- ************************************************************************************** -->
163 */
164 inline Real Gamma0(Real phi, Real theta) const noexcept {
165 return (Gamma0_ - 1) * fast_exp(theta * theta * norm_) + 1;
166 }
167
168 /// Duration of the ejecta in seconds
169 Real T0{1 * unit::sec};
170 /// Flag indicating if the ejecta spreads laterally during evolution
171 bool spreading{false};
172
173 private:
174 Real const norm_{0}; ///< Normalization factor for Gaussian distribution
175 Real const eps_k_{0}; ///< Peak energy per solid angle at center
176 Real const Gamma0_{1}; ///< Peak Lorentz factor at center
177};

201 : theta_c_(theta_c), eps_k_(E_iso / (4 * con::pi)), Gamma0_(Gamma0), k_(k), T0(T0), spreading(spreading) {}

205 * @brief Energy per solid angle as a function of phi and theta, with power-law falloff

208 * @return Energy per solid angle with power-law angular dependence
209 * <!-- ************************************************************************************** -->

211 inline Real eps_k(Real phi, Real theta) const noexcept { return eps_k_ / (1 + fast_pow(theta / theta_c_, k_)); }

232 Real const eps_k_{0}; ///< Energy per solid angle at the core

239 * @namespace math
240 * @brief Mathematical helper functions for constructing jet profiles.
241 * @details Provides a collection of functions for combining functions, constructing various injection profiles
242 * (e.g., tophat, Gaussian, power-law), and computing integrals. These functions are used to create
243 * complex jet profiles by combining spatial and temporal dependencies.
244 * <!-- ************************************************************************************** -->
245 */
246namespace math {
247 /**
248 * <!-- ************************************************************************************** -->
249 * @brief Combines a spatial function and a temporal function into one function of (phi, theta, t)
250 * @tparam F1 Type of the spatial function
251 * @tparam F2 Type of the temporal function
252 * @param f_spatial Function of (phi, theta)
253 * @param f_temporal Function of time
254 * @return Combined function where result is the product of both input functions
255 * <!-- ************************************************************************************** -->
256 */
257 template <typename F1, typename F2>
258 inline auto combine(F1 f_spatial, F2 f_temporal) noexcept {
259 return [=](Real phi, Real theta, Real t) noexcept { return f_spatial(phi, theta) * f_temporal(t); };
260 }
261
262 /**
263 * <!-- ************************************************************************************** -->
264 * @brief Creates a time-independent function from a spatial function by ignoring the time parameter t
265 * @tparam F1 Type of the spatial function
266 * @param f_spatial Function of (phi, theta)
267 * @return Function of (phi, theta, t) that ignores t and returns the result of f_spatial
268 * <!-- ************************************************************************************** -->
269 */
270 template <typename F1>
271 inline auto t_indep(F1 f_spatial) noexcept {
272 return [=](Real phi, Real theta, Real t) noexcept { return f_spatial(phi, theta); };
273 }

370/**
371 * <!-- ************************************************************************************** -->
372 * @brief Creates a Lorentz factor distribution using the Liang & Ghirlanda (2010) prescription
373 * @tparam F Type of the energy function
374 * @param energy_func Function that computes energy as a function of (phi, theta, t)
375 * @param e_max Maximum energy
376 * @param gamma_max Maximum Lorentz factor
377 * @param idx Power-law index relating energy to Lorentz factor
378 * @return Function that calculates Lorentz factor based on energy
379 * @details Returns a lambda function that computes a Lorentz factor using the Liang & Ghirlanda (2010)
380 * prescription. This is a commonly used model for the Lorentz factor distribution in GRB jets.
381 * The returned function calculates:
382 * e = energy_func(phi, theta, 0)
383 * u = (e / e_max)^idx * gamma_max
384 * return sqrt(1 + u^2)
385 * This creates a power-law relationship between the energy and Lorentz factor of the jet.
386 * <!-- ************************************************************************************** -->
387 */
388template <typename F>
389auto LiangGhirlanda2010(F energy_func, Real e_max, Real gamma_max, Real idx) {
390 return [=](Real phi, Real theta) noexcept {
391 // Get the energy at the given angle
392 Real e = energy_func(phi, theta);
393
394 // Calculate the velocity parameter u using power-law scaling
395 Real u = fast_pow(e / e_max, idx) * gamma_max;
396
397 // Convert to Lorentz factor
398 return std::sqrt(1 + u * u);
399 };
