17Real SmoothPowerLawSyn::log2_optical_thin(Real log2_nu) const noexcept {
18 if (log2_nu_m < log2_nu_c) {
19 return (log2_nu - log2_nu_m) / 3.0 +
20 log2_broken_power_ratio(log2_nu, log2_nu_m, diff_slope_m_slow_, smooth_m_slow_) +
21 log2_broken_power_ratio(log2_nu, log2_nu_c, diff_slope_c_slow_, smooth_c_slow_);
22 } else {

24 log2_broken_power_ratio(log2_nu, log2_nu_c, diff_slope_c_fast_, smooth_c_fast_) +
25 log2_broken_power_ratio(log2_nu, log2_nu_m, diff_slope_m_fast_, smooth_m_fast_);
26 }

29Real SmoothPowerLawSyn::log2_optical_thick(Real log2_nu) const noexcept {
30 const Real log2_x = log2_nu - log2_nu_m;
31 const Real s = -smooth_a_ * fast_exp2(2. / 3 * log2_x);
32
33 return 2.5 * log2_x + log2_softplus(-0.5 * log2_x + s);
34}

44Real SmoothPowerLawSyn::log2_optical_thin_sharp(Real log2_nu) const noexcept {
45 if (log2_nu_m < log2_nu_c) {
46 if (log2_nu < log2_nu_m) {

48 } else if (log2_nu < log2_nu_c) {

68Real SmoothPowerLawSyn::compute_log2_spectrum(Real log2_nu) const noexcept {
69 Real log2_f_thin = log2_optical_thin(log2_nu);
70 Real log2_f_thick = log2_optical_thick(log2_nu);

74void SmoothPowerLawSyn::build() noexcept {
75 log2_I_nu_max = fast_log2(I_nu_max);
76 log2_nu_m = fast_log2(nu_m);
77 log2_nu_c = fast_log2(nu_c);
78 log2_nu_a = fast_log2(nu_a);
79 log2_nu_M = fast_log2(nu_M);
80
81 // Uniform smoothing (spectral-index-dependent tuning disabled)
82 smooth_m_slow_ = 1;
83 smooth_c_slow_ = 1;
84 smooth_m_fast_ = 1;
85 smooth_c_fast_ = 1;
86 constexpr Real ln2 = std::numbers::ln2;
87 smooth_a_ = (3.5 * p - 1.5) / ln2;
88
89 // Precompute s*(beta_lo - beta_hi) for broken power law transitions
90 diff_slope_m_slow_ = smooth_m_slow_ * (0.5 * p - 1.0 / 6.0);
91 diff_slope_c_slow_ = smooth_c_slow_ * 0.5;
92 diff_slope_m_fast_ = smooth_m_fast_ * 0.5 * (p - 1.0);
93 diff_slope_c_fast_ = smooth_c_fast_ * 5.0 / 6.0;
94
95 inv_nu_M_ = 1.0 / nu_M;
96
97 // Compute normalization
98 if (nu_m < nu_c) {
99 log2_norm_ = 1.0 / smooth_m_slow_;
100 } else {
101 log2_norm_ = 1.0 / smooth_c_fast_;
102 }
103 log2_thick_norm_ = log2_optical_thin_sharp(log2_nu_a) - log2_optical_thick_sharp(log2_nu_a);
104}

106Real SmoothPowerLawSyn::compute_I_nu(Real nu) const noexcept {
107 if (nu <= nu_c) { // Below cooling frequency, simple scaling
108 return fast_exp(-nu * inv_nu_M_) * I_nu_max * compute_spectrum(nu);
109 } else {
110 return fast_exp(-nu * inv_nu_M_) * I_nu_max * compute_spectrum(nu) * inverse_compton_correction(*this, nu);
111 }
112}

114Real SmoothPowerLawSyn::compute_log2_I_nu(Real log2_nu) const noexcept {
115 constexpr Real log2e = std::numbers::log2e;
116 if (log2_nu <= log2_nu_c) { // Below cooling frequency, simple scaling
117 return log2_I_nu_max + compute_log2_spectrum(log2_nu) - log2e * inv_nu_M_ * fast_exp2(log2_nu);
118 } else {
119 const Real nu = fast_exp2(log2_nu);
120 return log2_I_nu_max + compute_log2_spectrum(log2_nu) - log2e * inv_nu_M_ * nu +
121 fast_log2(inverse_compton_correction(*this, nu));
122 }
