18/**
19 * <!-- ************************************************************************************** -->
20 * @struct PyMagnetar
21 * @brief Magnetar model for gamma-ray burst central engine energy injection.
22 * @details This class represents a magnetar spin-down model with characteristic luminosity, spin-down
23 * time scale, and power-law index. The magnetar provides continuous energy injection into
24 * the gamma-ray burst ejecta, following the relation: L(t) = L0 * (1 + t/t0)^(-q).
25 * This model is commonly used to explain X-ray plateaus and late-time energy injection
26 * in gamma-ray bursts.
27 * <!-- ************************************************************************************** -->
28 */
29struct PyMagnetar {
30 /**
31 * <!-- ************************************************************************************** -->
32 * @brief Construct a new PyMagnetar object with spin-down parameters.
33 * @details Initializes the magnetar model with the characteristic luminosity at the reference
34 * time, the spin-down time scale, and the power-law index for the energy injection.
35 * @param L0 Characteristic luminosity at reference time t = t0 [erg/s]
36 * @param t0 Spin-down time scale, time at which luminosity drops to L0/2^q [s]
37 * @param q Power-law index for spin-down evolution (typically 1-3, default 2)
38 * <!-- ************************************************************************************** -->
39 */
40 PyMagnetar(Real L0, Real t0, Real q = 2) : L0(L0), t0(t0), q(q) {}
41
42 Real L0; ///< Characteristic luminosity [erg/s]
43 Real t0; ///< Spin-down time scale [s]
44 Real q; ///< Power-law index for spin-down
45};

598template <typename Func>
599void PyModel::single_shock_emission(Shock const& shock, Coord const& coord, Array const& t_obs, Array const& nu_obs,
600 Observer& obs, PyRadiation rad, Flux& emission, Func&& flux_func) {
601 obs.observe(coord, shock, obs_setup.lumi_dist, obs_setup.z);
602
603 auto syn_e = generate_syn_electrons(shock);
604
605 auto syn_ph = generate_syn_photons(shock, syn_e);
606
607 if (rad.ssc_cooling) {
608 if (rad.kn) {
609 KN_cooling(syn_e, syn_ph, shock);
610 } else {
611 Thomson_cooling(syn_e, syn_ph, shock);
612 }
613 }
614
615 emission.sync = std::invoke(flux_func, obs, t_obs, nu_obs, syn_ph);
616
617 if (rad.ssc) {
618 auto IC_ph = generate_IC_photons(syn_e, syn_ph, rad.kn);
619 emission.ssc = std::invoke(flux_func, obs, t_obs, nu_obs, IC_ph);
620 }

624auto PyModel::compute_emission(Array const& t_obs, Array const& nu_obs, Func&& flux_func) -> PyFlux {
625 Coord coord = auto_grid(jet_, t_obs, this->theta_w, obs_setup.theta_obs, obs_setup.z, phi_resol, theta_resol,
626 t_resol, axisymmetric);
627
628 PyFlux flux;
629
630 Observer observer;
631
632 if (!rvs_rad_opt) {
633 auto fwd_shock = generate_fwd_shock(coord, medium_, jet_, fwd_rad.rad, rtol);
634
635 single_shock_emission(fwd_shock, coord, t_obs, nu_obs, observer, fwd_rad, flux.fwd,
636 std::forward<Func>(flux_func));
637
638 return flux;
639 } else {
640 auto rvs_rad = *rvs_rad_opt;
641 auto [fwd_shock, rvs_shock] = generate_shock_pair(coord, medium_, jet_, fwd_rad.rad, rvs_rad.rad, rtol);
642
643 single_shock_emission(fwd_shock, coord, t_obs, nu_obs, observer, fwd_rad, flux.fwd,
644 std::forward<Func>(flux_func));
645
646 single_shock_emission(rvs_shock, coord, t_obs, nu_obs, observer, rvs_rad, flux.rvs,
647 std::forward<Func>(flux_func));
648
649 return flux;
650 }
