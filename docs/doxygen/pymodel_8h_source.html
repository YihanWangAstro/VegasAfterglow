25 * <!-- ************************************************************************************** -->
26 * @struct PyMagnetar
27 * @brief Magnetar model for gamma-ray burst central engine energy injection.
28 * @details This class represents a magnetar spin-down model with characteristic luminosity, spin-down
29 * time scale, and power-law index. The magnetar provides continuous energy injection into
30 * the gamma-ray burst ejecta, following the relation: L(t) = L0 * (1 + t/t0)^(-q).
31 * This model is commonly used to explain X-ray plateaus and late-time energy injection
32 * in gamma-ray bursts.
33 * <!-- ************************************************************************************** -->
34 */
35struct PyMagnetar {
36 /**
37 * <!-- ************************************************************************************** -->
38 * @brief Construct a new PyMagnetar object with spin-down parameters.
39 * @details Initializes the magnetar model with the characteristic luminosity at the reference
40 * time, the spin-down time scale, and the power-law index for the energy injection.
41 * @param L0 Characteristic luminosity at reference time t = t0 [erg/s]
42 * @param t0 Spin-down time scale, time at which luminosity drops to L0/2^q [s]
43 * @param q Power-law index for spin-down evolution (typically 1-3, default 2)
44 * <!-- ************************************************************************************** -->
45 */
46 PyMagnetar(Real L0, Real t0, Real q = 2) : L0(L0), t0(t0), q(q) {}
47
48 Real L0; ///< Characteristic luminosity [erg/s]
49 Real t0; ///< Spin-down time scale [s]
50 Real q; ///< Power-law index for spin-down
51
52 [[nodiscard]] std::string repr() const {
53 char buf[128];
54 snprintf(buf, sizeof(buf), "Magnetar(L0=%.6g, t0=%.6g, q=%.6g)", L0, t0, q);
55 return buf;
56 }
57};

513/**
514 * <!-- ************************************************************************************** -->
515 * @class PyModel
516 * @brief Main gamma-ray burst afterglow model for comprehensive multi-wavelength predictions.
517 * @details This class provides the primary interface for gamma-ray burst afterglow modeling,
518 * combining jet structure, circumburst medium, observational setup, and radiation
519 * physics to compute multi-wavelength light curves, spectra, and internal shock
520 * evolution. It handles both forward and optional reverse shock emission with
521 * advanced radiation processes including synchrotron emission, synchrotron self-
522 * absorption, and inverse Compton scattering.
523 * <!-- ************************************************************************************** -->
524 */
525class PyModel {
526 public:
527 /**
528 * <!-- ************************************************************************************** -->
529 * @brief Construct complete afterglow model with physical and numerical parameters.
530 * @details Initializes the afterglow model by combining all physical components and
531 * setting numerical parameters for the simulation. The model can handle both
532 * axisymmetric and non-axisymmetric jet structures with customizable resolution.
533 * @param jet Ejecta object defining the jet structure and energy distribution
534 * @param medium Medium object representing the circumburst environment density profile
535 * @param observer Observer configuration including distance, redshift, and viewing angles
536 * @param fwd_rad Radiation parameters for forward shock microphysics
537 * @param rvs_rad Optional radiation parameters for reverse shock (default: none)
538 * @param resolutions Grid resolution tuple (phi_res, theta_res, time_res) in (deg⁻¹, deg⁻¹, decade⁻¹).
539 * The total grid points in each dimension are computed from these resolutions, but
540 * each has a minimum that cannot be reduced further: phi (min 1), theta (min 48), time (min 24).
541 * @param rtol Relative tolerance for numerical integration (default: 1×10⁻⁵)
542 * @param axisymmetric Whether to assume axisymmetric jet structure (default: true)
543 * <!-- ************************************************************************************** -->
544 */
545 PyModel(JetVariant jet, MediumVariant medium, PyObserver const& observer, PyRadiation const& fwd_rad,
546 std::optional<PyRadiation> const& rvs_rad = std::nullopt,

548 bool axisymmetric = true)
549 : jet_(std::move(jet)),
550 medium_(std::move(medium)),
551 obs_setup(observer),
552 fwd_rad(fwd_rad),
553 rvs_rad_opt(rvs_rad),
554 phi_resol(std::get<0>(resolutions)),
555 theta_resol(std::get<1>(resolutions)),
556 t_resol(std::get<2>(resolutions)),

563 /**
564 * <!-- ************************************************************************************** -->
565 * @brief Calculate specific flux at given times and frequencies.
566 * @details Computes the observed flux density at all combinations of specified times
567 * and frequencies, creating a complete flux grid. This is the primary method
568 * for generating light curves and spectra from the afterglow model.

780template <typename Func>
781void PyModel::single_shock_emission(Shock const& shock, Coord const& coord, Array const& t_obs, Array const& nu_obs,
782 Observer& obs, PyRadiation rad, Flux& emission, Func&& flux_func) {
783 {
784 AFTERGLOW_PROFILE_SCOPE(EAT_grid);
785 obs.observe(coord, shock, obs_setup.lumi_dist, obs_setup.z);
786 }
787
788 auto syn_e = [&] {
789 AFTERGLOW_PROFILE_SCOPE(syn_electrons);
790 return generate_syn_electrons(shock);
791 }();
792
793 auto syn_ph = [&] {
794 AFTERGLOW_PROFILE_SCOPE(syn_photons);
795 return generate_syn_photons(shock, syn_e);
796 }();
797
798 if (rad.ssc) {
799 AFTERGLOW_PROFILE_SCOPE(cooling);
800 if (rad.kn) {
801 KN_cooling(syn_e, syn_ph, shock);
802 } else {
803 Thomson_cooling(syn_e, syn_ph, shock);
804 }
805 }
806
807 {
808 AFTERGLOW_PROFILE_SCOPE(sync_flux);
809 emission.sync = std::invoke(flux_func, obs, t_obs, nu_obs, syn_ph);
810 }
811
812 if (rad.ssc) {
813 auto IC_ph = [&] {
814 AFTERGLOW_PROFILE_SCOPE(ic_photons);
815 return generate_IC_photons(syn_e, syn_ph, rad.kn, shock);
816 }();
817 {
818 AFTERGLOW_PROFILE_SCOPE(ssc_flux);
819 emission.ssc = std::invoke(flux_func, obs, t_obs, nu_obs, IC_ph);
820 }
821 }

824template <typename Func>
825auto PyModel::compute_emission(Array const& t_obs, Array const& nu_obs, Func&& flux_func) -> PyFlux {
826 AFTERGLOW_PROFILE_SCOPE(total);
827
828 PyFlux flux;
829 Observer observer;
830
831 if (!rvs_rad_opt) {
832 auto [coord, fwd_shock] = [&] {
833 AFTERGLOW_PROFILE_SCOPE(dynamics);
834 return solve_fwd_shock(jet_, medium_, t_obs, grid_config(32, 0.4), fwd_rad.rad, rtol);
835 }();
836 single_shock_emission(fwd_shock, coord, t_obs, nu_obs, observer, fwd_rad, flux.fwd,
837 std::forward<Func>(flux_func));
838 } else {
839 auto [coord, fwd_shock, rvs_shock] = [&] {
840 AFTERGLOW_PROFILE_SCOPE(dynamics);
841 return solve_shock_pair(jet_, medium_, t_obs, grid_config(36, 0.5), fwd_rad.rad, rvs_rad_opt->rad, rtol);
842 }();
843 single_shock_emission(fwd_shock, coord, t_obs, nu_obs, observer, fwd_rad, flux.fwd,
844 std::forward<Func>(flux_func));
845 single_shock_emission(rvs_shock, coord, t_obs, nu_obs, observer, *rvs_rad_opt, flux.rvs,
846 std::forward<Func>(flux_func));
847 }
848 return flux;
