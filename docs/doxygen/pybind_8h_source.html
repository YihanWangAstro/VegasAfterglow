36inline std::vector<size_t> logscale_screen(PyArray const& data, size_t num_order) {
37 const size_t total_size = data.size();
38
39 // Handle edge cases: empty or single-element arrays
40 if (total_size == 0) {
41 return {};
42 }
43 if (total_size == 1) {
44 return {0};
45 }
46
47 if (num_order == 0) {
48 std::vector<size_t> indices(total_size);
49 std::iota(indices.begin(), indices.end(), 0);
50 return indices;
51 }
52
53 const double log_start = std::log10(static_cast<double>(data(0)));
54 const double log_end = std::log10(static_cast<double>(data(total_size - 1)));
55 const double log_range = log_end - log_start;
56 const size_t total_points = static_cast<size_t>(std::ceil(log_range * static_cast<double>(num_order))) + 1;
57
58 std::vector<size_t> indices;
59 indices.reserve(total_points);
60
61 // Always include the first point
62 indices.push_back(0);
63
64 // Handle case where total_points <= 1 (avoid division by zero)
65 if (total_points <= 1) {
66 if (total_size > 1) {
67 indices.push_back(total_size - 1);
68 }
69 return indices;
70 }
71
72 const double step = log_range / static_cast<double>(total_points - 1);
73
74 for (size_t i = 1; i < total_points - 1; ++i) {
75 const double log_target = log_start + static_cast<double>(i) * step;
76 const double target_value = std::pow(10.0, log_target);
77
78 size_t best_idx = 1;
79 double min_diff = std::abs(static_cast<double>(data(1)) - target_value);
80
81 for (size_t j = 2; j < total_size - 1; ++j) {
82 const double diff = std::abs(static_cast<double>(data(j)) - target_value);
83 if (diff < min_diff) {
84 min_diff = diff;
85 best_idx = j;
86 }
87 }
88
89 if (std::ranges::find(indices, best_idx) == indices.end()) {
90 indices.push_back(best_idx);
91 }
92 }
93
94 // Always include the last point
95 if (total_size > 1) {
96 indices.push_back(total_size - 1);
97 }
98
99 std::ranges::sort(indices);
100 indices.erase(std::ranges::unique(indices).begin(), indices.end());
101
102 return indices;
103}
