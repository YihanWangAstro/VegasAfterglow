260/**
261 * <!-- ************************************************************************************** -->
262 * @brief Finds the root of a function using the bisection method
263 * @tparam Fun Type of the function
264 * @param f Function whose root we want to find
265 * @param low Lower bound of the search interval
266 * @param high Upper bound of the search interval
267 * @param eps Desired accuracy (default: 1e-6)
268 * @return Approximation of the root
269 * <!-- ************************************************************************************** -->
270 */
271template <typename Fun>
272auto root_bisect(Fun f, decltype(f(0)) low, decltype(f(0)) high, decltype(f(0)) eps = 1e-6) -> decltype(f(0)) {
273 using Scalar = decltype(f(0));
274 for (; (high - low) > std::fabs((high + low) * 0.5) * eps;) {
275 Scalar mid = 0.5 * (high + low);
276 if (f(mid) * f(high) > 0)
277 high = mid;
278 else
279 low = mid;
280 }
281 return 0.5 * (high + low);
282}
