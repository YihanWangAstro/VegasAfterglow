136/**
137 * <!-- ************************************************************************************** -->
138 * @brief Finds the root of a function using the bisection method
139 * @tparam Fun Type of the function
140 * @param f Function whose root we want to find
141 * @param low Lower bound of the search interval
142 * @param high Upper bound of the search interval
143 * @param eps Desired accuracy (default: 1e-6)
144 * @return Approximation of the root
145 * <!-- ************************************************************************************** -->
146 */
147template <typename Fun>
148auto root_bisect(Fun f, decltype(f(0)) low, decltype(f(0)) high, decltype(f(0)) eps = 1e-6) -> decltype(f(0)) {
149 using Scalar = decltype(f(0));
150 while ((high - low) > std::fabs((high + low) * 0.5) * eps) {
151 Scalar mid = 0.5 * (high + low);
152 if (f(mid) * f(high) > 0)
153 high = mid;
154 else
155 low = mid;
156 }
157 return 0.5 * (high + low);
158}

225/**
226 * <!-- ************************************************************************************** -->
227 * @brief Broadcasts computed values across grid based on shock symmetry.
228 * @tparam Array 3D array type (e.g., ICPhotonGrid or ElectronGrid)
229 * @tparam Shock Shock type containing symmetry information
230 * @param arr The 3D array to broadcast
231 * @param shock Shock object containing symmetry information and theta representatives
232 * @details This function handles the broadcasting of computed values from representative
233 * grid points to all grid points based on the detected symmetry level:
234 * - isotropic: Broadcasts from (0,0) to all (phi, theta)
235 * - piecewise: Broadcasts within theta groups, then across phi
236 * - phi_symmetric: Broadcasts only across phi direction
237 * <!-- ************************************************************************************** -->
238 */
239template <typename Array, typename Shock>
240void broadcast_symmetry(Array& arr, Shock const& shock) {
241 const size_t phi_size = arr.shape()[0];
242 const size_t theta_size = arr.shape()[1];
243
244 using SymmetryType = decltype(shock.symmetry);
245
246 if (shock.symmetry == SymmetryType::isotropic) {
247 for (size_t i = 0; i < phi_size; ++i)
248 for (size_t j = 0; j < theta_size; ++j)
249 if (i != 0 || j != 0)
250 xt::view(arr, i, j, xt::all()) = xt::view(arr, 0, 0, xt::all());
251 } else if (shock.symmetry == SymmetryType::piecewise) {
252 // Broadcast representative thetas to their groups
253 for (size_t r = 0; r < shock.theta_reps.size(); ++r) {
254 const size_t j_start = shock.theta_reps[r];
255 const size_t j_end = (r + 1 < shock.theta_reps.size()) ? shock.theta_reps[r + 1] : theta_size;
256 for (size_t j = j_start + 1; j < j_end; ++j)
257 xt::view(arr, 0, j, xt::all()) = xt::view(arr, 0, j_start, xt::all());
258 }
259 // Then broadcast phi=0 to all phi
260 for (size_t i = 1; i < phi_size; ++i)
261 for (size_t j = 0; j < theta_size; ++j)
262 xt::view(arr, i, j, xt::all()) = xt::view(arr, 0, j, xt::all());
263 } else if (shock.symmetry == SymmetryType::phi_symmetric) {
264 // Only broadcast phi direction
265 for (size_t i = 1; i < phi_size; ++i)
266 for (size_t j = 0; j < theta_size; ++j)
267 xt::view(arr, i, j, xt::all()) = xt::view(arr, 0, j, xt::all());
268 }
