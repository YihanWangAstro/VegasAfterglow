130/**
131 * <!-- ************************************************************************************** -->
132 * @brief Computes the Sedov length—a characteristic scale for blast wave deceleration
133 * @param E_iso Isotropic equivalent energy
134 * @param n_ism ISM number density
135 * @return Sedov length
136 * @details The Sedov length is a characteristic scale defined as the cube root of (E_iso / (ρc²)),
137 * where ρ is the ambient medium mass density
138 * <!-- ************************************************************************************** -->
139 */
140inline Real sedov_length(Real E_iso, Real n_ism) {
141 return std::cbrt(E_iso / (4 * con::pi / 3 * n_ism * con::mp * con::c2));
142}

147 * @param E_iso Isotropic equivalent energy
148 * @param n_ism ISM number density
149 * @param engine_dura Engine duration
150 * @return Reverse shock crossing radius
151 * <!-- ************************************************************************************** -->

153inline Real RS_crossing_radius(Real E_iso, Real n_ism, Real engine_dura) {
154 Real l = sedov_length(E_iso, n_ism);

193template <typename Ejecta>
194Real find_jet_edge(Ejecta const& jet, Real gamma_cut) {
195 if (jet.Gamma0(0, con::pi / 2) >= gamma_cut) {
196 return con::pi / 2; // If the Lorentz factor at pi/2 is above the cut, the jet extends to pi/2.
197 }
198 Real low = 0;
199 Real hi = con::pi / 2;
200 Real eps = 1e-9;
201 for (; hi - low > eps;) {
202 Real mid = 0.5 * (low + hi);
203 if (jet.Gamma0(0, mid) > gamma_cut) {
204 low = mid;
205 } else {
206 hi = mid;
207 }
208 }
209 return low;
210}

212template <typename Ejecta, typename Medium>
213Real jet_spreading_edge(Ejecta const& jet, Medium const& medium, Real phi, Real theta_min, Real theta_max, Real t0) {
214 Real step = (theta_max - theta_min) / 256;
215 Real theta_s = theta_min;
216 Real dp_min = 0;
217
218 for (Real theta = theta_min; theta <= theta_max; theta += step) {
219 // Real G = jet.Gamma0(phi, theta);
220 // Real beta0 = gamma_to_beta(G);
221 // Real r0 = beta0 * con::c * t0 / (1 - beta0);
222 // Real rho = medium.rho(phi, theta, 0);
223 Real th_lo = std::max(theta - step, theta_min);
224 Real th_hi = std::min(theta + step, theta_max);
225 Real dG = (jet.Gamma0(phi, th_hi) - jet.Gamma0(phi, th_lo)) / (th_hi - th_lo);
226 // Real drho = (medium.rho(phi, th_hi, r0) - medium.rho(phi, th_lo, r0)) / (th_hi - th_lo);
227 Real dp = dG; //(2 * G - 1) * rho * dG + (G - 1) * G * drho;
228
229 if (dp < dp_min) {
230 dp_min = dp;
231 theta_s = theta;
232 }
233 }
234 if (dp_min == 0) {
235 theta_s = theta_max;
236 }
237
238 return theta_s;
239}
