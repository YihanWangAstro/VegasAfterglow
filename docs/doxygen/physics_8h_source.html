132/**
133 * <!-- ************************************************************************************** -->
134 * @brief Computes the Sedov length—a characteristic scale for blast wave deceleration
135 * @param E_iso Isotropic equivalent energy
136 * @param n_ism ISM number density
137 * @return Sedov length
138 * @details The Sedov length is a characteristic scale defined as the cube root of (E_iso / (ρc²)),
139 * where ρ is the ambient medium mass density
140 * <!-- ************************************************************************************** -->
141 */
142inline Real sedov_length(Real E_iso, Real n_ism) {
143 return std::cbrt(E_iso / (4 * con::pi / 3 * n_ism * con::mp * con::c2));
144}

148 * @brief Returns the radius at which the reverse shock crosses, defined as the thick shell deceleration radius
149 * @param E_iso Isotropic equivalent energy

151 * @param engine_dura Engine duration
152 * @return Reverse shock crossing radius

218template <typename Ejecta>
219Real find_jet_edge(Ejecta const& jet, Real gamma_cut) {
220 if (jet.Gamma0(0, con::pi / 2) >= gamma_cut) {
221 return con::pi / 2; // If the Lorentz factor at pi/2 is above the cut, the jet extends to pi/2.
222 }
223 Real low = 0;
224 Real hi = con::pi / 2;
225 Real eps = 1e-9;
226 for (; hi - low > eps;) {
227 Real mid = 0.5 * (low + hi);
228 if (jet.Gamma0(0, mid) > gamma_cut) {
229 low = mid;
230 } else {
231 hi = mid;
232 }
233 }
234 return low;
235}

237template <typename Ejecta, typename Medium>
238Real jet_spreading_edge(Ejecta const& jet, Medium const& medium, Real phi, Real theta_min, Real theta_max, Real t0) {
239 Real step = (theta_max - theta_min) / 256;
240 Real theta_s = theta_min;
241 Real dp_min = 0;
242
243 for (Real theta = theta_min; theta <= theta_max; theta += step) {
244 // Real G = jet.Gamma0(phi, theta);
245 // Real beta0 = gamma_to_beta(G);
246 // Real r0 = beta0 * con::c * t0 / (1 - beta0);
247 // Real rho = medium.rho(phi, theta, 0);
248 Real th_lo = std::max(theta - step, theta_min);
249 Real th_hi = std::min(theta + step, theta_max);
250 Real dG = (jet.Gamma0(phi, th_hi) - jet.Gamma0(phi, th_lo)) / (th_hi - th_lo);
251 // Real drho = (medium.rho(phi, th_hi, r0) - medium.rho(phi, th_lo, r0)) / (th_hi - th_lo);
252 Real dp = dG; //(2 * G - 1) * rho * dG + (G - 1) * G * drho;
253
254 if (dp < dp_min) {
255 dp_min = dp;
256 theta_s = theta;
257 }
258 }
259 if (dp_min == 0) {
260 theta_s = theta_max;
261 }
262
263 return theta_s;
264}

266template <typename Ejecta>
267Array adaptive_theta_grid(Ejecta const& jet, Real theta_min, Real theta_max, Real theta_view, size_t theta_num) {
268 return xt::linspace(theta_min, theta_max, theta_num);
269 size_t num_samples = 500;
270 Array sample = xt::linspace(theta_min, theta_max, num_samples);
271 Array weights = xt::zeros<Real>({num_samples - 1});
272 Real sin_obs = std::sin(theta_view);
273 Real cos_obs = std::cos(theta_view);
274 Real cos_phi = 0;
275 for (size_t i = 1; i < num_samples; ++i) {
276 Real theta = 0.5 * (sample(i) + sample(i - 1));
277 Real gamma = jet.Gamma0(0, theta);
278 Real beta = gamma_to_beta(gamma);
279 // Real Doppler = 1 / (gamma * (1 - beta * std::cos(theta - theta_view)));
280 Real cosv = std::sin(theta) * cos_phi * sin_obs + std::cos(theta) * cos_obs;
281 Real Doppler = 1 / (gamma * (1 - beta * cosv));
282
283 // weights(i) = (Doppler * Doppler * Doppler * gamma * std::sin(theta));
284 weights(i) = (Doppler * std::sin(theta));
285 }
286
287 Array CFD = xt::zeros<Real>({num_samples});
288 CFD(0) = 0;
289 for (size_t i = 1; i < num_samples; ++i) {
290 CFD(i) = CFD(i - 1) + weights(i - 1) * (sample(i) - sample(i - 1));
291 }
292 CFD /= CFD(num_samples - 1);
293 auto targets = xt::linspace(0.0, 1.0, theta_num);
294 auto theta_new = xt::interp(targets, CFD, sample);
295 return theta_new;
296}

298template <typename Ejecta>
299Coord auto_grid(Ejecta const& jet, Array const& t_obs, Real theta_cut, Real theta_view, Real z, Real phi_resol,
300 Real theta_resol, Real t_resol, bool is_axisymmetric) {
301 // constexpr size_t min_grid_size = 24;
302 Coord coord;
303 coord.theta_view = theta_view;
304
305 size_t phi_num = std::max<size_t>(static_cast<size_t>(360 * phi_resol), 1);
306 coord.phi = xt::linspace(0., 2 * con::pi, phi_num); // Generate phi grid linearly spaced.
307
308 Real jet_edge = find_jet_edge(jet, con::Gamma_cut); // Determine the jet edge angle.
309 Real theta_min = 0;
310 Real theta_max = std::min(jet_edge, theta_cut);
311 size_t theta_num = std::max<size_t>(static_cast<size_t>((theta_max - theta_min) * 180 / con::pi * theta_resol), 24);
312 coord.theta = adaptive_theta_grid(jet, theta_min, theta_max, theta_view, theta_num); //
313
314 Real t_max = *std::max_element(t_obs.begin(), t_obs.end()); // Maximum observation time.
315 Real t_min = *std::min_element(t_obs.begin(), t_obs.end()); // Minimum observation time.
316 size_t t_num = std::max<size_t>(static_cast<size_t>(std::log10(t_max / t_min) * t_resol), 24);
317
318 size_t phi_size_needed = is_axisymmetric ? 1 : phi_num;
319 coord.t = xt::zeros<Real>({phi_size_needed, theta_num, t_num});
320 for (size_t i = 0; i < phi_size_needed; ++i) {
321 for (size_t j = 0; j < theta_num; ++j) {
322 Real b = gamma_to_beta(jet.Gamma0(coord.phi(i), coord.theta(j)));
323 Real theta_max = coord.theta(j) + theta_view;
324
325 Real t_start = 0.99 * t_min * (1 - b) / (1 - std::cos(theta_max) * b) / (1 + z);
326 Real t_end = 1.01 * t_max / (1 + z);
327 xt::view(coord.t, i, j, xt::all()) = xt::logspace(std::log10(t_start), std::log10(t_end), t_num);
328 }
329 }
330
331 return coord; // Construct coordinate object.
