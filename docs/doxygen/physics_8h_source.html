110/**
111 * <!-- ************************************************************************************** -->
112 * @brief Converts Lorentz factor (gamma) to a velocity fraction (beta)
113 * @param gamma Lorentz factor
114 * @return Velocity fraction (beta = v/c)
115 * <!-- ************************************************************************************** -->
116 */
117inline Real gamma_to_beta(Real gamma) {
118 return std::sqrt(gamma * gamma - 1) / gamma;
119}

121/**
122 * <!-- ************************************************************************************** -->
123 * @brief Computes adiabatic index as a function of the Lorentz factor
124 * @param gamma Lorentz factor
125 * @return Adiabatic index
126 * <!-- ************************************************************************************** -->
127 */
128inline Real adiabatic_idx(Real gamma) {
129 return 4.0 / 3.0 + 1 / (3 * gamma);
130}

132/**
133 * <!-- ************************************************************************************** -->
134 * @brief Computes the Sedov length—a characteristic scale for blast wave deceleration
135 * @param E_iso Isotropic equivalent energy
136 * @param n_ism ISM number density
137 * @return Sedov length
138 * @details The Sedov length is a characteristic scale defined as the cube root of (E_iso / (ρc²)),
139 * where ρ is the ambient medium mass density
140 * <!-- ************************************************************************************** -->
141 */
142inline Real sedov_length(Real E_iso, Real n_ism) {
143 return std::cbrt(E_iso / (4 * con::pi / 3 * n_ism * con::mp * con::c2));
144}

146/**
147 * <!-- ************************************************************************************** -->
148 * @brief Returns the radius at which the reverse shock crosses, defined as the thick shell deceleration radius
149 * @param E_iso Isotropic equivalent energy
150 * @param n_ism ISM number density
151 * @param engine_dura Engine duration
152 * @return Reverse shock crossing radius
153 * <!-- ************************************************************************************** -->
154 */
155inline Real RS_crossing_radius(Real E_iso, Real n_ism, Real engine_dura) {
156 const Real l = sedov_length(E_iso, n_ism);
157 return std::sqrt(std::sqrt(l * l * l * con::c * engine_dura));
158}
